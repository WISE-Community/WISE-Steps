  2 /**
  3  * The datagraphDispatcher catches events specific to authoring individual
  4  * datagraph steps and delegates them to the appropriate functions for
  5  * this view.
  6  */
  7 View.prototype.datagraphDispatcher = function(type,args,obj){
  8 	if(type=='datagraphDisplayOptionChanged'){
  9 		obj.DataGraphNode.displayOptionChanged();
 10 	} else if(type=='datagraphStartModeChanged'){
 11 		obj.DataGraphNode.startModeChanged();
 12 	} else if(type=='datagraphGraphOptionChanged'){
 13 		obj.DataGraphNode.graphOptionChanged(args[0]);
 14 	} else if(type=='datagraphPromptChanged'){
 15 		obj.DataGraphNode.updatePrompt();
 16 	} else if(type=='datagraphTableMetadataChanged'){
 17 		obj.DataGraphNode.tableMetadataChanged();
 18 	} else if(type=='datagraphEditableChanged'){
 19 		obj.DataGraphNode.editableChanged(args[0]);
 20 	} else if(type=='datagraphAddRow'){
 21 		obj.DataGraphNode.addRow();
 22 	} else if(type=='datagraphAddCol'){
 23 		obj.DataGraphNode.addCol();
 24 	} else if(type=='datagraphRemoveRow'){
 25 		obj.DataGraphNode.removeRow();
 26 	} else if(type=='datagraphRemoveCol'){
 27 		obj.DataGraphNode.removeCol();
 28 	} else if(type=='datagraphToggleSelected'){
 29 		obj.DataGraphNode.toggleSelected(args[0], args[1]);
 30 	} else if(type=='datagraphCellChanged'){
 31 		obj.DataGraphNode.cellChanged(args[0], args[1]);
 32 	} else if(type=='datagraphToggleEditable'){
 33 		obj.DataGraphNode.toggleEditable();
 34 	} else if(type=='datagraphInsertRowBefore'){
 35 		obj.DataGraphNode.insert(1,0);
 36 	} else if(type=='datagraphInsertRowAfter'){
 37 		obj.DataGraphNode.insert(1,1);
 38 	} else if(type=='datagraphInsertColBefore'){
 39 		obj.DataGraphNode.insert(2,0);
 40 	} else if(type=='datagraphInsertColAfter'){
 41 		obj.DataGraphNode.insert(2,1);
 42 	} else if(type=='datagraphStartOver'){
 43 		obj.DataGraphNode.startOver();
 44 	} else if(type=='datagraphSetQuantitative'){
 45 		obj.DataGraphNode.setQuantitative();
 46 	} else if(type=='datagraphSetQualitative'){
 47 		obj.DataGraphNode.setQualitative();
 48 	} else if(type=='datagraphAddIndependent'){
 49 		obj.DataGraphNode.addIndependent();
 50 	} else if(type=='datagraphRemoveIndependent'){
 51 		obj.DataGraphNode.removeIndependent();
 52 	} else if(type=='datagraphSetIndependent'){
 53 		obj.DataGraphNode.setIndependent();
 54 	} else if(type=='datagraphUnsetIndependent'){
 55 		obj.DataGraphNode.unsetIndependent();
 56 	} else if(type=='datagraphAddLabel'){
 57 		obj.DataGraphNode.addLabel();
 58 	} else if(type=='datagraphRemoveLabel'){
 59 		obj.DataGraphNode.removeLabel();
 60 	} else if(type=='datagraphSetLabel'){
 61 		obj.DataGraphNode.setLabel();
 62 	} else if(type=='datagraphUnsetLabel'){
 63 		obj.DataGraphNode.unsetLabel();
 64 	} else if(type=='datagraphBuildTable'){
 65 		obj.DataGraphNode.buildTable();
 66 	} else if(type=='datagraphCancelBuildTable'){
 67 		obj.DataGraphNode.cancelBuildTable();
 68 	} else if(type=='datagraphSubmitBuildTable'){
 69 		obj.DataGraphNode.submitBuildTable();
 70 	} else if(type=='datagraphGraphWidthChanged'){
 71 		obj.DataGraphNode.graphSizeChanged('Width');
 72 	} else if(type=='datagraphGraphHeightChanged'){
 73 		obj.DataGraphNode.graphSizeChanged('Height');
 74 	}
 75 };

114 /*
115  * add all the events to the vle so the vle will listen for these events
116  * and call the dispatcher function when the event is fired
117  */
118 for(var x=0; x<events.length; x++) {
119 	componentloader.addEvent(events[x], 'datagraphDispatcher');
120 };
