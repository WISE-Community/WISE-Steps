1 /**
  2  * Sets the MySystemNode type as an object of this view
  3  * @constructor
  4  * @author patrick lawler
  5  */
  6 View.prototype.MySystemNode = {};
  7 View.prototype.MySystemNode.commonComponents = ['Prompt', 'LinkTo'];
  8 	
  9 /**
 10  * Sets the view and content and then builds the page
 11  */
 12 View.prototype.MySystemNode.generatePage = function(view){
 13 	this.view = view;
 14 	this.content = this.view.activeContent.getContentJSON();
 15 	
 16 	this.buildPage();
 17 };
 18 
 19 /**
 20  * Get the array of common components which is an array with
 21  * string elements being the name of the common component
 22  */
 23 View.prototype.MySystemNode.getCommonComponents = function() {
 24 	return this.commonComponents;
 25 };
 26 
 27 /**
 28  * Builds the html elements needed to author a my system node
 29  */
 30 View.prototype.MySystemNode.buildPage = function(){
 31 	var parent = document.getElementById('dynamicParent');
 32 	
 33 	/* remove any old elements */
 34 	while(parent.firstChild){
 35 		parent.removeChild(parent.firstChild);
 36 	};
 37 	
 38 	/* create new elements */
 39 	var pageDiv = createElement(document, 'div', {id: 'dynamicPage', style:'width:100%;height:100%'});
 40 	var mainDiv = createElement(document, 'div', {id: 'mainDiv'});
 41 	var modulesDiv = createElement(document, 'div', {id: 'modulesDiv'});
 42 	var instructionsText = document.createTextNode("When entering image filenames, make sure to use the asset uploader on the main authoring page to upload your images.");
 43 	
 44 	/* append elements */
 45 	parent.appendChild(pageDiv);
 46 	pageDiv.appendChild(mainDiv);
 47 	mainDiv.appendChild(instructionsText);
 48 	mainDiv.appendChild(createBreak());
 49 	mainDiv.appendChild(document.createTextNode("Enter instructions -- text or html -- here."));
 50 	mainDiv.appendChild(createBreak());
 51 	mainDiv.appendChild(createElement(document, 'div', {id: 'promptContainer'}));
 52 	mainDiv.appendChild(createBreak());
 53 	mainDiv.appendChild(modulesDiv);
 54 	
 55 	this.generateModules();
 56 };
 57 
 58 View.prototype.MySystemNode.populatePrompt = function() {
 59 	$('#promptInput').val(this.content.prompt);
 60 };
 61 
 62 /**
 63  * Updates the html with the user entered prompt
 64  */
 65 View.prototype.MySystemNode.updatePrompt = function(){
 66 	this.content.prompt = document.getElementById('promptInput').value;
 67 	
 68 	/* fire source updated event */
 69 	this.view.eventManager.fire('sourceUpdated');
 70 };
 71 
 72 /**
 73  * Generates the modules creation elements
 74  */
 75 View.prototype.MySystemNode.generateModules = function(){
 76 	var parent = document.getElementById('modulesDiv');
 77 	
 78 	//remove old elements first
 79 	while(parent.firstChild){
 80 		parent.removeChild(parent.firstChild);
 81 	};
 82 	
 83 	parent.appendChild(createBreak());
 84 	
 85 	if(this.content.modules.length>0){
 86 		var modsText = document.createTextNode("Existing Modules");
 87 	} else {
 88 		var modsText = document.createTextNode("Create Modules");
 89 	};
 90 	
 91 	parent.appendChild(modsText);
 92 	parent.appendChild(createBreak());
 93 	
 94 	//create current mod elements
 95 	for(var a=0;a<this.content.modules.length;a++){
 96 		var modDiv = createElement(document, 'div', {id: 'modDiv_' + a});
 97 		var modText = document.createTextNode('Module');
 98 		var nameText = document.createTextNode("Name: ");
 99 		var imageText = document.createTextNode("Image: ");
100 		var nameInput = createElement(document, 'input', {id: 'nameInput_' + a, type: 'text', value: this.content.modules[a].name, onchange: 'eventManager.fire("mysystemFieldUpdated",["name","' + a + '"])'});
101 		var imageInput = createElement(document, 'input', {id: 'imageInput_' + a, type: 'text', value: this.content.modules[a].image, onchange: 'eventManager.fire("mysystemFieldUpdated",["image","' + a + '"])'});
102 		var removeButt = createElement(document, 'input', {type: 'button', id: 'removeButt', value: 'remove module', onclick: 'eventManager.fire("mysystemRemoveMod","' + a + '")'});
103 		
104 		parent.appendChild(modDiv);
105 		modDiv.appendChild(modText);
106 		modDiv.appendChild(createBreak());
107 		modDiv.appendChild(nameText);
108 		modDiv.appendChild(nameInput);
109 		modDiv.appendChild(createBreak());
110 		modDiv.appendChild(imageText);
111 		modDiv.appendChild(imageInput);
112 		modDiv.appendChild(createBreak());
113 		modDiv.appendChild(removeButt);
114 		modDiv.appendChild(createBreak());
115 		modDiv.appendChild(createBreak());
116 	};
117 	
118 	//create buttons to create new modules
119 	var createButt = createElement(document, 'input', {type:'button', value:'add new module', onclick:'eventManager.fire("mysystemAddNew")'});
120 	parent.appendChild(createButt);
121 };
122 
123 /**
124  * Updates a module's, at the given index, filed of the given name
125  * with the given value.
126  */
127 View.prototype.MySystemNode.fieldUpdated = function(name,ndx){
128 	this.content.modules[ndx][name] = document.getElementById(name + 'Input_' + ndx).value;
129 	
130 	/* for now, the icon is the same as the image */
131 	if(name=='image'){
132 		this.content.modules[ndx].icon = document.getElementById(name + 'Input_' + ndx).value;
133 	};
134 	
135 	/* fire source updated event */
136 	this.view.eventManager.fire('sourceUpdated');
137 };
138 
139 /**
140  * Removes a module from the modules
141  */
142 View.prototype.MySystemNode.removeMod = function(ndx){
143 	this.content.modules.splice(ndx, 1);
144 	this.generateModules();
145 	
146 	/* fire source updated event */
147 	this.view.eventManager.fire('sourceUpdated');
148 };
149 
150 /**
151  * Creates a new dummy module object and adds it to the mods array
152  */
153 View.prototype.MySystemNode.addNew = function(){
154 	this.content.modules.push({name:'', icon:'', image:'', xtype:'MySystemContainer', etype:'source', fields:{efficiency:'1'}});
155 	this.generateModules();
156 	
157 	/* fire source updated event */
158 	this.view.eventManager.fire('sourceUpdated');
159 };

161 /**
162  * Updates this content object when requested, usually when preview is to be refreshed
163  */
164 View.prototype.MySystemNode.updateContent = function(){
165 	/* update content object */
166 	this.view.activeContent.setContent(this.content);
167 };
