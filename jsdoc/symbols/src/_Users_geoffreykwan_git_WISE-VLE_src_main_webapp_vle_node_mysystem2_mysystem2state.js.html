  4  * This is the constructor for the state object that will be used to represent the
  5  * student work. An instance of this object will be created each time the student
  6  * submits an answer.
  7  *

  9  * is just used as an example. you can add any variables that will help you 
 10  * represent the student's work for your step type.
 11  * @constructor
 12  */
 13 function MYSYSTEM2STATE(response) {
 14   //remember the type to avoid corruption (i.e., loading a state saved by an openresponse WISE4 step)
 15   this.type = "MySystem2";
 16   
 17 	//the text response the student wrote
 18 	this.response = "";
 19 
 20 	if(response !== null) {
 21 		//set the response
 22 		this.response = response;
 23 	}
 24 }

 26 /**
 27  * This function is used to reload previous work the student submitted for the step.
 28  * The student work is retrieved and then this function is called to parse the student
 29  * work so that we can display the previous answer the student submitted.
 30  * 
 31  * note: you can change the variables in the stateJSONObj, the response 
 32  * variable is just used as an example. you can add any variables that will  
 33  * help you represent the student's work for your type of step.
 34  * 
 35  * @param stateJSONObj a JSONObject representing the student work
 36  * @return a MYSYSTEM2STATE object
 37  */
 38 MYSYSTEM2STATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 39 	//obtain the student work from the JSONObject
 40 	var response = stateJSONObj.response;
 41 	
 42 	//create a state object with the student work
 43 	var mysystemState = new MYSYSTEM2STATE(response);
 44 	
 45 	//return the state object
 46 	return mysystemState;
 47 };
