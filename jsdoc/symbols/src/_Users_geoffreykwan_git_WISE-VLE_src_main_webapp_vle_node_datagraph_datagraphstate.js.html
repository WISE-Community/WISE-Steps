1 /**
  2  * Datagraph state object
  3  * @constructor
  4  * @author patrick lawler
  5  */
  6 function DATAGRAPHSTATE(args){
  7 	this.timestamp = args[1];
  8 	this.table = args[0];
  9 	this.tableString;
 10 	
 11 	//set timestamp if not provided
 12 	if(!this.timestamp){
 13 		this.timestamp = Date.parse(new Date());
 14 	};
 15 	
 16 	//convert json table to string
 17 	if(this.table){
 18 		this.tableString = $.stringify(this.table);
 19 	};
 20 };
 21 
 22 /**
 23  * Given a @param stateJSONObj, creates, populates and returns a DATAGRAPHSTATE obj
 24  */
 25 DATAGRAPHSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 26 	return new DATAGRAPHSTATE([stateJSONObj.table, stateJSONObj.timestamp]);
 27 };
 28 
 29 /**
 30  * Returns the student edited table as a string
 31  */
 32 DATAGRAPHSTATE.prototype.getDataXML = function() {
 33 	return this.tableString + "<timestamp>" + this.timestamp + "</timestamp>";
 34 };
 35 
 36 /**
 37  * Returns the student edited table as a string
 38  */
 39 DATAGRAPHSTATE.prototype.getStudentWork = function() {
 40 	return this.tableString;
 41 }

 46 };
