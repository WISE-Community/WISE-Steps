 27 /**
 28  * This is the constructor for the state object that will be used to represent the
 29  * student work. An instance of this object will be created each time the student
 30  * submits an answer.
 31  * 
 32  * xTODO: rename TEMPLATESTATE
 33  * 
 34  * note: you can change the variables in this constructor, the response variable
 35  * is just used as an example. you can add any variables that will help you 
 36  * represent the student's work for your step type.
 37  * @constructor
 38  */
 39 function ExplanationBuilderState(explanationIdeas, answer, timestamp) {
 40 	this.explanationIdeas = [];
 41 	this.answer = '';
 42 	this.timestamp;
 43 	var selected = [];
 44 	
 45 	if(explanationIdeas != null) {
 46 		this.explanationIdeas = explanationIdeas;
 47 	}
 48 	
 49 	if(answer != null) {
 50 		this.answer = answer;
 51 	}
 52 	
 53 	if(timestamp != null) {
 54 		this.timestamp = timestamp;
 55 	}
 56 };

 72 	}
 73 };
