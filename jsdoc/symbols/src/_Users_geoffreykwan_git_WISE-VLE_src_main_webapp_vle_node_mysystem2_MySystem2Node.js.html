 81 /**
 82  * Renders the student work into the div. The grading tool will pass in a
 83  * div id to this function and this function will insert the student data
 84  * into the div.
 85  * 
 86  * @param divId the id of the div we will render the student work into
 87  * @param nodeVisit the student work
 88  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
 89  * div ids use this to prevent DOM conflicts such as when the show all work div
 90  * uses the same ids as the show flagged work div
 91  * @param workgroupId the id of the workgroup this work belongs to
 92  * 
 93  * Note: you may need to add code to this function if the student
 94  * data for your step is complex or requires additional processing.
 95  * look at SensorNode.renderGradingView() as an example of a step that
 96  * requires additional processing
 97  */
 98 Mysystem2Node.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
 99 	//Get the latest student state object for this step
100 	var MYSYSTEM2STATE = nodeVisit.getLatestWork();
101 	
102 	/*
103 	 * get the step work id from the node visit in case we need to use it in
104 	 * a DOM id. we don't use it in this case but I have retrieved it in case
105 	 * someone does need it. look at SensorNode.js to view an example of
106 	 * how one might use it.
107 	 */
108 	var stepWorkId = nodeVisit.id;
109 	
110 	// get student work (string)
111 	var studentWork = MYSYSTEM2STATE.getStudentWork().response;
112 	
113 	// get content
114     var content = this.getContent().getContentString();
115 
116 	// get content baseurl (e.g. 'http://localhost:8080/curriculum/897')
117 	var contentBaseUrl = this.view.config.getConfigParam('getContentBaseUrl');	
118 
119     // prepend contentbaseurl in front of "assets" to content
120     content = this.view.utils.prependContentBaseUrlToAssets(contentBaseUrl,content);
121     
122     // prepend contentbaseurl in front of "assets" to student's work
123     var studentWork = this.view.utils.prependContentBaseUrlToAssets(contentBaseUrl, studentWork);
124 	
125 	// put the student work and content in a hidden element
126     // add enlarge link to show student's diagram in a popup window
127 	var divContent = "<a class='msEnlarge' style='text-decoration:underline; color:blue;'" +
128 		"onclick='var newWindow=window.open(\"/vlewrapper/vle/node/mysystem2/mysystem2.html\"); newWindow.divId=\""+divId+"\"'>enlarge</a>" +
129     	"<span id='content_"+divId+"' style='display:none'>"+content+"</span>" +
130     	"<span id='studentwork_"+divId+"' style='display:none'>"+studentWork+"</span>";
131 	
132 	$('#' + divId).html(divContent);
133 };
