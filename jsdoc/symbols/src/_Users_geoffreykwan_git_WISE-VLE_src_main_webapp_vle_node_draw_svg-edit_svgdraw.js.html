 20 	this.defaultDescription = ""; // string to hold starting description text
 21 	this.instructions = ""; // string to hold prompt/instructions text

 44 SVGDRAW.prototype.loadModules = function(jsonfilename, context) {

 52 	}

 54 		context.defaultSnapshots = data.defaultSnapshots;
 55 	}
 56 	if(data.snapshots_active){
 57 		context.snapshotsActive = data.snapshots_active;
 58 	}
 59 	if(data.snapshots_max){
 60 		context.snapshots_max = data.snapshots_max;
 61 	}
 62 	if(data.description_active){
 63 		context.descriptionActive = data.description_active;
 64 	}
 65 	if(data.description_default) {
 66 		context.defaultDescription = data.description_default;
 67 	}
 68 	if(data.prompt){
 69 		context.instructions = data.prompt;
 70 	}
 71 	if(data.svg_background){
 72 		context.defaultImage = data.svg_background;
 73 	}
 74 	if(data.toolbar_options){
 75 		context.toolbarOptions = data.toolbar_options;
 76 	}
 77 	
 78 	var myDataService = new VleDS(vle);
 79  	// or var myDataService = new DSSService(read,write);
 80 	context.setDataService(myDataService);
 81 	context.load();   // load preview data, if any, or load default background
 82 };

191 	if(data.selected > -1){
192 		svgEditor.active = data.selected;
193 		svgEditor.selected = true;
194 		setTimeout(function(){
195 			for(var i=0; i<svgEditor.snapshots.length; i++){
196 				if(svgEditor.snapshots[i].id == svgEditor.active){
197 					svgEditor.index = i;
198 					svgEditor.warningStackSize = 0;
199 					var resetUndoStack = svgCanvas.getPrivateMethods().resetUndoStack;
200 					resetUndoStack();
201 					$("#tool_undo").addClass("tool_button_disabled").addClass("disabled");
202 					// scroll snap images panel depending on selected snapshot
203 					var page = Math.floor(svgEditor.index/3);
204 					$("#snap_images").attr({ scrollTop: page * 375 });
205 					svgEditor.snapCheck();
206 					break;
207 				}
208 				
209 			}
210 		},1000);
211 		svgEditor.initSnap = false;
212 	} else {
213 		$("#snap_images").attr({ scrollTop: 0 });

515 	  VleDS = function(_vle){
516 	    this.data = "";
517 	    this.annotations = "";
518 	    this.vle = _vle;
519 	    this.vleNode=_vle.getCurrentNode();
520 	    this.lz77 = new LZ77();
521 	  };
522 
523 	  VleDS.prototype = {
524 	    save: function(_data) {
525     		var data = _data;
526     		/* compress nodeState data */
527 			data = "--lz77--" + this.lz77.compress($.stringify(_data));
528 			this.vle.saveState(data,this.vleNode);
529 	        this.data = data;
530 	    },
531 
532 	    load: function(context,callback) {
533 	    	//this.data = this.vle.getLatestStateForCurrentNode();
534 			var data = this.vle.getLatestStateForCurrentNode();
535 			/* decompress nodeState data */
536 			if(typeof data == "string"){ // check whether data has been previously compressed (for backwards compatibilty)
537 				if (data.match(/^--lz77--/)) {
538 					//alert('match');
539 					data = data.replace(/^--lz77--/,"");
540 					data = $.parseJSON(this.lz77.decompress(data));
541 				}
542 			}
543 			this.data = data;
544 			callback(this.data,context);
545 	    },
546 	    
547 	    loadAnnotations: function(context,callback) {
548 	    	//this.annotations = this.vle.get
549 	    },
550 	    toString: function() {
551 	      return "VLE Data Service (" + this.vle + ")";
552 	    }
553 	  };
