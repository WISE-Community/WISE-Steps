 29 SensorNode.prototype.onExit = function() {
 30 	//check if the content panel has been set
 31 	if(this.contentPanel) {
 32 		
 33 		if(this.contentPanel.save) {
 34 			//tell the content panel to save
 35 			this.contentPanel.save();
 36 		}
 37 		
 38 		try {
 39 			/*
 40 			 * check if the onExit function has been implemented or if we
 41 			 * can access attributes of this.contentPanel. if the user
 42 			 * is currently at an outside link, this.contentPanel.onExit
 43 			 * will throw an exception because we aren't permitted
 44 			 * to access attributes of pages outside the context of our
 45 			 * server.
 46 			 */
 47 			if(this.contentPanel.onExit) {
 48 				try {
 49 					//run the on exit cleanup
 50 					this.contentPanel.onExit();					
 51 				} catch(err) {
 52 					//error when onExit() was called, e.g. mysystem editor undefined
 53 				}
 54 			}	
 55 		} catch(err) {
 56 			/*
 57 			 * an exception was thrown because this.contentPanel is an
 58 			 * outside link. we will need to go back in the history
 59 			 * and then trying to render the original node.
 60 			 */
 61 			history.back();
 62 			//setTimeout(function() {thisObj.render(this.ContentPanel)}, 500);
 63 		}
 64 	}
 65 };

 67 /**
 68  * Renders the student work into the div
 69  * @param divId the id of the div we will render the student work into
 70  * @param nodeVisit the student work
 71  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
 72  * div ids use this to prevent DOM conflicts such as when the show all work div
 73  * uses the same ids as the show flagged work div
 74  * @param workgroupId the id of the workgroup this work belongs to
 75  */
 76 SensorNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
 77 	//create a SENSOR object that we will use to perform all the graphing logic for us
 78 	var sensor = new SENSOR(this, this.view);
 79 	
 80 	if(childDivIdPrefix == null) {
 81 		//the default child div id prefix will be "" if none is provided
 82 		childDivIdPrefix = "";
 83 	}
 84 	
 85 	//get the step work id from the node visit
 86 	var stepWorkId = nodeVisit.id;
 87 	
 88 	/*
 89 	 * get the student work, in this case the student work is
 90 	 * the sensor state
 91 	 */
 92 	var sensorState = nodeVisit.getLatestWork();
 93 	
 94 	//set the sensor state into our sensor object
 95 	sensor.sensorState = sensorState;
 96 	
 97 	/*
 98 	 * get the data array from the sensor state in the format
 99 	 * that we can send to flot
100 	 */
101 	var graphDataArray = sensor.generateGraphDataArray(sensorState);
102 	
103 	//get the graph parameters from the content
104 	var graphParams = sensor.parseGraphParams(this.content.getContentJSON().graphParams);
105 
106 	//create the sensor graph div that we will use to display the graph
107 	var sensorGraphDiv = createElement(document, 'div', {id: childDivIdPrefix + 'sensorGraphDiv_' + stepWorkId, style:'width:400px;height:200px;'});
108 	
109 	//create the div that will display the check boxes to filter the lines (if this graph has multiple lines, if not, this will be empty)
110 	var sensorGraphCheckBoxesDiv = createElement(document, 'div', {id: childDivIdPrefix + 'sensorGraphCheckBoxesDiv_' + stepWorkId});
111 	
112 	//create the div that will display the student annotations for the graph
113 	var sensorAnnotationsDiv = createElement(document, 'div', {id: childDivIdPrefix + 'sensorAnnotationsDiv_' + stepWorkId});
114 	
115 	//create the response div that we will use to display what the student typed
116 	var sensorResponseDiv = createElement(document, 'div', {id: childDivIdPrefix + 'sensorResponseDiv_' + stepWorkId});
117 	
118 	//add all the divs to the main work div 
119 	$('#' + divId).append(sensorGraphDiv);
120 	$('#' + divId).append(sensorGraphCheckBoxesDiv);
121 	$('#' + divId).append(sensorAnnotationsDiv);
122 	$('#' + divId).append(sensorResponseDiv);
123 	
124 	//plot the graph in the sensor graph div
125 	sensor.plotData(sensorGraphDiv.id, sensorGraphCheckBoxesDiv.id);
126 	
127 	/*
128 	 * used to hide or show the annotation tool tips. if the teacher has
129 	 * their mouse in the graph div we will hide the annotation tool tips
130 	 * so that they don't block them from viewing the plot points.
131 	 * when the mouse cursor is outside of the graph div we will show the
132 	 * annotation tool tips for them to view.
133 	 */
134 	$("#" + sensorGraphDiv.id).bind('mouseover', (function(event) {
135 		$(".activeAnnotationToolTip").hide();
136 	}));
137 	$("#" + sensorGraphDiv.id).bind('mouseleave', (function(event) {
138 		$(".activeAnnotationToolTip").show();
139 	}));
140 	
141 	//get the annotations as a string
142 	var annotationsHtml = sensorState.getAnnotationsHtml();
143 	
144 	//set the annotations text
145 	$('#' + sensorAnnotationsDiv.id).html(annotationsHtml);
146 	
147 	//get the student response that was typed
148 	var response = sensorState.response;
149 	
150 	//replace \n with <br> so that the line breaks are displayed for the teacher
151 	response = this.view.replaceSlashNWithBR(response);
152 	
153 	//insert the response the student typed
154 	$('#' + sensorResponseDiv.id).html(response);
155 };
