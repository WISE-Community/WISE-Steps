  2 /**
  3  * Handles events triggered when a user is authoring a sensor step
  4  */
  5 View.prototype.sensorDispatcher = function(type,args,obj){
  6 	if(type=='sensorUpdateSensorType'){
  7 		obj.SensorNode.updateSensorType();
  8 	} else if(type=='sensorUpdateXUnits'){
  9 		obj.SensorNode.updateXUnits();
 10 	} else if(type=='sensorUpdateXMin'){
 11 		obj.SensorNode.updateXMin();
 12 	} else if(type=='sensorUpdateXMax'){
 13 		obj.SensorNode.updateXMax();
 14 	} else if(type=='sensorUpdateYUnits'){
 15 		obj.SensorNode.updateYUnits();
 16 	} else if(type=='sensorUpdateYMin'){
 17 		obj.SensorNode.updateYMin();
 18 	} else if(type=='sensorUpdateYMax'){
 19 		obj.SensorNode.updateYMax();
 20 	} else if(type=='sensorUpdateShowGraphOptions'){
 21 		obj.SensorNode.updateShowGraphOptions();
 22 	} else if(type=='sensorUpdateEnableCreatePrediction'){
 23 		obj.SensorNode.updateEnableCreatePrediction();
 24 	} else if(type=='sensorUpdateEnableSensor'){
 25 		obj.SensorNode.updateEnableSensor();
 26 	} else if(type=='sensorUpdateShowVelocity'){
 27 		obj.SensorNode.updateShowVelocity();
 28 	} else if(type=='sensorUpdateShowAcceleration'){
 29 		obj.SensorNode.updateShowAcceleration();
 30 	} else if(type=='sensorUpdateRequirePredictionBeforeEnter'){
 31 		obj.SensorNode.updateRequirePredictionBeforeEnter();
 32 	} else if(type=='sensorUpdateGraphTitle'){
 33 		obj.SensorNode.updateGraphTitle();
 34 	} else if(type=='sensorUpdateLockPredictionOnCollectionStart'){
 35 		obj.SensorNode.updateLockPredictionOnCollectionStart();
 36 	} else if(type=='sensorUpdateDataCollectionTimeLimit'){
 37 		obj.SensorNode.updateDataCollectionTimeLimit();
 38 	} else if(type=='sensorUpdateAllowUpdateAxisRange'){
 39 		obj.SensorNode.updateAllowUpdateAxisRange();
 40 	}
 41 };
 42 
 43 //this list of events
 44 var events = [
 45 	'sensorUpdateSensorType',
 46 	'sensorUpdateXUnits',
 47 	'sensorUpdateXMin',
 48 	'sensorUpdateXMax',
 49 	'sensorUpdateYUnits',
 50 	'sensorUpdateYMin',
 51 	'sensorUpdateYMax',
 52 	'sensorUpdateShowGraphOptions',
 53 	'sensorUpdateEnableCreatePrediction',
 54 	'sensorUpdateEnableSensor',
 55 	'sensorUpdateShowVelocity',
 56 	'sensorUpdateShowAcceleration',
 57 	'sensorUpdateRequirePredictionBeforeEnter',
 58 	'sensorUpdateGraphTitle',
 59 	'sensorUpdateLockPredictionOnCollectionStart',
 60 	'sensorUpdateDataCollectionTimeLimit',
 61 	'sensorUpdateAllowUpdateAxisRange'
 62 ];
 63 
 64 /*
 65  * add all the events to the vle so the vle will listen for these events
 66  * and call the dispatcher function when the event is fired
 67  */
 68 for(var x=0; x<events.length; x++) {
 69 	componentloader.addEvent(events[x], 'sensorDispatcher');
 70 };
 71 
 72 //used to notify scriptloader that this script has finished loading
 73 if(typeof eventManager != 'undefined'){
 74 	eventManager.fire('scriptLoaded', 'vle/node/sensor/sensorEvents.js');
 75 };
