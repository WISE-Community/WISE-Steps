 29 /**
 30  * This is the constructor for the object that will perform the logic for
 31  * the step when the students work on it. An instance of this object will
 32  * be created in the .html for this step (look at template.html)
 33  * 
 34  * TODO: rename Template
 35  * 
 36  * @constructor
 37  */
 38 function Template(node) {
 39 	this.node = node;
 40 	this.view = node.view;
 41 	this.content = node.getContent().getContentJSON();
 42 	
 43 	if(node.studentWork != null) {
 44 		this.states = node.studentWork; 
 45 	} else {
 46 		this.states = [];  
 47 	};
 48 };
 49 
 50 /**
 51  * This function renders everything the student sees when they visit the step.
 52  * This includes setting up the html ui elements as well as reloading any
 53  * previous work the student has submitted when they previously worked on this
 54  * step, if any.
 55  * 
 56  * TODO: rename Template
 57  * 
 58  * note: you do not have to use 'promptDiv' or 'studentResponseTextArea', they
 59  * are just provided as examples. you may create your own html ui elements in
 60  * the .html file for this step (look at template.html).
 61  */
 62 Template.prototype.render = function() {
 63 	//display any prompts to the student
 64 	$('#promptDiv').html(this.content.prompt);
 65 	
 66 	//load any previous responses the student submitted for this step
 67 	var latestState = this.getLatestState();
 68 	
 69 	if(latestState != null) {
 70 		/*
 71 		 * get the response from the latest state. the response variable is
 72 		 * just provided as an example. you may use whatever variables you
 73 		 * would like from the state object (look at templateState.js)
 74 		 */
 75 		var latestResponse = latestState.response;
 76 		
 77 		//set the previous student work into the text area
 78 		$('#studentResponseTextArea').val(latestResponse); 
 79 	}
 80 };
 81 
 82 /**
 83  * This function retrieves the latest student work
 84  * 
 85  * TODO: rename Template
 86  * 
 87  * @return the latest state object or null if the student has never submitted
 88  * work for this step
 89  */
 90 Template.prototype.getLatestState = function() {
 91 	var latestState = null;
 92 	
 93 	//check if the states array has any elements
 94 	if(this.states != null && this.states.length > 0) {
 95 		//get the last state
 96 		latestState = this.states[this.states.length - 1];
 97 	}
 98 	
 99 	return latestState;
100 };
101 
102 /**
103  * This function retrieves the student work from the html ui, creates a state
104  * object to represent the student work, and then saves the student work.
105  * 
106  * TODO: rename Template
107  * 
108  * note: you do not have to use 'studentResponseTextArea', they are just 
109  * provided as examples. you may create your own html ui elements in
110  * the .html file for this step (look at template.html).
111  */
112 Template.prototype.save = function() {
113 	//get the answer the student wrote
114 	var response = $('#studentResponseTextArea').val();
115 	
116 	/*
117 	 * create the student state that will store the new work the student
118 	 * just submitted
119 	 * 
120 	 * TODO: rename TemplateState
121 	 * 
122 	 * make sure you rename TemplateState to the state object type
123 	 * that you will use for representing student data for this
124 	 * type of step. copy and modify the file below
125 	 * 
126 	 * vlewrapper/WebContent/vle/node/template/templateState.js
127 	 * 
128 	 * and use the object defined in your new state.js file instead
129 	 * of TemplateState. for example if you are creating a new
130 	 * quiz step type you would copy the file above to
131 	 * 
132 	 * vlewrapper/WebContent/vle/node/quiz/quizState.js
133 	 * 
134 	 * and in that file you would define QuizState and therefore
135 	 * would change the TemplateState to QuizState below
136 	 */
137 	var templateState = new TemplateState(response);
138 	
139 	/*
140 	 * fire the event to push this state to the global view.states object.
141 	 * the student work is saved to the server once they move on to the
142 	 * next step.
143 	 */
144 	eventManager.fire('pushStudentWork', templateState);
145 
146 	//push the state object into this or object's own copy of states
147 	this.states.push(templateState);
148 };
149

161 }
