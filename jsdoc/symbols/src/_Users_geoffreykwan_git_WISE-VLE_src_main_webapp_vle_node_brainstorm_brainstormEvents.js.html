  2 /**
  3  * The brainstormDispatcher catches events specific to authoring individual
  4  * brainstorm steps and delegates them to the appropriate functions for
  5  * this view.
  6  */
  7 View.prototype.brainstormDispatcher = function(type,args,obj){
  8 	if(type=='brainstormUpdateExpectedLines'){
  9 		obj.BrainstormNode.updateExpectedLines();
 10 	} else if(type=='brainstromUpdateTitle'){
 11 		obj.BrainstormNode.updateTitle();
 12 	} else if(type=='brainstormUpdateGated'){
 13 		obj.BrainstormNode.updateGated(args[0]);
 14 	} else if(type=='brainstormUpdateDisplayName'){
 15 		obj.BrainstormNode.updateDisplayName(args[0]);
 16 	} else if(type=='brainstormUpdateRichText'){
 17 		obj.BrainstormNode.updateRichText(args[0]);
 18 	} else if(type=='brainstormUpdatePollEnded'){
 19 		obj.BrainstormNode.updatePollEnded(args[0]);
 20 	} else if(type=='brainstormUpdateInstantPoll'){
 21 		obj.BrainstormNode.updateInstantPoll(args[0]);
 22 	} else if(type=='brainstormStarterChanged'){
 23 		obj.BrainstormNode.starterChanged();
 24 	} else if(type=='brainstormStarterUpdated'){
 25 		obj.BrainstormNode.starterUpdated();
 26 	} else if(type=='brainstormUpdatePrompt'){
 27 		obj.BrainstormNode.updatePrompt();
 28 	} else if(type=='brainstormCreateNewResponse'){
 29 		obj.BrainstormNode.createNewResponse();
 30 	} else if(type=='brainstormRemoveResponse'){
 31 		obj.BrainstormNode.removeResponse();
 32 	} else if(type=='brainstormResponseNameChanged'){
 33 		obj.BrainstormNode.responseNameChanged(args[0]);
 34 	} else if(type=='brainstormResponseValueChanged'){
 35 		obj.BrainstormNode.responseValueChanged(args[0]);
 36 	} else if(type=='brainstormResponseSelected'){
 37 		obj.BrainstormNode.responseSelected(args[0]);
 38 	} else if(type=='brainstormUseServerUpdated') {
 39 		obj.BrainstormNode.useServerUpdated(args[0]);
 40 	};
 41 };
 42 
 43 //this list of events
 44 var events = [
 45 	'brainstormUpdateExpectedLines',
 46 	'brainstromUpdateTitle',
 47 	'brainstormUseServerUpdated',
 48 	'brainstormUpdateGated',
 49 	'brainstormUpdateDisplayName',
 50 	'brainstormUpdateRichText',
 51 	'brainstormUpdatePollEnded',
 52 	'brainstormUpdateInstantPoll',
 53 	'brainstormStarterChanged',
 54 	'brainstormStarterUpdated',
 55 	'brainstormUpdatePrompt',
 56 	'brainstormCreateNewResponse',
 57 	'brainstormRemoveResponse',
 58 	'brainstormResponseNameChanged',
 59 	'brainstormResponseValueChanged',
 60 	'brainstormResponseSelected'
 61 ];
 62 
 63 /*
 64  * add all the events to the vle so the vle will listen for these events
 65  * and call the dispatcher function when the event is fired
 66  */
 67 for(var x=0; x<events.length; x++) {
 68 	componentloader.addEvent(events[x], 'brainstormDispatcher');
 69 };
 70 
 71 //used to notify scriptloader that this script has finished loading
 72 if(typeof eventManager != 'undefined'){
 73 	eventManager.fire('scriptLoaded', 'vle/node/brainstorm/brainstormEvents.js');
 74 };
