1 /**
  2  * Object for storing state information of MultipleChoice item.
  3  * @constructor
  4  */
  5 function BRANCHSTATE(args) {
  6 	this.type = "branch";
  7 	this.isCorrect = null;
  8 	
  9 	//stores the human readable value of the choice chosen
 10 	this.response = new Array();
 11 
 12 	//use the current time or the argument timestamp passed in
 13 	if(args){
 14 		if(args[0]){
 15 			this.timestamp = args[0];
 16 		};
 17 		
 18 		if(args[1]){
 19 			this.choices = args[1];
 20 		};
 21 		
 22 		if(args[2]){
 23 			this.score = args[2];
 24 		}
 25 	};
 26 	
 27 	if(!this.timestamp){
 28 		this.timestamp = new Date().getTime();
 29 	};
 30 	
 31 	if(!this.choices){
 32 		this.choices = new Array();
 33 	};
 34 };
 35 
 36 /*
 37  * Add a choice that the student chose
 38  */
 39 BRANCHSTATE.prototype.addChoice = function(choice) {
 40 	this.choices.push(choice);
 41 };
 42 
 43 /*
 44  * Add the human readable value of the choice chosen
 45  */
 46 BRANCHSTATE.prototype.addResponse = function(response) {
 47 	this.response.push(response);
 48 };
 49 
 50 BRANCHSTATE.prototype.print = function() {
 51 };
 52 
 53 /**
 54  * Takes in a state JSON object and returns an BRANCHSTATE object
 55  * @param stateJSONObj a state JSON object
 56  * @return an BRANCHSTATE object
 57  */
 58 BRANCHSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 59 	//create a new BRANCHSTATE object
 60 	var branchState = new BRANCHSTATE();
 61 	
 62 	//set the attributes of the BRANCHSTATE object
 63 	branchState.isCorrect = stateJSONObj.isCorrect;
 64 	branchState.timestamp = stateJSONObj.timestamp;
 65 	branchState.choices = stateJSONObj.choices;
 66 	branchState.score = stateJSONObj.score;
 67 	
 68 	/*
 69 	 * an array containing the human readable value of the choice(s)
 70 	 * chosen by the student
 71 	 */
 72 	branchState.response = stateJSONObj.response;
 73 
 74 	//return the BRANCHSTATE object
 75 	return branchState;
 76 };
 77 
 78 /**
 79  * Returns human readable form of the choices the student chose
 80  */
 81 BRANCHSTATE.prototype.getHumanReadableForm = function() {
 82 	var humanReadableText = "isCorrect: " + this.isCorrect;
 83 	humanReadableText += "choices: " + this.choices;
 84 	humanReadableText += "score: " + this.score;
 85 	return humanReadableText;
 86 };
 87 
 88 /**
 89  * Returns the human readable choices the student chose
 90  * @return a string containing the human readable choices
 91  * 		the student chose. if the step is check box type
 92  * 		the choices chosen will be separated by a comma
 93  */
 94 BRANCHSTATE.prototype.getStudentWork = function() {
 95 	var studentWork = "";
 96 	
 97 	//check if there were any choices chosen
 98 	if(this.response) {
 99 		//loop through the array of choices
100 		for(var x=0; x<this.response.length; x++) {
101 			if(studentWork != "") {
102 				//separate each choice with a comma
103 				studentWork += ", ";
104 			}
105 			
106 			//add the choice to the student work
107 			studentWork += this.response[x];
108 		}
109 		
110 		if(this.score){
111 			studentWork += " score: " + this.score;
112 		}
113 	}
114 	
115 	return studentWork;
116 };
