  2 /**
  3  * @constructor
  4  * @param node
  5  * @param view
  6  * @returns
  7  */
  8 function MC(node, view) {
  9 	this.node = node;
 10 	this.view = view;
 11 	this.content = node.getContent().getContentJSON();
 12 	this.choices = [];
 13 	this.attempts = [];
 14 	this.stages = [];
 15 	
 16 	if(node.studentWork != null) {
 17 		this.states = node.studentWork;
 18 		
 19 		this.attempts = this.states.slice();
 20 	} else {
 21 		this.states = [];  
 22 	}
 23 	
 24 	/* add each choice object from the content to the choices array */
 25 	for(var a=0;a<this.content.assessmentItem.interaction.choices.length;a++){
 26 		this.choices.push(this.content.assessmentItem.interaction.choices[a]);
 27 	}
 28 	
 29 	//boolean to prevent shuffling after each answer submit
 30 	this.previouslyRendered = false;
 31 };

 45 /**
 46  * Loads state from the view
 47  */
 48 MC.prototype.loadState = function() {
 49 	for (var i=0; i < this.view.getVLEState().visitedNodes.length; i++) {
 50 		var nodeVisit = this.view.getVLEState().visitedNodes[i];
 51 		if (nodeVisit.getNodeId() == this.node.id) {
 52 			for (var j=0; j<nodeVisit.nodeStates.length; j++) {
 53 				this.states.push(nodeVisit.nodeStates[j]);
 54 			};
 55 		};
 56 	};
 57 };

 59 /**
 60  * Get the latest state
 61  * @returns the latest state
 62  */
 63 MC.prototype.getLatestState = function(){
 64 	var latestState = null;
 65 	
 66 	if(this.states && this.states.length>0){
 67 		latestState = this.states[this.states.length -1];
 68 	};
 69 
 70 	return latestState;
 71 };

 73 /**
 74  * Get the student's latest submission for this node that has work. 
 75  * The node is specific to a student.
 76  * @param nodeId the id of the node we want the student's work from
 77  * @return the newest NODE_STATE for this node
 78  */
 79 MC.prototype.getLatestStateFromNodeId = function(nodeId) {
 80 	var nodeVisits = this.view.getVLEState().getNodeVisitsByNodeId(nodeId);
 81 	
 82 	/*
 83 	 * loop through all the nodeVisits and find the latest nodeVisit
 84 	 * that has content in the nodeStates
 85 	 */
 86 	for(var x=0; x<nodeVisits.length; x++) {
 87 		//loop through the nodeVisits starting from the end
 88 		var nodeVisit = nodeVisits[nodeVisits.length - (x + 1)];
 89 		if(nodeVisit != null) {
 90 			//an array of nodeStates
 91 			var nodeStates = nodeVisit.nodeStates;
 92 			
 93 			//check if there is anything in the nodeStates
 94 			if(nodeStates != null && nodeStates.length > 0) {
 95 				//get the latest nodeState
 96 				var nodeState = nodeStates[nodeStates.length - 1];
 97 				return nodeState;
 98 			};
 99 		};
100 	};
101 	return null;
102 };

104 //gets and returns a choice object given the choice's identifier
105 MC.prototype.getChoiceByIdentifier = function(identifier) {
106 	for (var i=0;i<this.choices.length;i++) {
107 		if (this.removeSpace(this.choices[i].identifier) == identifier) {
108 			return this.choices[i];
109 		};
110 	};
111 	return null;
112 };

114 /**
115  * Render the MC
116  */
117 MC.prototype.render = function() {
118 	/* set the question type title */
119 	$('#questionType').html((this.node.getType()=='ChallengeNode') ? 'Challenge Question' : 'Multiple Choice');
120 	
121 	/* render the prompt */
122 	$('#promptDiv').html(this.content.assessmentItem.interaction.prompt);
123 
124 	/* remove buttons */
125 	var radiobuttondiv = document.getElementById('radiobuttondiv');
126 	while(radiobuttondiv.hasChildNodes()) {
127 		radiobuttondiv.removeChild(radiobuttondiv.firstChild);
128 	}
129 	
130 	/* 
131 	 * if shuffle is enabled, shuffle the choices when they enter the step
132 	 * but not each time after they submit an answer
133 	 */
134 	if(this.content.assessmentItem.interaction.shuffle && !this.previouslyRendered){
135 		this.choices.shuffle();
136 	}
137 	
138 	/* set variable whether this multiplechoice should be rendered with radio buttons or checkboxes */
139 	if(this.content.assessmentItem.interaction.maxChoices==1){
140 		var type = 'radio';
141 	} else {
142 		var type = 'checkbox';
143 	}
144 	
145 	/* render the choices */
146 	for(var i=0;i<this.choices.length;i++) {
147 		var choiceHTML = '<table><tbody><tr><td><input type="' + type + '" name="radiobutton" id="' + this.removeSpace(this.choices[i].identifier) + 
148 			'" value="' + this.removeSpace(this.choices[i].identifier) + '" class="' + type + '"/></td><td><div id="choicetext:' + this.removeSpace(this.choices[i].identifier) + 
149 			'">' + this.choices[i].text + '</div></td><td><div id="feedback_' + this.removeSpace(this.choices[i].identifier) + '" name="feedbacks"></div></td></tr></tbody></table>';
150 		
151 		$('#radiobuttondiv').append(choiceHTML);
152 		$('#' + this.removeSpace(this.choices[i].identifier)).click(function(){enableCheckAnswerButton('true');});
153 		
154 		if(this.selectedInSavedState(this.choices[i].identifier)){
155 			$('#' + this.removeSpace(this.choices[i].identifier)).attr('checked', true);
156 		}
157 	}
158 		
159 	//addClassToElement("checkAnswerButton", "disabledLink");
160 	$('#checkAnswerButton').parent().addClass('ui-state-disabled');
161 	//addClassToElement("tryAgainButton", "disabledLink");
162 	$('#tryAgainButton').parent().addClass('ui-state-disabled');
163 	clearFeedbackDiv();
164 	
165 	if (this.content.assessmentItem.responseDeclaration.correctResponse.length<1){
166 		// if there is no correct answer to this question (ie, when they're filling out a form),
167 		// change button to say "save answer" and "edit answer" instead of "check answer" and "try again"
168 		// and don't show the number of attempts.
169 		document.getElementById("checkAnswerButton").innerHTML = "Save Answer";
170 		document.getElementById("tryAgainButton").innerHTML = "Edit Answer";
171 	} else {
172 		displayNumberAttempts("This is your", "attempt", this.attempts);
173 	};
174 	
175 	//get the latest state
176 	var latestState = this.getLatestState();
177 	
178 	if(latestState != null && latestState.isCorrect) {
179 		//the student previously answered the question correctly
180 		
181 		//display the message that they correctly answered the question
182 		var resultMessage = this.getResultMessage(latestState.isCorrect);
183 		
184 		//check if scoring is enabled
185 		if(this.isChallengeScoringEnabled()) {
186 			//display the score they received
187 			var score = this.getScore(this.attempts.length);
188 			resultMessage += " You received " + score + " point(s).";
189 		}
190 		
191 		$('#resultMessageDiv').html(resultMessage);
192 	} else {
193 		//student has not answered this question correctly
194 		
195 		//check if challenge question is enabled
196 		if(this.isChallengeEnabled()) {
197 			//challenge question is enabled so we will create the constraint
198 			eventManager.fire('addConstraint',{type:'WorkOnXBeforeAdvancingConstraint', x:{id:this.node.id, mode:'node'}, id:this.node.utils.generateKey(20), workCorrect:true, buttonName:"Check Answer"});			
199 		}
200 	}
201 	
202 	if(this.node.getType()=='ChallengeNode') {
203 		//check if scoring is enabled
204 		if(this.isChallengeScoringEnabled()) {
205 			//display the current possible score table
206 			var scoreMessage = this.getCurrentPossibleScoreTableHtml();
207 			
208 			//display the score
209 			$('#scoreDiv').html(scoreMessage);				
210 		}
211 	}
212 	
213 	//turn this flag on so that the step does not shuffle again during this visit
214 	this.previouslyRendered = true;
215 	
216 	this.node.view.eventManager.fire('contentRenderComplete', this.node.id, this.node);
217 };

219 /**
220  * Get the table that displays the current possible scores
221  */
222 MC.prototype.getCurrentPossibleScoreTableHtml = function() {
223 	var html = "";
224 	
225 	//check if there is an attempts object
226 	if(this.content.assessmentItem.interaction.attempts != null) {
227 		
228 		//get the latest state
229 		var latestState = this.getLatestState();
230 		
231 		//get the scores
232 		var scores = this.content.assessmentItem.interaction.attempts.scores;
233 		
234 		//get the number of attempts the student has made
235 		var numAttempts = this.attempts.length;
236 		
237 		if(latestState != null && !latestState.isCorrect) {

244 		}
245 		
246 		//generate the current possible score table

312 /**
313  * Returns true if the choice with the given id is correct, false otherwise.

316 	/* if no correct answers specified by author, then always return true */
317 	if(this.content.assessmentItem.responseDeclaration.correctResponse.length==0){
318 		return true;
319 	};
320

324 			return true;
325 		};
326 	};
327 	return false;

330 /**
331  * Checks Answer and updates display with correctness and feedback
332  * Disables "Check Answer" button and enables "Try Again" button
333  */
334 MC.prototype.checkAnswer = function() {
335 	//if (hasClass("checkAnswerButton", "disabledLink")) {
336 	if ($('#checkAnswerButton').parent().hasClass('ui-state-disabled')) {
337 		return;
338 	}
339 
340 	//clear the previous result message
341 	$('#resultMessageDiv').html('');
342 	
343 	this.attempts.push(null);
344 	
345 	var radiobuttondiv = document.getElementById('radiobuttondiv');
346 	var inputbuttons = radiobuttondiv.getElementsByTagName('input');
347 	var mcState = (this.node.getType()=='MultipleChoiceNode') ? new MCSTATE() : (this.node.getType()=='BranchNode' ? new BRANCHSTATE() : new CHALLENGESTATE());
348 	var isCorrect = true;
349 	
350 	if(!this.enforceMaxChoices(inputbuttons)){
351 		return;
352 	}
353 	
354 	enableRadioButtons(false);        // disable radiobuttons
355 	//addClassToElement("checkAnswerButton", "disabledLink");
356 	$('#checkAnswerButton').parent().addClass('ui-state-disabled'); // disable checkAnswerButton
357 	//removeClassFromElement("tryAgainButton", "disabledLink");
358 	$('#tryAgainButton').parent().removeClass('ui-state-disabled'); // show try again button
359 	
360 	for (var i=0;i<inputbuttons.length;i++) {
361 		var checked = inputbuttons[i].checked;		
362 		var choiceIdentifier = inputbuttons[i].getAttribute('id');  // identifier of the choice that was selected
363 		// use the identifier to get the correctness and feedback
364 		var choice = this.getChoiceByIdentifier(choiceIdentifier);
365 
366 		if (checked) {
367 			if (choice) {
368 				document.getElementById('feedback_' + choiceIdentifier).innerHTML = choice.feedback;
369 
370 				if(this.node.getType()=='ChallengeNode') {
371 					//highlight the feedback in yellow if this is a challenge question
372 					$('#feedback_' + choiceIdentifier).css('background-color', 'yellow');
373 				}
374 				
375 				var choiceTextDiv = document.getElementById("choicetext:" + choiceIdentifier);
376 				if (this.isCorrect(choice.identifier)) {
377 					choiceTextDiv.setAttribute("class", "correct");
378 				} else {
379 					choiceTextDiv.setAttribute("class", "incorrect");
380 					isCorrect = false;
381 				}
382 				
383 				mcState.addChoice(choice.identifier);
384 				
385 				//add the human readable value of the choice chosen
386 				mcState.addResponse(choice.text);
387 			} else {
388 				this.node.view.notificationManager('error retrieving choice by choiceIdentifier', 3);
389 			}
390 		} else {
391 			if(this.isCorrect(choice.identifier)){
392 				isCorrect = false;
393 			}
394 		}
395 	}
396 	
397 	mcState.isCorrect = isCorrect;
398 	
399 	/* If this is a challenge node, we need to get the score and message associated
400 	 * with the number of attempts and whether this attempt is correct or not. We
401 	 * also need to disable the try again button. */
402 	if(this.node.getType()=='ChallengeNode'){
403 		
404 		if(this.isChallengeScoringEnabled()) {
405 			//get the scores
406 			var scores = this.content.assessmentItem.interaction.attempts.scores;
407 			
408 			//get the number of attempts the student has made
409 			var numAttempts = this.attempts.length;
410 			
411 			if(!isCorrect) {
412 				/*
413 				 * the student has not answered the question correctly so their
414 				 * possible current possible score is if they answer it on their
415 				 * next attempt
416 				 */ 
417 				numAttempts += 1;
418 			}
419 			
420 			//get the current possible score table
421 			var scoreMessage = getCurrentPossibleScoreTable(numAttempts, scores);
422 			
423 			//display the score
424 			$('#scoreDiv').html(scoreMessage);			
425 		}
426 		
427 		//get the challenge message
428 		var resultMessage = this.getResultMessage(isCorrect);
429 
430 		//check if scoring is enabled
431 		if(this.isChallengeScoringEnabled()) {
432 			if(isCorrect) {
433 				//get the score
434 				var score = this.getScore(this.attempts.length);
435 				
436 				/* add the score to the state */
437 				mcState.score = score;
438 				
439 				//display the score they received
440 				resultMessage += " You received " + score + " point(s).";
441 			} else {
442 				//student answered incorrectly
443 				mcState.score = 0;
444 			}
445 		}
446 		
447 		if(!isCorrect) {
448 			/*
449 			 * the student answered incorrectly so we will make the 
450 			 * background yellow since we will also be highlighting
451 			 * the associated step in the menu yellow
452 			 */
453 			resultMessage = "<table style='background-color:yellow' align='center'><tr><td>" + resultMessage + "</td></tr></table>";
454 		}
455 		
456 		/* set feedback message */
457 		$('#resultMessageDiv').html(resultMessage);
458 		
459 		/* disable the try again button */
460 		//$('#tryAgainButton').addClass('disabledLink');
461 		$('#tryAgainButton').parent().addClass('ui-state-disabled');
462 	} else if(this.node.getType()=='BranchNode'){
463 		/* this is a branch node, we will update the constraints so that students can
464 		 * navigate to the appropriate branch given their answer, we also cannot allow
465 		 * students to return and change their answer, so we need to create a constraint
466 		 * so that students won't be able to change anything on this step */
467 		this.node.view.eventManager.fire('addConstraint', {type:'NotVisitableXConstraint', x:{id:this.node.id, mode:'node'}, status:1, menuStatus:0, msg:'You can only answer this question once.'});
468 		
469 		/* remove the notvisitablex constraints for the appropriate branch based
470 		 * on the student response
471 		 *
472 		 * the value will have the spaces removed, so we want to get the value
473 		 * of the identifier as specified in the content. */
474 		var checkedId = this.resolveIdentifier($('#radiobuttondiv input:radio:checked').val());
475 		for(var v=0;v<this.content.branches.length;v++){
476 			/* if the determined choiceId is in the choiceIds of this branch, then
477 			 * we need to remove the specified constraints for this branch */
478 			if(this.content.branches[v].choiceIds.indexOf(checkedId) != -1){
479 				for(var w=0;w<this.content.branches[v].constraintIds.length;w++){
480 					this.node.view.eventManager.fire('removeConstraint', this.content.branches[v].constraintIds[w]);
481 				}
482 			}
483 		}
484 		
485 		/* disable the try again button */
486 		//$('#tryAgainButton').addClass('disabledLink');
487 		$('#tryAgainButton').parent().addClass('ui-state-disabled');
488 	} else if(isCorrect){
489 		//the student answered correctly
490 		
491 		//get the congratulations message and display it
492 		var resultMessage = this.getResultMessage(isCorrect);
493 		document.getElementById('resultMessageDiv').innerHTML = resultMessage;
494 	}
495 	
496 	//fire the event to push this state to the global view.states object
497 	eventManager.fire('pushStudentWork', mcState);
498 	
499 	//push the state object into this mc object's own copy of states
500 	this.states.push(mcState);
501 };

503 /**
504  * Returns true iff this.maxChoices is less than two or
505  * the number of checkboxes equals this.maxChoices. Returns
506  * false otherwise.
507  */
508 MC.prototype.enforceMaxChoices = function(inputs){
509 	var maxChoices = parseInt(this.content.assessmentItem.interaction.maxChoices);
510 	if(maxChoices>1){
511 		var countChecked = 0;
512 		for(var x=0;x<inputs.length;x++){
513 			if(inputs[x].checked){
514 				countChecked += 1;
515 			};
516 		};
517 		
518 		if(countChecked>maxChoices){
519 			this.node.view.notificationManager.notify('You have selected too many. Please select only ' + maxChoices + ' choices.',3);
520 			return false;
521 		} else if(countChecked<maxChoices){
522 			this.node.view.notificationManager.notify('You have not selected enough. Please select ' + maxChoices + ' choices.',3);
523 			return false;
524 		};
525 	};
526 	return true;
527 };

550 /**
551  * Given whether this attempt is correct, adds any needed linkTo and
552  * constraints and returns a message string.
553  * 
554  * @param boolean - isCorrect
555  * @return string - html response
556  */
557 MC.prototype.getResultMessage = function(isCorrect){
558 	/* we need to retrieve the attempt object corresponding to the current number of attempts */
559 	var attempt = this.content.assessmentItem.interaction.attempts;
560 	
561 	/* if this attempt is correct, then we only need to return a msg */
562 	if(isCorrect){
563 		return "You have successfully completed this question!";
564 	} else {
565 		/* this is not correct, so we need to set up a linkTo and constraint
566 		 * and return a message with the linkTo if a step has been specified
567 		 * to navigate to otherwise, we need to return an empty string */
568 		if(attempt.navigateTo && attempt.navigateTo != ''){
569 			var msg = 'Please review ';
570 			var position = this.node.view.getProject().getPositionById(attempt.navigateTo);
571 			var linkNode = this.node.view.getProject().getNodeById(attempt.navigateTo);
572 			var stepNumberAndTitle = this.node.view.getProject().getStepNumberAndTitle(attempt.navigateTo);
573 			
574 			/* create the linkTo and add it to the message */
575 			var linkTo = {key:this.node.utils.generateKey(),nodePosition:position};
576 			this.node.addLink(linkTo);
577 			msg += '<a style=\"color:blue;text-decoration:underline;font-weight:bold;cursor:pointer\" onclick=\"node.linkTo(\'' + linkTo.key + '\')\">Step ' + stepNumberAndTitle + '</a> before trying again.';
578 			
579 			//create the message that will display in the alert
580 			var optsMsg = 'You must visit "Step ' + stepNumberAndTitle + '" before trying this step again.';
581 			
582 			/* create the constraint to disable this step until students have gone to
583 			 * the step specified by this attempt */
584 			this.node.view.eventManager.fire('addConstraint', {type:'VisitXBeforeYConstraint', x:{id:attempt.navigateTo, mode:'node'}, y:{id:this.node.id, mode:'node'}, status: 1, menuStatus:0, effective: Date.parse(new Date()), id:this.node.utils.generateKey(20), msg:optsMsg});
585 			
586 			return msg;
587 		} else {
588 			return '';
589 		}
590 	}
591 };

602  * in the content. We need to do this because when setting ids in the html, we

604  * authoring tool may have included spaces.

612 			return this.choices[a].identifier;
