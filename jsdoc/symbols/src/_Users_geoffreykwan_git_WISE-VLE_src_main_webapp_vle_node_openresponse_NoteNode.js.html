 11 /**
 12  * @constructor
 13  * @extends Node
 14  * @param nodeType
 15  * @param view
 16  * @returns {NoteNode}
 17  */
 18 function NoteNode(nodeType, view) {
 19 	this.view = view;
 20 	this.type = nodeType;
 21 	this.prevWorkNodeIds = [];
 22 	this.studentWork;
 23 	this.exportableToNodes = new Array(			
 24 			"NoteNode", 
 25 			"OpenResponseNode", 
 26 			"SVGDrawNode");		
 27 };
 28 
 29 /**
 30  * Note nodes ignore the content panel and just renders in the global note panel.
 31  * Unlike other nodes, the note does not need to wait for scripts to load. The baseHtmlContent
 32  * for a note is just a div that gets written within another div, so the content should
 33  * be loaded during the render.
 34  * 
 35  * @param contentPanel
 36  * @param studentWork
 37  */
 38 NoteNode.prototype.render = function(contentPanel, studentWork){
 39 	this.studentWork = studentWork;
 40 	
 41 	/* set the baseHtmlContent if it has not yet been set up */
 42 	if(!this.baseHtmlContent){
 43 		this.baseHtmlContent = this.view.getHTMLContentTemplate(this);
 44 	}
 45 	
 46 	/* Set self as the active note node. Only one note can be active at a time */
 47 	$('#notePanel').html(this.injectBaseRef(this.injectKeystrokeManagerScript(this.view.injectVleUrl(this.baseHtmlContent.getContentString()))));
 48 	this.view.activeNote = new OPENRESPONSE(this, this.view);
 49 	
 50 	/* disable rich text editing for notes for now */
 51 	this.view.activeNote.content.isRichTextEditorAllowed = false;
 52 	
 53 	/* render the content for this node */
 54 	this.view.activeNote.render();
 55 	
 56 	/* show the note panel */
 57 	$('#notePanel').dialog('open');
 58 	
 59 	//apply iframe draggable fix
 60 	$(".ui-draggable").draggable( "option", "iframeFix", true );
 61 	$( ".ui-draggable" ).resizable( "option", "ghost", true );
 62 };
 63 
 64 /**
 65  * Called when the step is exited. This is used for auto-saving.
 66  */
 67 NoteNode.prototype.onExit = function() {
 68 	/* check if there is an active note and tell it to save */
 69 	if(this.view.activeNote) {
 70 		this.view.activeNote.save();
 71 	}
 72 };

 74 /**
 75  * Returns true iff this node can export work to the specified node.
 76  * @param exportToNode node to export work into
 77  * @return true/false
 78  */
 79 NoteNode.prototype.canExportWork = function(exportToNode) {
 80 	return this.exportableToNodes &&
 81 		this.exportableToNodes.indexOf(exportToNode.type) > -1;
 82 };
 83 
 84 /**
 85  * Returns a string of the work so that it can be imported by the specified exportToNode
 86  * @param exportToNode node that will import the return value of this method
 87  * @return null if this node cannot export work to the exportToNode
 88  */
 89 NoteNode.prototype.exportWork = function(exportToNode) {	
 90 	if (this.canExportWork(exportToNode)) {
 91 	    var nodeVisitArray = this.view.state.getNodeVisitsByNodeId(this.id);
 92 	    if (nodeVisitArray.length > 0) {
 93 	        var states = [];
 94 	        var latestNodeVisit = nodeVisitArray[nodeVisitArray.length -1];
 95 	        for (var i = 0; i < nodeVisitArray.length; i++) {
 96 	            var nodeVisit = nodeVisitArray[i];
 97 	            for (var j = 0; j < nodeVisit.nodeStates.length; j++) {
 98 	                states.push(nodeVisit.nodeStates[j]);
 99 	            }
100 	        }
101 	        var latestState = states[states.length - 1];
102 	        var studentWork = latestState.getStudentWork();
103 	        
104 	        if (exportToNode.type == "SVGDrawNode") {
105 	        	var svgString = '<text x="250" y="150" font-family="Verdana" font-size="35" fill="black" >'
106 				+ studentWork
107 				+ '</text>';
108 				return svgString;
109 	        } else {
110 	        	return studentWork;
111 	        }
112 	    };			
113 	};
114 	return null;
115 };
116 
117 /**
118  * Imports and inserts the work from the specified importFromNode
119  * @param importFromNode node that will export the data for this node to import
120  * @return
121  */
122 NoteNode.prototype.importWork = function(importFromNode) {
123 	var studentWork = importFromNode.exportWork(this);
124 	if (studentWork != null) {
125 		if(this.view && this.view.activeNote) {
126 			this.view.activeNote.appendResponse(studentWork);
127 		};
128 	};
129 };

131 NoteNode.prototype.getHTMLContentTemplate = function() {
132 	return createContent('node/openresponse/note.html');
133 };
