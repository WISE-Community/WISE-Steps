 11 /**
 12  * @constructor
 13  * @extends Node
 14  * @param nodeType
 15  * @param view
 16  * @returns {HtmlNode}
 17  */
 18 function HtmlNode(nodeType, view) {
 19 	this.view = view;
 20 	this.type = nodeType;
 21 	this.audios = [];
 22 	this.contentBase;
 23 	this.audioSupported = true;

 28 HtmlNode.prototype.setHtmlContent = function(htmlContent) {

 43 HtmlNode.prototype.exportNode = function() {
 44 	var exportXML = "";
 45 	
 46 	exportXML += this.exportNodeHeader();
 47 	
 48 	exportXML += "<content><![CDATA[";
 49 	exportXML += this.element.getElementsByTagName("content")[0].firstChild.nodeValue;
 50 	exportXML += "]]></content>";
 51 	
 52 	exportXML += this.exportNodeFooter();
 53 	
 54 	return exportXML;
 55 };
