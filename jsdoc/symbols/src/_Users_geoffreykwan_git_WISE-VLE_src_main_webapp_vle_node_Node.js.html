 62 /**
 63  * Retrieves the question/prompt the student reads for this step.
 64  * @return a string containing the prompt. (the string may be an
 65  * html string)
 66  */
 67 Node.prototype.getPrompt = function() {
 68 	var prompt = "";
 69 	
 70 	if(this.content != null) {
 71 		//get the content for the node
 72 		var contentJSON = this.content.getContentJSON();
 73 
 74 		if(contentJSON != null) {
 75 			//see if the node content has an assessmentItem
 76 			if(contentJSON.assessmentItem != null) {
 77 				//obtain the prompt
 78 				var assessmentItem = contentJSON.assessmentItem;
 79 				var interaction = assessmentItem.interaction;
 80 				prompt = interaction.prompt;	
 81 			}
 82 			// check if node is an SVGDrawNode
 83 			if(this.type=='SVGDrawNode'){
 84 			//obtain the prompt
 85 				if(contentJSON.prompt != null){
 86 					prompt = contentJSON.prompt;
 87 				}
 88 			}
 89 		}
 90 	}
 91 	
 92 	//return the prompt
 93 	return prompt;
 94 };

143 Node.prototype.getNodeById = function(nodeId) {
144 	if (this.id == nodeId) {
145 		return this;
146 	} else if (this.children.length == 0) {
147 		return null;
148 	} else {
149 		var soFar = false;
150 		for (var i=0; i < this.children.length; i++) {
151 			soFar = soFar || this.children[i].getNodeById(nodeId);
152 		};
153 		return soFar;
154 	};

190 /**
191  * Renders itself to the specified content panel
192  */
193 Node.prototype.render = function(contentPanel, studentWork, disable) {
194 	this.studentWork = studentWork;
195 	
196 	/* clean up any disabled panel that might exist from previous render */
197 	$('#disabledPanel').remove();
198 	
199 	/*check if the user had clicked on an outside link in the previous step
200 	 */
201 	if(this.handlePreviousOutsideLink(this, contentPanel)) {
202 		/*
203 		 * the user was at an outside link so the function
204 		 * handlePreviousOutsideLink() has taken care of the
205 		 * rendering of this node
206 		 */
207 		return;
208 	}
209 	
210 	/* if no content panel specified use default */
211 	if(contentPanel){
212 		/* make sure we use frame window and not frame element */
213 		this.contentPanel = window.frames[contentPanel.name];
214 	} else if(contentPanel == null) {
215 		/* use default ifrm */
216 		this.contentPanel = window.frames['ifrm'];
217 	}
218 
219 	/* 
220 	 * if node is not self rendering which means it is a node that
221 	 * requires an html file and a content file
222 	 */
223 	if(!this.selfRendering){
224 		/*
225 		 * check to see if this contentpanel has already been rendered
226 		 * also check if loadContentAfterScriptsLoad is null, this happens
227 		 * if the step contained a link to the internet and the student
228 		 * clicked on the link to load the page from the internet
229 		 */
230 		if(this.contentPanel.nodeId!=this.id || this.contentPanel.loadContentAfterScriptsLoad == null){
231 			if(!this.baseHtmlContent){
232 				this.baseHtmlContent = this.view.getHTMLContentTemplate(this);
233 			}
234 			
235 			/* make nodeId available to the content panel, and hence, the html */
236 			this.contentPanel.nodeId = this.id;
237 			
238 			/* inject urls and write html to content panel */
239 			this.contentPanel.document.open();
240 			this.contentPanel.document.write(this.injectBaseRef(this.injectKeystrokeManagerScript(this.baseHtmlContent.getContentString())));
241 			this.contentPanel.document.close();
242 		} else {
243 			/* already rendered, just load content */
244 			this.contentPanel.loadContentAfterScriptsLoad(this);
245 		}
246 	} else {
247 		/* if baseHtmlContent has not already been created, create it now */
248 		if(!this.baseHtmlContent){
249 			this.baseHtmlContent = createContent(this.view.getProject().makeUrl(this.content.getContentJSON().src, this));
250 			
251 			/* change filename url for the modules if this is a MySystemNode */
252 			if(this.type == 'MySystemNode'){
253 				this.baseHtmlContent.setContent(this.updateJSONContentPath(this.view.getConfig().getConfigParam('getContentBaseUrl'), this.baseHtmlContent.getContentString()));
254 			}
255 		}
256 		
257 		//write the content into the contentPanel, this will render the html in that panel
258 		this.contentPanel.document.open();
259 		this.contentPanel.document.write(this.injectBaseRef(this.injectKeystrokeManagerScript(this.baseHtmlContent.getContentString())));
260 		this.contentPanel.document.close();
261 	}
262 	
263 	if(this.contentPanel != null) {
264 		//set the event manager into the content panel so the html has access to it
265 		this.contentPanel.eventManager = eventManager;
266 		this.contentPanel.nodeId = this.id;
267 		this.contentPanel.node = this;
268 		this.contentPanel.scriptloader = this.view.scriptloader;
269 		
270 		if(this.type == 'MySystemNode' || this.type == 'SVGDrawNode' || 
271 				this.type == 'OpenResponseNode' || this.type == 'HtmlNode' ||
272 				this.type == 'MWNode') {
273 			this.contentPanel.vle = this.view;
274 		}
275 	}
276 	
277 	/* if there is a disable constraint, we want to set a semi-transparent panel over the content div */
278 	if(disable==1){
279 		/* get the position, height and width of the content panel */
280 		var panelPosition = $('#projectRightLowerBox').offset();
281 		var panelHeight = $('#projectRightLowerBox').height() + 2;
282 		var panelWidth = $('#projectRightLowerBox').width() + 2;
283 		
284 		/* create the disabledPanel and append it to the given document */
285 		var dynamicPanel = $('<div id="disabledPanel"></div>').css({opacity: 0.361, height:panelHeight, width:panelWidth, background:'#000', position:'absolute', 'z-index':999, top:panelPosition.top, left:panelPosition.left}).fadeIn(300);
286 		$('body').append(dynamicPanel);
287 	}
288 	
289 	if(this.view.config.getConfigParam('theme') == 'UCCP') {
290 		/*
291 		 * if this is a UCCP project then we will post the current step
292 		 * for BlueJ interaction purposes. we need to post the current step
293 		 * even when it is not a BlueJ step because we need to know when
294 		 * they are not on a BlueJ step.
295 		 */
296 		this.extraData = "";
297 		var blueJProjectPath = this.content.getContentJSON().blueJProjectPath;
298 		
299 		if(blueJProjectPath != null) {
300 			this.extraData = blueJProjectPath;
301 		}
302 		this.view.postCurrentStep(this);
303 	}
304 };

349 					}

351 				
352 				//add the constraint
353 				this.view.eventManager.fire('addConstraint',{type:'WorkOnXBeforeAdvancingConstraint', x:{id:this.id, mode:'node'}, id:this.utils.generateKey(20), updateAfterAdd: true, buttonName: buttonName});

504 /**
505  * Injects base ref in the head of the html if base-ref is not found, and returns the result
506  * @param content
507  * @return the content with the injected base ref
508  */
509 Node.prototype.injectBaseRef = function(content) {
510 	if (content.search(/<base/i) > -1) {
511 		// no injection needed because base is already in the html
512 		return content;
513 	} else {
514 		// NATE did this...  to check for node specific base urls
515 		// var contentBaseUrl = "";
516 		var cbu = "";   
517 		
518 		if (this.ContentBaseUrl) {
519 			// NATE screwed with this also
520 			cbu = this.ContentBaseUrl;
521 		} else if(this.type == "HtmlNode") {
522 			//get the content base url e.g. "http://localhost:8080/curriculum/667/"
523 			if(this.view.authoringMode) {
524 				//we are in the step authoring mode 
525 				cbu = this.getAuthoringModeContentBaseUrl();
526 			} else {
527 				//we are in the student vle
528 				cbu = this.view.getConfig().getConfigParam('getContentBaseUrl');					
529 			}
530 		} else {
531 			//set the content base url to the step type folder path 
532 			
533 			//get the content url e.g. "node/openresponse/openresponse.html"
534 			var contentUrl = this.baseHtmlContent.getContentUrl();
535 			
536 			//get the content url path e.g. "node/openresponse"
537 			contentUrlPath = contentUrl.substring(0, contentUrl.lastIndexOf('/'));
538 			
539 			//get the window location e.g. "http://localhost:8080/vlewrapper/vle/vle.html"
540 			var loc = window.location.toString();
541 			
542 			//get the vle location e.g. "http://localhost:8080/vlewrapper/vle/"
543 			var vleLoc = loc.substring(0, loc.indexOf('/vle/')) + '/vle/';
544 			
545 			//create the base href path e.g. "http://localhost:8080/vlewrapper/vle/node/openresponse/"
546 			cbu = vleLoc + contentUrlPath + '/';
547 		}
548 
549 		//add any missing html, head or body tags
550 		content = this.addMissingTags(content);
551 		
552 		//create the base tag
553 		var baseRefTag = "<base href='" + cbu + "'/>";
554 
555 		//get the content in all lowercase
556 		var contentToLowerCase = content.toLowerCase();
557 		
558 		//get the index of the open head tag
559 		var indexOfHeadOpenTag = contentToLowerCase.indexOf("<head>");
560 		
561 		var newContent = "";
562 		
563 		//check if there is an open head tag
564 		if(indexOfHeadOpenTag != -1) {
565 			//insert the base tag after the head open tag
566 			newContent = this.insertString(content, indexOfHeadOpenTag + "<head>".length, baseRefTag);
567 		} else {
568 			newContent = content;
569 		}
570 		
571 		//return the updated content
572 		return newContent;
573 	}
574 };

576 /**
577  * Add any missing html, head or body tags
578  * @param content the html content
579  * @return the html content with html, head, and body tags inserted
580  * if necessary
581  */
582 Node.prototype.addMissingTags = function(content) {
583 	/*
584 	 * add tags in this order for simplicity
585 	 * <body>
586 	 * </body>
587 	 * <head>
588 	 * </head>
589 	 * <html>
590 	 * </html>
591 	 */
592 	
593 	//check if the content contains the body open tag
594 	if(!this.containsBodyOpenTag(content)) {
595 		/*
596 		 * get the content all in lower case so we can search for the positions of tags
597 		 * by comparing them to lower case tags
598 		 */
599 		var contentToLowerCase = content.toLowerCase();
600 		
601 		//get the index of the html open tag
602 		var indexOfHtmlOpenTag = contentToLowerCase.indexOf("<html>");
603 		
604 		//get the index of the head close tag
605 		var indexOfHeadCloseTag = contentToLowerCase.indexOf("</head>");
606 		
607 		if(indexOfHeadCloseTag != -1) {
608 			//head close tag was found so we will insert '<body>' right after it
609 			content = this.insertString(content, indexOfHeadCloseTag + "</head>".length, "<body>");
610 		} else if(indexOfHtmlOpenTag != -1) {
611 			/*
612 			 * head close tag was not found and html open tag was found so we will 
613 			 * insert '<body>' right after the html open tag
614 			 */
615 			content = this.insertString(content, indexOfHtmlOpenTag + "<html>".length, "<body>");
616 		} else {
617 			/*
618 			 * html open and head close tags were not found so we will just add '<body>'
619 			 * to the beginning of the content
620 			 */
621 			content = "<body>" + content;
622 		}
623 	}
624 	
625 	//check if the content contains the body close tag
626 	if(!this.containsBodyCloseTag(content)) {
627 		/*
628 		 * get the content all in lower case so we can search for the positions of tags
629 		 * by comparing them to lower case tags
630 		 */
631 		var contentToLowerCase = content.toLowerCase();
632 		
633 		//get the index of the html close tag
634 		var indexOfHtmlCloseTag = contentToLowerCase.indexOf("</html>");
635 		
636 		if(indexOfHtmlCloseTag != -1) {
637 			//html close tag was found so we will insert '</body>' right before it
638 			content = this.insertString(content, indexOfHtmlCloseTag, "</body>");
639 		} else {
640 			//html close tag was not found so we will just add '</body>' to the end of the content
641 			content = content + "</body>";
642 		}
643 	}
644 
645 	//check if the content contains the head open tag
646 	if(!this.containsHeadOpenTag(content)) {
647 		/*
648 		 * get the content all in lower case so we can search for the positions of tags
649 		 * by comparing them to lower case tags
650 		 */
651 		var contentToLowerCase = content.toLowerCase();
652 		
653 		//get the index of the html open tag
654 		var indexOfHtmlOpenTag = contentToLowerCase.indexOf("<html>");
655 		
656 		if(indexOfHtmlOpenTag != -1) {
657 			//html open tag was found so we will insert '<head>' right after it
658 			content = this.insertString(content, indexOfHtmlOpenTag + "<html>".length, "<head>");
659 		} else {
660 			/*
661 			 * html open tag was not found so we will just add '<head>' to the
662 			 * beginning of the content
663 			 */
664 			content = "<head>" + content;			
665 		}
666 	}
667 	
668 	//check if the content contains the head close tag
669 	if(!this.containsHeadCloseTag(content)) {
670 		/*
671 		 * get the content all in lower case so we can search for the positions of tags
672 		 * by comparing them to lower case tags
673 		 */
674 		var contentToLowerCase = content.toLowerCase();
675 		
676 		//get the index of the body close tag (body open tag should always exist)
677 		var indexOfBodyOpenTag = contentToLowerCase.indexOf("<body>");
678 		
679 		if(indexOfBodyOpenTag != -1) {
680 			//we need to insert '</head>' right before the open body tag
681 			content = this.insertString(content, indexOfBodyOpenTag, "</head>");
682 		}
683 	}
684 
685 	//check if the content contains the html open tag
686 	if(!this.containsHtmlOpenTag(content)) {
687 		//add the html open tag to the beginning of the content
688 		content = "<html>" + content;
689 	}
690 	
691 	//check if the content contains the html close tag
692 	if(!this.containsHtmlCloseTag(content)) {
693 		//add the html close tag to the end of the content
694 		content = content + "</html>";
695 	}
696 	
697 	return content;
698 };

700 /**
701  * Insert a string into the content at the given position
702  * @param content the html content
703  * @param position the position to insert the string intot he content
704  * @param stringToInsert the string to insert into the content
705  * @return the content with the string inserted into it
706  */
707 Node.prototype.insertString = function(content, position, stringToInsert) {
708 	//get everything before the position
709 	var beginning = content.substring(0, position);
710 	
711 	//get everything after the position
712 	var end = content.substring(position);
713 	
714 	//combine everything with the stringToInsert inbetween
715 	var newContent = beginning + stringToInsert + end;
716 	
717 	return newContent;
718 };

720 /**
721  * Check if the tags in the array are all found in the content.
722  * If one of the tags are not found, we will return false.
723  * @param content the html content
724  * @param tags an array of html tags to search for
725  * @return true if we found all the tags, false if we are missing
726  * any tag
727  */
728 Node.prototype.containsTags = function(content, tags) {
729 	if(content != null) {
730 		//make the content lowercase so we can compare the lowercase tags
731 		var contentToLowerCase = content.toLowerCase();
732 		
733 		//loop through all the tags
734 		for(var x=0; x<tags.length; x++) {
735 			//get a tag
736 			var tag = tags[x];
737 			
738 			if(tag != null) {
739 				//make the tag lower case
740 				tag = tag.toLowerCase();
741 				
742 				//check if we found the tag 
743 				if(contentToLowerCase.indexOf(tag) == -1) {
744 					//we did not find the tag
745 					return false;
746 				}
747 			}
748 		}
749 	}
750 	
751 	//we found all the tags
752 	return true;
753 };
