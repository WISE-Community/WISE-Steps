  2 /**
  3  * Retrieves the previous work the student submitted
  4  * @return the latest student work
  5  */
  6 function getDrawingData() {
  7 	if(typeof vle != 'undefined'){
  8 		return vle.getLatestHtmlState();
  9 	};
 10 };
 11 
 12 /**
 13  * Obtain the drawing data from the applet and save it into
 14  * the vle state
 15  */
 16 function saveDrawingData() {
 17 	/*
 18 	 * obtain the drawing data from the applet. this is performed
 19 	 * by calling the Java function getDrawingData(). the applet
 20 	 * name is 'wisedraw' so we can access the applet from
 21 	 * document.wisedraw and then with a handle on the applet
 22 	 * we can call getDrawingData()
 23 	 */
 24 	var drawingData = document.wisedraw.getDrawingData();
 25 	
 26 	//save the drawing data into the vle state
 27 	if(typeof vle != 'undefined'){
 28 		vle.saveHtmlState(drawingData);
 29 	};
 30 };
 31 
 32 /**
 33  * This is called when the applet has loaded. It will set
 34  * the initial drawing data into the applet. We have to do it
 35  * this way because there is a bug for Mac OSX/Firefox which
 36  * returns null when we call getMember() from within Java, so
 37  * we can't pull the drawing data from the vle state from Javascript
 38  * to Java. Due to this, we need to push the drawing data from 
 39  * Javascript to Java by having Javascript set the drawing data
 40  * into the Java. The applet refreshes the drawing when this is
 41  * called so the image is updated.
 42  */
 43 function doneLoading() {
 44 	document.wisedraw.setInitialDrawingData(getDrawingData());
 45 };
 46 
 47 /**
 48  * Called when the step is exited such as when the user moves
 49  * to another step. This allows us to save the student drawing
 50  * when they leave the step, even if they haven't clicked on
 51  * the save button.
 52  */
 53 function onExit() {
 54 	saveDrawingData();
 55 };
