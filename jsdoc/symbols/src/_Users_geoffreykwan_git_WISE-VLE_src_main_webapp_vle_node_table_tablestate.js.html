 27 /**
 28  * This is the constructor for the state object that will be used to represent the
 29  * student work. An instance of this object will be created each time the student
 30  * submits an answer.
 31  * 
 32  * TODO: rename TEMPLATESTATE
 33  * 
 34  * note: you can change the variables in this constructor, the response variable
 35  * is just used as an example. you can add any variables that will help you 
 36  * represent the student's work for your step type.
 37  * 
 38  * @constructor
 39  */
 40 function TableState(response, tableData) {
 41 	//the text response the student wrote
 42 	this.response = "";
 43 	
 44 	//the table data that contains the values in the table as well as settings
 45 	this.tableData = [];
 46 
 47 	if(response != null) {
 48 		//set the response
 49 		this.response = response;
 50 	}
 51 	
 52 	if(tableData != null) {
 53 		//set the table data
 54 		this.tableData = tableData;
 55 	}
 56 };

 58 /**
 59  * This function is used to reload previous work the student submitted for the step.
 60  * The student work is retrieved and then this function is called to parse the student
 61  * work so that we can display the previous answer the student submitted.
 62  * 
 63  * TODO: rename TEMPLATESTATE
 64  * 
 65  * note: you can change the variables in the stateJSONObj, the response 
 66  * variable is just used as an example. you can add any variables that will  
 67  * help you represent the student's work for your type of step.
 68  * 
 69  * @param stateJSONObj a JSONObject representing the student work
 70  * @return a TEMPLATESTATE object
 71  */
 72 TableState.prototype.parseDataJSONObj = function(stateJSONObj) {
 73 	//obtain the student response from the JSONObject
 74 	var response = stateJSONObj.response;
 75 	
 76 	//obtain the student table data
 77 	var tableData = stateJSONObj.tableData;
 78 	
 79 	/*
 80 	 * create a state object with the student work
 81 	 * TODO: rename TEMPLATESTATE
 82 	 */
 83 	var tableState = new TableState(response, tableData);
 84 	
 85 	//return the state object
 86 	return tableState;
 87 };

102 /**
103  * Get the html for the table. This table is only for display purposes
104  * and will be completely static. The grading tool utilizes this function.
105  * @return the html for the table
106  */
107 TableState.prototype.getTableHtml = function() {
108 	var numRows = 0;
109 	var numColumns = 0;
110 	
111 	/*
112 	 * get the first column of the table data so we can find out
113 	 * how many rows are in the table
114 	 */
115 	var firstColumn = this.tableData[0];
116 	
117 	if(firstColumn != null) {
118 		//get the number of rows
119 		numRows = firstColumn.length;		
120 	}
121 	
122 	//get the number of columns
123 	numColumns = this.tableData.length;
124 	
125 	var tableHtml = "";
126 	
127 	//make the table
128 	tableHtml += "<table style='border-width:1px; border-style:outset'>"
129 	
130 	//loop through all the rows
131 	for(var y=0; y<numRows; y++) {
132 	
133 		tableHtml += "<tr>";
134 		
135 		//loop through all the columns
136 		for(var x=0; x<numColumns; x++) {
137 			tableHtml += "<td style='border-width:1px; border-style:inset; width:20px; height:20px'>";
138 			
139 			if(this.tableData != null && this.tableData[x] != null &&
140 					this.tableData[x][y] != null && this.tableData[x][y].text != null) {
141 				
142 				//get the text that we will display in the cell
143 				var cellText = this.tableData[x][y].text;
144 				
145 				if(cellText == null || cellText == '') {
146 					//set the text to   if it is null or empty string
147 					cellText = ' ';
148 				}
149 				
150 				//add the cell text
151 				tableHtml += cellText;
152 			}
153 			
154 			tableHtml += "</td>";
155 		}
156 		
157 		tableHtml += "</tr>";
158 	}
159 	
160 	tableHtml += "</table>";
161 	
162 	return tableHtml;
163 };
