1 /**
  2  * @constructor
  3  * @param args
  4  * @returns
  5  */
  6 function ASSESSMENTLISTSTATE(args) {
  7 	this.type = "assessmentlist";
  8 	this.submit = false;
  9 	this.assessments = new Array();
 10 
 11 	//use the current time or the argument timestamp passed in
 12 	if(args){
 13 		if(args[0]){
 14 			this.timestamp = args[0];
 15 		};
 16 		
 17 		if(args[1]){
 18 			this.choices = args[1];
 19 		};
 20 	};
 21 	
 22 	if(!this.timestamp){
 23 		this.timestamp = new Date().getTime();
 24 	};
 25 	
 26 };
 27 
 28 /**
 29  * Takes in a state JSON object and returns an ASSESSMENTLISTSTATE object
 30  * @param stateJSONObj a state JSON object
 31  * @return an ASSESSMENTLISTSTATE object
 32  */
 33 ASSESSMENTLISTSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 34 	//create a new ASSESSMENTLISTSTATE object
 35 	var alState = new ASSESSMENTLISTSTATE();
 36 	
 37 	//set the attributes of the ASSESSMENTLISTSTATE object
 38 	alState.assessments = stateJSONObj.assessments;
 39 	alState.timestamp = stateJSONObj.timestamp;
 40 	alState.submit = stateJSONObj.submit;
 41 	alState.isSubmit = stateJSONObj.isSubmit;
 42 	alState.locked = stateJSONObj.locked;
 43 	
 44 	//return the MCSTATE object
 45 	return alState;
 46 };
 47 
 48 /**
 49  * Returns the human readable student's answer
 50  * @return a string containing the human readable answer
 51  */
 52 ASSESSMENTLISTSTATE.prototype.getStudentWork = function(showAutoScoreResult) {
 53 	var studentWorkSoFar = "";
 54 	var autoScoreTotalScore = 0;   // total auto scored points the student earned
 55 	var autoScoreTotalMaxScore = 0;   // total auto scored points possible
 56 	
 57 	//check if there were any responses
 58 	if(this.assessments) {
 59 		//loop through the array of assessments
 60 		for(var x=0; x<this.assessments.length; x++) {
 61 			if(studentWorkSoFar != "") {
 62 				//separate each response
 63 				studentWorkSoFar += "<br/><br/>";
 64 			}
 65 			
 66 			//add the response to the student work
 67 			studentWorkSoFar += "Part " + (x+1) + ": <br/>";
 68 			var assessment = this.assessments[x];
 69 
 70 			if (assessment.type && assessment.response) {
 71 				if (assessment.type == "radio") {
 72 					studentWorkSoFar += assessment.response.text;
 73 					if (assessment.response.autoScoreResult && showAutoScoreResult) {
 74 						// append results from auto score
 75 						var autoScoreResult = assessment.response.autoScoreResult;
 76 						studentWorkSoFar += "<br/>Auto Score Results:<br/>";
 77 						if (autoScoreResult.isCorrect) {
 78 							studentWorkSoFar += "Student got this question CORRECT";
 79 						} else {
 80 							studentWorkSoFar += "Student got this question INCORRECT";							
 81 						}
 82 						var studentScore = autoScoreResult.choiceScore ? autoScoreResult.choiceScore : 0;
 83 						var maxScore = autoScoreResult.maxScore ? autoScoreResult.maxScore : 0;
 84 						studentWorkSoFar += " and received ";
 85 						studentWorkSoFar += studentScore;
 86 						studentWorkSoFar += " points out of ";
 87 						studentWorkSoFar += maxScore;
 88 						
 89 						// update total scores
 90 						autoScoreTotalScore += studentScore;
 91 						autoScoreTotalMaxScore += maxScore;
 92 					}
 93 				} else if (assessment.type == "text") {
 94 					studentWorkSoFar += assessment.response.text;
 95 				}
 96 			}
 97 		}
 98 	}
 99 	
100 	// append autoScore result summary at the end, if request
101 	if (showAutoScoreResult && autoScoreTotalMaxScore > 0) {
102 		studentWorkSoFar += "<br/><br/>";
103 		studentWorkSoFar += "Auto Score Results Summary: ";
104 		studentWorkSoFar += "Student got " + autoScoreTotalScore + " points out of " + autoScoreTotalMaxScore;
105 	}
106 	
107 	return studentWorkSoFar;
108 };
