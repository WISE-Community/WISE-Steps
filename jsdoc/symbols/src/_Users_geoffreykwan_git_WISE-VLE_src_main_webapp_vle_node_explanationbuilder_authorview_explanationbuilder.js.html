  6 View.prototype.ExplanationBuilderNode = {};

  8 /*
  9  * Add the name of the common component that this step will use. The
 10  * common components will be handled by the authoring tool. You will
 11  * need to create div elements with the appropriate id for the
 12  * authoring tool to insert the component into. Any additional custom
 13  * authoring components specific to your step type will be written 
 14  * by you in the generatePage() function. You may leave the array
 15  * empty if you are not using any common components.
 16  * 
 17  * Here are the available common components
 18  * 'Prompt'
 19  * 'LinkTo'
 20  * 'StudentResponseBoxSize'
 21  * 'RichTextEditorToggle'
 22  * 'StarterSentenceAuthoring'
 23  * 
 24  * If you use a common components, you must create a div with the
 25  * appropriate id, here are the respective ids
 26  * 'promptContainer'
 27  * (LinkTo does not require a div)
 28  * 'studentResponseBoxSizeContainer'
 29  * 'richTextEditorToggleContainer'
 30  * 'starterSentenceAuthoringContainer'
 31  * 
 32  * 
 33  * xTODO: rename TemplateNode
 34  */
 35 View.prototype.ExplanationBuilderNode.commonComponents = [];
 36 
 37 /**
 38  * Generates the authoring page. This function will create the authoring
 39  * components such as textareas, radio buttons, check boxes, etc. that
 40  * authors will use to author the step. For example if the step has a
 41  * text prompt that the student will read, this function will create
 42  * a textarea that will allow the author to type the text that the
 43  * student will see. You will also need to populate the textarea with
 44  * the pre-existing prompt if the step has been authored before.
 45  * 
 46  * xTODO: rename TemplateNode
 47  */
 48 View.prototype.ExplanationBuilderNode.generatePage = function(view){
 49 	this.view = view;
 50 	
 51 	//get the content of the step
 52 	this.content = this.view.activeContent.getContentJSON();
 53 	
 54 	//get the html element that all the authoring components will be located
 55 	var parent = document.getElementById('dynamicParent');
 56 	
 57 	/*
 58 	 * wipe out the div that contains the authoring components because it
 59 	 * may still be populated with the authoring components from a previous
 60 	 * step the author has been authoring since we re-use the div id
 61 	 */
 62 	parent.removeChild(document.getElementById('dynamicPage'));
 63 
 64 	//create a new div that will contain the authroing components
 65 	var pageDiv = createElement(document, 'div', {id:'dynamicPage', style:'width:100%;height:100%'});
 66 	
 67 	//create the label for the textarea that the author will write the prompt in
 68 	var promptText = document.createTextNode("Prompt for Student:");
 69 	
 70 	/*
 71 	 * create the textarea that the author will write the prompt in
 72 	 * 
 73 	 * onkeyup will fire the 'templateUpdatePrompt' event which will
 74 	 * be handled in the <new step type name>Events.js file
 75 	 * 
 76 	 * For example if you are creating a quiz step you would look in
 77 	 * your quizEvents.js file.
 78 	 * 
 79 	 * when you add new authoring components you will need to create
 80 	 * new events in the <new step type name>Events.js file and then
 81 	 * create new functions to handle the event
 82 	 */
 83 	var promptTextArea = createElement(document, 'textarea', {id: 'promptTextArea', rows:'10', cols:'85', onkeyup:"eventManager.fire('explanationBuilderUpdatePrompt')"});
 84 	
 85 	//create the label for the textarea that the author will write the instructions in
 86 	var instructionsText = document.createTextNode("Instructions for Student:");
 87 	
 88 	//get the instructions
 89 	var instructionsValue = this.content.instructions;
 90 	
 91 	if(instructionsValue == null) {
 92 		//set instructions to empty string if instructions is not in the content
 93 		instructionsValue = "";
 94 	}
 95 	
 96 	//create the textarea that the author will write the instructions in
 97 	var instructionsTextArea = createElement(document, 'textarea', {id: 'instructionsTextArea', rows:'10', cols:'85', onkeyup:"eventManager.fire('explanationBuilderUpdateInstructions')"});
 98 
 99 	//populate the instructions text area
100 	instructionsTextArea.value = instructionsValue;
101 	
102 	//get the existing background url
103 	var background = this.content.background;
104 	
105 	//the label for the background url input
106 	var backgroundImageUrlLabel = document.createTextNode("Background Image Url");
107 	var localImageLabel = document.createTextNode(" - image you have uploaded to the project, e.g. assets/image.jpg");
108 	var absoluteImageLabel = document.createTextNode(" - image from the internet, e.g. http://www.website.com/image.jpg");
109 	var maxImageSizeLabel = document.createTextNode(" - max image size that you will be able to view is 485x315");
110 	
111 	//the text input for the background url
112 	var backgroundImageUrl = createElement(document, 'input', {type: 'text', id: 'backgroundImageUrl', name: 'backgroundImageUrl', value: background, size:60, onchange: 'eventManager.fire("explanationBuilderUpdateBackgroundImageUrl")'});
113 	
114 	//add the authoring components to the page
115 	pageDiv.appendChild(promptText);
116 	pageDiv.appendChild(createBreak());
117 	pageDiv.appendChild(promptTextArea);
118 	pageDiv.appendChild(instructionsText);
119 	pageDiv.appendChild(createBreak());
120 	pageDiv.appendChild(instructionsTextArea);	
121 	pageDiv.appendChild(createBreak());
122 	pageDiv.appendChild(createBreak());
123 	pageDiv.appendChild(createBreak());
124 	pageDiv.appendChild(backgroundImageUrlLabel);
125 	pageDiv.appendChild(createBreak());
126 	pageDiv.appendChild(localImageLabel);
127 	pageDiv.appendChild(createBreak());
128 	pageDiv.appendChild(absoluteImageLabel);
129 	pageDiv.appendChild(createBreak());
130 	pageDiv.appendChild(maxImageSizeLabel);
131 	pageDiv.appendChild(createBreak());
132 	pageDiv.appendChild(backgroundImageUrl);
133 	
134 	//add the page to the parent
135 	parent.appendChild(pageDiv);
136 	
137 	//populate the prompt if this step has been authored before
138 	this.populatePrompt();
139 };

161 /**
162  * Populate the authoring textarea where the user types the prompt that
163  * the student will read
164  * 
165  * xTODO: rename TemplateNode
166  */
167 View.prototype.ExplanationBuilderNode.populatePrompt = function() {

170 };

172 /**
173  * Updates the content's prompt to match that of what the user input
174  * 
175  * xTODO: rename TemplateNode
176  */
177 View.prototype.ExplanationBuilderNode.updatePrompt = function(){
178 	/* update content */
179 	this.content.prompt = document.getElementById('promptTextArea').value;
180 	
181 	/*
182 	 * fire source updated event, this will update the preview
183 	 */
184 	this.view.eventManager.fire('sourceUpdated');
185 };
