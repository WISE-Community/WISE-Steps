 26 /**
 27  * STATE
 28  *   buckets
 29  * @constructor
 30  */
 31 function MSSTATE() {
 32 	this.type = "ms";
 33 	this.sourceBucket = null;
 34     this.buckets = [];
 35     this.score = null;
 36     this.isCorrect = null;
 37 }

 47 /**
 48  * Create a MSSTATE that we can jsonify. This means removing
 49  * cycling references such as the the choices pointing to
 50  * the bucket that they are in, and the bucket pointing
 51  * to the choices that are in it
 52  * @return
 53  */
 54 MSSTATE.prototype.getJsonifiableState = function() {
 55 	//make a new MSSTATE
 56 	var msState = new MSSTATE();
 57 	
 58 	//set the buckets into our new MSSTATE
 59 	msState.buckets = this.buckets;
 60 	
 61 	//loop through all the buckets
 62 	for(var x=0; x<msState.buckets.length; x++) {
 63 		//get a bucket
 64 		var bucket = msState.buckets[x];
 65 		
 66 		//loop through all the choices that are in the bucket
 67 		for(var y=0; y<bucket.choices.length; y++) {
 68 			//get a choice
 69 			var choice = bucket.choices[y];
 70 			
 71 			//remove the reference to the bucket
 72 			choice.bucket = null;
 73 		}
 74 	}
 75 	
 76 	if(this.score != null) {
 77 		//set the score if available
 78 		msState.score = this.score;
 79 	}
 80 	
 81 	if(this.isCorrect != null) {
 82 		//set isCorrect
 83 		msState.isCorrect = this.isCorrect;
 84 	}
 85 	
 86 	//return the MSSTATE
 87 	return msState;
 88 };

 90 /**
 91  * Creates a MSSTATE from the stateJSONObj
 92  * @param stateJSONObj a JSON object representing a MSSTATE
 93  * @return a populated MSSTATE
 94  */
 95 MSSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 96 	//create a new MSSTATE object
 97 	var msState = new MSSTATE();
 98 
 99 	//get the buckets from the json
100 	msState.buckets = stateJSONObj.buckets;
101 	
102 	//get the score from the json
103 	msState.score = stateJSONObj.score;
104 	
105 	//get whether the state is correct
106 	msState.isCorrect = stateJSONObj.isCorrect;
107 	
108 	//return the MCSTATE object
109 	return msState;
110 };
