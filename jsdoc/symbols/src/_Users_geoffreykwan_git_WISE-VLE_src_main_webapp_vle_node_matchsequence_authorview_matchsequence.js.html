228 /**
229  * Generates a table of choices vs. containters which will allow for editing
230  * creating feedback that choice in that container
231  */
232 View.prototype.MatchSequenceNode.generateFeedbackTable = function(){
233 	var parent = document.getElementById('feedbackDiv');
234  	var old = document.getElementById('containerChoiceTable');
235  	
236  	//wipe out old
237  	if(old){
238  		parent.removeChild(old);
239  	};
240  	
241  	//create new
242 	var choices = this.getChoices();
243 	var containers = this.getFields();
244  	var table = createElement(document, 'table', {id: 'containerChoiceTable'});
245  	var headerRow = createElement(document, 'tr', {id: 'headerRow'});
246  	var headerLabel = createElement(document, 'td', {id: 'headerLabel'});
247  	var containerLabelDiv = createElement(document, 'div', {id: 'containerLabelDiv', align: 'right'});
248  	var choiceLabelDiv = createElement(document, 'div', {id: 'choiceLabelDiv'});
249 				 	
250  	containerLabelDiv.innerHTML = 'Containers:';
251  	choiceLabelDiv.innerHTML = 'Choices';
252  	
253  	headerLabel.appendChild(containerLabelDiv);
254  	headerLabel.appendChild(choiceLabelDiv);
255  	headerRow.appendChild(headerLabel);
256  	for(var a=0;a<containers.length;a++){
257  		var containerName = createElement(document, 'td', {id: 'containerName_' + containers[a].identifier});
258  		containerName.innerHTML = containers[a].name;
259  		headerRow.appendChild(containerName);
260  	};
261  	
262  	table.appendChild(headerRow);
263  	
264  	for(var b=0;b<choices.length;b++){
265  		var row = createElement(document, 'tr', {id: 'row_' + b});
266  		var choiceLabelTD = createElement(document, 'td', {id: 'choiceLabelTD_' + b});
267  		
268  		choiceLabelTD.innerHTML = choices[b].value;
269  		row.appendChild(choiceLabelTD);
270  		for(var c=0;c<containers.length;c++){
271  			var choiceId = choices[b].identifier;
272  			var containerId = containers[c].identifier;
273  			var val = this.getValueByChoiceAndFieldIdentifiers(choiceId, containerId);
274  			var feedbackTD = createElement(document, 'td', {id: 'cell_' + b + '_' + c, onclick: 'eventManager.fire("msEditIndividualFeedback",["' + choiceId + '","' + containerId + '"])'});
275  			
276  			if(val){ //feedback exists
277  				var html;
278  				if(val.isCorrect){
279  					feedbackTD.setAttribute('bgcolor', '#99FFCC');
280  					html = val.feedback;
281  				} else {
282  					feedbackTD.setAttribute('bgcolor', '#FF9999');
283  					html = val.feedback;
284  				};
285  				feedbackTD.innerHTML = html;
286  			} else { //feedback not there
287  				feedbackTD.setAttribute('bgcolor', '#FF9999');
288  				feedbackTD.innerHTML = 'Create Feedback';
289  			};
290  			
291  			row.appendChild(feedbackTD);
292  		};
293  		
294  		table.appendChild(row);
295  	};
296  	
297  	parent.appendChild(table);
298 };

320 /**
321  * Saves the user entered feedback and refreshes the table
322  */
323 View.prototype.MatchSequenceNode.saveFeedback = function(){
324 	var feedback = this.getValueByChoiceAndFieldIdentifiers(this.currentChoiceId, this.currentContainerId);
325  	
326  	if(!feedback){
327  		this.createNewFeedback(this.currentChoiceId, this.currentContainerId);
328  		feedback = this.getValueByChoiceAndFieldIdentifiers(this.currentChoiceId, this.currentContainerId);
329  	};
330  	this.updateFeedback(this.currentChoiceId, this.currentContainerId, document.getElementById('feedbackEditInput').value);
331  	
332  	this.generateFeedbackTable();
333  	document.getElementById('feedbackEditInput').value = '';
334  	hideElement('feedbackEditDiv');
335  	
336  	/* fire source updated event */
337 	this.view.eventManager.fire('sourceUpdated');
338 };

352 /**
353  * Removes previous containerSelect element (if exists) and
354  * generates a new one based on the gapMultiple elements
355  * defined in xmlPage
356  */
357 View.prototype.MatchSequenceNode.generateContainers = function(){
358  	var parent = document.getElementById('containerTable');
359  	var fields = this.getFields();
360  	
361  	//remove old tr elements
362  	var rows = document.getElementsByName('containerRow');
363  	for(var v=rows.length - 1;v>=0;v--){
364  		parent.removeChild(rows[v]);
365  	};
366  	
367  	for(var u=0;u<fields.length;u++){
368  		var row = createElement(document, 'tr', {name: 'containerRow'});
369  		row.appendChild(this.generateContainer(fields[u], u));
370  		parent.appendChild(row);
371  	};
372  };

417 /**
418  * Given a choice (gapText) and its associated gapIdentifier, generates
419  * and returns a table row with the choice Text
420  */
421 View.prototype.MatchSequenceNode.generateChoice = function(choice,fieldIdentifier){
422 	var identifier = choice.identifier;
423 	var row = createElement(document, 'tr', {id: 'choiceRow_'+ identifier});
424 	var td = createElement(document, 'td', {id: 'choiceTD_' + identifier});
425 	var radioChoice = createElement(document, 'input', {type: 'radio', name: 'radioChoice_' + fieldIdentifier, onfocus: 'eventManager.fire("msChoiceSelected",["' + identifier + '", "' + fieldIdentifier + '"])', value: identifier});
426 	var textChoice = createElement(document, 'input', {type: 'text', id: 'textChoice_' + identifier, onfocus: 'eventManager.fire("msChoiceSelected",["' + identifier + '", "' + fieldIdentifier + '"])', onchange: 'eventManager.fire("msChoiceTextUpdated","' + identifier + '")'});
427 	var ordered = createElement(document, 'input', {type: 'text', size: '1', id: 'orderChoice_' + identifier, onfocus: 'eventManager.fire("msChoiceSelected",["' + identifier + '", "' + fieldIdentifier + '"])', onkeyup: 'eventManager.fire("msOrderUpdated","' + identifier + '")'});

429 	
430 	textChoice.value = choice.value;
431 	
432 	td.appendChild(radioChoice);
433 	td.appendChild(textChoice);
434 	
435 	if(this.getOrdered()){
436 		ordered.value = this.getOrderPositionByChoiceIdentifier(identifier);
437 		td.appendChild(textOrder);
438 		td.appendChild(ordered);
439 	};
440 	
441 	row.appendChild(td);
442 	return row;
443 };

987 /**
988  * returns true if there is < 2 order values or if all of the values
989  * are exactly 1 more than the previous
990  */
991 View.prototype.MatchSequenceNode.sequential = function(){
992 	var containers = this.getFields();
993 	for(var u=0;u<containers.length;u++){
994 		var choices = this.getChoicesByContainerIdentifier(containers[u].identifier);
995 		var order = [];
996 		for(var v=0;v<choices.length;v++){
997 			var val = this.getValueByChoiceAndFieldIdentifiers(choices[v].identifier, containers[u].identifier).order;
998 			order.push(val);
999 		};
1000 		order.sort(this.sort);
1001 		for(var w=0;w<order.length;w++){
1002 			if(w!=0){
1003 				if(parseInt(order[w])!=(parseInt(order[w-1]) + 1)){
1004 					return false;
1005 				};
1006 			};
1007 		};
1008 	};
1009 	return true;
1010 };
