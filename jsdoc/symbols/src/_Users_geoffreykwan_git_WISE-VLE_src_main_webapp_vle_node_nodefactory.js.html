  9 /*
 10  * an array that contains the accepted node type names. additional
 11  * values in the array will be dynamically added when we load the
 12  * step/node types.
 13  */
 14 NodeFactory.acceptedTagNames = ['node', 'sequence'];
 15 
 16 /*
 17  * an array that contains the constructors for creating new nodes.
 18  * this array will be dynamically populated when we load the
 19  * step/node types. 
 20  */
 21 NodeFactory.nodeConstructors = {};
 22 
 23 /**
 24  * Create a node object
 25  * @param jsonNode the JSON representation of the node
 26  * @param view
 27  * @return a new node object
 28  */
 29 NodeFactory.createNode = function(jsonNode, view){
 30 	var nodeType = null;
 31 	
 32 	//get the node type
 33 	if(jsonNode.type){
 34 		nodeType = jsonNode.type;
 35 	}
 36 	
 37 	//check if the node type is acceptable
 38 	if (NodeFactory.acceptedTagNames.indexOf(nodeType) > -1) {
 39 		if (nodeType == "sequence") {
 40 			//node is a sequence node
 41 			
 42 			//create a new sequence
 43 			var sequenceNode = new Node("sequence", view);
 44 			
 45 			//set the attributes of the sequence
 46 			sequenceNode.id = jsonNode.identifier;
 47 			sequenceNode.title = jsonNode.title;
 48 			
 49 			//return the sequence
 50 			return sequenceNode;
 51 		} else {
 52 			//node is a step node
 53 			
 54 			//get the constructor for the node type
 55 			var nodeConstructor = NodeFactory.nodeConstructors[nodeType];
 56 			
 57 			if(nodeConstructor != null) {
 58 				//create a new node object with the constructor
 59 				return new nodeConstructor(nodeType, view);
 60 			} else {
 61 				return new Node(null, view);
 62 			};
 63 		};
 64 	};
 65 };
 66 
 67 /**
 68  * Adds a tag name to the acceptedTagNames array
 69  * @param tagName a node type
 70  * e.g. 'HtmlNode'
 71  */
 72 NodeFactory.addAcceptedTagName = function(tagName) {
 73 	//check if the tag name has already been added
 74 	if(NodeFactory.acceptedTagNames.indexOf(tagName) == -1) {
 75 		//tag name has not been added so we will add it now
 76 		NodeFactory.acceptedTagNames.push(tagName);		
 77 	}
 78 };
 79 
 80 /**
 81  * Adds a constructor to the nodeConstructors array
 82  * @param nodeType the node type
 83  * @param nodeConstructor the constructor for the node type
 84  */
 85 NodeFactory.addConstructor = function(nodeType, nodeConstructor) {
 86 	//check if we have already added a constructor for the node type
 87 	if(NodeFactory.nodeConstructors[nodeType] == null) {
 88 		//we have not added a constructor for the node type yet so we will now
 89 		NodeFactory.nodeConstructors[nodeType] = nodeConstructor;
 90 	}
 91 };
 92 
 93 /**
 94  * Adds the node type to the acceptedTagNames array and the
 95  * constructor to the nodeConstructors array.
 96  * @param nodeType the node type
 97  * @param nodeConstructor the constructor for the node type
 98  */
 99 NodeFactory.addNode = function(nodeType, nodeConstructor) {
100 	//add the node type to the acceptedTagNames array
101 	NodeFactory.addAcceptedTagName(nodeType);
102 	
103 	//add the constructor to the nodeConstructors array
104 	NodeFactory.addConstructor(nodeType, nodeConstructor);
105 };
106 
107 /**
108  * Get an array of the available node types
109  * @return an array containing the node types as strings
110  */
111 NodeFactory.getNodeTypes = function() {
112 	var nodeTypes = [];
113 	
114 	//loop through the array of acceptedTagNames
115 	for(var x=0; x<this.acceptedTagNames.length; x++) {
116 		var tagName = this.acceptedTagNames[x];
117 		
118 		//check that the tagName is not 'node' or 'sequence'
119 		if(tagName != 'node' && tagName != 'sequence') {
120 			nodeTypes.push(tagName);
121 		}
122 	}
123 	
124 	return nodeTypes;
125 };
