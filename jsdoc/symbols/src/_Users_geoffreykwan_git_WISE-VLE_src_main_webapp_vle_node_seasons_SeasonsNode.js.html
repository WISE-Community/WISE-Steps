112 /**
113  * This function is called when the student exits the step. It is mostly
114  * used for error checking.
115  * 
116  * TODO: rename SeasonsNode
117  * 
118  * Note: In most cases you will not have to change anything here.
119  */
120 SeasonsNode.prototype.onExit = function() {
121 	//check if the content panel has been set
122 	if(this.contentPanel) {
123 		
124 		if(this.contentPanel.save) {
125 			//tell the content panel to save
126 			this.contentPanel.save();
127 		}
128 		
129 		try {
130 			/*
131 			 * check if the onExit function has been implemented or if we
132 			 * can access attributes of this.contentPanel. if the user
133 			 * is currently at an outside link, this.contentPanel.onExit
134 			 * will throw an exception because we aren't permitted
135 			 * to access attributes of pages outside the context of our
136 			 * server.
137 			 */
138 			if(this.contentPanel.onExit) {
139 				try {
140 					//run the on exit cleanup
141 					this.contentPanel.onExit();					
142 				} catch(err) {
143 					//error when onExit() was called, e.g. mysystem editor undefined
144 				}
145 			}	
146 		} catch(err) {
147 			/*
148 			 * an exception was thrown because this.contentPanel is an
149 			 * outside link. we will need to go back in the history
150 			 * and then trying to render the original node.
151 			 */
152 			history.back();
153 		}
154 	}
155 };

157 /**
158  * Renders the student work into the div. The grading tool will pass in a
159  * div id to this function and this function will insert the student data
160  * into the div.
161  * 
162  * @param divId the id of the div we will render the student work into
163  * @param nodeVisit the student work
164  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
165  * div ids use this to prevent DOM conflicts such as when the show all work div
166  * uses the same ids as the show flagged work div
167  * @param workgroupId the id of the workgroup this work belongs to
168  * 
169  * TODO: rename SeasonsNode
170  * Note: you may need to add code to this function if the student
171  * data for your step is complex or requires additional processing.
172  * look at SensorNode.renderGradingView() as an example of a step that
173  * requires additional processing
174  */
175 SeasonsNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
176 	/*
177 	 * Get the latest student state object for this step
178 	 * TODO: rename seasonsState to reflect your new step type
179 	 * 
180 	 * e.g. if you are creating a quiz step you would change it to quizState
181 	 */
182 	var seasonsState = nodeVisit.getLatestWork();
183 	
184 	/*
185 	 * get the step work id from the node visit in case we need to use it in
186 	 * a DOM id. we don't use it in this case but I have retrieved it in case
187 	 * someone does need it. look at SensorNode.js to view an example of
188 	 * how one might use it.
189 	 */
190 	var stepWorkId = nodeVisit.id;
191 	
192 	/*
193 	 * TODO: rename seasonsState to match the variable name you
194 	 * changed in the previous line above
195 	 */
196 	var studentWork = seasonsState.getStudentWork();
197 	
198 	//put the student work into the div
199 	$('#' + divId).html(studentWork);
200 };
