  3 /**
  4  * This is the constructor for the object that will perform the logic for
  5  * the step when the students work on it. An instance of this object will
  6  * be created in the .html for this step (look at mysystem.html)
  7  * @constructor
  8  */
  9 function Mysystem2(node,view) {
 10   this.node = node;
 11   this.content = node.getContent().getContentJSON();
 12 
 13   this.domIO = document.getElementById('my_system_state');
 14   
 15   if (typeof node.studentWork != 'undefined' && node.studentWork !== null) {
 16     this.states = node.studentWork; 
 17     
 18     MySystem.registerExternalSaveFunction(this.save, this);
 19     this.node.view.eventManager.subscribe('processPostResponseComplete', this.saveSuccessful);
 20     // MySystem is set by default to save every 20 seconds. If we want to change that frequency, we can call
 21     // MySystem.setAutoSaveFrequency(20000)
 22   } 
 23   else {
 24     this.states = [];
 25   }
 26 }

 28 /**
 29  * This function renders everything the student sees when they visit the step.
 30  * This includes setting up the html ui elements as well as reloading any
 31  * previous work the student has submitted when they previously worked on this
 32  * step, if any.
 33  */
 34 Mysystem2.prototype.render = function() {
 35   var latestState = this.getLatestState();
 36   if (latestState !== null) {
 37     /*
 38      * get the response from the latest state. the response variable is
 39      * just provided as an example. you may use whatever variables you
 40      * would like from the state object (look at templatestate.js)
 41      */
 42     var latestResponse = latestState.response;
 43     this.domIO.textContent = latestResponse;
 44   }
 45   
 46   // It turns out that sometimes when firebug is enabled and reloading
 47   // the page and switching back to the step: The SC.onReady.done method is never called
 48   // which should mean the jquery ready method is never called either.  
 49   // It seems this has to do with how the step iframe is setup. Its contents are injected
 50   // instead of loading it from a url.  So far the approach below seems to fix this problem
 51   // and not cause other problems.
 52   if(!SC.isReady) {
 53     SC.onReady.done();
 54   }
 55 
 56   var lastRenewal = 0;
 57   if (typeof eventManager != 'undefined') {
 58     // watch for changes to the student data and renew the session whenever it changes
 59     $('#my_system_state').bind("DOMSubtreeModified", function() {
 60       var now = new Date().getTime();
 61       if (now - lastRenewal > 15000) {  // only renew at most once every 15 seconds
 62         SC.Logger.log("renewing session");
 63         eventManager.fire('renewSession');
 64         lastRenewal = now;
 65       }
 66     });
 67   }
 68 
 69   if (this.content) {
 70     MySystem.loadWiseConfig(this.content,latestState);
 71   }
 72   if (latestState) {
 73     MySystem.updateFromDOM();
 74   }
 75 };

 77 /**
 78  * This function retrieves the latest student work
 79  *
 80  * @return the latest state object or null if the student has never submitted
 81  * work for this step
 82  */
 83 Mysystem2.prototype.getLatestState = function() {
 84   var latestState = null,
 85       state,
 86       i,
 87       numberOfOwnProperties = function (obj) {
 88         var p, n = 0;
 89 
 90         for (p in obj) {
 91           if (obj.hasOwnProperty(p)) n++;
 92         }
 93         return n;
 94       };
 95       
 96   if (this.states) {
 97     for (i = this.states.length - 1; i >= 0; i--) {
 98       state = this.states[i];
 99       // because of WISE4 corruption issues, reject states that may have been saved to our states array
100       // by non-MySystem steps such as openresponse steps
101       if (state.type === "MySystem2" || (typeof state.type === 'undefined' && numberOfOwnProperties(state) === 1)) {
102         latestState = state;
103         break;
104       }
105     }
106   }
107   
108   return latestState;
109 };

118 /**
119  * This function retrieves the student work from the html ui, creates a state
120  * object to represent the student work, and then saves the student work.
121  * 
122  * note: you do not have to use 'studentResponseTextArea', they are just 
123  * provided as examples. you may create your own html ui elements in
124  * the .html file for this step (look at mysystem.html).
125  */
126 Mysystem2.prototype.save = function() {
127   // We use a simple dom element for our data passing
128   var response =this.domIO.textContent;
129   console.log("Saving MySystem");
130   
131   /*
132    * create a student state object that will store the new work the student
133    * just submitted
134    */
135   var state = new MYSYSTEM2STATE(response);
136   
137   /*
138    * fire the event to push this state to the global view.states object.
139    * the student work is saved to the server once they move on to the
140    * next step.
141    */
142   eventManager.fire('pushStudentWork', state);
143 
144   // push the state object into this or object's own copy of states
145   this.states.push(state);
146   
147   // save back to the server. In a single node visit, this will use the same
148   // node visit id each time, so it will save the (growing) stack back to the same
149   // place each time. 
150   this.node.view.postCurrentNodeVisit(this.node.view.state.getCurrentNodeVisit());
151 };

153 // we subscribe this function to the eventManager's processPostResponseComplete event,
154 // so it ought to get called after postCurrentNodeVisit receives a success message from the post
155 Mysystem2.prototype.saveSuccessful = function () {
156   // it would be great if we could unsubscribe from the event manager when we leave the step.
157   // as we can't, our subscription hangs around, even if we are in some other step.
158   // Check we have a MySystem first
159   if (window['MySystem'] !== undefined){
160     window['MySystem'].externalSaveSuccessful(true);
161   }
162 };
