 15 /**
 16  * Sets the view and content and then builds the page
 17  */
 18 View.prototype.Mysystem2Node.generatePage = function(view){
 19   this.view = view;
 20   this.content = this.view.activeContent.getContentJSON();
 21   if (typeof this.content == 'undefined') {
 22     this.content = {};
 23   }
 24 
 25   this.buildPage();
 26 
 27   var iframe = createElement(document, 'iframe', {
 28     src: '/vlewrapper/vle/node/mysystem2/authoring/index.html',
 29     width: '99%',
 30     style: 'display: block; height: 100%;',
 31     id: 'mysystem2-authoring-iframe',
 32     onload: 'eventManager.fire("mysystem2AuthoringIFrameLoaded");'
 33   });
 34 
 35   var parent = document.getElementById('dynamicPage');
 36   parent.appendChild(iframe);
 37   parent.appendChild(this.getBuildInfoDiv());
 38 };

 40 View.prototype.Mysystem2Node.AuthoringIFrameLoaded = function(){
 41   var iframe = document.getElementById('mysystem2-authoring-iframe').contentWindow;
 42   
 43   iframe.MSA.setupParentIFrame(this.content, this, function (){
 44     /* fire source updated event */
 45     this.view.eventManager.fire('sourceUpdated');
 46   });
 47 };

 49 View.prototype.Mysystem2Node.getBuildInfoDiv = function() {

 51   var git_sha_div         = createElement(document, 'div', {id: 'git_sha_div'    }) ;
 52   var git_time_div        = createElement(document, 'div', {id: 'git_time_div'   }) ;
 53   var git_branch_div      = createElement(document, 'div', {id: 'git_branch_div' }) ;
 54   var sc_build_time_div   = createElement(document, 'div', {id: 'sc_build_time'  }) ;
 55   var sc_build_number_div = createElement(document, 'div', {id: 'sc_build_number'}) ;
 56 
 57   var git_sha         = document.createTextNode("commit sha  : 72cb000dc1b4f9b91484b0a023fde690a9eb045e ");
 58   var git_time        = document.createTextNode("commit time : Fri Sep 23 18:55:43 2011 -0400 ");
 59   var git_branch      = document.createTextNode("git branch  : (HEAD, usingVisibilityBindingsInInspector, master) ");
 60   var sc_build_time   = document.createTextNode("build time  : 2011-09-23 19:00:42 -0400 ");
 61   var sc_build_number = document.createTextNode("build no.   : b7496a1eec79ffecb2969797c585818a31348044 ");
 62   
 63   git_sha_div.appendChild(git_sha);
 64   git_time_div.appendChild(git_time);
 65   git_branch_div.appendChild(git_branch);
 66   sc_build_time_div.appendChild(sc_build_time);
 67   sc_build_number_div.appendChild(sc_build_number);
 68 
 69   metaDiv.appendChild(git_sha_div);
 70   metaDiv.appendChild(git_time_div);
 71   metaDiv.appendChild(git_branch_div);
 72   metaDiv.appendChild(sc_build_number_div);
 73   metaDiv.appendChild(sc_build_time_div);

 75 };

 85 /**
 86  * Builds the html elements needed to author a my system node
 87  */
 88 View.prototype.Mysystem2Node.buildPage = function(){
 89   var parent = document.getElementById('dynamicParent');
 90 
 91   /* remove any old elements */
 92   while(parent.firstChild){
 93     parent.removeChild(parent.firstChild);
 94   }
 95 
 96   /* create new elements */
 97   var pageDiv = createElement(document, 'div', {id: 'dynamicPage', style:'width:100%;height:100%'});
 98 	var mainDiv = createElement(document, 'div', {id: 'mainDiv'});
 99   var instructionsText = document.createTextNode("When entering image filenames, make sure to use the asset uploader on the main authoring page to upload your images.");
100 
101   /* append elements */
102   parent.appendChild(pageDiv);
103   pageDiv.appendChild(mainDiv);
104   mainDiv.appendChild(instructionsText);
105   mainDiv.appendChild(createBreak());
106   mainDiv.appendChild(document.createTextNode("Enter instructions -- text or html -- here."));
107   mainDiv.appendChild(createBreak());
108   mainDiv.appendChild(createElement(document, 'div', {id: 'promptContainer'}));
109   mainDiv.appendChild(createBreak());
110 };
