  2 /**
  3  * A representation of a Brainstorm which can be either 
  4  * open_response or single_choice (not implemented) and
  5  * can run with a server or serverless
  6  * @constructor
  7  * @author: patrick lawler
  8  */
  9 function BRAINSTORM(node){
 10 	this.node = node;
 11 	this.content = node.getContent().getContentJSON();
 12 	this.states = [];
 13 	this.recentResponses = new Array();
 14 	this.subscribed = false;
 15 	
 16 	if(node.studentWork != null) {
 17 		this.states = node.studentWork; 
 18 	} else {
 19 		this.states = [];  
 20 	};
 21 
 22 };

249 /**
250  * This function is used by array.sort(function) which sorts an array
251  * using the function that is passed in.
252  * This function looks at the timestamp attribute of objects and 
253  * sorts them in chronological order, from oldest to newest.
254  */
255 function sortByTimestamp(object1, object2) {
256 	//get the timestamp values
257 	var timestamp1 = object1.timestamp;
258 	var timestamp2 = object2.timestamp;
259 	
260 	/*
261 	 * return a negative value if 1 comes before 2
262 	 * return 0 if values are the same
263 	 * return a positive value if 1 comes after 2
264 	 */
265 	return timestamp1 - timestamp2;
266 }

268 /**
269  * The callback function for displaying class mate posts. After it displays
270  * all the class mate posts, we will display the student's recent post from
271  * the recentResponses array.
272  * @param responseText the response text from the async request
273  * @param responseXML the response xml from the async request
274  * @param handlerArgs the extra arguments used by this function
275  */
276 function getClassmateResponsesCallback(responseText, responseXML, handlerArgs) {
277 	
278 	if(responseText) {
279 		//the responseText should be json
280 		
281 		//parse the json
282 		visits = $.parseJSON(responseText);
283 		
284 		//used for adding responses to the student UI
285 		var frameDoc = handlerArgs.frameDoc;
286 		var responsesParent = frameDoc.getElementById('responses');
287 		
288 		/*
289 		 * the array that holds objects that represent a response. the
290 		 * object contains a userId, responseText, timestamp 
291 		 */
292 		var responseStates = new Array();
293 		
294 		//loop through the visits
295 		for(var x=0; x<visits.length; x++) {
296 			//obtain a visit
297 			var visitObj = visits[x];
298 			
299 			//create a node visit object from the json data
300 			var nodeVisitObj = NODE_VISIT.prototype.parseDataJSONObj(visitObj.data, handlerArgs.vle);
301 			
302 			//set the id of the node visit object
303 			nodeVisitObj.id = visitObj.stepWorkId;
304 			
305 			//obtain the userId
306 			var userId = visitObj.userId;
307 			
308 			//loop through the states in the visit
309 			for(var y=0; y<nodeVisitObj.nodeStates.length; y++) {
310 				//obtain a state
311 				var nodeState = nodeVisitObj.nodeStates[y];
312 				
313 				/*
314 				 * create an object that will contain the userId, responseText,
315 				 * and timestamp
316 				 */ 
317 				var responseState = new Object();
318 				responseState.userId = userId;
319 				responseState.responseText = nodeState.getStudentWork();
320 				responseState.timestamp = nodeState.timestamp;
321 				
322 				//add the responseState object to the array
323 				responseStates.push(responseState);
324 			}
325 		}
326 		
327 		//sort the array by timestamp using the function we wrote
328 		responseStates.sort(sortByTimestamp);
329 		
330 		//loop through the responseStates
331 		for(var z=0; z<responseStates.length; z++) {
332 			//obtain a responseState
333 			var responseState = responseStates[z];
334 			
335 			//obtain the userId and responseText
336 			var userId = responseState.userId;
337 			var responseText = responseState.responseText;
338 			
339 			//create a DOM id for the response we will add to the student's UI
340 			var localResponseId = "response" + responsesParent.childNodes.length;
341 			
342 			//add the response to the UI
343 			BRAINSTORM.prototype.addStudentResponse(frameDoc, responseText, userId, localResponseId, handlerArgs.vle);
344 		}
345 	} else {
346 		/*
347 		 * obtain the frameDoc from the handlerArgs. the frameDoc is the
348 		 * dom object for the brainstorm html interface
349 		 */
350 		var frameDoc = handlerArgs.frameDoc;
351 		
352 		/*
353 		 * retrieve the response(s) the student has posted during this current
354 		 * node visit
355 		 */ 
356 		var recentResponses = handlerArgs.recentResponses;
357 		
358 		//the student's vle
359 		var vle = handlerArgs.vle;
360 		
361 		//obtain the dom object that holds all the responses
362 		var responsesParent = frameDoc.getElementById('responses');
363 		
364 		/*
365 		 * node_visits are wrapped in a workgroup tag, the same workgroup may show
366 		 * up multiple times in the xml if that workgroup posted multiple times
367 		 */
368 		var workgroups = responseXML.getElementsByTagName("workgroup");
369 
370 		//loop through all the workgroups
371 		for(var x=0; x<workgroups.length; x++) {
372 			//obtain the userId (same as workgroupId)
373 			var userId = workgroups[x].attributes.getNamedItem("userId").nodeValue;
374 			
375 			//the data is the node state xml text
376 			var data = workgroups[x].getElementsByTagName("data")[0];
377 			if(data != null && data != "") {
378 				/*
379 				 * obtain all the responses from the node state data. each node
380 				 * state can have multiple response tags if the student posted
381 				 * multiple times in a single node visit
382 				 */
383 				var responses = data.getElementsByTagName("response");
384 
385 				//loop through the responses in this node visit
386 				for(var y=0; y<responses.length; y++) {
387 					//obtain the text the student wrote and posted
388 					var postText = responses[y].firstChild.nodeValue;
389 
390 					//create a unique dom id for the response
391 					var localResponseId = "response" + responsesParent.childNodes.length;
392 
393 					//add the posted response to the user interface
394 					BRAINSTORM.prototype.addStudentResponse(frameDoc, postText, userId, localResponseId, vle);
395 				};
396 			};
397 		};
398 		
399 		BRAINSTORM.prototype.showRecentResponses(frameDoc, recentResponses, responsesParent, vle);
400 	};
401 };

403 /**
404  * This displays the responses made by the author/teacher. This
405  * is used in server as well as serverless mode.
406  * @param frameDoc the dom object that contains all the brainstorm
407  * 		elements
408  */
409 BRAINSTORM.prototype.showCannedResponses = function(frameDoc){
410 	/* get parent */
411 	var responsesParent = frameDoc.getElementById('responses');
412 	
413 	/* remove any old children */
414 	while(responsesParent.firstChild){
415 		responsesParent.removeChild(responsesParent.firstChild);
416 	};
417 	
418 	if (this.content.cannedResponses) {
419 		/* create new response elements for each response in canned responses and append to parent */
420 		for(var p=0;p<this.content.cannedResponses.length;p++){
421 			var response = createElement(frameDoc, 'div', {rows: '7', cols:  '100', disabled: true, id: this.content.cannedResponses[p].name});
422 			var responseTitle = createElement(frameDoc, 'div', {id: 'responseTitle_' + this.content.cannedResponses[p].name});
423 			responseTitle.innerHTML = 'Posted By:  ' + this.content.cannedResponses[p].name;
424 			responseTitle.appendChild(createElement(frameDoc, 'br'));
425 			responseTitle.appendChild(response);
426 			responseTitle.setAttribute('class', 'responseTitle');
427 			response.innerHTML = this.content.cannedResponses[p].response;
428 			response.setAttribute('class', 'responseTextArea');
429 
430 			responsesParent.appendChild(responseTitle);
431 			responsesParent.appendChild(createElement(frameDoc, 'br'));
432 		};
433 	};
434 	
435 };

561 /**
562  * Add the response to the brainstorm display. This function is used to display
563  * canned responses, classmate responses, and recent responses.
564  * @param frameDoc the dom object that contains all the brainstorm display elements
565  * @param responseText the text a student has posted in response to the question
566  * @param userId the id of the student who made this post
567  * @param localResponseId a local dom id for the response element we are going to make
568  * @param vle the vle of the student who is logged in
569  */
570 BRAINSTORM.prototype.addStudentResponse = function(frameDoc, responseText, userId, localResponseId, vle) {
571 	//obtain the dom object that holds all the responses
572 	var responsesParent = frameDoc.getElementById('responses');
573 	
574 	//create the response and response title elements
575 	var responseTextArea = createElement(frameDoc, 'div', {rows: '7', cols:  '80', disabled: true, id: localResponseId});
576 	var responseTitle = createElement(frameDoc, 'div', {id: 'responseTitle_' + localResponseId});
577 	
578 	//set the html for the response title
579 	responseTitle.innerHTML = 'Posted By:  ' + vle.getUserAndClassInfo().getUserNameByUserId(userId);
580 	responseTitle.appendChild(createElement(frameDoc, 'br'));
581 	responseTitle.appendChild(responseTextArea);
582 	responseTitle.setAttribute('class', 'responseTitle');
583 	
584 	//set the student text in the response textarea
585 	responseTextArea.innerHTML = responseText;

587 
588 	//add the whole response element into the parent container
589 	responsesParent.appendChild(responseTitle);

608  * Attempts to retrieve the contentbase url from project and inject
609  * it into the content.
610  */
611 BRAINSTORM.prototype.injectBaseRef = function(content){
612 	if (content.search(/<base/i) > -1) {
613 		// no injection needed because base is already in the html
614 		return content;
615 	} else {
616 	    // NATE!!
617 	    var cbu = contentBaseUrl;
618 	    if (this.node.ContentBaseUrl) {
619 	        cbu = this.node.ContentBaseUrl;
620 	    }
621 		var domain = 'http://' + window.location.toString().split("//")[1].split("/")[0];
622 		
623 		if(window.parent.vle){
624 			var baseRefTag = "<base href='" + window.parent.vle.project.cbu + "'/>";
625 		} else if(typeof vle!='undefined'){
626 			var baseRefTag = "<base href='" + vle.project.cbu + "'/>";
627 		} else {
628 			return content;
629 		};
630 		
631 		var headPosition = content.indexOf("<head>");
632 		var newContent = content.substring(0, headPosition + 6);  // all the way up until ...<head>
633 		newContent += baseRefTag;
634 		newContent += content.substring(headPosition+6);
635 
636 		return newContent;
637 	};	
638 };

640 /**
641  * Places the starter sentence, if provided, at the top of the
642  * response and appends any of the student's work after it.
643  */
644 BRAINSTORM.prototype.showStarter = function(){
645 	if(this.content.starterSentence.display != '0'){
646 
647 		//get the response box element
648 		var responseBox = document.getElementById('studentResponse');
649 		
650 		//update normally if rich text editor is not available
651 		if(!this.richTextEditor){
652 			responseBox.value = this.content.starterSentence.sentence + '\n\n' + responseBox.value;
653 		} else {//otherwise, we need to set it in the editor instance
654 			this.richTextEditor.setContent(this.content.starterSentence.sentence + '<br/><br/>' + this.richTextEditor.getContent());
655 		};
656 		
657 		//link clicked, so remove it
658 		document.getElementById('starterParent').innerHTML = '';
659 	} else {
660 		this.node.view.notificationManager.notify("There is no starter sentence specified for this step", 3);
661 	};
662 };

669 BRAINSTORM.prototype.refreshResponses = function(frameDoc) {

676 	//show the canned responses
