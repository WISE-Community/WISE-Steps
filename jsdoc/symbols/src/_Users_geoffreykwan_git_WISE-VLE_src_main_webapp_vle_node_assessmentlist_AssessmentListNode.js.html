 23 /**
 24  * Sets up a WorkOnXConstraint before rendering so that students will
 25  * not be able to navigate to any other step before completing work on
 26  * this step if that was specified in the content.
 27  * 
 28  * @param contentPanel
 29  * @param studentWork
 30  */
 31 AssessmentListNode.prototype.render = function(contentPanel,studentWork, disable){
 32 	/* add a new constraint for this assessment list if the content specifies that
 33 	 * student must complete work before exiting to another step */
 34 	if(this.content.getContentJSON().isMustCompleteAllPartsBeforeExit){
 35 		this.view.eventManager.fire('addConstraint',{type:'WorkOnXConstraint', x:{id:this.id, mode:'node'}, id:this.utils.generateKey(20)});
 36 	}
 37 	
 38 	/* call super */
 39 	Node.prototype.render.call(this, contentPanel, studentWork, disable);
 40 };

 42 /**
 43  * Renders the student work into the div. The grading tool will pass in a
 44  * div id to this function and this function will insert the student data
 45  * into the div.
 46  * 
 47  * @param divId the id of the div we will render the student work into
 48  * @param nodeVisit the student work
 49  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
 50  * div ids use this to prevent DOM conflicts such as when the show all work div
 51  * uses the same ids as the show flagged work div
 52  * @param workgroupId the id of the workgroup this work belongs to
 53  */
 54 AssessmentListNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
 55     // Get the latest student state object for this step
 56 	var assessmentListState = nodeVisit.getLatestState();
 57 	
 58 	// get human readable work string
 59 	var showAutoScoreResult = true;
 60 	var readableStudentWork = assessmentListState.getStudentWork(showAutoScoreResult);
 61 	$('#' + divId).html(readableStudentWork);
 62 };

 65 /**
 66  * Override of Node.isCompleted
 67  * Get whether the step is completed or not
 68  * @return a boolean value whether the step is completed or not
 69  */
 70 AssessmentListNode.prototype.isCompleted = function() {
 71 	var nodeVisitsForThisNode = this.view.state.getNodeVisitsByNodeId(this.id);
 72 	if (nodeVisitsForThisNode != null) {
 73 		for (var i=0; i < nodeVisitsForThisNode.length; i++) {
 74 			var nodeVisitForThisNode = nodeVisitsForThisNode[i];
 75 			if (nodeVisitForThisNode.nodeStates != null) {
 76 				for (var k=0;k<nodeVisitForThisNode.nodeStates.length;k++) {
 77 					var nodeState = nodeVisitForThisNode.nodeStates[k];
 78 					if (nodeState.isSubmit || nodeState.submit) {
 79 						return true;
 80 					}
 81 				}
 82 			}
 83 		}
 84 	}
 85 	return false;
 86 };

 88 /**
 89  * Returns the prompt for this node by loading the content and then
 90  * obtaining it from the object
 91  * @return the prompt for this node
 92  */
 93 AssessmentListNode.prototype.getPrompt = function() {
 94 	//get the content for the node
 95 	var contentJSON = this.content.getContentJSON();
 96 
 97 	var prompt = null;
 98 	
 99 	//see if the node content has an assessmentItem
100 	if(contentJSON.prompt != null) {
101 		//obtain the prompt
102 		prompt = contentJSON.prompt + "<br/>";	
103 	}
104 	// add the prompts from each part
105 	for (var i=0; i < contentJSON.assessments.length; i++) {
106 		var assessment = contentJSON.assessments[i];
107 		prompt += "Part " + (i+1) + ": " + assessment.prompt;
108 		// if radio assessment, also show choices
109 		if (assessment.type == "radio") {
110 			prompt += "<br/>    Choices:<br/>";
111 				for (var x=0; x<assessment.choices.length; x++) {
112 					prompt += "        " + assessment.choices[x].text;
113 					if (x != (assessment.choices.length - 1)) {
114 						prompt += "<br/>";
115 					}
116 				}
117 		}
118 		prompt += "<br/>";
119 	}
120 				
121 	//return the prompt
122 	return prompt;
123 };

126  * This is called when the node is exited
127  * @return
128  */
129 AssessmentListNode.prototype.onExit = function() {
130 	//check if the content panel has been set
131 	if(this.contentPanel) {
132 		try {
133 			/*
134 			 * check if the onExit function has been implemented or if we
135 			 * can access attributes of this.contentPanel. if the user
136 			 * is currently at an outside link, this.contentPanel.onExit
137 			 * will throw an exception because we aren't permitted
138 			 * to access attributes of pages outside the context of our
139 			 * server.
140 			 */
141 			if(this.contentPanel.onExit) {
142 				try {
143 					//run the on exit cleanup
144 					this.contentPanel.onExit();					
145 				} catch(err) {
146 					//error when onExit() was called, e.g. mysystem editor undefined
147 				}
148 			}	
149 		} catch(err) {
150 			/*
151 			 * an exception was thrown because this.contentPanel is an
152 			 * outside link. we will need to go back in the history
153 			 * and then trying to render the original node.
154 			 */
155 			history.back();
156 			//setTimeout(function() {thisObj.render(this.ContentPanel)}, 500);
157 		}
158 	}	
159 };
