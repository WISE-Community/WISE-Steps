1 /**
  2  * Sets the OutsideUrlNode type as an object of this view
  3  * @constructor
  4  * @author patrick lawler
  5  */
  6 View.prototype.OutsideUrlNode = {};
  7 
  8 View.prototype.OutsideUrlNode.commonComponents = [];
  9 
 10 View.prototype.OutsideUrlNode.generatePage = function(view){
 11 	this.view = view;
 12 	
 13 	var parent = document.getElementById('dynamicParent');
 14 	
 15 	/* wipe out old element */
 16 	parent.removeChild(document.getElementById('dynamicPage'));
 17 	
 18 	/* create new elements */
 19 	var pageDiv = createElement(document, 'div', {id:'dynamicPage', style:'width:100%;height:100%'});
 20 	var tInput = createElement(document, 'input', {id:'tInput', type:'text', size:'50', onkeyup:'eventManager.fire("sourceUpdated")'});
 21 	var urlText = document.createTextNode("URL: ");
 22 	
 23 	parent.appendChild(pageDiv);
 24 	pageDiv.appendChild(urlText);
 25 	pageDiv.appendChild(tInput);
 26 	
 27 	tInput.value = this.view.activeContent.getContentJSON().url;
 28 };
 29 
 30 /**
 31  * Updates this content object when requested, usually when preview is to be refreshed
 32  */
 33 View.prototype.OutsideUrlNode.updateContent = function(){
 34 	/* update content object */
 35 	this.view.activeContent.getContentJSON().url = document.getElementById('tInput').value;
 36 };
 37 
 38 /**
 39  * Get the array of common components which is an array with
 40  * string elements being the name of the common component
 41  */
 42 View.prototype.OutsideUrlNode.getCommonComponents = function() {
 43 	return this.commonComponents;
 44 };
 45 
 46 //used to notify scriptloader that this script has finished loading
 47 if(typeof eventManager != 'undefined'){
 48 	eventManager.fire('scriptLoaded', 'vle/node/outsideurl/authorview_outsideurl.js');
 49 };
