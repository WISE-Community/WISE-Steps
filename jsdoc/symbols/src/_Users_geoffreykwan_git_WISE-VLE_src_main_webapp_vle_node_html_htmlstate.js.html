  2 /**
  3  * @constructor
  4  * @extends Node
  5  * @param state
  6  * @param timestamp
  7  * @returns
  8  */
  9 function HTMLSTATE(state, timestamp) {
 10 	this.type = "html";
 11 	this.data = state;
 12 	
 13 	if(timestamp == null) {
 14 		this.timestamp = new Date().getTime();
 15 	} else {
 16 		this.timestamp = timestamp;
 17 	}
 18 };
 19 
 20 /**
 21  * Takes in a state JSON object and returns an HTMLSTATE object
 22  * @param stateJSONObj a state JSON object
 23  * @return a HTMLSTATE object
 24  */
 25 HTMLSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 26 	//create a new HTMLSTATE object
 27 	var htmlState = new HTMLSTATE();
 28 	
 29 	//set the attributes of the HTMLSTATE object
 30 	htmlState.data = stateJSONObj.data;
 31 	htmlState.timestamp = stateJSONObj.timestamp;
 32 	
 33 	//return the HTMLSTATE object
 34 	return htmlState;
 35 }
