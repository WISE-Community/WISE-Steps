  2 /**
  3  * The drawingDispatcher catches events specific to authoring individual
  4  * draw node steps and delegates them to the appropriate functions for
  5  * this view.
  6  */
  7 View.prototype.fillinDispatcher = function(type,args,obj){
  8 	if(type=='fillinTextUpdated'){
  9 		obj.FillinNode.fillinTextUpdated();
 10 	} else if(type=='fillinCreateFillin'){
 11 		obj.FillinNode.createFillin();
 12 	} else if(type=='fillinRemoveFillin'){
 13 		obj.FillinNode.removeFillin();
 14 	} else if(type=='fillinClick'){
 15 		obj.FillinNode.fillinClick(args[0], args[1]);
 16 	} else if(type=='fillinChangeSelected'){
 17 		obj.FillinNode.changeSelected(args[0]);
 18 	} else if(type=='fillinAddNewAllowable'){
 19 		obj.FillinNode.addNewAllowable(args[0]);
 20 	} else if(type=='fillinEntryChanged'){
 21 		obj.FillinNode.entryChanged(args[0]);
 22 	} else if(type=='fillinRemoveAllowable'){
 23 		obj.FillinNode.removeAllowable(args[0], args[1]);
 24 	};
 25 };
 26 
 27 //this list of events
 28 var events = [
 29 	'fillinTextUpdated',
 30 	'fillinCreateFillin',
 31 	'fillinRemoveFillin',
 32 	'fillinClick',
 33 	'fillinChangeSelected',
 34 	'fillinAddNewAllowable',
 35 	'fillinEntryChanged',
 36 	'fillinRemoveAllowable'
 37 ];
 38 
 39 /*
 40  * add all the events to the vle so the vle will listen for these events
 41  * and call the dispatcher function when the event is fired
 42  */
 43 for(var x=0; x<events.length; x++) {
 44 	componentloader.addEvent(events[x], 'fillinDispatcher');
 45 };
 46 
 47 //used to notify scriptloader that this script has finished loading
 48 if(typeof eventManager != 'undefined'){
 49 	eventManager.fire('scriptLoaded', 'vle/node/fillin/fillinEvents.js');
 50 };
