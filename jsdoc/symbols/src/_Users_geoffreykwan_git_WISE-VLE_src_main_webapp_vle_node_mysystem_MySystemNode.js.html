 26 /**
 27  * Overrides Node.getPrompt() function
 28  * Retrieves the question/prompt the student reads for this step.
 29  * @return a string containing the prompt. (the string may be an
 30  * html string)
 31  */
 32 MySystemNode.prototype.getPrompt = function() {
 33 	var prompt = "";
 34 	
 35 	if(this.content != null) {
 36 		//get the content for the node
 37 		var contentJSON = this.content.getContentJSON();
 38 
 39 		//see if the node content has an assessmentItem
 40 		if(contentJSON != null && contentJSON.prompt != null) {
 41 			prompt = contentJSON.prompt;	
 42 		}
 43 	}
 44 	
 45 	//return the prompt
 46 	return prompt;
 47 };

 58 /**
 59  * This is called when the node is exited
 60  * @return
 61  */
 62 MySystemNode.prototype.onExit = function() {
 63 	//check if the content panel has been set
 64 	if(this.contentPanel) {
 65 		try {
 66 			/*
 67 			 * check if the onExit function has been implemented or if we
 68 			 * can access attributes of this.contentPanel. if the user
 69 			 * is currently at an outside link, this.contentPanel.onExit
 70 			 * will throw an exception because we aren't permitted
 71 			 * to access attributes of pages outside the context of our
 72 			 * server.
 73 			 */
 74 			if(this.contentPanel.onExit) {
 75 				try {
 76 					//run the on exit cleanup
 77 					this.contentPanel.onExit();					
 78 				} catch(err) {
 79 					//error when onExit() was called, e.g. mysystem editor undefined
 80 				}
 81 			}	
 82 		} catch(err) {
 83 			/*
 84 			 * an exception was thrown because this.contentPanel is an
 85 			 * outside link. we will need to go back in the history
 86 			 * and then trying to render the original node.
 87 			 */
 88 			history.back();
 89 			//setTimeout(function() {thisObj.render(this.ContentPanel)}, 500);
 90 		}
 91 	}
 92 };
