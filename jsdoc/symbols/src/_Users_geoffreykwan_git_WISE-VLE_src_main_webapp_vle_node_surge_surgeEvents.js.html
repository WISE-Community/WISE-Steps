  2 /*
  3  * This handles events and calls the appropriate function to handle
  4  * the event.
  5  */
  6 View.prototype.surgeDispatcher = function(type,args,obj){
  7 	/*
  8 	 * check to see if the event name matches 
  9 	 */ 
 10 	if(type == 'surgeUpdateLevelString') {
 11 		obj.SurgeNode.updateLevelString(args);
 12 	} else if (type == 'surgeImportLevelStringToEditor') {
 13 		obj.SurgeNode.importLevelStringToEditor(args);
 14 	}
 15 };
 16 
 17 /*
 18  * this is a list of events that can be fired. when the event is fired,
 19  * the dispatcher function above will be called and then call the
 20  * appropriate function to handle the event.
 21  */
 22 var events = [
 23 	'surgeUpdateLevelString',
 24 	'surgeImportLevelStringToEditor'
 25 ];
 26 
 27 /*
 28  * add all the events to the vle so the vle will listen for these events
 29  * and call the dispatcher function when the event is fired
 30  */
 31 for(var x=0; x<events.length; x++) {
 32 	componentloader.addEvent(events[x], 'surgeDispatcher');
 33 };
 34 
 35 //used to notify scriptloader that this script has finished loading
 36 if(typeof eventManager != 'undefined'){
 37 	eventManager.fire('scriptLoaded', 'vle/node/surge/surgeEvents.js');
 38 };
