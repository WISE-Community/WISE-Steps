  9 /**
 10  * Generates the authoring page for open response node types
 11  */
 12 View.prototype.OpenResponseNode.generatePage = function(view){
 13 	this.view = view;
 14 	this.content = this.view.activeContent.getContentJSON();
 15 	
 16 	var parent = document.getElementById('dynamicParent');
 17 	
 18 	//wipe out old
 19 	parent.removeChild(document.getElementById('dynamicPage'));
 20 	
 21 	//create new
 22 	var pageDiv = createElement(document, 'div', {id:'dynamicPage', style:'width:100%;height:100%'});
 23 	var promptText = document.createTextNode("Question for Student:");
 24 	var linesText = document.createTextNode("Size of Student Response Box (# rows):");
 25 	var richTextEditorText = document.createTextNode("Use Rich Text Editor");
 26 	
 27 	pageDiv.appendChild(createBreak());
 28 	pageDiv.appendChild(createElement(document, 'div', {id: 'studentResponseBoxSizeContainer'}));
 29 	pageDiv.appendChild(createBreak());
 30 	pageDiv.appendChild(createBreak());
 31 	pageDiv.appendChild(createElement(document, 'div', {id: 'richTextEditorToggleContainer'}));
 32 	pageDiv.appendChild(createBreak());
 33 	pageDiv.appendChild(createBreak());
 34 	pageDiv.appendChild(createElement(document, 'div', {id: 'starterSentenceAuthoringContainer'}));
 35 	pageDiv.appendChild(createBreak());
 36 	pageDiv.appendChild(createBreak());
 37 
 38 	//display any additional text entry areas for peer/teacher review if necessary
 39 	if(this.view.activeNode.peerReview) {
 40 		pageDiv.appendChild(this.generatePeerReview(this.view.activeNode.peerReview));
 41 		pageDiv.appendChild(createBreak());
 42 		pageDiv.appendChild(createBreak());		
 43 	} else if(this.view.activeNode.teacherReview) {
 44 		pageDiv.appendChild(this.generateTeacherReview(this.view.activeNode.teacherReview));
 45 		pageDiv.appendChild(createBreak());
 46 		pageDiv.appendChild(createBreak());
 47 	}
 48 	
 49 	pageDiv.appendChild(promptText);
 50 	pageDiv.appendChild(createBreak());
 51 	pageDiv.appendChild(createElement(document, 'div', {id: 'promptContainer'}));
 52 	
 53 	parent.appendChild(pageDiv);
 54 };

 73 /**
 74  * Generates the starter sentence input options for this open response
 75  */
 76 View.prototype.OpenResponseNode.generateStarter = function(){
 77 	//create div for starterSentence options
 78 	var starterDiv = createElement(document, 'div', {id: 'starterDiv'});
 79 	
 80 	//create starter sentence options
 81 	var noStarterInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio', onclick: 'eventManager.fire("openResponseStarterOptionChanged")', value: '0'});
 82 	var starterOnClickInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio', onclick: 'eventManager.fire("openResponseStarterOptionChanged")', value: '1'});
 83 	var starterImmediatelyInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio', onclick: 'eventManager.fire("openResponseStarterOptionChanged")', value: '2'});
 84 	var starterSentenceInput = createElement(document, 'textarea', {id: 'starterSentenceInput', cols: '60', rows: '4', wrap: 'soft', onchange: 'eventManager.fire("openResponseStarterSentenceUpdated")'});
 85 	var noStarterInputText = document.createTextNode('Do not use starter sentence');
 86 	var starterOnClickInputText = document.createTextNode('Starter sentence available upon request');
 87 	var starterImmediatelyInputText = document.createTextNode('Starter sentence shows immediately');
 88 	var starterSentenceText = document.createTextNode('Starter sentence: ');
 89 	
 90 	starterDiv.appendChild(noStarterInput);
 91 	starterDiv.appendChild(noStarterInputText);
 92 	starterDiv.appendChild(createBreak());
 93 	starterDiv.appendChild(starterOnClickInput);
 94 	starterDiv.appendChild(starterOnClickInputText);
 95 	starterDiv.appendChild(createBreak());
 96 	starterDiv.appendChild(starterImmediatelyInput);
 97 	starterDiv.appendChild(starterImmediatelyInputText);
 98 	starterDiv.appendChild(createBreak());
 99 	starterDiv.appendChild(starterSentenceText);
100 	starterDiv.appendChild(starterSentenceInput);
101 	
102 	//set value of textarea
103 	starterSentenceInput.value = this.content.starterSentence.sentence;
104 	
105 	//set appropriate radio button and enable/disable textarea
106 	var displayOption = this.content.starterSentence.display;
107 	
108 	if(displayOption=='0'){
109 		starterSentenceInput.disabled = true;
110 		noStarterInput.checked = true;
111 	} else if(displayOption=='1'){
112 		starterOnClickInput.checked = true;
113 	} else if(displayOption=='2'){
114 		starterImmediatelyInput.checked = true;
115 	};
116 	
117 	return starterDiv;
118 };

237 /**
238  * Updates this content when the starter sentence option has changed.
239  */
240 View.prototype.OpenResponseNode.starterChanged = function(){
241 	var options = document.getElementsByName('starterRadio');
242 	var optionVal;
243 	
244 	/* get the checked option and update the content's starter sentence attribute */
245 	for(var q=0;q<options.length;q++){
246 		if(options[q].checked){
247 			this.content.starterSentence.display = options[q].value;
248 			if(options[q].value=='0'){
249 				document.getElementById('starterSentenceInput').disabled = true;
250 			} else {
251 				document.getElementById('starterSentenceInput').disabled = false;
252 			};
253 		};
254 	};
255 	
256 	/* fire source updated event */
257 	this.view.eventManager.fire('sourceUpdated');
258 };

271 /**
272  * Generates and returns an HTML Input Element of type checkbox 
273  * used to determine whether a rich text editor should be used.
274  */
275 View.prototype.OpenResponseNode.generateRichText = function(){
276 	var richTextChoice = createElement(document, 'input', {id: 'richTextChoice', type: 'checkbox', onclick: 'eventManager.fire("openResponseUpdateRichText")'});
277 	
278 	/* set whether this input is checked */
279 	richTextChoice.checked = this.content.isRichTextEditorAllowed;
280 	
281 	//disable the checkbox if the step is for a peer/teacher review sequence
282 	if(this.view.activeNode.peerReview || this.view.activeNode.teacherReview) {
283 		richTextChoice.disabled = true;
284 	}
285 	
286 	return richTextChoice;
287 };
