 10 	this.content = node.getContent().getContentJSON();
 11 	
 12 	if(node.studentWork != null) {

 15 		this.states = [];  
 16 	};
 17 	
 18 	this.showTopScore = true;
 19 };

 44 /**
 45  * Check the scores for all the steps that have the given tag and occur
 46  * before the current step in the project
 47  * @param tagName the tag name
 48  * @param functionArgs the arguments to this function
 49  * @returns the results from the check, the result object
 50  * contains a pass field and a message field
 51  */
 52 Surge.prototype.checkScoreForTags = function(tagName, functionArgs) {
 53 	//default values for the result
 54 	var result = {
 55 		pass:true,
 56 		message:''
 57 	};
 58 	
 59 	//get the minimum required score
 60 	var minScore = functionArgs[0];
 61 	
 62 	//array to accumulate the nodes that the student has not completed with a high enough score
 63 	var nodesFailed = [];
 64 
 65 	//the node ids of the steps that come before the current step and have the given tag
 66 	var nodeIds = this.view.getProject().getPreviousNodeIdsByTag(tagName, this.node.id);
 67 	
 68 	if(nodeIds != null) {
 69 		//loop through all the node ids that come before the current step and have the given tag
 70 		for(var x=0; x<nodeIds.length; x++) {
 71 			//get a node id
 72 			var nodeId = nodeIds[x];
 73 			
 74 			if(nodeId != null) {
 75 				//get the latest work for the node
 76 				var latestWork = this.view.state.getLatestWorkByNodeId(nodeId);
 77 				
 78 				if(latestWork != "") {
 79 					//get the top score for the step
 80 					var score = latestWork.response.topScore;
 81 					
 82 					if(score < minScore) {
 83 						//the score is not high enough
 84 						nodesFailed.push(nodeId);
 85 					}
 86 				} else {
 87 					/*
 88 					 * the student does not have any work for the step so we
 89 					 * will add it to our array
 90 					 */
 91 					nodesFailed.push(nodeId);
 92 				}
 93 			}
 94 		}
 95 	}
 96 	
 97 	if(nodesFailed.length != 0) {
 98 		//the student has failed at least one of the steps
 99 		
100 		//create the message to display to the student
101 		var message = "You must obtain a score higher than " + minScore + " on these steps before you can work on this step<br><br>";
102 		
103 		//loop through all the failed steps
104 		for(var x=0; x<nodesFailed.length; x++) {
105 			var nodeId = nodesFailed[x];
106 			
107 			//get the step number and title for the failed step
108 			var stepNumberAndTitle = this.view.getProject().getStepNumberAndTitle(nodeId);
109 			
110 			//add the step number and title to the message
111 			message += stepNumberAndTitle + "<br>";
112 		}
113 		
114 		//set the fields in the result
115 		result.pass = false;
116 		result.message = message;		
117 	}
118 	
119 	return result;
120 };

122 /**
123  * Get the accumulated score for all the steps with the given tag
124  * @param tagName the tag name
125  * @param functionArgs the arguments to this function (this is not actually used in this function)
126  * @return the accumulated score for all the steps that are tagged
127  */
128 Surge.prototype.getAccumulatedScoreForTags = function(tagName, functionArgs) {
129 	var result = 0;
130 	
131 	//get all the node ids with the given tag
132 	var nodeIds = this.view.getProject().getNodeIdsByTag(tagName);
133 
134 	if(nodeIds != null) {
135 		//loop through all the node ids
136 		for(var x=0; x<nodeIds.length; x++) {
137 			//get a node id
138 			var nodeId = nodeIds[x];
139 			
140 			if(nodeId != null) {
141 				//get the latest work for the node
142 				var latestWork = this.view.state.getLatestWorkByNodeId(nodeId);
143 				
144 				if(latestWork != "" && latestWork.response != null) {
145 					
146 					if(latestWork.response.topScore != null) {
147 						//get the top score for the step
148 						var score = latestWork.response.topScore;
149 						
150 						//try to convert the score to a number in case it is stored as a string
151 						score = parseFloat(score);
152 						
153 						//check if the score is a number
154 						if(!isNaN(score)) {
155 							//the score is a number
156 							result += score;							
157 						}
158 					}
159 				}
160 			}
161 		}
162 	}
163 	
164 	return result;
165 };

167 /**
168  * This function renders everything the student sees when they visit the step.
169  * This includes setting up the html ui elements as well as reloading any
170  * previous work the student has submitted when they previously worked on this
171  * step, if any.
172  */
173 Surge.prototype.render = function() {
174 	//whether we want to allow the student to work on this step
175 	var enableStep = true;
176 	
177 	/*
178 	 * a message to display to the student at the top of the step
179 	 * usually used to display error messages when they need to
180 	 * complete a previous step before being able to work on the
181 	 * current step
182 	 */
183 	var message = '';
184 	
185 	//the accumulated score among a family tag of steps
186 	var acuumulatedScore = 0;
187 	
188 	//process the tag maps if we are not in authoring mode
189 	if(this.view.authoringMode == null || !this.view.authoringMode) {
190 		
191 		//the tag maps
192 		var tagMaps = this.node.tagMaps;
193 		
194 		//check if there are any tag maps
195 		if(tagMaps != null) {
196 			
197 			//loop through all the tag maps
198 			for(var x=0; x<tagMaps.length; x++) {
199 				
200 				//get a tag map
201 				var tagMapObject = tagMaps[x];
202 				
203 				if(tagMapObject != null) {
204 					//get the variables for the tag map
205 					var tagName = tagMapObject.tagName;
206 					var functionName = tagMapObject.functionName;
207 					var functionArgs = tagMapObject.functionArgs;
208 					
209 					if(functionName == "checkScore") {
210 						//we will check the score for the steps that are tagged
211 						
212 						//get the result of the check
213 						var result = this.checkScoreForTags(tagName, functionArgs);
214 						enableStep = result.pass;
215 						message = result.message;
216 					} else if(functionName == "getAccumulatedScore") {
217 						//we will get the accumulated score for the steps that are tagged
218 						
219 						//get the accumulated score
220 						accumulatedScore = this.getAccumulatedScoreForTags(tagName, functionArgs);
221 						
222 						//display the accumulated score to the student
223 						$('#accumulatedScoreDiv').html('Accumulated Score: ' + accumulatedScore);
224 					} else if(functionName == "checkCompleted") {
225 						//we will check that all the steps that are tagged have been completed
226 						
227 						//get the result of the check
228 						var result = checkCompletedForTags(this, tagName, functionArgs);
229 						enableStep = result.pass;
230 						message = result.message;
231 					}
232 				}
233 			}
234 		}
235 	}
236 	
237 	if(enableStep) {
238 		//the student is able to work on this step
239 		
240 		//display any prompts to the student
241 		$('#promptDiv').html(this.content.prompt);
242 		
243 		var	swfHtml = '<object classid="clsid:d27cdb6e-ae6d-11cf-96b8-444553540000" codebase="http://download.macromedia.com/pub/shockwave/cabs/flash/swflash.cab#version=9,0,0,0" width="770" height="480" id="surge" align="middle">'
244 			+ '<param name="allowScriptAccess" value="sameDomain" />'
245 			+ '<param name="allowFullScreen" value="false" />'
246 			+ '<param name="wmode" value="opaque" />'
247 			+ '<param name="movie" value="surge.swf" /><param name="quality" value="high" /><param name="bgcolor" value="#ffffff" />'
248 			+ '<embed src="surge.swf" wmode="opaque" quality="high" bgcolor="#ffffff" width="770" height="480" name="surge" align="middle" allowScriptAccess="sameDomain" allowFullScreen="false" type="application/x-shockwave-flash" pluginspage="http://www.macromedia.com/go/getflashplayer" />'
249 			+ '</object>';
250 		
251 		$('#swfDiv').html(swfHtml);	
252 		
253 		if(this.showTopScore) {
254 			//show the top score for the current step
255 			this.displayTopScore();
256 		}
257 	}
258 	
259 	//if there is a message, display it to the student
260 	$('#messageDiv').html(message);
261 };

263 /**
264  * Display the top score to the student
265  */
266 Surge.prototype.displayTopScore = function() {
267 	//get the latest state
268 	var latestState = this.getLatestState();
269 	var topScoreMessage = '';
270 	
271 	//make the message to display the top score
272 	if(latestState != null) {
273 		topScoreMessage = 'Top Score: ' + latestState.response.topScore;
274 	} else {
275 		topScoreMessage = 'Top Score: 0';
276 	}
277 	
278 	//display the top score to the student
279 	$('#topScoreDiv').html(topScoreMessage);
280 };

312 /**
313  * This function retrieves the latest student work
314  *
315  * @return the latest state object or null if the student has never submitted
316  * work for this step
317  */
318 Surge.prototype.getLatestState = function() {
319 	var latestState = null;
320 	
321 	//check if the states array has any elements
322 	if(this.states != null && this.states.length > 0) {
323 		//get the last state
324 		latestState = this.states[this.states.length - 1];
325 	}
326 	
327 	return latestState;
328 };

330 /**
331  * This function retrieves the student work from the html ui, creates a state
332  * object to represent the student work, and then saves the student work.
333  * 
334  * note: you do not have to use 'studentResponseTextArea', they are just 
335  * provided as examples. you may create your own html ui elements in
336  * the .html file for this step (look at surge.html).
337  */
338 Surge.prototype.save = function(st) {
339 	
340 	var stateJSON = JSON.parse(st);
341 	
342 	/*
343 	 * create the student state that will store the new work the student
344 	 * just submitted
345 	 */
346 	var surgeState = new SurgeState(stateJSON);
347 	
348 	/*
349 	 * fire the event to push this state to the global view.states object.
350 	 * the student work is saved to the server once they move on to the
351 	 * next step.
352 	 */
353 	eventManager.fire('pushStudentWork', surgeState);
354 
355 	//push the state object into this or object's own copy of states
356 	this.states.push(surgeState);
357 	
358 	/*
359 	 * process the student work to see if we need to display a bronze,
360 	 * silver, or gold star next to the step in the nav menu
361 	 */
362 	this.node.processStudentWork(surgeState);
363 	
364 	if(this.showTopScore) {
365 		//show the top score for the current step
366 		this.displayTopScore();
367 	}
368 	
369 	/*
370 	 * post the current node visit to the db immediately without waiting
371 	 * for the student to exit the step.
372 	 */
373 	this.node.view.postCurrentNodeVisit(this.node.view.state.getCurrentNodeVisit());
374 };
