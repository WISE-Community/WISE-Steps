  2 /**
  3  * @constructor
  4  * @param node
  5  * @param view
  6  * @returns
  7  */
  8 function ASSESSMENTLIST(node, view) {
  9 	this.node = node;
 10 	this.view = view;
 11 	this.content = node.getContent().getContentJSON();
 12 	if(node.studentWork != null) {
 13 		this.states = node.studentWork; 
 14 	} else {
 15 		this.states = [];  
 16 	};
 17 	
 18 	//check if there is an associated node whose work we might display in this step
 19 	if(this.node.associatedStartNode != null) {
 20 		//get the node id for the associated node that the student will be reviewing work for
 21 		this.associatedStartNodeId = this.node.associatedStartNode;
 22 		
 23 		//get the associated node object
 24 		this.associatedStartNode = this.view.getProject().getNodeById(this.associatedStartNodeId);
 25 		
 26 		if(this.associatedStartNode != null) {
 27 			//get the content for the associated node
 28 			this.associatedStartNodeContent = this.associatedStartNode.getContent().getContentJSON();			
 29 		}
 30 	}
 31 	
 32 	//check if this step is being used for a peer review
 33 	if(this.node.peerReview != null) {
 34 		//values to be set after we retrieve the other student work that this student will be reviewing
 35 		this.otherStudentWorkgroupId = null;
 36 		this.otherStudentStepWorkId = null;
 37 		this.otherStudentNodeVisit = null;
 38 		this.showAuthorContent = false;
 39 
 40 		//this will store the peer review as an annotation
 41 		this.annotation = null;	
 42 		
 43 		if(this.node.peerReview == 'annotate') {
 44 			this.openPercentageTrigger = this.content.openPercentageTrigger;
 45 			this.openNumberTrigger = this.content.openNumberTrigger;
 46 			this.openLogicTrigger = this.content.openLogicTrigger;
 47 		}
 48 	}
 49 	
 50 	if(this.node.peerReview != null || this.node.teacherReview != null) {
 51 		//tell the node that it is part of a review sequence
 52 		this.node.setIsPartOfReviewSequence();
 53 		
 54 		//get the custom message to display to the student when the step is not open to work on
 55 		this.stepNotOpenCustomMessage = this.content.stepNotOpenCustomMessage;
 56 	}
 57 
 58 	//check if this step is locked
 59 	if(this.view != null && this.view.isLatestNodeStateLocked && this.view.isLatestNodeStateLocked(this.node.id)) {
 60 		//set this flag for future look up
 61 		this.locked = true;
 62 		
 63 		//tell the node that the student has completed it
 64 		this.node.setCompleted();
 65 	}
 66 	
 67 	if (this.isSubmitted()) {
 68 		//tell the node that the student has completed it
 69 		this.node.setCompleted();
 70 	}
 71 
 72 	this.stateChanged = false;
 73 };

113 /**
114  * Make the request to retrieve the other student work
115  */
116 ASSESSMENTLIST.prototype.retrieveOtherStudentWork = function() {
117 	//get the url
118 	var getPeerReviewUrl = this.view.getConfig().getConfigParam('getPeerReviewUrl');
119 	
120 	//get the parameters to retrieve the other student work
121 	var action = "studentRequest";
122 	var runId = this.view.getConfig().getConfigParam('runId');
123 	var workgroupId = this.view.getUserAndClassInfo().getWorkgroupId();
124 	var periodId = this.view.getUserAndClassInfo().getPeriodId();
125 	var nodeId = this.associatedStartNodeId;
126 	var openPercentageTrigger = this.openPercentageTrigger;
127 	var openNumberTrigger = this.openNumberTrigger;
128 	var openLogicTrigger = this.openLogicTrigger;
129 	var peerReviewAction = "annotate";
130 	var classmateWorkgroupIds = this.view.getUserAndClassInfo().getWorkgroupIdsInClass().toString();
131 	
132 	//compile the parameters into an object for cleanliness
133 	var getPeerReviewUrlArgs = {
134 			action:action,
135 			runId:runId,
136 			workgroupId:workgroupId,
137 			periodId:periodId,
138 			nodeId:nodeId,
139 			openPercentageTrigger:openPercentageTrigger,
140 			openNumberTrigger:openNumberTrigger,
141 			openLogicTrigger:openLogicTrigger,
142 			peerReviewAction:peerReviewAction,
143 			classmateWorkgroupIds:classmateWorkgroupIds
144 	};
145 	
146 	//make the request
147 	this.view.connectionManager.request('GET', 1, getPeerReviewUrl, getPeerReviewUrlArgs, this.retrieveOtherStudentWorkCallback, [this]);
148 };

150 /**
151  * Parses the other student work response and then renders everything in the vle.
152  * Everything includes, the prompt for the associated node, the other student work,
153  * and the text box for this current to write their peer review.
154  * @param text a JSON string containing a NodeVisit from the other student
155  * @param xml
156  * @param args contains the PEERREVIEWANNOTATE object so we can have access to it
157  */
158 ASSESSMENTLIST.prototype.retrieveOtherStudentWorkCallback = function(text, xml, args) {
159 	//get the or object
160 	var thisAl = args[0];
161 	
162 	//clear this variable to make sure we don't use old data
163 	thisAl.otherStudentNodeVisit = null;
164 	
165 	//check if there was any text response
166 	if(text != null && text != "") {
167 		//there was text returned so we will parse it, the text should be a NodeVisit in JSON form
168 		var peerWorkToReview = $.parseJSON(text);
169 		
170 		var peerWorkText = "";
171 		
172 		//handle error cases
173 		if(peerWorkToReview.error) {
174 			if(peerWorkToReview.error == 'peerReviewUserHasNotSubmittedOwnWork') {
175 				//the user has not submitted work for the original step
176 				thisAl.onlyDisplayMessage('<p>To start this step you must first submit a response in step <b><a style=\"color:blue\" onclick=\"eventManager.fire(\'renderNode\', [\'' + thisAl.view.getProject().getPositionById(thisAl.associatedStartNode.id) + '\']) \">' + thisAl.view.getProject().getStepNumberAndTitle(thisAl.associatedStartNode.id) + '</a></b> (link).</p>');
177 			} else if(peerWorkToReview.error == 'peerReviewNotAbleToAssignWork' || peerWorkToReview.error == 'peerReviewNotOpen') {
178 				/*
179 				 * server was unable to assign student any work to review, most likely because there was no available work to assign
180 				 * or
181 				 * the peer review has not opened yet
182 				 */
183 				
184 				var startNodeTitle = "";
185 				if(thisAl.associatedStartNode != null) {
186 					//get the step number and node title for the start node
187 					startNodeTitle = thisAl.view.getProject().getStepNumberAndTitle(thisAl.associatedStartNode.id);
188 				}
189 				
190 				if(thisAl.stepNotOpenCustomMessage != null && thisAl.stepNotOpenCustomMessage != "") {
191 					//use the custom authored message
192 					thisAl.onlyDisplayMessage(thisAl.stepNotOpenCustomMessage.replace(/associatedStartNode.title/g, startNodeTitle));
193 				} else {
194 					//use the default message
195 					thisAl.onlyDisplayMessage('<p>This step is not available yet.</p></p><p>More of your peers need to submit a response for step <b>"' + startNodeTitle + '"</b>. <br/>You will then be assigned a response to review.</p><p>Please return to this step again in a few minutes.</p>');					
196 				}
197 			}
198 			
199 			//check if we should show the authored work
200 			if(peerWorkToReview.error == 'peerReviewShowAuthoredWork') {
201 				//show the authored work for the student to review
202 				peerWorkText = thisAl.content.authoredWork;
203 				thisAl.showAuthorContent = true;
204 			} else {
205 				return;
206 			}
207 		} else {
208 			//set the variables for the other student
209 			thisAl.otherStudentWorkgroupId = peerWorkToReview.workgroupId;
210 			thisAl.otherStudentStepWorkId = peerWorkToReview.stepWorkId;
211 			thisAl.otherStudentNodeVisit = peerWorkToReview.nodeVisit;
212 			
213 			peerWorkText = thisAl.associatedStartNode.getPeerReviewOtherStudentWork(thisAl.otherStudentNodeVisit);
214 		}
215 		
216 		//reaplce \n with <br>
217 		peerWorkText = thisAl.replaceSlashNWithBR(peerWorkText);
218 		
219 		//show regular divs such as prompt, starter, and response and populate them
220 		thisAl.showDefaultDivs();
221 		//thisAl.showDefaultValues();
222 		
223 		//set more informative labels
224 		$('#promptLabelDiv').html('instructions');
225 		$('#responseLabelDiv').html('your feedback for <i>Team Anonymous</i>:');
226 		
227 		//display the prompt
228 		$('#originalPromptTextDiv').html(thisAl.associatedStartNode.getPeerReviewPrompt());
229 		$('#originalPromptDisplayDiv').show();
230 		
231 		/*
232 		 * display the other student's work or a message saying there is no other student work
233 		 * available yet
234 		 */
235 		$('#associatedWorkLabelDiv').html('work submitted by <i>Team Anonymous</i>:');		
236 		$('#associatedWorkTextDiv').html(peerWorkText);
237 		$('#associatedWorkDisplayDiv').show();
238 		
239 		//set the response if there were previous revisions 
240 		//thisAl.setResponse();
241 	}
242 	
243 	/*
244 	 * perform any final tasks after we have finished retrieving
245 	 * any other work and have displayed it to the student
246 	 */
247 	thisAl.doneRendering();
248 };

300 /**
301  * Render the AssessmentList
302  */
303 ASSESSMENTLIST.prototype.render = function() {
304 	//create the submit button
305 	$("#submitButtonDiv").html('<input id="submitButton" type="button" onclick="submit()" value="Submit the Questionnaire"></input>');
306 	
307 	if(this.node.peerReview == 'annotate' || this.node.teacherReview == 'annotate') {
308 		/*
309 		 * this is a peer or teacher review annotate node (part 2 of a review sequence)
310 		 * so we will create a save draft button
311 		 */
312 		if ($("#saveDraftButton").size() == 0) {
313 			$("#submitButtonDiv").before('<div class="buttonDiv ui-button ui-widget ui-state-default ui-corner-all ui-button-text-only"><a id="saveDraftButton" onClick="javascript:save();">SAVE DRAFT</a></div> ');
314 			
315 			/*
316 			 * disable the save draft button initially. the save
317 			 * draft button will be enabled once the student 
318 			 * modifies any of the assessment parts
319 			 */
320 			this.setSaveDraftUnavailable();
321 		}
322 	}
323 	
324 	/* if student has already done this step
325 	 * && isLockAfterSubmit is true
326 	 * && is displayAnswerSubmit is false,
327 	 * don't show the step, just display
328 	 * the fact that they've already completed it.
329 	 */
330 	if (this.content.isLockAfterSubmit
331 			&& !this.content.displayAnswerAfterSubmit
332 			&& this.isSubmitted()) {
333 		
334 		/* if this is part 2 of peer-review sequence and the student has completed it */
335 		if((this.node.peerReview == 'annotate' || this.node.teacherReview == 'annotate') && this.isSubmitted()) {
336 			//display this message in the step frame
337 			this.onlyDisplayMessage('<p>You have successfully reviewed the work submitted by <i>Team Anonymous</i>.</p><p>Well done!</p>');
338 		} else {
339 			//this is a regular assessment list so we will just lock the screen
340 			this.lockScreen();
341 		}
342 		return;
343 	};
344 	
345 	if((this.node.peerReview == 'annotate' || this.node.teacherReview == 'annotate') && this.content.isLockAfterSubmit) {
346 		//this is a peer or teacher review annotate step (part 2 of a review sequence)
347 		if(this.associatedStartNode != null) {
348 			if(this.node.peerReview != null) {
349 				//this is a peer review
350 				
351 				if(this.view.authoringMode) {
352 					/*
353 					 * we are in authoring mode so we will just display default values
354 					 * so the author can get an idea of how it will look to the students
355 					 */
356 					
357 					//set more informative labels
358 					$('#promptLabelDiv').html('instructions');
359 					
360 					//display the prompt
361 					$('#originalPromptTextDiv').html('[Prompt from the first peer review step will display here]');
362 					$('#originalPromptDisplayDiv').show();
363 					
364 					/*
365 					 * display the other student's work or a message saying there is no other student work
366 					 * available yet
367 					 */
368 					$('#associatedWorkLabelDiv').html('work submitted by <i>Team Anonymous</i>:');		
369 					$('#associatedWorkTextDiv').html('[Work from a random classmate will display here]');
370 					$('#associatedWorkDisplayDiv').show();
371 				} else {
372 					//this is the step where the student writes comments on their classmates work
373 					this.retrieveOtherStudentWork();					
374 				}
375 			} else if(this.node.teacherReview != null) {
376 				//this is a teacher review
377 				
378 				//this is the step where the student annotates the authored work
379 				this.displayTeacherWork();
380 			}
381 		}
382 	}
383 	
384 	/* render the overall prompt for the whole step */
385 	$("#promptDiv").html(this.content.prompt);
386 
387 	var assessmentHTML = "";
388 	/* for each assessment in the list, render them, including the state, if any */
389 	for (var i=0; i<this.content.assessments.length; i++) {
390 		var assessment = this.content.assessments[i];
391 		assessmentHTML += this.getHTML(assessment,i);
392 	};
393 	$("#assessmentsDiv").html(assessmentHTML);
394 	
395 	/* if student has already done this step, don't show the step, just display
396 	 * the fact that they've already completed it.
397 	 */
398 	if (this.isSubmitted() && this.content.isLockAfterSubmit) {
399 		this.lockScreen();
400 		return;
401 	};
402 	
403 	/*
404 	 * if this step does not lock, we will set submit button to be unavailable.
405 	 * when the student changes any of their answers in any field, the submit
406 	 * button will become available.
407 	 */
408 	if(!this.content.isLockAfterSubmit) {
409 		this.setSaveUnavailable();
410 	}
411 	
412 	this.node.view.eventManager.fire('contentRenderComplete', this.node.id, this.node);
413 };

415 ASSESSMENTLIST.prototype.submit = function() {
416 	var allCompleted = this.isAllPartsCompleted();
417 	
418 	if (allCompleted) {
419 		if (this.content.isLockAfterSubmit) {
420 			doLockStep=confirm("Click 'OK' to save and lock this step.  Your data will be saved and you will not be able to make any more changes.\nIf you want to keep working on this step, click 'Cancel'.");
421 			if (doLockStep==true) { 
422 				//disable the submit and save draft buttons
423 				this.setSaveUnavailable();
424 				this.setSaveDraftUnavailable();
425 				var isSubmit = true;
426 				this.save(isSubmit);
427 				this.lockScreen();
428 				
429 				// if this is a peer review 'annotate' part, also post the answer as annotation.
430 				if(this.node.peerReview != null) {
431 					//this is a peer review of a previous step
432 
433 					if(this.node.peerReview == 'annotate' && !this.showAuthorContent) {
434 						//send the annotation for the peer review
435 						var response = this.getResponsesToImportantParts();
436 						this.postAnnotation(response);
437 					}
438 				}
439 				
440 			} 
441 		} else {
442 			//disable the submit and save draft buttons
443 			this.setSaveUnavailable();
444 			this.setSaveDraftUnavailable();
445 			var isSubmit = true;
446 			this.save(isSubmit);
447 		};
448 	} else {
449 		/* all not completed yet, notify user and have them finish */
450 		alert("Please answer all the questions before submitting this questionnaire.");
451 	};
452 };

454 /**
455  * Obtains the student's responses to the important parts in the review sequence.
456  */
457 ASSESSMENTLIST.prototype.getResponsesToImportantParts = function() {
458 	var response = "";
459 	/* for each assessment in the list, render them, including the state, if any */
460 	for (var i=0; i<this.content.assessments.length; i++) {
461 		var assessment = this.content.assessments[i];
462 		if (assessment.isImportantReviewSequencePart) {
463 			var lastResponse = this.getLastSavedResponse(assessment);
464 			if (assessment.type == "radio") {
465 				lastResponse = this.getChoiceText(assessment,lastResponse);
466 			};
467 			response += lastResponse + "<br/><br/>";
468 		}
469 	};
470 	return response;
471 };

486 /**
487  * Send the response back as an annotation. Used for Peer Review. 
488  * @param response
489  */
490 ASSESSMENTLIST.prototype.postAnnotation = function(response) {
491 	//obtain the parameters needed to post the annotation
492 	var runId = this.view.getConfig().getConfigParam('runId');
493 	var nodeId = this.otherStudentNodeVisit.nodeId;
494 	var toWorkgroup = this.otherStudentWorkgroupId;
495 	var fromWorkgroup = this.view.getUserAndClassInfo().getWorkgroupId();
496 	var type = "comment";
497 	var value = response;
498 	var stepWorkId = this.otherStudentStepWorkId;
499 	var action = "peerReviewAnnotate";
500 	var periodId = this.view.getUserAndClassInfo().getPeriodId();
501 	
502 	//get the url
503 	var postAnnotationsUrl = this.view.getConfig().getConfigParam('postAnnotationsUrl');
504 	
505 	//compile the args into an object for cleanliness
506 	var postAnnotationsUrlArgs = {runId:runId,
507 								  nodeId: nodeId,
508 								  toWorkgroup:toWorkgroup,
509 								  fromWorkgroup:fromWorkgroup,
510 								  annotationType:type,
511 								  value:encodeURIComponent(value),
512 								  stepWorkId: stepWorkId,
513 								  action:action,
514 								  periodId:periodId};
515 	
516 	//create the view's annotations object if it does not exist
517 	if(this.view.annotations == null) {
518 		this.view.annotations = new Annotations();
519 	}
520 	
521 	//create the annotation locally to keep our local copy up to date
522 	var annotation = new Annotation(runId, nodeId, toWorkgroup, fromWorkgroup, type, value, stepWorkId);
523 	
524 	//add the annotation to the view's annotations
525 	this.view.annotations.updateOrAddAnnotation(annotation);
526 	
527 	//a callback function that does nothing
528 	var postAnnotationsCallback = function(text, xml, args) {};
529 	
530 	//post the annotation to the server
531 	this.view.connectionManager.request('POST', 1, postAnnotationsUrl, postAnnotationsUrlArgs, postAnnotationsCallback);
532 };

534 /**
535  * Returns true iff all parts have been completed 
536  */
537 ASSESSMENTLIST.prototype.isAllPartsCompleted = function() {
538 	var totalRadioCount = $('.choicesDiv').length; /* total number of radio assessments, including follow-ups */
539 	var radioNumChecked = 0;
540 	/* check that all radio button assessment have been answered */
541 	$('input:radio:checked').each(function() {
542 		radioNumChecked++;
543 	});
544 
545 	if (totalRadioCount != radioNumChecked) {
546 		return false;
547 	}
548 
549 	var textBoxesAllComplete = true;
550 	/* check that all textboxes have been populated */
551 	$(".textboxTextArea").each(function() { 
552 		if ($(this).val() == null || $(this).val() == "") {
553 			textBoxesAllComplete = false;
554 	    }
555 	}); 
556 	return textBoxesAllComplete;

559 /**
560  * Saves the current page.
561  */
562 ASSESSMENTLIST.prototype.save = function(isSubmit) {
563 	/*
564 	 * we need to save a new state if the state has changed
565 	 * or if the student is submitting their final answer
566 	 */
567 	if(this.stateChanged || isSubmit) {
568 		var alState = new ASSESSMENTLISTSTATE();
569 		alState.isSubmit = isSubmit;
570 		
571 		if(isSubmit) {
572 			alState.locked = true;			
573 			// denote that this step is completed.		
574 			this.node.setCompleted();
575 		}
576 		
577 		for (var i=0; i<this.content.assessments.length; i++) {
578 			var assessment = this.content.assessments[i];
579 			var assessmentState = {};
580 			assessmentState.id = assessment.id;
581 			assessmentState.type = assessment.type;
582 			assessmentState.response = this.getResponse(assessment);
583 			alState.assessments.push(assessmentState);
584 		};
585 		
586 		//fire the event to push this state to the global view.states object
587 		eventManager.fire('pushStudentWork', alState);
588 		
589 		this.states.push(alState);
590 		
591 		//disable the save draft button
592 		this.setSaveDraftUnavailable();	
593 		
594 		this.stateChanged = false;
595 	}
596 };

598 /**
599  * Returns student's current response to be saved for the specified assessment object.
600  * @param assessmentJSON JSON obj of assessment item.
601  *   Currently, we only support assessments of type 
602  *   - "radio" (for multiple choice)
603  *   - "text" (for open response)
604  * @return html string
605  */
606 ASSESSMENTLIST.prototype.getResponse = function(assessmentJSON) {
607 	if (assessmentJSON.type == "text") {
608 		var text = $("#"+assessmentJSON.id+"textbox").val();
609 		if (text != null && text != "") {
610 			var response = {};
611 			response.text = text;
612 			return response;
613 		} else {
614 			return null;
615 		}
616 	} else if  (assessmentJSON.type == "radio") {
617 		var maxScoreSoFar = 0;
618 		if (assessmentJSON.isAutoScoreEnabled) {
619 			// compute total possible score for this item by summing its choice scores
620 			for (var x=0; x < assessmentJSON.choices.length; x++) {				
621 				if (assessmentJSON.choices[x].choiceScore && maxScoreSoFar < assessmentJSON.choices[x].choiceScore) {
622 					maxScoreSoFar = assessmentJSON.choices[x].choiceScore;
623 				}
624 			}
625 		}
626 		var choiceId = $("input[name='" + assessmentJSON.id + "']:checked").val();
627 		if (choiceId != null && choiceId != "") {
628 			var response = {};
629 			response.id = choiceId;
630 			// get text of choice from assessment obj
631 			for (var x=0; x < assessmentJSON.choices.length; x++) {
632 				if (assessmentJSON.choices[x].id == choiceId) {
633 					response.text = assessmentJSON.choices[x].text;
634 					// if this radio item has a correct answer to it, add how the student did to the response
635 					if (assessmentJSON.isAutoScoreEnabled) {
636 						response.autoScoreResult = {
637 								isCorrect:assessmentJSON.choices[x].isCorrect,
638 								choiceScore:assessmentJSON.choices[x].choiceScore,
639 								maxScore:maxScoreSoFar
640 						};
641 						// include total possible score
642 					}
643 				}
644 			}
645 			return response; 
646 		} else {
647 			return null;
648 		}
649 	};
650 	return "";
651 };

653 /**
654  * Returns HTML string to display in the assessmentlist page for the specified assessment
655  * @param assessmentJSON JSON obj of assessment item.
656  *   Currently, we only support assessments of type 
657  *   - "radio" (for multiple choice)
658  *   - "text" (for open response)
659  *   @index: 0-based index of this assessment in the whole step.
660  * @return html string
661  */
662 ASSESSMENTLIST.prototype.getHTML = function(assessmentJSON,index) {
663 	var html = "<div id='"+assessmentJSON.id+"Div' class='"+assessmentJSON.type+" assessment'>" +
664 	    "<div id='"+assessmentJSON.id+"PromptDiv' class='promptDiv'><span class='partSpan'>" + (index+1) + ".  </span>"+assessmentJSON.prompt+"</div>";
665 	if (assessmentJSON.type == "text") {
666 		html += this.getHTMLText(assessmentJSON);
667 	} else if  (assessmentJSON.type == "radio") {
668 		html += this.getHTMLRadio(assessmentJSON);
669 	};
670 	html += "</div>";
671 	return html;
672 };

674 /**
675  * Returns HTML String to display a radio button assessment, like multiple choice. The assessment object
676  * is passed in as JSON object, and is of type radio.
677  * @param radioJSON
678  * @return
679  */
680 ASSESSMENTLIST.prototype.getHTMLRadio = function(radioJSON) {
681 	var html = "";
682 	
683 	if(radioJSON.choices.length > 0) {
684 		html = "<div id='"+radioJSON.id+"choicesDiv' class='choicesDiv'>";
685 		var lastChosenChoiceId = this.getLastSavedResponse(radioJSON);
686 		for (var i=0;i<radioJSON.choices.length;i++) {
687 			var choice = radioJSON.choices[i];
688 			html+="<div id='"+choice.id+"choiceDiv' class='choiceDiv'>";
689 			if (lastChosenChoiceId != null && lastChosenChoiceId == choice.id) {
690 			    html += "<input class='interactable' type='radio' name='"+radioJSON.id+"' value='"+choice.id+"' onchange='javascript:assessmentListChanged()' checked>";
691 			} else {
692 			    html += "<input class='interactable' type='radio' name='"+radioJSON.id+"' value='"+choice.id+"' onchange='javascript:assessmentListChanged()'>";
693 			}
694 			html += "<span class='choicetextSpan'>"+choice.text+"</span>"
695 			    +"</div>";
696 		};
697 		html += "</div>";
698 	}
699 
700 	return html;
701 };

703 /**
704  * Returns HTML String to display a textbox assessment, like open response. The assessment object
705  * is passed in as JSON object, and is of type text.
706  * @param textJSON
707  * @return
708  */
709 ASSESSMENTLIST.prototype.getHTMLText = function(textJSON) {
710 	var html = "<div id='"+textJSON.id+"textboxDiv' class='textboxDiv'>";
711 	var starterprompt = "";
712 	/* display starter prompt if specified */
713 	if (textJSON.starter != null && textJSON.starter.text != null) {
714 		if (textJSON.starter.display == 1 || textJSON.starter.display == 2) {
715 			starterprompt = textJSON.starter.text;
716 		};
717 	};
718 
719 	html += "<textarea id='"+textJSON.id+"textbox' class='textboxTextArea interactable' onkeypress='javascript:assessmentListChanged()'>";
720 	var lastSavedResponse = this.getLastSavedResponse(textJSON);
721 	if (lastSavedResponse != null) {
722 		html += lastSavedResponse;
723 	} else {
724 		html += starterprompt;
725 	}
726 	html += "</textarea>";
727 	html += "</div>";	
728 	return html;
729 };

731 /**
732  * Returns true iff student has "submitted" (ie locked) 
733  * this step, so no more editing can be done.
734  * @param assessmentJSON
735  * @return
736  */
737 ASSESSMENTLIST.prototype.isSubmitted = function() {
738 	if (this.states && this.states.length > 0) {
739 		for (var i=0; i < this.states.length; i++) {
740 			if (this.states[i].isSubmit || this.states[i].submit) {
741 				return true;
742 			}
743 		}
744 	}
745 	return false;
746 };

748 /**
749  * Returns last saved response for given assessment part.
750  */
751 ASSESSMENTLIST.prototype.getLastSavedResponse = function(assessmentJSON) {
752 	if (this.states && this.states.length > 0) {
753 		var latestState = this.states[this.states.length-1];
754 		for (var i=0; i<latestState.assessments.length; i++) {
755 			if (latestState.assessments[i].id == assessmentJSON.id) {
756 				//check if the response is null
757 				if(latestState.assessments[i].response != null) {
758 					//the student does have a previous response/answer
759 					if (assessmentJSON.type == "radio") {
760 						return latestState.assessments[i].response.id;
761 					} else if (assessmentJSON.type == "text") {
762 						return latestState.assessments[i].response.text;
763 					};
764 				} else {
765 					//the student does not have a previous response/answer
766 					return null;
767 				}
768 			};
769 		};
770 	};
771 	return null;
772 };

873 /**
874  * Replace \n with <br> so that the new lines are displayed to the
875  * students
876  * @param response an array containing the response string or just
877  * the response string
878  * @return the response string with all \n replaced with <br>
879  */
880 ASSESSMENTLIST.prototype.replaceSlashNWithBR = function(response) {
881 	var responseString = '';
882 	
883 	//check if the response is an array
884 	if(response.constructor.toString().indexOf('Array') != -1) {
885 		//the response is an array so we will obtain the string that is in it
886 		responseString = response[0];
887 	} else {
888 		//the response is a string
889 		responseString = response;
890 	}
891 	
892 	//replace \n with <br>
893 	return responseString.replace(/\n/g, '<br>');
894 };
