  2 /**
  3  * The multiplechoiceDispatcher catches events specific to authoring individual
  4  * multiple choice steps and delegates them to the appropriate functions for
  5  * this view.
  6  */
  7 View.prototype.multiplechoiceDispatcher = function(type,args,obj){
  8 	if(type=='mcCreateNewChoice'){
  9 		obj.MultipleChoiceNode.createNewChoice();
 10 	} else if(type=='mcClearCorrectChoice'){
 11 		obj.MultipleChoiceNode.clearCorrectChoice();
 12 	} else if(type=='mcShuffleChange'){
 13 		obj.MultipleChoiceNode.shuffleChange(args[0]);
 14 	} else if(type=='mcFeedbackOptionChange'){
 15 		obj.MultipleChoiceNode.feedbackOptionChange(args[0]);
 16 	} else if(type=='mcNumChoiceChanged'){
 17 		obj.MultipleChoiceNode.numChoiceChanged();
 18 	} else if(type=='mcXmlUpdated'){
 19 		obj.MultipleChoiceNode.updatePrompt();
 20 	} else if(type=='mcCorrectChoiceChange'){
 21 		obj.MultipleChoiceNode.correctChoiceChange(args[0]);
 22 	} else if(type=='mcRemoveChoice'){
 23 		obj.MultipleChoiceNode.removeChoice(args[0]);
 24 	} else if(type=='challengeNavigateToChanged'){
 25 		obj.MultipleChoiceNode.challengeNavigateToChanged();
 26 	} else if(type=='challengeAddNew'){
 27 		obj.MultipleChoiceNode.addNewAttemptScore();
 28 	} else if(type=='challengeRemoveLast'){
 29 		obj.MultipleChoiceNode.removeLastAttemptScore();
 30 	} else if(type=='challengeScoreChanged'){
 31 		obj.MultipleChoiceNode.scoreChanged(args[0]);
 32 	} else if(type=='branchCreateNewBranch'){
 33 		obj.MultipleChoiceNode.createNewBranch();
 34 	} else if(type=='branchRemoveBranch'){
 35 		obj.MultipleChoiceNode.removeBranch(args[0]);
 36 	} else if(type=='branchRemoveChoice'){
 37 		obj.MultipleChoiceNode.branchRemoveChoice(args[0],args[1]);
 38 	} else if(type=='branchAssociateAnswer'){
 39 		obj.MultipleChoiceNode.associateAnswer(args[0]);
 40 	} else if(type=='branchRemoveNode'){
 41 		obj.MultipleChoiceNode.removeNode(args[0],args[1]);
 42 	} else if(type=='branchAssociateNode'){
 43 		obj.MultipleChoiceNode.associateNode(args[0]);
 44 	} else if(type=='branchSelectAssociateAnswer'){
 45 		obj.MultipleChoiceNode.selectedAssociateAnswer(args[0]);
 46 	} else if(type=='branchSelectAssociateNode'){
 47 		obj.MultipleChoiceNode.selectedAssociateNode(args[0]);
 48 	}
 49 };
 50 
 51 //this list of events
 52 var events = [
 53 	'mcCreateNewChoice',
 54 	'mcClearCorrectChoice',
 55 	'mcShuffleChange',
 56 	'mcFeedbackOptionChange',
 57 	'mcNumChoiceChanged',
 58 	'mcXmlUpdated',
 59 	'mcCorrectChoiceChange',
 60 	'mcRemoveChoice',
 61 	'challengeNavigateToChanged',
 62 	'challengeAddNew',
 63 	'challengeRemoveLast',
 64 	'challengeScoreChanged',
 65 	'branchCreateNewBranch',
 66 	'branchRemoveBranch',
 67 	'branchRemoveChoice',
 68 	'branchAssociateAnswer',
 69 	'branchRemoveNode',
 70 	'branchAssociateNode',
 71 	'branchSelectAssociateAnswer',
 72 	'branchSelectAssociateNode'
 73 ];
 74 
 75 /*
 76  * add all the events to the vle so the vle will listen for these events
 77  * and call the dispatcher function when the event is fired
 78  */
 79 for(var x=0; x<events.length; x++) {
 80 	componentloader.addEvent(events[x], 'multiplechoiceDispatcher');
 81 };
 82 
 83 //used to notify scriptloader that this script has finished loading
 84 if(typeof eventManager != 'undefined'){
 85 	eventManager.fire('scriptLoaded', 'vle/node/multiplechoice/multipleChoiceEvents.js');
 86 };
