  9 /**
 10  * Generates the authoring page for open response node types
 11  */
 12 View.prototype.MultipleChoiceNode.generatePage = function(view){
 13 	this.view = view;
 14 	this.content = this.view.activeContent.getContentJSON();
 15 	this.nodeUtils = new Node().utils;
 16 	
 17 	var parent = document.getElementById('dynamicParent');
 18 	
 19 	//wipe out old
 20 	parent.removeChild(document.getElementById('dynamicPage'));
 21 	
 22 	//create new
 23 	var pageDiv = createElement(document, 'div', {id:'dynamicPage', style:'width:100%;height:100%'});
 24 	var answerText = document.createTextNode('Answers & Feedback:');
 25 	var shuffleText = document.createTextNode('Shuffle answers before next try');
 26 	var feedbackText = document.createTextNode('Feedback Options');
 27 	var challengeText = document.createTextNode('Challenge Question Setup');
 28 	var branchText = document.createTextNode('Branching Setup');
 29 	
 30 	pageDiv.appendChild(shuffleText);
 31 	pageDiv.appendChild(createBreak());
 32 	pageDiv.appendChild(this.generateShuffle());
 33 	pageDiv.appendChild(createBreak());
 34 	pageDiv.appendChild(feedbackText);
 35 	pageDiv.appendChild(createBreak());
 36 	pageDiv.appendChild(this.generateFeedbackOptions());
 37 	
 38 	/* if this is a branch node, we always want the bullets to
 39 	 * show up for a choice, so do not allow athors to modify it */
 40 	if(this.view.activeNode.getType()!='BranchNode'){
 41 		pageDiv.appendChild(createBreak());
 42 		pageDiv.appendChild(this.generateNumChoiceOption());
 43 	} 
 44 	
 45 	pageDiv.appendChild(createBreak());
 46 	pageDiv.appendChild(createBreak());
 47 	pageDiv.appendChild(document.createTextNode("Edit prompt:"));
 48 	pageDiv.appendChild(createElement(document, 'div', {id:'promptContainer'}));
 49 	pageDiv.appendChild(createBreak());
 50 	pageDiv.appendChild(createBreak());
 51 	
 52 	/* if the activeNode is a challengeNode, insert the challenge authoring options here */
 53 	if(this.view.activeNode.getType()=='ChallengeNode'){
 54 		pageDiv.appendChild(challengeText);
 55 		pageDiv.appendChild(this.generateChallengeSetup());
 56 		pageDiv.appendChild(createBreak());
 57 		pageDiv.appendChild(createBreak());
 58 	}
 59 	
 60 	pageDiv.appendChild(answerText);
 61 	pageDiv.appendChild(this.generateAnswers());
 62 	pageDiv.appendChild(createBreak());
 63 	pageDiv.appendChild(createElement(document, 'input', {type: "button", id: "createNewButton", value: "Create New Answer", onclick: 'eventManager.fire("mcCreateNewChoice")'}));
 64 	pageDiv.appendChild(createElement(document, 'input', {type: "button", value: "Reset Correct Answer selection", onclick: 'eventManager.fire("mcClearCorrectChoice")'}));
 65 	
 66 	parent.appendChild(pageDiv);
 67 	
 68 	/* if the activeNode is a BranchNode, insert the branch authoring here */
 69 	if(this.view.activeNode.getType()=='BranchNode'){
 70 		this.generateBranchSetup();
 71 	}
 72 	
 73 	/* generate drag and drop */
 74 	this.generateDD();
 75 };

155 /**
156  * Returns an element that contains the possible answers to the prompt for
157  * this content with the appopriate feedback element
158  */
159 View.prototype.MultipleChoiceNode.generateAnswers = function(){
160 	var answerElements = [];
161 	var answerDiv = createElement(document, 'div', {id: 'answerDiv', name: 'answerDiv'});
162 	var answerUL = createElement(document, 'ul', {id: 'answerUL', 'class': 'container'});
163 	
164 	
165 	var answers = this.content.assessmentItem.interaction.choices;
166 	for(var h=0;h<answers.length;h++){
167 		var answerText = answers[h].text;
168 		var answerTextLabel = document.createTextNode('Answer: ');
169 		var feedbackTextLabel = document.createTextNode('Feedback:');
170 		feedback = this.generateFeedback(answers[h], h);
171 		options = this.generateOptions(h, answers.length);
172 		var answerLI = createElement(document, 'li', {id: 'answerLI_' + h, name: 'answerLI', 'class': 'draggable'});
173 		var answer = createElement(document, 'input', {type: 'text', key: answers[h].identifier, name: "answerInput", value: answerText, wrap: 'hard', onkeyup: 'eventManager.fire("mcXmlUpdated")'});
174 		answerLI.appendChild(answerTextLabel);
175 		answerLI.appendChild(answer);
176 		answerLI.appendChild(createBreak());
177 		answerLI.appendChild(feedbackTextLabel);
178 		answerLI.appendChild(feedback);
179 		answerLI.appendChild(createBreak());
180 		answerLI.appendChild(options);
181 		answerUL.appendChild(answerLI);
182 	};
183 	
184 	answerDiv.appendChild(answerUL);
185 	return answerDiv;
186 };

234 /**
235  * Generates answers in xmlPage based on the choice elements on this page and the corresponding correctResponse values
236  */
237 View.prototype.MultipleChoiceNode.updateAnswer = function(parent){
238 	var answers = document.getElementsByName('answerLI');
239 	this.content.assessmentItem.interaction.choices = [];
240 	this.content.assessmentItem.responseDeclaration.correctResponse = [];
241 	
242 	for(var d=0;d<answers.length;d++){
243 		/* create new choice object, setting values for the fixed, identifier and text fields */
244 		var choice = {feedback:'', fixed:true, identifier: answers[d].childNodes[1].getAttribute('key'), text: answers[d].childNodes[1].value};
245 		
246 		/* set value for feedback field */
247 		if(answers[d].childNodes[4]!=null && answers[d].childNodes[4].getAttribute('type')=='text'){
248 			choice.feedback = answers[d].childNodes[4].value;
249 		};
250 		
251 		/* set the correct response if this choice has been chosen as the correct response */
252 		if(this.content.assessmentItem.interaction.hasInlineFeedback){
253 			var checked = answers[d].getElementsByTagName('input')[2].checked;
254 		} else {
255 			var checked = answers[d].getElementsByTagName('input')[1].checked;
256 		};
257 		
258 		if(checked){
259 			this.addCorrectChoice(choice.identifier);
260 		};
261 		
262 		this.content.assessmentItem.interaction.choices.push(choice);
263 	};
264 };

382 	var empty = false;
383 	
384 	for(var c=0;c<choices.length;c++){
385 		if(choices[c].value==null || choices[c].value==""){
386 			empty = true;
387 		};
388 	};
389

391 		this.view.notificationManager.notify("No choice fields are allowed to be empty or null. Exiting...", 3);

393 	};
394 	return true;

448 /**
449  * When needed, generates the challenge setup for authors to author ChallengeNodes.
450  */
451 View.prototype.MultipleChoiceNode.generateChallengeSetup = function(){
452 	/* create challenge setup structure */
453 	var challengeDiv = createElement(document, 'div', {id:'challengeDiv'});
454 	var navigateToDiv = createElement(document, 'div', {id:'navigateToDiv'});
455 	var attemptsDiv = createElement(document, 'div', {id:'attemptsDiv'});
456 	challengeDiv.appendChild(navigateToDiv);
457 	challengeDiv.appendChild(createBreak());
458 	challengeDiv.appendChild(attemptsDiv);
459 	
460 	/* create the navigateTo elements */
461 	var navToText = document.createTextNode('Select the step that students should review before being allowed to try again.');
462 	var navToSelect = createElement(document,'select', {id:'navigateToSelect', onchange:'eventManager.fire("challengeNavigateToChanged")'});
463 	var nodeIds = this.view.getProject().getNodeIds();
464 	var noneOption = createElement(document, 'option', {value:''});
465 	noneOption.text = '-- none --';
466 	navToSelect.appendChild(noneOption);
467 	
468 	/* add all of the nodes in the project */
469 	for(var a=0;a<nodeIds.length;a++){
470 		var nodeId = nodeIds[a];
471 		var stepNumberAndTitle = this.view.getProject().getStepNumberAndTitle(nodeId);
472 		var currentOption = createElement(document, 'option', {value:nodeId});
473 		currentOption.text = stepNumberAndTitle;
474 		navToSelect.appendChild(currentOption);
475 		
476 		/* check to see if this should be the selected node */
477 		if(this.content.assessmentItem.interaction.attempts.navigateTo==nodeIds[a]){
478 			navToSelect.selectedIndex = a + 1;
479 		}
480 	}
481 	
482 	challengeDiv.appendChild(navToText);
483 	challengeDiv.appendChild(createBreak());
484 	challengeDiv.appendChild(navToSelect);
485 	challengeDiv.appendChild(createBreak());
486 	challengeDiv.appendChild(createBreak());
487 	
488 	/* create the attempts elements */
489 	var attemptsText = document.createTextNode('Specify the score for each attempt by the student.');
490 	var attemptsTable = createElement(document, 'table', {id:'scoreAttemptsTable'});
491 	var ath = createElement(document, 'thead', {id:'scoreAttemptsTableHead'});
492 	var athRow = createElement(document, 'tr', {id:'scoreAttemptsHeaderRow'});
493 	var attemptTH = createElement(document, 'th', {id:'attemptTH'});
494 	var scoreTH = createElement(document, 'th', {id:'scoreHeadTH'});
495 	var atb = createElement(document, 'tbody', {id:'scoreAttemtpsTableBody'});
496 	var addNewButton = createElement(document, 'input', {type:'button', value:'Add new attempt/score', onclick:'eventManager.fire("challengeAddNew")'});
497 	var removeLastButton = createElement(document, 'input', {type:'button', value:'Remove last attempt/score', onclick:'eventManager.fire("challengeRemoveLast")'});
498 	
499 	challengeDiv.appendChild(attemptsText);
500 	challengeDiv.appendChild(createBreak());
501 	challengeDiv.appendChild(attemptsTable);
502 	challengeDiv.appendChild(addNewButton);
503 	challengeDiv.appendChild(removeLastButton);
504 	
505 	attemptsTable.appendChild(ath);
506 	attemptsTable.appendChild(atb);
507 	ath.appendChild(athRow);
508 	athRow.appendChild(attemptTH);
509 	athRow.appendChild(scoreTH);
510 	
511 	attemptTH.innerHTML = 'Attempt #';
512 	scoreTH.innerHTML = 'Score';
513 	
514 	/* add attempts/scores that are specified in the content */
515 	for(var attempt in this.content.assessmentItem.interaction.attempts.scores){
516 		var score = this.content.assessmentItem.interaction.attempts.scores[attempt];
517 		var currentRow = createElement(document, 'tr', {id:'attemptRow_' + attempt});
518 		var attemptTD = createElement(document, 'td', {id:'attemptTD_' + attempt});
519 		var scoreTD = createElement(document, 'td', {id:'scoreTD_' + attempt});
520 		var scoreInput = createElement(document, 'input', {type:'text', id:'scoreInput_' + attempt, size:3, onkeyup:'eventManager.fire("challengeScoreChanged","' + attempt + '")', value: score});
521 		
522 		atb.appendChild(currentRow);
523 		currentRow.appendChild(attemptTD);
524 		currentRow.appendChild(scoreTD);
525 		attemptTD.innerHTML = attempt;
526 		scoreTD.appendChild(scoreInput);
527 	}
528 	
529 	return challengeDiv;
530 };

588 View.prototype.MultipleChoiceNode.scoreChanged = function(attemptNum){

718 /**
719  * Returns a list of all Ids of the given type that are already in use for branching.
720  * 
721  * @param  string - type
722  * @return array - used ids of given type
723  */
724 View.prototype.MultipleChoiceNode.getAllUsedIdsOf = function(type){
725 	var returnIds = [];
726 	
727 	/* loop through all specified branches and concat the Ids of the given type */
728 	for(var d=0;d<this.content.branches.length;d++){
729 		if(type=='choice'){
730 			returnIds = returnIds.concat(this.content.branches[d].choiceIds);
731 		} else if(type=='branch'){
732 			returnIds.push(this.content.branches[d].id);
733 		} else if(type=='node'){
734 			returnIds = returnIds.concat(this.content.branches[d].branchIds);
735 		}
736 	}
737 	
738 	/* return the concatenated ids */
739 	return returnIds;
740 };

742 /**
743  * Returns an array of all choices that are not currently used by
744  * any of the branches.
745  * 
746  * @return array - choice
747  */
748 View.prototype.MultipleChoiceNode.getUnusedChoices = function(){
749 	var returnChoices = [];
750 	var choices = this.content.assessmentItem.interaction.choices;
751 	var usedChoiceIds = this.getAllUsedIdsOf('choice');
752 	
753 	for(var f=0;f<choices.length;f++){
754 		if(usedChoiceIds.indexOf(choices[f].identifier) == -1){
755 			returnChoices.push(choices[f]);
756 		}
757 	}
758 	
759 	return returnChoices;
760 };
