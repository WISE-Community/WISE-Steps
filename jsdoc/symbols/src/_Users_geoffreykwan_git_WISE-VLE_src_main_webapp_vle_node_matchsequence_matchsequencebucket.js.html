 26 /**
 27  * MatchSequence Bucket object
 28  *
 29  * SAMPLE bucketDOM:
 30  *   <gapMultiple identifier="yes" ordinal="false" numberOfEntries="0"><html><body class="slot">Items that contain one or more computers:)</body></html></gapMultiple>
 31  * @constructor
 32  */
 33 function MSBUCKET(bucketObj) {
 34   if (bucketObj) {
 35     this.isTargetBucket = true;
 36     this.obj = bucketObj;
 37     this.identifier = this.obj.identifier;
 38     this.text = this.obj.name;   // bucket header that students will see, can be html
 39     this.choices = [];
 40   } else {
 41     this.isTargetBucket = false;
 42     this.identifier = "sourceBucket";
 43     this.text = "Choices";
 44     this.choices = [];
 45   }
 46 }
 47 
 48 /**
 49  * Shuffles the choices in the bucket
 50  */
 51 MSBUCKET.prototype.shuffle = function(){
 52 	this.choices.shuffle();
 53 };
 54 
 55 /**
 56  * If prototype 'shuffle' for array is not found, create it
 57  */
 58 if(!Array.shuffle){
 59 	Array.prototype.shuffle = function (){ 
 60         for(var rnd, tmp, i=this.length; i; rnd=parseInt(Math.random()*i), tmp=this[--i], this[i]=this[rnd], this[rnd]=tmp);
 61     };
 62 };
