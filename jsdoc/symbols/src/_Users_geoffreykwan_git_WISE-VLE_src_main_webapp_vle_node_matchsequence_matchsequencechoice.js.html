 26 /**
 27  * MatchSequence Choice object
 28  * 
 29  * Params:
 30  *   choiceDOM
 31  *   bucket - bucket to which this choice belongs
 32  *
 33  * SAMPLE choiceDOM:
 34  *   <gapText identifier="ps3" matchMax="1"><html><body class="choice">Playstation 3</body></html></gapText>
 35  * @constructor
 36  */
 37 function MSCHOICE(choiceObj, bucket) {
 38     if (choiceObj) {
 39       this.obj = choiceObj;
 40       this.identifier = choiceObj.identifier;
 41       this.text = choiceObj.value;    // choice text that students will see, can be html
 42       this.bucket = bucket;
 43     } else {
 44       this.bucket = bucket;
 45     };
 46 };
