 78 /**
 79  * Renders the student work into the div. The grading tool will pass in a
 80  * div id to this function and this function will insert the student data
 81  * into the div.
 82  * 
 83  * @param divId the id of the div we will render the student work into
 84  * @param nodeVisit the student work
 85  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
 86  * div ids use this to prevent DOM conflicts such as when the show all work div
 87  * uses the same ids as the show flagged work div
 88  * @param workgroupId the id of the workgroup this work belongs to
 89  * 
 90  * Note: you may need to add code to this function if the student
 91  * data for your step is complex or requires additional processing.
 92  * look at SensorNode.renderGradingView() as an example of a step that
 93  * requires additional processing
 94  */
 95 FlashNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
 96 	var gradingText = "";
 97 	
 98 	//Get the latest student state object for this step
 99 	var flashState = nodeVisit.getLatestWork().response.data;
100 	
101 	/*
102 	 * get the step work id from the node visit in case we need to use it in
103 	 * a DOM id. we don't use it in this case but I have retrieved it in case
104 	 * someone does need it. look at SensorNode.js to view an example of
105 	 * how one might use it.
106 	 */
107 	var stepWorkId = nodeVisit.id;
108 	
109 	//var studentWork = flashState.getStudentWork();
110 	var studentWork = JSON.stringify(flashState);
111 	// print out the studentWork string
112 	gradingText += studentWork;
113 	
114 	//put the student work into the div
115 	$('#' + divId).html(gradingText);
116 };

118 /**
119 * Gets the swf file for the Flash node and inserts into the div, loading student data.
120 * The grading tool will pass in a div id to this function and this function will insert the student data
121 * into the div.
122 * 
123 * @param divId the id of the div we will render the student work into
124 * @param nodeVisit the student work
125 * @param childDivIdPrefix (optional) a string that will be prepended to all the 
126 * div ids use this to prevent DOM conflicts such as when the show all work div
127 * uses the same ids as the show flagged work div
128 * @param workgroupId the id of the workgroup this work belongs to
129 */
130 FlashNode.prototype.renderGradingViewFlash = function(divId, nodeVisit, childDivIdPrefix, workgroupId, nodeContent) {
131 	var gradingText = '';
132 	
133 	if(typeof nodeVisit.getLatestWork().response.data != "undefined"){
134 		gradingText += '<div id="alternateContent"><a href="http://www.adobe.com/go/getflashplayer">'+
135 			'<img src="http://www.adobe.com/images/shared/download_buttons/get_flash_player.gif" alt="Get Adobe Flash player" /></a></div>';
136 	
137 		//Get the latest student state object for this step
138 		var flashState = nodeVisit.getLatestWork().response.data;
139 		
140 		/*
141 		 * get the step work id from the node visit in case we need to use it in
142 		 * a DOM id. we don't use it in this case but I have retrieved it in case
143 		 * someone does need it. look at SensorNode.js to view an example of
144 		 * how one might use it.
145 		 */
146 		var stepWorkId = nodeVisit.id;
147 		
148 		//var studentWork = flashState.getStudentWork();
149 		var studentWork = JSON.stringify(flashState);
150 		
151 		//put the alternate content div into the DOM
152 		$('#' + divId).html(gradingText);
153 		
154 		var width = nodeContent.width;
155 		var height = nodeContent.height;
156 		
157 		// shrink flash dimensions to fit in grading display (TODO: provide enlarge button?)
158 		if(width > 500){
159 			height = height*500/width;
160 			width = 500;
161 		}
162 		var minPlayerVersion = nodeContent.minPlayerVersion;
163 		var activity_uri = nodeContent.activity_uri;
164 		var flashvars = nodeContent.flashvars;
165 		
166 		// add latest student data to flashvars (processed by Flash on load)
167 		flashvars.studentData = studentWork;
168 		
169 		// embed the swf in the DOM
170 		var params = {};
171 		params.quality = "high";
172 		params.wmode = "opaque";
173 		params.allowscriptaccess = "sameDomain";
174 		var attributes = {};
175 		attributes.id = "flashContent";
176 		attributes.name = "flashContent";
177 		attributes.styleclass = "flashContent";
178 		swfobject.embedSWF(activity_uri, "alternateContent", width, height, minPlayerVersion, "/vlewrapper/vle/swfobject/expressInstall.swf", flashvars, params, attributes);
179 	} else {
180 		gradingText += 'Error: Student data not found. Check Flash file to ensure export format is correct.';
181 		
182 		//put the error string into the div
183 		$('#' + divId).html(gradingText);
184 	}
185 		
186 };
