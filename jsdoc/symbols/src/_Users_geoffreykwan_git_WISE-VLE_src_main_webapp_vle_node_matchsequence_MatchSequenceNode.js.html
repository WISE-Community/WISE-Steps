 38 /**
 39  * Renders the student work into the div. The grading tool will pass in a
 40  * div id to this function and this function will insert the student data
 41  * into the div.
 42  * 
 43  * @param divId the id of the div we will render the student work into
 44  * @param nodeVisit the student work
 45  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
 46  * div ids use this to prevent DOM conflicts such as when the show all work div
 47  * uses the same ids as the show flagged work div
 48  * @param workgroupId the id of the workgroup this work belongs to
 49  */
 50 MatchSequenceNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
 51 	//create the match sequence object so we can reference the content later
 52 	var matchSequence = new MS(this, this.view);
 53 	
 54 	//get the latest state object
 55 	var state = nodeVisit.getLatestWork();
 56 	
 57 	var text = "";
 58 	
 59 	//loop through all the target buckets
 60 	for(var h=0;h<state.buckets.length;h++){
 61 		var bucket = state.buckets[h];
 62 		//get the text for the bucket
 63 		var bucketText = bucket.text;
 64 
 65 		/*
 66 		 * each bucket will be represented as following
 67 		 * 
 68 		 * ([bucketText]: choice1Text, choice2Text)
 69 		 */
 70 		text += "([" + bucketText + "]: ";
 71 		
 72 		var choicesText = "";
 73 		
 74 		//loop through the choices
 75 		for(var g=0;g<bucket.choices.length;g++){
 76 			//if this is not the first choice, add a comma to separate them
 77 			if(choicesText != "") {
 78 				choicesText += ", ";
 79 			}
 80 			
 81 			//add the bucket text
 82 			choicesText += bucket.choices[g].text;
 83 		};
 84 		
 85 		text += choicesText;
 86 		
 87 		//close the bucket and add a new line for easy reading
 88 		text += ")<br>";
 89 	};
 90 	
 91 	if(state.score != null) {
 92 		//get the max score
 93 		var maxScore = matchSequence.getMaxPossibleScore();
 94 		
 95 		text += "<br>";
 96 		text += "Auto-Graded Score: " + state.score + "/" + maxScore;
 97 	}
 98 	
 99 	$('#' + divId).append(text);
100 };
