 29 /**
 30  * This is the constructor for the object that will perform the logic for
 31  * the step when the students work on it. An instance of this object will
 32  * be created in the .html for this step (look at template.html)
 33  * @constructor
 34  * xTODO: rename TEMPLATE
 35  */
 36 function ExplanationBuilder(node, view) {
 37 	this.node = node;
 38 	this.view = view;
 39 	this.content = node.getContent().getContentJSON();
 40 
 41 	if(node.studentWork != null) {
 42 		this.states = node.studentWork; 
 43 	} else {
 44 		this.states = [];  
 45 	};
 46 
 47 	//var studentExplanation;
 48 	this.ideaBasket;
 49 	this.basketChanged = false;
 50 	this.stateChanged = false;
 51 	
 52 	this.question = '';
 53 	this.instructions = '';
 54 	this.bg = '';
 55 	this.latestState;
 56 };

 58 /**
 59  * This function renders everything the student sees when they visit the step.
 60  * This includes setting up the html ui elements as well as reloading any
 61  * previous work the student has submitted when they previously worked on this
 62  * step, if any.
 63  * 
 64  * xTODO: rename TEMPLATE
 65  * 
 66  * note: you do not have to use 'promptDiv' or 'studentResponseTextArea', they
 67  * are just provided as examples. you may create your own html ui elements in
 68  * the .html file for this step (look at template.html).
 69  */
 70 ExplanationBuilder.prototype.render = function() {
 71 	//get the question or prompt the student will read
 72 	var question = this.content.prompt;
 73 	
 74 	//get the instructions the stuent will read
 75 	var instructions = this.content.instructions;
 76 	
 77 	//get the background image that will be displayed in the drop area
 78 	var bg = this.content.background;
 79 	
 80 	if(question != null){
 81 		this.question = question;
 82 		$('#questionText').text(question);
 83 	}
 84 	
 85 	if(instructions != null) {
 86 		this.instructions = instructions;
 87 		$('#instructions').html(instructions);
 88 	}
 89 	
 90 	if(bg){
 91 		this.bg = bg;
 92 		$('#explanationIdeas').css('background-image','url(' + bg + ')');
 93 		$('#explanationIdeas').css('background-repeat','no-repeat');
 94 		$('#explanationIdeas').css('background-position','left top');
 95 	}
 96 
 97 	//initialize the UI and load the idea basket
 98 	this.initializeUI();
 99 };

169 /**
170  * This function retrieves the latest student work
171  * 
172  * xTODO: rename TEMPLATE
173  * 
174  * @return the latest state object or null if the student has never submitted
175  * work for this step
176  */
177 ExplanationBuilder.prototype.getLatestState = function() {
178 	var latestState = null;
179 
180 	//check if the states array has any elements
181 	if(this.states != null && this.states.length > 0) {
182 		//get the last state
183 		latestState = this.states[this.states.length - 1];
184 	}
185 
186 	return latestState;
187 };

189 /**
190  * This function retrieves the student work from the html ui, creates a state
191  * object to represent the student work, and then saves the student work.
192  * 
193  * xTODO: rename TEMPLATE
194  * 
195  * note: you do not have to use 'studentResponseTextArea', they are just 
196  * provided as examples. you may create your own html ui elements in
197  * the .html file for this step (look at template.html).
198  */
199 ExplanationBuilder.prototype.save = function() {
200 	//if the vle does not have the basket we will not save the student work for this step
201 	if(this.view.ideaBasket) {
202 		//the vle has the basket so we will save the student work
203 		
204 		//get the answer the student wrote
205 		var answer = $('#explanationText').val();
206 		
207 		//get the last state that was submitted (the previous time the student worked on this step)
208 		if(this.latestState != null) {
209 			if(this.latestState.answer != answer) {
210 				//the answer has changed since last time
211 				this.stateChanged = true;
212 			}
213 		}
214 		
215 		//check if the student has changed anything in this step
216 		if(this.stateChanged == true) {
217 			/*
218 			 * create the student state that will store the new work the student
219 			 * just submitted
220 			 * 
221 			 * xTODO: rename TEMPLATESTATE
222 			 * 
223 			 * make sure you rename TEMPLATESTATE to the state object type
224 			 * that you will use for representing student data for this
225 			 * type of step. copy and modify the file below
226 			 * 
227 			 * vlewrapper/WebContent/vle/node/template/templatestate.js
228 			 * 
229 			 * and use the object defined in your new state.js file instead
230 			 * of TEMPLATESTATE. for example if you are creating a new
231 			 * quiz step type you would copy the file above to
232 			 * 
233 			 * vlewrapper/WebContent/vle/node/quiz/quizstate.js
234 			 * 
235 			 * and in that file you would define QUIZSTATE and therefore
236 			 * would change the TEMPLATESTATE to QUIZSTATE below
237 			 */
238 			var explanationBuilderState = new ExplanationBuilderState(this.explanationIdeas, $('#explanationText').val(), Date.parse(new Date()));
239 
240 			/*
241 			 * fire the event to push this state to the global view.states object.
242 			 * the student work is saved to the server once they move on to the
243 			 * next step.
244 			 */
245 			eventManager.fire('pushStudentWork', explanationBuilderState);
246 
247 			//push the state object into this or object's own copy of states
248 			this.states.push(explanationBuilderState);
249 		}
250 		
251 		if(this.basketChanged == true) {
252 			//save the basket since it has been changed
253 			this.ideaBasket.saveIdeaBasket(this.view);
254 			
255 			//set this back to false now that we have saved it
256 			this.basketChanged = false;
257 		}		
258 	}
259 };

370 ExplanationBuilder.prototype.load = function(question, instructions, bg, explanationIdeas, answer){
371 	if(question != null){
372 		this.question = question;
373 		$('#questionText').text(question);
374 		//localStorage.question = question;
375 	}
376 	
377 	if(instructions != null) {
378 		this.instructions = instructions;
379 		$('#instructions').html(instructions);
380 	}
381 	
382 	if(bg){
383 		this.bg = bg;
384 		$('#explanationIdeas').css('background-image','url(' + bg + ')');
385 		$('#explanationIdeas').css('background-repeat','no-repeat');
386 		$('#explanationIdeas').css('background-position','left top');
387 		//localStorage.bg = bg;
388 	}
389 
390 	if(answer){
391 		this.answer = answer;
392 		$('#explanationText').val(answer);
393 		//localStorage.answer = answer;
394 	} else {
395 		//localStorage.answer = '';
396 	}
397 
398 	// clear out existing rows and explanation ideas
399 	$('#activeIdeas tbody tr').each(function(){
400 		$(this).remove();
401 	});
402 
403 	$('.exIdea').remove();
404 
405 	if(this.view.authoringMode) {
406 		//we are in preview step so we will create a dummy idea basket
407 		this.ideaBasket = new IdeaBasket('{"ideas":[],"deleted":[],"nextIdeaId":1,"id":-1,"runId":-1,"workgroupId":-1,"projectId":-1}');
408 	}
409 	
410 	if(this.ideaBasket == null) {
411 		//we do not have the basket
412 		
413 		/*
414 		 * display a message to the student and disable the buttons 
415 		 * and textarea so the student can't work on the step
416 		 */
417 		alert("Error: Failed to retrieve Idea Basket, you will not be able to work on this step, reload this step or refresh the VLE to try to load it again", 3);
418 		$('#addNew').attr('disabled', 'disabled');
419 		$('#save').attr('disabled', 'disabled');
420 		$('#explanationText').attr('disabled', 'disabled');
421 	} else {
422 		//we have the basket
423 		
424 		//loop through all the ideas in the basket
425 		for(var i=0; i<this.ideaBasket.ideas.length; i++){
426 			//add the idea to the display where we show all the ideas in the basket
427 			this.addRow(this.ideaBasket.ideas[i],true);
428 		}
429 		
430 		if(explanationIdeas){
431 			//loop through all the ideas that the student has used (dragged to the right)
432 			for (var a=0; a<explanationIdeas.length; a++){
433 				//get the attributes
434 				var id = explanationIdeas[a].id;
435 				var left = explanationIdeas[a].xpos;
436 				var top = explanationIdeas[a].ypos;
437 				var color = explanationIdeas[a].color;
438 				var lastAcceptedText = null;
439 				
440 				if(explanationIdeas[a].lastAcceptedText){
441 					//get the last accepted text
442 					lastAcceptedText = explanationIdeas[a].lastAcceptedText;
443 				} else {
444 					//get the text from the basket
445 					var idea = this.ideaBasket.getIdeaById(id);
446 					var ideaText = idea.text;
447 					explanationIdeas[a].lastAcceptedText = ideaText;
448 					lastAcceptedText = ideaText;
449 				}
450 				
451 				//determine if the idea is active or in the trash
452 				var isActive = this.ideaBasket.isIdeaActive(id);
453 				
454 				//add the explanation idea to 
455 				this.addExpIdea(this,true,isActive,id,left,top,color,lastAcceptedText);
456 			}
457 			
458 			this.explanationIdeas = explanationIdeas;
459 		}
460 	}
461 
462 	$('#colorPicker').hide();
463 	$('.exIdea').removeClass('selected');
464 
465 	this.selected = [];
466 
467 	this.init(this);
468 };

471 	//get the values for the current step

480 	this.basketChanged = true;

482 	//this.updateOrder();

651 /**
652  * Check if the idea is being used in any explanation builder steps (excluding the current step),
653  * if it is, we will display a confirmation popup that asks the

663 ExplanationBuilder.prototype.checkIfIdeaUsed = function(id) {
664 	// TODO: ignore current node in this check

1123 ExplanationBuilder.prototype.addStepUsedIn = function(ideaId, nodeId) {
1124 	//get the idea
1125 	var idea = this.ideaBasket.getIdeaById(ideaId);
1126 	
1127 	if(idea != null) {
1128 		//get the array of steps that the idea is used in

1130

1135 		
1136 		var stepAlreadyAdded = false;

1140 			//get a nodeId

1144 				//the nodeId is already in the array

1146 			}
1147 		}

1165

1177 			//get a nodeId

1260 				$('#editOtherSource').hide();

1262 			}
1263 			$('#editTags').val(ideas[i].tags);
1264 			$("input[name='editFlag']").each(function(){
1265 				if($(this).attr('value')==ideas[i].flag){
1266 					$(this).attr('checked', true);
1267 				} else {
1268 					$(this).attr('checked', false);
1269 				}
1270 			});
1271 			break;
1272 		}
1273 	}
