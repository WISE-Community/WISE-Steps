  2 /**
  3  * @constructor
  4  * @param state
  5  * @param timestamp
  6  * @returns
  7  */
  8 function MYSYSTEMSTATE(state, timestamp) {
  9 	this.type = "mysystem";
 10 	this.data = state;
 11 	
 12 	if(timestamp == null) {
 13 		this.timestamp = new Date().getTime();
 14 	} else {
 15 		this.timestamp = timestamp;
 16 	}
 17 };

 19 /**
 20  * Gets the xml format for the student data
 21  * @return an xml string with the student data
 22  */
 23 MYSYSTEMSTATE.prototype.getDataXML = function() {
 24 	var dataXML = "<data>" + this.data + "</data>";
 25 	dataXML += "<timestamp>" + this.timestamp + "</timestamp>";
 26 	return dataXML;
 27 }

 29 /**
 30  * Creates a state object from an xml object
 31  * @param stateXML an xml object
 32  * @return an MYSYSTEMSTATE object
 33  */
 34 MYSYSTEMSTATE.prototype.parseDataXML = function(stateXML) {
 35 	//obtain the data element
 36 	var dataElement = stateXML.getElementsByTagName("data")[0];
 37 	
 38 	//obtain the timestamp element
 39 	var timestampElement = stateXML.getElementsByTagName("timestamp")[0];
 40 	
 41 	//check if both elements exist
 42 	if(dataElement != null && timestampElement != null) {
 43 		//obtain the values for the data and timestamp
 44 		var data = dataElement.textContent;
 45 		var timestamp = timestampElement.textContent;
 46 		
 47 		//create an MYSYSTEMSTATE
 48 		var state = new MYSYSTEMSTATE(data, timestamp);
 49 		return state;
 50 	} else {
 51 		return null;
 52 	}
 53 }

 55 /**
 56  * Takes in a state JSON object and returns an MYSYSTEMSTATE object
 57  * @param stateJSONObj a state JSON object
 58  * @return a MYSYSTEMSTATE object
 59  */
 60 MYSYSTEMSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 61 	//create a new MYSYSTEMSTATE object
 62 	var mysystemState = new MYSYSTEMSTATE();
 63 	
 64 	//set the attributes of the MYSYSTEMSTATE object
 65 	mysystemState.data = stateJSONObj.data;
 66 	mysystemState.timestamp = stateJSONObj.timestamp;
 67 	
 68 	//return the MYSYSTEMSTATE object
 69 	return mysystemState;
 70 }
