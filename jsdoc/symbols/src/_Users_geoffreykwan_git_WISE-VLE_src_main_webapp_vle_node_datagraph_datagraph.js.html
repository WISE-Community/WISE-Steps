1 /**
  2  * @constructor
  3  * @param node
  4  * @returns
  5  */
  6 function DATAGRAPH(node){
  7 	this.node = node;
  8 	this.content = node.getContent().getContentJSON();
  9 	this.graphType;
 10 	this.mode;
 11 	this.table;
 12 	this.organizer;
 13 	this.range = {x:{low:0,high:0},y:{low:0,high:0}};
 14 	
 15 	if(this.state){
 16 		this.table = this.state.table;
 17 	} else {
 18 		this.table = this.copyTable();
 19 	}
 20 	
 21 	this.organizer = new DataOrganizer();
 22 	
 23 	if(this.content.options.display.start=='0'){
 24 		this.mode = '0';
 25 	} else if(this.content.options.display.which=='1' || this.content.options.display.which=='2'){
 26 		this.mode = '1';
 27 	} else {
 28 		this.mode = '2';
 29 	}
 30 };

 32 /**
 33  * Depending on the mode, renders the appropriate view.
 34  */
 35 DATAGRAPH.prototype.render = function(){
 36 	if(this.mode == '0'){
 37 		this.renderData();
 38 	} else if(this.mode == '1'){
 39 		this.renderGraph();
 40 	} else {
 41 		this.renderData(true);
 42 		this.renderGraph(true);
 43 	}
 44 	
 45 	this.node.view.eventManager.fire('contentRenderComplete', this.node.id, this.node);
 46 };

 90 /**
 91  * Renders the graph view.
 92  */
 93 DATAGRAPH.prototype.renderGraph = function(both){
 94 	if(!both){
 95 		$('#tableDiv').hide();
 96 	}
 97 	
 98 	/* show graph div */
 99 	$('#graphDiv').show();
100 	
101 	/* set prompt */
102 	$('#dgPromptDiv').html(this.content.prompt);
103 	
104 	/* hide title */
105 	$('#displayTitleDiv').hide();
106 	
107 	/* hide save and restore buttons */
108 	if(!both){
109 		$('#saveButt,#restoreButt').hide();
110 	}
111 	
112 	/* set toggle button with correct text */
113 	$('#switchButt').val('Display Data Table View');
114 	
115 	/* make toggle invisible if only one mode specified */
116 	if(!(this.content.options.display.which == '2' || this.content.options.display.which == '3')){
117 		$('#switchButt').hide();
118 	} else {
119 		$('#switchButt').show();
120 	}
121 	
122 	/* set up the editables based on the content */
123 	this.renderEditables();
124 	
125 	/* set graph options visible, then set individual options based on json settings */
126 	$('#graphOptions').show();
127 	
128 	/* retrieve the range if is at the default value (0 for both low and high) */
129 	if(this.range.y.low==0 && this.range.y.high==0 && this.range.x.low==0 && this.range.x.high==0){
130 		this.getRange();
131 	}
132 	
133 	/* set the range options */
134 	if(this.content.options.graph.range){
135 		$('#rangeTable').show();
136 		
137 		/* set the range elements */
138 		$('#yFromRange').val(this.range.y.low);
139 		$('#yToRange').val(this.range.y.high);
140 		$('#xFromRange').val(this.range.x.low);
141 		$('#xToRange').val(this.range.x.high);
142 	} else {
143 		$('#rangeTable').hide();
144 	}
145 	
146 	/* set the graph options */
147 	$('#graphTypeSelect').children().remove();
148 	$('#graphTypeSelect').show();
149 	
150 	for(var r in this.content.options.graph){
151 		if(this.content.options.graph[r] && r != 'axis' && r != 'range'){
152 			if(!this.graphType){
153 				this.graphType = r;
154 			}
155 			
156 			$('#graphTypeSelect').append('<option id="' + r + '_Opt" value="' + r + '">' + r + '</option>');
157 		}
158 	}
159 	
160 	/* set the graph width and height */
161 	$('#graphContainer,#graphTitleDiv,#graphYLabel,#graphXLabel').css('width', this.table.graphWidth);
162 	$('#graphContainer').css('height', this.table.graphHeight);
163 	
164 	/* set the current selected option to the current type */
165 	$('#graphTypeSelect').val(this.graphType);
166 
167 	/* draws actual graph */
168 	$.plot($('#graphContainer'), this.organizer.getData(this.table), this.getOptions());
169 	
170 	/* set the x and y labels and the graph title */
171 	$('#graphTitleDiv').html(this.table.title);
172 	$('#graphXLabel').html(this.table.xLabel);
173 	$('#graphYLabel').html(this.table.yLabel);
174 };

207 /**
208  * Generates the html table from the json table
209  */
210 DATAGRAPH.prototype.buildDataTable = function(){
211 	/* clear div */
212 	$('#tableDiv').children().remove();
213 	$('#tableDiv').show();
214 	
215 	/* create basic table structure */
216 	$('#tableDiv').append('<table id="dataTable"><thead id="dataTableHead"></thead><tbody id="dataTableBody"></tbody></table>');
217 	
218 	/* fillin table structure based on content */
219 	for(var a=0;a<this.table.rows.length;a++){
220 		$('#dataTableBody').append('<tr id="row_' + a + '"></tr>');
221 		for(var b=0;b<this.table.rows[a].cols.length;b++){
222 			var cell = this.table.rows[a].cols[b];
223 			$('#row_' + a).append('<td><input type="text" id="cell_in_' + a + '_' + b + '" onchange="updateCell(\'' + a + '\',\'' + b + '\')"></input></td>');
224 
225 			/* set the value of the input if specified */
226 			if(cell.value){
227 				$('#cell_in_' + a + '_' + b).val(cell.value);
228 			}
229 			
230 			/* disable input if student is not allowed to edit the field */
231 			if(!cell.options.editable){
232 				$('#cell_in_' + a + '_' + b).attr('disabled', true);
233 			}
234 		}
235 	}
236 };

249 /**
250  * Given the x and y coordinates, updates the associated value in this
251  * table with the user specified value.
252  * 
253  * @param x
254  * @param y
255  */
256 DATAGRAPH.prototype.updateCell = function(x, y){
257 	var cell = this.getCell(x,y);
258 	var val = $('#cell_in_' + x + '_' + y).val();
259 	
260 	/* validate input before updating, notify user if input invalid */
261 	if(cell.options.isLabel || !isNaN(val)){
262 		cell.value = val;
263 	} else {
264 		$('#cell_in_' + x + '_' + y).val(cell.value);
265 		this.node.view.notificationManager.notify('The cell you are entering data into only accepts numbers.',3);
266 	}

314 /**
315  * Gets the highest and the lowest values of the data that is
316  * currently in the table.
317  */
318 DATAGRAPH.prototype.getRange = function(){
319 	var yLow, yHigh, xLow, xHigh;
320 	
321 	for(var h=0;h<this.table.rows.length;h++){
322 		for(var i=0;i<this.table.rows[h].cols.length;i++){
323 			var cell = this.table.rows[h].cols[i];
324 			/* ignore non-numeric values */
325 			if(!isNaN(cell.value)){
326 				
327 				/* if this is not the independent column, then the values will affect
328 				 * the y range, if it is an indpendent column and is not qualitative, then
329 				 * the values will affect the x range. If it is independent and qualitative 
330 				 * or there is no independent variable, then the values of the x range are
331 				 * set later */
332 				if(i != this.table.independentIndex){
333 					/* high value not set, set it now */
334 					if(!yHigh){
335 						yHigh = cell.value;
336 					}
337 					
338 					/* low value not set, set it now */
339 					if(!yLow){
340 						yLow = cell.value;
341 					}
342 					
343 					/* set the high and low ranges to whichever is higher or lower respectively */
344 					yHigh = Math.max(yHigh, cell.value);
345 					yLow = Math.min(yLow, cell.value);
346 				} else if(!this.table.isQualitative){
347 					/* high value not set, set it now */
348 					if(!xHigh){
349 						xHigh = cell.value;
350 					}
351 					
352 					/* low value not set, set it now */
353 					if(!xLow){
354 						xLow = cell.value;
355 					}
356 					
357 					/* set the high and low ranges to whichever is higher or lower respectively */
358 					xHigh = Math.max(xHigh, cell.value);
359 					xLow = Math.min(xLow, cell.value);
360 				}
361 			}
362 		}
363 	}
364 	
365 	/* set the high point to 25 percent higher than the total range */
366 	this.range.y.high = yHigh + ((yHigh - yLow) * 0.25);
367 	this.range.y.low = yLow;
368 	
369 	/* If there is no independent variable or the x-axis is specified as qualitative
370 	 * in the content, then the range of the x-axis is based on the number of rows in
371 	 * the table, otherwise, we will use the values we set above */
372 	if(this.table.independentIndex == -1 || this.table.isQualitative){
373 		this.range.x.low = 0;
374 		this.range.x.high = this.table.rows.length - 1;
375 	} else {
376 		this.range.x.low = xLow;
377 		this.range.x.high = xHigh;
378 	}
379 };

381 /**
382  * Returns a flot formatted options object based on the current state
383  * 
384  * @return object - options
385  */
386 DATAGRAPH.prototype.getOptions = function(){
387 	var options = {
388 			lines: {show: false},
389 			points: {show: false},
390 			bars: {show: false},
391 			yaxis: {
392 				min: Math.min(0,this.range.y.low),
393 				max: this.range.y.high
394 			},
395 			xaxis: {
396 				min: this.range.x.low,
397 				max: this.range.x.high,
398 				ticks:[]
399 			}
400 	};
401 	
402 	/* set which type of graph to display */
403 	if(this.graphType=='bar'){
404 		options.bars.show = true;
405 	} else if(this.graphType=='line'){
406 		options.lines.show = true;
407 	} else if(this.graphType=='linePoint'){
408 		options.points.show = true;
409 		options.lines.show = true;
410 	} else if(this.graphType=='point'){
411 		options.points.show = true;
412 	}
413 	
414 	/* check to see if there is anything in the table */
415 	if(this.table.rows.length > 0){
416 		/* set the x-axis labels based on the table state */
417 		if(this.table.independentIndex < 0){
418 			/* Create ticks and/or labels based on number of data values for each column.
419 			 * Subtract one if there is a label. */
420 			var ticks = this.table.rows.length - ((this.table.rows[0].cols[0].options.isLabel) ? 1 : 0);
421 			for(var i=0;i<ticks;i++){
422 				options.xaxis.ticks.push(i);
423 			}
424 		} else if(this.table.isQualitative) {
425 			/* This is a qualitative graph, the ticks are sequential starting from zero for line and
426 			 * point type graphs, 0.5 for bar chart and the label is the value of each independent col
427 			 * cell, excluding any label cell. */
428 			var col = [];
429 			for(var j=0;j<this.table.rows.length;j++){
430 				if(j != this.table.titleIndex){
431 					col.push(this.table.rows[j].cols[this.table.independentIndex].value);
432 				}
433 			}
434 			
435 			for(var k=0;k<col.length;k++){
436 				options.xaxis.ticks.push([k + ((this.graphType == 'bar') ? 0.5 : 0),col[k]]);
437 			}
438 		} else {
439 			/* This is a quantitative graph, make the ticks null if it is a so that flot figures
440 			 * it all out from the data. */
441 			options.xaxis.ticks = null;
442 		}
443 	}
444 	
445 	return options;
446 };

452 function DataOrganizer(){

454 };

463 /**
464  * Parses the table and returns a formatted data object compatible
465  * with flot.
466  * 
467  * @param table
468  * @return object - data
469  */
470 DataOrganizer.prototype.getData = function(table){
471 	if(table){
472 		this.table = table;
473 	}
474 	
475 	if(!this.table){
476 		throw 'No table set in organizer, unable to parse the table.';
477 	} else {
478 		return this.parseTable();
479 	}
480 };
481 
482 /**
483  * Walks through the table and pushs the data into column objects, then
484  * creates and returns a data object compatible with flot.
485  * 
486  * @return object - data
487  */
488 DataOrganizer.prototype.parseTable = function(){
489 	var cols = [];
490 	
491 	/* walk through the cells, adding their data and option info to the appropriate columns */
492 	for(var c=0;c<this.table.rows.length;c++){
493 		for(var d=0;d<this.table.rows[c].cols.length;d++){
494 			/* get the cell */
495 			var cell = this.table.rows[c].cols[d];
496 			
497 			/* create a column object if none yet exists for this column */
498 			if(!cols[d]){
499 				cols[d] = this.getColumnTemplate();
500 			}
501 			
502 			/* If this is the label row, set the column label to this value. If a value
503 			 * already exists, throw an error. There should be only one label/column */
504 			if(c == this.table.titleIndex){
505 				if(cols[d].label){
506 					throw 'Invalid table structure, label conflict. Cannot continue!';
507 				} else {
508 					cols[d].label = cell.value;
509 				}
510 			} else {
511 				/* If this is not the independent column or if the independent column is not
512 				 * qualitative, then the cell value must be a number. Validate that this is the
513 				 * case. If not, set the value to null and notify the user, otherwise, set the
514 				 * value.
515 				 */
516 				if((this.table.independentIndex != d && isNaN(cell.value)) || (this.table.independentIndex == d && !this.table.isQualitative && isNaN(cell.value))){
517 					cols[d].data.push(null);
518 					this.node.view.notificationManager.notify('The cell at x:' + c + ' y:' + d + ' with the value: ' + cell.value + ' is not a valid number. Ignoring data.', 3);
519 				} else {
520 					cols[d].data.push(cell.value);
521 				}
522 			}
523 		}
524 	}
525 	
526 	/* Create the data object based on the created columns data. */
527 	var data = [];
528 	
529 	/* cycle through the cols and create the series for each */
530 	for(var f=0;f<cols.length;f++){
531 		/* the independent values are the x value for each series, so we do not
532 		 * want to process this as a series if it has been specified */
533 		if(f != this.table.independentIndex){
534 			/* get a new series template */
535 			var series = this.getSeriesTemplate();
536 			
537 			/* set the label for this series */
538 			series.label = cols[f].label;
539 			
540 			/* If there is no independent variable, or the independent variable is qualitative, the x-values
541 			 * for each series should be sequential starting at 0, otherwise, we need to use the quantitative
542 			 * values specified in the independent series. */
543 			for(var g=0;g<cols[f].data.length;g++){
544 				if(this.table.independentIndex < 0 || this.table.isQualitative){
545 					series.data[g] = [g, cols[f].data[g]];
546 				} else {
547 					series.data[g] = [cols[this.table.independentIndex].data[g], cols[f].data[g]];
548 				}
549 			}
550 			
551 			/* add the series to the data */
552 			data.push(series);
553 		}
554 	}
555 	
556 	/* if this is a quantitative graph, flot plots in the order of the data,
557 	 * so we need to sort it before returning */
558 	if(!this.table.isQualitative){
559 		for(var h=0;h<data.length;h++){
560 			data[h].data.sort(function(arr1,arr2){
561 				if(parseInt(arr1[0]) < parseInt(arr2[0])){
562 					return -1;
563 				} else if(parseInt(arr1[0])==parseInt(arr2[0])){
564 					return 0;
565 				} else {
566 					return 1;
567 				}
568 			});
569 		}
570 	}
571 	
572 	return data;
573 };
