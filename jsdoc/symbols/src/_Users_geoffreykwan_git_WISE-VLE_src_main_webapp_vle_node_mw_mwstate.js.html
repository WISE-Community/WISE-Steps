  3  * @constructor
  4  * @param state
  5  * @param timestamp
  6  * @returns

  8 function MWSTATE(state, timestamp) {
  9 	this.type = "mw";
 10 	this.data = state;
 11 	
 12 	if(timestamp == null) {
 13 		this.timestamp = new Date().getTime();
 14 	} else {
 15 		this.timestamp = timestamp;
 16 	}
 17 };

 19 /**
 20  * Takes in a state JSON object and returns an MWSTATE object
 21  * @param stateJSONObj a state JSON object
 22  * @return an MWSTATE object
 23  */
 24 MWSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 25 	//create a new MWSTATE object
 26 	var state = new MWSTATE();
 27 	
 28 	//set the attributes of the MWSTATE object
 29 	state.data = stateJSONObj.data;
 30 	state.timestamp = stateJSONObj.timestamp;
 31 	
 32 	//return the MWSTATE object
 33 	return state;
 34 };
