253 /**
254  * Generates the actual table based on its representation in the JSON content
255  */
256 View.prototype.DataGraphNode.generateTableDiv = function(){
257 	/* clean up any existing table elements and generate new table structure */
258 	$('#datagraphTableDiv').remove();
259 	$('#tableEditingDiv').append('<div id="datagraphTableDiv"></div>');
260 	$('#datagraphTableDiv').append('<table id="datagraphDataTable"><thead id="tHead"></thead><tbody id="tBody"></tbody></table>');
261 	
262 	/* if table elements have been specified in the content append them to the table,
263 	 * prepend an additional row to create the column tabs and prepend a column for
264 	 * the row tabs */
265 	if(this.content.table.rows.length>0){
266 		for(var a=-1;a<this.content.table.rows.length;a++){
267 			$('#tBody').append('<tr id="datagraphRow_' + a + '"></tr>');
268 			
269 			if(a==-1){
270 				/* this is the corner/column tab row, create the table elements
271 				 * for the corner and column tabs */
272 				for(var b=-1;b<this.content.table.rows[0].cols.length;b++){
273 					if(b==-1){
274 						/* this is the corner tab which selects all the table cells */
275 						$('#datagraphRow_' + a).append('<td><div id="tab_' + a + '_' + b + '" class="cornerTab tableCell" onclick="eventManager.fire(\'datagraphToggleSelected\',[\'' + a + '\',\'' + b + '\'])">Toggle All</div></td>');
276 					} else {
277 						/* this is a column tab which selects all cells in this column */
278 						$('#datagraphRow_' + a).append('<td><div id="tab_' + a + '_' + b + '" class="columnTab tableCell column_' + b + '" onclick="eventManager.fire(\'datagraphToggleSelected\',[\'' + a + '\',\'' + b + '\'])">Toggle Column</div></td>');
279 					}
280 				}				
281 			} else {
282 				for(var b=-1;b<this.content.table.rows[a].cols.length;b++){
283 					/* if this is the first column, this will be a tab td */
284 					if(b==-1){
285 						/* this is a row tab which selects all cells in this row */
286 						$('#datagraphRow_' + a).append('<td><div id="tab_' + a + '_' + b + '" class="rowTab tableCell row_' + a + '" onclick="eventManager.fire(\'datagraphToggleSelected\',[\'' + a + '\',\'' + b + '\'])">Toggle Row</div></td>');

301 /**
302  * Given the x,y coordinates of a cell, adds the appropriate classes

304  */
305 View.prototype.DataGraphNode.addCellClasses = function(a,b){
306 	/* all cells get the datagraphCell class */
307 	$('#datagraphCell_' + a + '_' + b).addClass('datagraphCell');
308 	
309 	/* add noneditable class to cells that specify so in the content */
310 	if(!this.content.table.rows[a].cols[b].options.editable){
311 		$('#datagraphCellInput_' + a + '_' + b).addClass('noneditable');
312 	}
313 	
314 	/* check for label row */
315 	if(a==this.content.table.titleIndex){
316 		$('#datagraphCell_' + a + '_' + b).addClass('label');
317 	}
318 	
319 	/* check for independent variable column */
320 	if(b==this.content.table.independentIndex){
321 		$('#datagraphCell_' + a + '_' + b).addClass('independent');
322 	}
323 	
324 	/* check for top left corner */
325 	if(a==0 && b==0){
326 		$('#datagraphCell_' + a + '_' + b).addClass('topLeftCorner');
327 	}
328 	
329 	/* check for bottom left corner */
330 	if(a==this.content.table.rows.length - 1 && b==0){
331 		$('#datagraphCell_' + a + '_' + b).addClass('bottomLeftCorner');
332 	}
333 	
334 	/* check for top right corner */
335 	if(a==0 && b==this.content.table.rows[0].cols.length - 1){
336 		$('#datagraphCell_' + a + '_' + b).addClass('toprightCorner');
337 	}
338 	
339 	/* check for bottom right corner */
340 	if(a==this.content.table.rows.length - 1 && b==this.content.table.rows[0].cols.length - 1){

391 /**
392  * Toggles selected cells based on cell clicked, current selected items
393  * and whether the shift key was held while clicking.
394  * 
395  * @param x - the row number
396  * @param y - the column number
397  */
398 View.prototype.DataGraphNode.toggleSelected = function(x, y){
399 	/* if shift key is down when clicked, then we do not want to clear any
400 	 * of the currently selected items, otherwise we do.
401 	 */
402 	if(!this.view.keystrokeManager.isShiftkeydown()){
403 		$('.activeSelect').not('#datagraphCellInput_' + x + '_' + y).not('#tab_' + x + '_' + y).removeClass('activeSelect');
404 	}
405 	
406 	/* determine id based on x and y args, then toggle the activeSelect class, if
407 	 * this is a tab cell, select/deslect all items in the corresponding row/column */
408 	var id = ((x < 0 || y < 0) ? '#tab_' : '#datagraphCellInput_') + x + '_' + y;
409 	if($(id).hasClass('activeSelect')){
410 		/* remove the class from the clicked element */
411 		$(id).removeClass('activeSelect');
412 		
413 		/* if this is a tab element remove activeSelect for all cells in column/row */
414 		if(x < 0 && y < 0){
415 			/* this is the ALL tab */
416 			$('.tableCell').removeClass('activeSelect');
417 		} else if(x < 0){
418 			/* this is a column tab */
419 			$('.column_' + y).removeClass('activeSelect');
420 		} else if(y < 0){
421 			/* this is a row tab */
422 			$('.row_' + x).removeClass('activeSelect');
423 		}
424 	} else {
425 		/* add the class to the clicked element */
426 		$(id).addClass('activeSelect');
427 		
428 		/* if this is a tab element, add the activeSelect for all cells in column/row */
429 		if(x < 0 && y < 0){
430 			/* this is the ALL tab */
431 			$('.tableCell').addClass('activeSelect');
432 		} else if(x < 0){
433 			/* this is a column tab */
434 			$('.column_' + y).addClass('activeSelect');
435 		} else if(y < 0){
436 			/* this is a row tab */
437 			$('.row_' + x).addClass('activeSelect');
438 		}
439 	}
440 	
441 	/* regenerate the options based on the table state and selected cells/columns/rows */
442 	this.generateSelectionOptions();
443 };

445 /**
446  * Generates and displays option elements based on the state of the 
447  * table and the selected elements.
448  */
449 View.prototype.DataGraphNode.generateSelectionOptions = function(){
450 	var opts = this.getSelectionOptionsObject();
451 	
452 	/* remove any existing options */
453 	$('#selectOptionsOptions').html('');
454 	
455 	/* check for no data in the table */
456 	if(this.content.table.rows.length == 0){
457 		opts.table.template.use = true;
458 		opts.independent.addIndependent.use = true;
459 		opts.label.addLabel.use = true;
460 		opts.row_column.addRow.use = true;
461 		opts.row_column.addColumn.use = true;
462 		
463 		/* use the options object to create the selection options */
464 		this.createSelectionOptions(opts);
465 		
466 		return;
467 	}
468 	
469 	/* add options that are always available */
470 	opts.table.startOver.use = true;
471 	opts.row_column.addRow.use = true;
472 	opts.row_column.addColumn.use = true;
473 	
474 	/* check for title row */
475 	if(this.content.table.titleIndex==-1){
476 		/* no title row */
477 		opts.label.addLabel.use = true;
478 	}
479 	
480 	/* check for independent column */
481 	if(this.content.table.independentIndex==-1){
482 		/* no independent column */
483 		opts.independent.addIndependent.use = true;
484 	}
485 	
486 	/* check for qualitative/quantitative */
487 	if(this.content.table.isQualitative){
488 		/* is a qualitative graph */
489 		opts.independent.setQuantitative.use = true;
490 	} else {
491 		/* is a quantative graph */
492 		opts.independent.setQualitative.use = true;
493 	}	
494 	
495 	/* check to see if nothing is selected, validate selected and set options 
496 	 * based on what is selected */
497 	if($('.activeSelect').not('.cornerTab,.columnTab,.rowTab').size() > 0){
498 		opts.cell.editable.use = true;
499 		
500 		/* find the rows and columns based on the currently selected */
501 		var uniqueRows = [];
502 		var uniqueCols = [];
503 		$('.activeSelect').not('.cornerTab,.columnTab,.rowTab').each(function(ndx){

1173  * Gets the user input and validates them, builds a table based on the input
1174  * and refreshes the table and preview.
1175  */
1176 View.prototype.DataGraphNode.submitBuildTable = function(){

1178 	var lab = parseInt($('#buildLabelSelect').val());
1179 	var ind = parseInt($('#independentSelect').val());
1180 	var type = $('#tableTypeSelect').val();
1181 	var rows = parseInt($('#buildRowsInput').val()) + lab;
1182 	var cols = parseInt($('#buildColsInput').val()) + ind;

1184 	/* validate that the rows and columns were specified and

1188 		return;

1190

1195 	}
1196 	
1197 	/* set the table with the template */
1198 	this.content.table = this.getTableTemplate();
1199 	
1200 	/* create table rows and columns */

1204 	
1205 	/* setup label row if needed */
1206 	if(lab==1){

1208 			this.content.table.rows[0].cols[j].options.isLabel = true;

1210 		this.content.table.titleIndex = 0;
1211 	}
1212 	
1213 	/* setup independent column if needed */
1214 	if(ind==1){
1215 		if(type==1){
1216 			for(var l=0;l<this.content.table.rows.length;l++){
1217 				this.content.table.rows[l].cols[0].options.isLabel = true;
1218 			}
1219 		}
1220 		
1221 		this.content.table.independentIndex = 0;
1222 	}
1223 	
1224 	/* set up type */
1225 	if(type==1){
1226 		this.content.table.isQualitative = true;
1227 	} else {
1228 		this.content.table.isQualitative = false;
1229 	}
1230
