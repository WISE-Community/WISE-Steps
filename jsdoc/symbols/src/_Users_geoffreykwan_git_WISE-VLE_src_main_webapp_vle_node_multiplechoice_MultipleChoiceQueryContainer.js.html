1 /**
  2  * MultipleChoiceQueryContainer
  3  * 
  4  * An object that contains query entries for a specific MultipleChoice node.
  5  */
  6 
  7 /**
  8  * Constructor
  9  * @param nodeId the nodeId for a specific node
 10  * @param prompt the prompt for this multiple choice question
 11  * @param choiceIdToValue an array that contains all the choiceIds, the
 12  * 		index/key of the array is the choiceId, the value is the user
 13  * 		displayed value
 14  */
 15 function MultipleChoiceQueryContainer(nodeId, prompt, choiceIdToValue) {
 16 	this.queryEntryArray = new Array();
 17 	this.nodeId = nodeId;
 18 	this.prompt = prompt;
 19 	
 20 	//an array that has (key, value) = (choiceId, choiceValue)
 21 	this.choiceIdToValue = choiceIdToValue;
 22 	this.choiceIdCount = new Array();
 23 	
 24 	for(var choiceId in this.choiceIdToValue) {
 25 		this.choiceIdCount[choiceId] = 0;
 26 	}
 27 }
 28 
 29 /**
 30  * Adds a query entry into this container
 31  * @param queryEntry a MultipleChoiceQueryEntry
 32  * @return
 33  */
 34 MultipleChoiceQueryContainer.prototype.addQueryEntry = function(queryEntry) {
 35 	this.queryEntryArray.push(queryEntry);
 36 	
 37 	//update the count for the choiceId that was chosen
 38 	this.choiceIdCount[queryEntry.choiceId] = this.choiceIdCount[queryEntry.choiceId] + 1;
 39 }
 40 
 41 /**
 42  * Retrieves the number of students that chose the choiceId
 43  * @param choiceId the id for the choice
 44  * @return the number of students that chose the choiceId
 45  */
 46 MultipleChoiceQueryContainer.prototype.retrieveCount = function(choiceId) {
 47 	return this.choiceIdCount[choiceId];
 48 }
 49 
 50 /**
 51  * Retrieves all the choiceIds for this node/question
 52  * @return an array containing all the choiceIds
 53  */
 54 MultipleChoiceQueryContainer.prototype.retrieveChoiceIds = function() {
 55 	var choices = new Array();
 56 	for(var choice in this.choiceIdToValue) {
 57 		choices.push(choice);
 58 	}
 59 	return choices;
 60 }
 61 
 62 /**
 63  * Returns an html string that displays accumulated data as well as 
 64  * individual student data for this node/step
 65  * @return an html string that displays aggregate data and individual 
 66  * 		student data
 67  */
 68 MultipleChoiceQueryContainer.prototype.printContainer = function() {
 69 	var print = "";
 70 	var total = 0;
 71 	
 72 	print += this.prompt;
 73 	
 74 	/*
 75 	 * display answers and totals and percentages that students chose e.g.
 76 	 * 
 77 	 * Accumulated Data
 78 	 * Bluefin tuna: 2 [33.333333333333336%]
 79 	 * Leopard Shark: 3 [50%]
 80 	 * Blue rockfish: 0 [0%]
 81 	 * Halibut: 1 [16.666666666666668%]
 82 	 * Total: 6
 83 	 */
 84 	print += "<hr>";
 85 	print += "<b>Accumulated Data</b>";
 86 	print += "<br>";
 87 	
 88 	var choiceIds = this.retrieveChoiceIds();
 89 	
 90 	/*
 91 	 * calculate the total number of choices students have chosen by
 92 	 * accumulating all the counts for all the choiceIds
 93 	 */
 94 	for(var x=0; x<choiceIds.length; x++) {
 95 		var choiceId = choiceIds[x];
 96 		total += this.retrieveCount(choiceId);
 97 	}
 98 	
 99 	/*
100 	 * loop through all the choices and retrieve the count for each
101 	 * choiceId and also calculate the percentage for the choiceId
102 	 * by dividing it with the total from above.
103 	 */
104 	for(var z=0; z<choiceIds.length; z++) {
105 		var choiceId2 = choiceIds[z];
106 		if (choiceId2 != "indexOf") {
107 			
108 		var choiceCount = this.retrieveCount(choiceId2);
109 		var percentage = 0;
110 		if(total != 0) {
111 			percentage = (choiceCount * 100 / total);
112 		}
113 		print += this.choiceIdToValue[choiceId2] + ": " + choiceCount + " [" + percentage + "%]";
114 		print += "<br>";
115 		}
116 	}
117 	
118 	print += "Total: " + total;
119 	print += "<br><hr><br>";
120 	
121 	/*
122 	 * display the answers for everyone that answered this question e.g.
123 	 * 
124 	 * Individual Data
125 	 * [User 1] answered Bluefin tuna
126 	 * [User 2] answered Leopard Shark
127 	 * [User 3] answered Bluefin tuna
128 	 * [User 4] answered Leopard Shark
129 	 * [User 5] answered Leopard Shark
130 	 * [User 6] answered Halibut
131 	 */
132 	print += "<b>Individual Data</b>";
133 	print += "<br>";
134 	
135 	//loop through all the query entries
136 	for(var y=0; y<this.queryEntryArray.length; y++) {
137 		//display the data in the query entry
138 		print += this.queryEntryArray[y].printEntry();
139 		print += "<br>";
140 	}
141 	
142 	return print;
143 }
