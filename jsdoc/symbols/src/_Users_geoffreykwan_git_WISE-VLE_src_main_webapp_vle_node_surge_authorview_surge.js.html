 33 /**

 36  * authors will use to author the step. For example if the step has a
 37  * text prompt that the student will read, this function will create
 38  * a textarea that will allow the author to type the text that the
 39  * student will see. You will also need to populate the textarea with

135 /**
136  * Imports content.levelString into the leveleditor. Will be called by an event later
137  * @param levelString
138  * @return
139  */
140 View.prototype.SurgeNode.importLevelStringToEditor = function() {	
141 	var levelString = this.content.levelString;
142 	thisMovie("leveleditor").editorImport(levelString);
143 };

183 /**
184  * Updates this content object when requested, usually when preview is to be refreshed
185  */
186 View.prototype.SurgeNode.updateContent = function(){
187 	/* update content object */
188 	this.view.activeContent.setContent(this.content);
189 };

192 /**
193  * Updates the content's level string to match that of what the user input
194  */
195 View.prototype.SurgeNode.updateLevelString = function(levelStringIn){
196 	/* update content */
197 	if (levelStringIn != null) {
198 		$('#levelStringTextArea').val(levelStringIn);
199 	} 
200 	
201 	// get the level content from the editor
202 	
203 	this.content.levelString = $('#levelStringTextArea').val();
204 	
205 	/*
206 	 * fire source updated event, this will update the preview
207 	 */
208 	this.view.eventManager.fire('sourceUpdated');
209 };
