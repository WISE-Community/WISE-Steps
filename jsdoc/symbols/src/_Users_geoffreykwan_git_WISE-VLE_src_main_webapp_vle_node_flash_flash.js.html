1 /**
  2  * This is the constructor for the object that will perform the logic for
  3  * the step when the students work on it. An instance of this object will
  4  * be created in the .html for this step (look at surge.html)
  5  * @constructor
  6  */
  7 function Flash(node) {
  8 	this.node = node;
  9 	this.view = node.view;
 10 	this.content = node.getContent().getContentJSON();
 11 	
 12 	if(node.studentWork != null) {
 13 		this.states = node.studentWork; 
 14 	} else {
 15 		this.states = [];  
 16 	};
 17 };

 19 //identify the Flash applet in the DOM - Provided by Adobe on a section on their site about the AS3 ExternalInterface usage.
 20 Flash.prototype.thisMovie = function(movieName) {
 21     if(navigator.appName.indexOf("Microsoft") != -1) {
 22         return window[movieName];
 23     } else {
 24         return document[movieName];
 25     }
 26 };

 28 // Call as3 function in identified Flash applet
 29 Flash.prototype.sendStateToFlash = function(value) {
 30 	swfobject.getObjectById("flashContent").importData(value);
 31 };
 32 
 33 //Call as3 function in identified Flash applet
 34 Flash.prototype.getStateFromFlash = function() {
 35 	return swfobject.getObjectById("flashContent").exportData();
 36 };

 38 /**
 39  * This function renders everything the student sees when they visit the step.
 40  * This includes setting up the html ui elements as well as reloading any
 41  * previous work the student has submitted when they previously worked on this
 42  * step, if any.
 43  */
 44 Flash.prototype.render = function() {
 45 	//display any prompts to the student
 46 	$('#promptDiv').html(this.content.prompt);
 47 	
 48 	var width = this.content.width;
 49 	var height = this.content.height;
 50 	var minPlayerVersion = this.content.minPlayerVersion;
 51 	var activity_uri = this.content.activity_uri;
 52 	
 53 	// flashvars can be used by authors to send starting parameters to the swf on load
 54 	var flashvars = this.content.flashvars;
 55 	
 56 	// if data logging is enabled, add latest student work to the flashvars
 57 	if(this.content.enableData){
 58 		var lastState = '';
 59 		// get latest student state
 60 		if (flash.getLatestState() != null) {
 61 			lastState = JSON.stringify(flash.getLatestState().response.data);
 62 		}
 63 		// add student data to flashvars (needs to be processed by Flash on load)
 64 		flashvars.studentData = lastState;
 65 	}
 66 
 67 	var params = {};
 68 	params.quality = "high";
 69 	params.wmode = "opaque";
 70 	params.allowscriptaccess = "sameDomain";
 71 	var attributes = {};
 72 	attributes.id = "flashContent";
 73 	attributes.name = "flashContent";
 74 	attributes.styleclass = "flashContent";
 75 	swfobject.embedSWF(activity_uri, "alternateContent", width, height, minPlayerVersion, "/vlewrapper/vle/swfobject/expressInstall.swf", flashvars, params, attributes);	
 76 };
 77 
 78 /**
 79  * This function retrieves the latest student work in WISE
 80  *
 81  * @return the latest state object or null if the student has never submitted
 82  * work for this step
 83  */
 84 Flash.prototype.getLatestState = function() {
 85 	var latestState = null;
 86 	
 87 	//check if the states array has any elements
 88 	if(this.states != null && this.states.length > 0) {
 89 		//get the last state
 90 		latestState = this.states[this.states.length - 1];
 91 	}
 92 	
 93 	return latestState;
 94 };

 96 /**
 97  * This function retrieves the student work from the flash swf, creates a state
 98  * object to represent the student work, and then saves the student work.
 99  * 
100  * note: you do not have to use 'studentResponseTextArea', they are just 
101  * provided as examples. you may create your own html ui elements in
102  * the .html file for this step (look at surge.html).
103  */
104 Flash.prototype.save = function() {
105 	if(this.content.enableData){
106 		var studentData = flash.getStateFromFlash();
107 		
108 		if(typeof studentData != "undefined"){
109 			var stateJSON = JSON.parse(studentData);
110 			
111 			/*
112 			 * create the student state that will store the new work the student
113 			 * just submitted
114 			 */
115 			var flashState = new FlashState(stateJSON);
116 			
117 			/*
118 			 * fire the event to push this state to the global view.states object.
119 			 * the student work is saved to the server once they move on to the
120 			 * next step.
121 			 */
122 			eventManager.fire('pushStudentWork', flashState);
123 
124 			//push the state object into this or object's own copy of states
125 			this.states.push(flashState);
126 		}
127 	}
128 };
