  8 /*
  9  * Add the name of the common component that this step will use. The
 10  * common components will be handled by the authoring tool. You will
 11  * need to create div elements with the appropriate id for the
 12  * authoring tool to insert the component into. Any additional custom
 13  * authoring components specific to your step type will be written

 28  * 'studentResponseBoxSizeContainer'
 29  * 'richTextEditorToggleContainer'
 30  * 'starterSentenceAuthoringContainer'
 31  * 
 32  * 
 33  * TODO: rename TemplateNode

 35 View.prototype.IdeaBasketNode.commonComponents = [];

 37 /**
 38  * Generates the authoring page. This function will create the authoring
 39  * components such as textareas, radio buttons, check boxes, etc. that
 40  * authors will use to author the step. For example if the step has a
 41  * text prompt that the student will read, this function will create
 42  * a textarea that will allow the author to type the text that the
 43  * student will see. You will also need to populate the textarea with
 44  * the pre-existing prompt if the step has been authored before.
 45  * 
 46  * TODO: rename TemplateNode
 47  */
 48 View.prototype.IdeaBasketNode.generatePage = function(view){
 49 	this.view = view;
 50 	
 51 	//get the content of the step
 52 	this.content = this.view.activeContent.getContentJSON();
 53 	
 54 	//get the html element that all the authoring components will be located
 55 	var parent = document.getElementById('dynamicParent');
 56 	
 57 	/*
 58 	 * wipe out the div that contains the authoring components because it
 59 	 * may still be populated with the authoring components from a previous
 60 	 * step the author has been authoring since we re-use the div id
 61 	 */
 62 	parent.removeChild(document.getElementById('dynamicPage'));
 63 
 64 	//create a new div that will contain the authroing components
 65 	var pageDiv = createElement(document, 'div', {id:'dynamicPage', style:'width:100%;height:100%'});
 66 	
 67 	//create the label for the textarea that the author will write the prompt in
 68 	var promptText = document.createTextNode("Prompt for Student:");
 69 	
 70 	/*
 71 	 * create the textarea that the author will write the prompt in
 72 	 * 
 73 	 * onkeyup will fire the 'templateUpdatePrompt' event which will
 74 	 * be handled in the <new step type name>Events.js file
 75 	 * 
 76 	 * For example if you are creating a quiz step you would look in
 77 	 * your quizEvents.js file.
 78 	 * 
 79 	 * when you add new authoring components you will need to create
 80 	 * new events in the <new step type name>Events.js file and then
 81 	 * create new functions to handle the event
 82 	 */
 83 	var promptTextArea = createElement(document, 'textarea', {id: 'promptTextArea', rows:'20', cols:'85', onkeyup:"eventManager.fire('ideaBasketUpdatePrompt')"});
 84 	
 85 	//add the authoring components to the page
 86 	pageDiv.appendChild(promptText);
 87 	pageDiv.appendChild(createBreak());
 88 	pageDiv.appendChild(promptTextArea);
 89 	pageDiv.appendChild(createBreak());
 90 
 91 	//add the page to the parent
 92 	parent.appendChild(pageDiv);
 93 	
 94 	//populate the prompt if this step has been authored before
 95 	this.populatePrompt();
 96 };

108 /**

110  * 
111  * TODO: rename TemplateNode

113 View.prototype.IdeaBasketNode.updateContent = function(){
114 	/* update content object */
115 	this.view.activeContent.setContent(this.content);
116 };
117 
118 /**
119  * Populate the authoring textarea where the user types the prompt that
120  * the student will read
121  * 
122  * TODO: rename TemplateNode
123  */
124 View.prototype.IdeaBasketNode.populatePrompt = function() {
125 	//get the prompt from the content and set it into the authoring textarea
126 	$('#promptTextArea').val(this.content.prompt);
127 };
128 
129 /**
130  * Updates the content's prompt to match that of what the user input
131  * 
132  * TODO: rename TemplateNode
133  */
134 View.prototype.IdeaBasketNode.updatePrompt = function(){
135 	/* update content */
136 	this.content.prompt = $('#promptTextArea').val();
137 	
138 	/*
139 	 * fire source updated event, this will update the preview
140 	 */
141 	this.view.eventManager.fire('sourceUpdated');
142 };
