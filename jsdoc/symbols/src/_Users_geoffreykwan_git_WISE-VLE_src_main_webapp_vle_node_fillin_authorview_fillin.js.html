 74 /**
 75  * Generates the fulltext from the content and sets the textarea's value with the full text
 76  */
 77 View.prototype.FillinNode.generateFullText = function(){
 78 	this.fullText = "";
 79 	this.fillin = [];
 80 	this.fillinIndexes = [];
 81 	
 82 	var currentIndex = 0;
 83 	var html = '';
 84 	
 85 	for(var x=0;x<this.content.assessmentItem.interaction.length;x++){
 86 		if(this.content.assessmentItem.interaction[x].type=='htmltext'){
 87 			html += this.spaceMaker(this.content.assessmentItem.interaction[x].text);
 88 			currentIndex += this.spaceMaker(this.content.assessmentItem.interaction[x].text).length;
 89 		} else if(this.content.assessmentItem.interaction[x].type=='textEntryInteraction'){
 90 			var indexes = currentIndex + "|";
 91 			this.fillin.push(this.content.assessmentItem.interaction[x].responseIdentifier);
 92 			html += this.retrieveInteractionText(this.content.assessmentItem.interaction[x].responseIdentifier);
 93 			currentIndex += this.retrieveInteractionText(this.content.assessmentItem.interaction[x].responseIdentifier).length;
 94 			indexes += currentIndex;
 95 			this.fillinIndexes.push(indexes);
 96 		};
 97 	};
 98 	
 99 	this.fullText = html;
100 	return html;
101 };

177 /**
178  * Removes fillin responseDeclaration and textEntryInteraction with
179  * the given identifier
180  */
181 View.prototype.FillinNode.removeFillin = function(){
182 	var identifier = this.getSelectedIdentifier();
183 	
184 	if(identifier){
185 		var declarations = this.content.assessmentItem.responseDeclarations;
186 		var foundIndex;
187 		
188 		/* find the right location to remove because the remainder will need to be updated */
189 		for(var h=0;h<declarations.length;h++){
190 			if(declarations[h].identifier==identifier){ 
191 				/* remove declaration */
192 				declarations.splice(h,1);
193 				
194 				/* remove associated interaction */
195 				this.content.assessmentItem.interaction.splice(this.content.assessmentItem.interaction.indexOf(this.getInteraction(identifier)),1);
196 				foundIndex = h;
197 			};
198 		};
199 		
200 		/* now update remaining identifiers */
201 		for(var t=foundIndex;t<declarations.length;t++){
202 			this.decrementIdentifiers(declarations[t]);
203 		};
204 		
205 		/* regenerate page */
206 		this.regeneratePage();
207 		
208 		/* fire source updated event */
209 		this.view.eventManager.fire('sourceUpdated');
210 	} else {
211 		this.view.notificationManager.notify('Please select a fillin that you wish to remove first', 3);
212 	};
213 };

399 /**
400  * Creates a new fillin based on the selected text in the promptInput text area
401  */
402 View.prototype.FillinNode.createFillin = function(){
403 	var start = document.getElementById('promptInput').selectionStart;
404 	var end = document.getElementById('promptInput').selectionEnd;
405 	
406 	/* make sure there are no overlaps */
407 	if(this.overlaps(start, end, true)){
408 		this.view.notificationManager.notify('The existing selection overlaps with another fillin. Either edit the existing fillin or remove it before proceeding. Exiting...', 3);
409 		return;
410 	};
411 	
412 	/* make sure text is selected */
413 	if(start==end){
414 		this.view.notificationManager.notify('Please select some text before creating a fillin. Exiting...', 3);
415 		return;
416 	};
417 	
418 	/* determine the location to insert new responseDeclaration in content */
419 	var location = 0;
420 	for(var k=0;k<this.fillinIndexes.length;k++){
421 		var startEnd = this.fillinIndexes[k].split('|');
422 		location = k;
423 		if(start<startEnd[0]){
424 			break;
425 		};
426 		if(k==this.fillinIndexes.length - 1){
427 			location = k + 1;
428 		};
429 	};
430 	
431 	/* then update content - textentry interaction, responsedeclaration */
432 	var identifier = 'response_' + location;
433 	this.createResponseDeclaration(identifier, this.fullText.substring(start, end), location);
434 	this.createTextInteraction(identifier, start, end, this.fullText.substring(start, end));
435 	
436 	/* regenerate page */
437 	this.regeneratePage();
438 	
439 	/* fire source updated event */
440 	this.view.eventManager.fire('sourceUpdated');
441 };

443 /**
444  * Creates a responseDeclaration given an identifier and a correctResponse
445  */
446 View.prototype.FillinNode.createResponseDeclaration = function(identifier, correctResponse, location){
447 	var declarations = this.content.assessmentItem.responseDeclarations;
448 	var interactions = this.getTextEntryInteractions();
449 	
450 	//get nextNode to insert responseDeclaration in right place and change identifiers for 
451 	//existing declarations appropriately as well as their associated textEntryInteractions
452 	if(declarations.length>0 && location < declarations.length){
453 		for(var s=location;s<declarations.length;s++){
454 			declarations[s].identifier = 'response_' + (s + 1);
455 			interactions[s].responseIdentifier = 'response_' + (s + 1);
456 		};
457 	};
458 	
459 	/* create declaration with appropriate values */
460 	var declaration = {correctResponses:[{response:correctResponse, value:'1'}], identifier:identifier};
461 	this.content.assessmentItem.responseDeclarations.splice(location,0,declaration);
462 };

478 /**
479  * Given an identifier that is associated with a responseDeclaration, the start and end indexes
480  * of the characters in the full text that were selected by the user and the actual text between
481  * those points, creates a textEntryInteraction element in the xmlPage at the appropriate point
482  * and modifies the existing elements to accomodate the changes.
483  */
484 View.prototype.FillinNode.createTextInteraction = function(identifier, start, end, fillinText){
485 	var runningText = '';
486 	for(var x=0;x<this.content.assessmentItem.interaction.length;x++){
487 		if(this.content.assessmentItem.interaction[x].type=='htmltext'){
488 			/* grab current text */
489 			var currentText = this.spaceMaker(this.content.assessmentItem.interaction[x].text);
490 			/* if start and end is included - then we need to modify existing element and insert new element here */
491 			if((runningText + currentText).length > start){
492 				/* create new textinteraction */
493 				var interaction = {type:'textEntryInteraction', responseIdentifier:identifier, expectedLength:(end-start)+2};
494 				
495 				/* create htmltext to hold any of the text after the new textinteraction text */
496 				var remainder = {type:'htmltext',text:currentText.substring(end - runningText.length, (runningText + currentText).length)};
497 				
498 				/* set the current html text to hold the text before the new textentryinteraction item */
499 				this.content.assessmentItem.interaction[x].text = currentText.substring(0, start - runningText.length);
500 				
501 				/* insert right after this one */
502 				this.content.assessmentItem.interaction.splice(x+1,0,interaction,remainder);
503 				return;
504 			} else {
505 				runningText += currentText;
506 			};
507 		} else if(this.content.assessmentItem.interaction[x].type=='textEntryInteraction'){
508 			runningText += this.retrieveInteractionText(this.content.assessmentItem.interaction[x].responseIdentifier);
509 		};
510 	};
511 };

513 /**
514  * if the provided start or end overlaps with an existing fillin
515  * this function returns true, otherwise, returns false
516  */
517 View.prototype.FillinNode.overlaps = function(start, end, create){
518 	var runningText = '';
519 	var difference = this.getDifference();
520 	var realStart;
521 	var realEnd;
522 	
523 	if(difference > 0){
524 		/* deleting text */
525 		realStart = start;
526 		realEnd = end + difference;
527 	} else {
528 		/* adding text */
529 		realStart = start + difference - 1;
530 		realEnd = end - 1;
531 	};
532 	
533 	if(difference==0 && !create){
534 		/* no change */
535 		return false;
536 	} else {
537 		/* changed look for overlap */
538 		for(var x=0;x<this.content.assessmentItem.interaction.length;x++){
539 			if(this.content.assessmentItem.interaction[x].type=='htmltext'){
540 				runningText += this.spaceMaker(this.content.assessmentItem.interaction[x].text);
541 			} else if(this.content.assessmentItem.interaction[x].type=='textEntryInteraction'){
542 				var currentText = this.retrieveInteractionText(this.content.assessmentItem.interaction[x].responseIdentifier);
543 				var fullLength = (runningText + currentText).length;
544 				if((runningText.length > realStart && runningText.length < realEnd) || (fullLength > realStart && runningText < realEnd)){
545 					return true;
546 				};
547 				if(realStart >= runningText.length && realStart < fullLength){
548 					return true;
549 				};
550 				runningText += currentText;
551 			};
552 		};
553 	};
554 	return false;
555 };

564 /**
565  * Determines what user input has changed for the fillin and updates the content accordingly.
566  */
567 View.prototype.FillinNode.fillinTextUpdated = function(){
568 	if(this.validate()){
569 		var difference = this.getDifference();
570 		if(difference==0){
571 			/* do nothing, no changes */
572 			return;
573 		} else if(difference>0){
574 			/* handle text removed case */
575 			var start = document.getElementById('promptInput').selectionStart;
576 			var end = start + difference;
577 			var runningText = '';
578 			for(var x=0;x<this.content.assessmentItem.interaction.length;x++){
579 				if(this.content.assessmentItem.interaction[x].type=='htmltext'){
580 					var currentText = this.spaceMaker(this.content.assessmentItem.interaction[x].text);
581 					if(start <= (runningText + currentText).length){
582 						/* this is where the change was made, update text */
583 						if(end <= (runningText + currentText).length){
584 							/* entire deletion occurs within this node */
585 							this.content.assessmentItem.interaction[x].text = currentText.substring(0, start - runningText.length) + currentText.substring(end - runningText.length, currentText.length);
586 							this.charCount -= difference;
587 							
588 							/* regenerate full text */
589 							this.generateFullText();
590 							
591 							/* fire source updated event */
592 							this.view.eventManager.fire('sourceUpdated');
593 							return;
594 						} else {
595 							/* deletion goes beyond this node */
596 							this.content.assessmentItem.interaction[x].text = currentText.substring(0, start - runningText.length);
597 							runningText += currentText;
598 							start = runningText.length;
599 						};
600 					} else {
601 						/* keep looking, the change was not here */
602 						runningText += currentText;
603 					};
604 				} else if(this.content.assessmentItem.interaction[x].type=='textEntryInteraction'){ //should never be the case that changed text is here.
605 					runningText += this.retrieveInteractionText(this.content.assessmentItem.interaction[x].responseIdentifier);
606 				};
607 			};
608 			/* set character count variable */
609 			this.charCount -= difference;
610 			
611 			/* generate the full text */
612 			this.generateFullText();
613 			
614 			/* fire source updated event */
615 			this.view.eventManager.fire('sourceUpdated');
616 		} else {
617 			/* handle text added case */
618 			var end = document.getElementById('promptInput').selectionStart;
619 			var start = end + difference;
620 			var newText = document.getElementById('promptInput').value.substring(start, end); //get new chars
621 			var runningText = '';
622 			if(this.content.assessmentItem.interaction.length>0){
623 				for(var x=0;x<this.content.assessmentItem.interaction.length;x++){
624 					if(this.content.assessmentItem.interaction[x].type=='htmltext'){
625 						if(this.content.assessmentItem.interaction[x].text && this.content.assessmentItem.interaction[x].text != ''){
626 							var currentText = this.spaceMaker(this.content.assessmentItem.interaction[x].text);
627 							if(start <= (runningText + currentText).length){
628 								/* this is where the change was made, update text */
629 								this.content.assessmentItem.interaction[x].text = currentText.substring(0, start - runningText.length) + newText + currentText.substring(start - runningText.length, currentText.length);
630 								
631 								/* update character count variable */
632 								this.charCount += newText.length;
633 								
634 								/* generate full text */
635 								this.generateFullText();
636 								
637 								/* fire source updated event */
638 								this.view.eventManager.fire('sourceUpdated');
639 								return;
640 							} else {
641 								runningText += currentText;
642 							};
643 						} else {
644 							/* no data in prompt - if this is the location add it here */
645 							this.content.assessmentItem.interaction[x].text = newText;
646 							
647 							/* update character count variable */
648 							this.charCount += newText.length;
649 							
650 							/* generate full text */
651 							this.generateFullText();
652 							
653 							/* fire source updated event */
654 							this.view.eventManager.fire('sourceUpdated');
655 							return;
656 						};
657 					} else if(this.content.assessmentItem.interaction[x].type=='textEntryInteraction'){
658 						runningText += this.retrieveInteractionText(this.content.assessmentItem.interaction[x].responseIdentifier);
659 					};
660 				};
661 			} else {
662 				/* this is the first added */
663 				this.content.assessmentItem.interaction.push({type:'htmltext', text:newText});
664 				
665 				/* update character count variable */
666 				this.charCount += newText.length;
667 				
668 				/* generate full text */
669 				this.generateFullText();
670 				
671 				/* fire source updated event */
672 				this.view.eventManager.fire('sourceUpdated');
673 				return;
674 			};
675 		};
676 	} else {
677 		this.regeneratePage();
678 	};
679 };
