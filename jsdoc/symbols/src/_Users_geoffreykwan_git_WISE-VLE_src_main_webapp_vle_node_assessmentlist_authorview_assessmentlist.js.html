 48 /**
 49  * Generates the options for an assessment list node
 50  */
 51 View.prototype.AssessmentListNode.generateOptions = function(){
 52 	var optionsHtml = '<table id="assessmentOptionsTable"><thead><tr><td>Display Answer After Submit</td><td>Lock After Submit</td><td>Complete All Before Exit</td></tr></thead>' +

 56 		'<label for="lockRadioNo"><input type="radio" name="lockRadio" id="lockRadioNo" value="false" onclick="eventManager.fire(\'assessmentOptionChanged\',\'lock\')"/>No</label></td>' +
 57 		'<td><label for="completeRadioYes"><input type="radio" name="completeRadio" id="completeRadioYes" value="true" onclick="eventManager.fire(\'assessmentOptionChanged\',\'complete\')"/>Yes</label></br>' +
 58 		'<label for="completeRadioNo"><input type="radio" name="completeRadio" id="completeRadioNo" value="false" onclick="eventManager.fire(\'assessmentOptionChanged\',\'complete\')"/>No</label></td></tr></tbody></table>';
 59 	
 60 	$('#assessmentOptionsContainer').append(optionsHtml);
 61 	
 62 	$('input[name=displayRadio]').filter('[value=' + this.content.displayAnswerAfterSubmit + ']').attr('checked', true);
 63 	$('input[name=lockRadio]').filter('[value=' + this.content.isLockAfterSubmit + ']').attr('checked', true);
 64 	$('input[name=completeRadio]').filter('[value=' + this.content.isMustCompleteAllPartsBeforeExit + ']').attr('checked', true);
 65 	
 66 	if(this.view.activeNode.peerReview) {
 67 		/*
 68 		 * this step is in a peer review sequence and requires
 69 		 * specific assessment list options to be set. we will disable
 70 		 * these options so that the author can't modify what they
 71 		 * have been set to.
 72 		 */
 73 		$('input[name=displayRadio]').attr('disabled', true);
 74 		$('input[name=lockRadio]').attr('disabled', true);
 75 		$('input[name=completeRadio]').attr('disabled', true);
 76 	}

 79 /**
 80  * Given the option type, updates the corresponding option in the content
 81  * with the user specified value.
 82  * 
 83  * @param String - type
 84  */
 85 View.prototype.AssessmentListNode.optionChanged = function(type){
 86 	var val = $('input[name=' + type + 'Radio]:checked').val();
 87 	val = (val==="true") ? true : false;
 88 	
 89 	if(type=='display'){
 90 		this.content.displayAnswerAfterSubmit = val;
 91 	} else if(type=='lock'){
 92 		this.content.isLockAfterSubmit = val;
 93 	} else if(type=='complete'){
 94 		this.content.isMustCompleteAllPartsBeforeExit = val;
 95 	}
 96 	
 97 	/* fire source updated event */
 98 	this.view.eventManager.fire('sourceUpdated');
 99 };

365 /**
366  * Generates the starter sentence input options for this open response item
367  */
368 View.prototype.AssessmentListNode.generateStarter = function(ndx){
369 	//create div for starterSentence options
370 	var starterDiv = createElement(document, 'div', {id: 'starterDiv_'+ndx});
371 	
372 	//create starter sentence options
373 	var noStarterInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio_'+ndx, onclick: "eventManager.fire('assessmentlistStarterOptionChanged','"+ ndx +"')", value: '0'});
374 	var starterOnClickInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio_'+ndx, onclick: "eventManager.fire('assessmentlistStarterOptionChanged','"+ ndx +"')", value: '1'});
375 	var starterImmediatelyInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio_'+ndx, onclick: "eventManager.fire('assessmentlistStarterOptionChanged','"+ ndx +"')", value: '2'});
376 	var starterSentenceInput = createElement(document, 'textarea', {id: 'starterSentenceInput_'+ndx, cols: '60', rows: '4', wrap: 'soft', onchange: "eventManager.fire('assessmentlistStarterSentenceUpdated','"+ ndx +"')"});
377 	var noStarterInputText = document.createTextNode('Do not use starter sentence');
378 	var starterOnClickInputText = document.createTextNode('Starter sentence available upon request');
379 	var starterImmediatelyInputText = document.createTextNode('Starter sentence shows immediately');
380 	var starterSentenceText = document.createTextNode('Starter sentence: ');
381 	
382 	starterDiv.appendChild(noStarterInput);
383 	starterDiv.appendChild(noStarterInputText);
384 	starterDiv.appendChild(createBreak());
385 	starterDiv.appendChild(starterOnClickInput);
386 	starterDiv.appendChild(starterOnClickInputText);
387 	starterDiv.appendChild(createBreak());
388 	starterDiv.appendChild(starterImmediatelyInput);
389 	starterDiv.appendChild(starterImmediatelyInputText);
390 	starterDiv.appendChild(createBreak());
391 	starterDiv.appendChild(starterSentenceText);
392 	starterDiv.appendChild(starterSentenceInput);
393 	
394 	//set value of textarea
395 	starterSentenceInput.value = this.content.assessments[ndx].starter.text;
396 	
397 	//set appropriate radio button and enable/disable textarea
398 	var displayOption = this.content.assessments[ndx].starter.display;
399 	
400 	if(displayOption=='0'){
401 		starterSentenceInput.disabled = true;
402 		noStarterInput.checked = true;
403 	} else if(displayOption=='1'){
404 		starterOnClickInput.checked = true;
405 	} else if(displayOption=='2'){
406 		starterImmediatelyInput.checked = true;
407 	};
408 	
409 	return starterDiv;
410 };
