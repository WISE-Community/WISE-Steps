1 /**
  2  * The constructor for the cargraph state
  3  * @constructor
  4  * @param response the text response the student typed
  5  * @param annotationArray an array containing objects that represent annotations
  6  * for points on the graph 
  7  * @param timestamp the time the cargraph state was created
  8  * @param xMin the min x value
  9  * @param xMax the max x value
 10  * @param yMin the min y value
 11  * @param yMax the max y value
 12  * 
 13  * example of what a state looks like: 
 14  * {
 15  *   "response":"",
 16  *   "annotationArray":[
 17  *     {"seriesName":"greenCar","dataIndex":3,"dataText":"0.18 min, 14.94 km","annotationText":"green .18 min.","x":0.18,"y":14.94},
 18  *     {"seriesName":"yellowCar","dataIndex":8,"dataText":"0.4 min, 7.04 km","annotationText":"yellow .4 min","x":0.4,"y":7.04}
 19  *   ]
 20  *   "predictionArray":[
 21  *     {"id":"greenCar","predictions":[{"x":0,"y":11.03},{"x":0.01,"y":12.35},{"x":0.02,"y":14.72},{"x":0.03,"y":15.82},{"y":16.27,"x":0.04},{"x":0.06,"y":8.15},{"x":0.12,"y":12.94},{"y":10.73,"x":0.13},{"x":0.17,"y":15.82},{"x":0.18,"y":16.41},{"x":0.45,"y":1.65},{"x":0.47,"y":2.98},{"x":0.48,"y":4.01},{"x":0.5,"y":5.34}]},
 22  *     {"id":"yellowCar","predictions":[{"x":0.07,"y":17.3},{"x":0.1,"y":14.57},{"x":0.25,"y":6.89},{"x":0.26,"y":6.15}
 23  *   ],
 24  *   "timestamp":1303406004861
 25  * }
 26  */
 27 function CARGRAPHSTATE(response, annotationArray, predictionArray, timestamp, xMin, xMax, yMin, yMax, predictionLocked) {
 28 	//the text response the student wrote
 29 	this.response = "";
 30 	
 31 	//an array of annotations the student has created
 32 	this.annotationArray = [];
 33 	
 34 	//an array of prediction objects
 35 	// [
 36 	//	{id:"car1",predictions:[[0,1],[1,3],[2,5]]}, 
 37 	//	{id:"car2",predictions:[[0,0],[1,5],[5,3]]}
 38 	// ]
 39 	this.predictionArray = [];
 40 	
 41 	if(response != null) {
 42 		//set the response if it was provided to the constructor
 43 		this.response = response;
 44 	}
 45 		
 46 	if(annotationArray != null) {
 47 		//set the data array if it was provided to the constructor
 48 		this.annotationArray = annotationArray;
 49 	}
 50 	
 51 	if(predictionArray != null) {
 52 		//set the data array if it was provided to the constructor
 53 		this.predictionArray = predictionArray;
 54 	}
 55 	
 56 	if(timestamp == null) {
 57 		//set the timestamp to the current time if timestamp is not provided
 58 		this.timestamp = new Date().getTime();
 59 	} else {
 60 		this.timestamp = timestamp;
 61 	}
 62 	
 63 	//set the axis range values
 64 	this.xMin = xMin;
 65 	this.xMax = xMax;
 66 	this.yMin = yMin;
 67 	this.yMax = yMax;
 68 	
 69 	//set whether the prediction is locked
 70 	this.predictionLocked = predictionLocked;
 71 };

 73 /**
 74  * Creates a CARGRAPHSTATE object from the data in the JSON object
 75  * @param stateJSONObj a JSON object containing data for a CARGRAPHSTATE
 76  * @return a CARGRAPHSTATE object with the attributes populated
 77  */
 78 CARGRAPHSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 79 	//get the student response
 80 	var response = stateJSONObj.response;
 81 	
 82 	//get the annotation array
 83 	var annotationArray = stateJSONObj.annotationArray;
 84 	
 85 	//get the prediction array
 86 	var predictionArray = stateJSONObj.predictionArray;
 87 	
 88 	//get the timestamp
 89 	var timestamp = stateJSONObj.timestamp;
 90 	
 91 	//get the axis range values
 92 	var xMin = stateJSONObj.xMin;
 93 	var xMax = stateJSONObj.xMax;
 94 	var yMin = stateJSONObj.yMin;
 95 	var yMax = stateJSONObj.yMax;
 96 	
 97 	//get whether the prediction has been locked
 98 	var predictionLocked = stateJSONObj.predictionLocked;
 99 
100 	//create a CARGRAPHSTATE object
101 	var cargraphState = new CARGRAPHSTATE(response, annotationArray, predictionArray, timestamp, xMin, xMax, yMin, yMax, predictionLocked);
102 	
103 	return cargraphState;
104 };

134 /**
135  * Get an annotation object given the seriesName and dataIndex
136  * @param seriesName the name of the graph line
137  * e.g.
138  * 'distance'
139  * 'velocity'
140  * 'acceleration'
141  * 'temperature'
142  * @param dataIndex the index on the graph line
143  * @return an annotation object or null if none was found to match
144  */
145 CARGRAPHSTATE.prototype.getAnnotationBySeriesDataIndex = function(seriesName, dataIndex) {
146 	var annotation = null;
147 	
148 	//loop through all the annotations
149 	for(var x=0; x<this.annotationArray.length; x++) {
150 		//get an annotation
151 		var tempAnnotation = this.annotationArray[x];
152 		
153 		//check if the series name and data index match
154 		if(tempAnnotation.seriesName == seriesName && tempAnnotation.dataIndex == dataIndex) {
155 			//values match so we have found the annotation we want
156 			annotation = tempAnnotation;
157 			
158 			//jump out of the for loop
159 			break;
160 		}
161 	}
162 	
163 	return annotation;
164 };

269 /**
270  * Get the human readable form of the annotations as a string
271  * @return a string containing the human readable form of the annotations
272  * e.g.
273  * distance [11.61 s, 0.72 m]: this is where it was the highest
274  * velocity [13.87 s, -0.09 m/s]: this is where it was the fastest
275  */
276 CARGRAPHSTATE.prototype.getAnnotationsHtml = function() {
277 	var annotationsHtml = "<br>";
278 	
279 	//loop through all the annotations
280 	for(var x=0; x<this.annotationArray.length; x++) {
281 		//get an annotation
282 		var annotation = this.annotationArray[x];
283 		var dataText = annotation.dataText;
284 		var annotationText = annotation.annotationText;
285 		var seriesName = annotation.seriesName;
286 		
287 		/*
288 		 * get the human readable form of the annotation
289 		 * e.g.
290 		 * distance [11.61 s, 0.72 m]: this is where it was the highest
291 		 */
292 		annotationsHtml += seriesName + " [" + dataText + "]: " + annotationText;
293 		annotationsHtml += "<br>";
294 	}
295 	
296 	return annotationsHtml;
297 };

299 CARGRAPHSTATE.prototype.getPredictionObjByPredictionId = function(predictionId) {
300 	for (var i=0; i < this.predictionArray.length; i++) {
301 		if (this.predictionArray[i].id == predictionId) {

303 		}
304 	}

308 CARGRAPHSTATE.prototype.setPredictionsForPredictionId = function(predictionId, predictionArray) {
309 	for (var i=0; i < this.predictionArray.length; i++) {
310 		if (this.predictionArray[i].id == predictionId) {
311 			this.predictionArray[i].predictions = predictionArray;
312 		}
313 	}
314 };

316 /**
317  * Adds an element into the prediction array
318  * @param x the x value of the point
319  * @param y the y value of the point
320  */
321 CARGRAPHSTATE.prototype.predictionReceived = function(predictionId, x, y) {
322 	if (this.getPredictionObjByPredictionId(predictionId) == null) {
323 		this.predictionArray.push({id:predictionId, predictions:[]});
324 	}
325 	var predictionArray = this.getPredictionObjByPredictionId(predictionId).predictions;
326 	//remove any existing point with the same x value
327 	for(var i=0; i<predictionArray.length; i++) {
328 		var predictionPoint = predictionArray[i];
329 		
330 		if(predictionPoint.x == x) {
331 			predictionArray.splice(i, 1);
332 			break;
333 		}
334 	}
335 	
336 	//create an object that contains the x, y points
337 	var predictionData = {
338 			x: x,
339 			y: y
340 	};
341 
342 	//add the element to the array
343 	predictionArray.push(predictionData);
344 	
345 	//sort the array by the x value
346 	predictionArray.sort(this.sortPredictionArray);
347 	
348 	// save 
349 	this.setPredictionsForPredictionId(predictionId, predictionArray);	
350 };

352 /**
353  * Remove the element with the given index from the prediction array
354  * @param index the index to remove
355  */
356 CARGRAPHSTATE.prototype.predictionRemoved = function(predictionId,index) {	
357 	var predictions = this.getPredictionObjByPredictionId(predictionId).predictions;
358 	this.setPredictionsForPredictionId(predictionId, predictions.splice(index,1));
359 	//this.predictions.predictionArrayId.splice(index, 1);
360 };

362 /**
363  * A function used to sort arrays in ascending x values. The elements 
364  * in the array contain x and y fields.
365  * e.g.
366  * {
367  * 		x:1.4,
368  * 		y:8.45
369  * }
370  * @param point1 an element in the array
371  * @param point2 an element in the array
372  * @return negative value if point1.x is smaller than point2.x
373  * or
374  * positive value if point1.x is larger than point2.x
375  * or
376  * 0 if the values are the same
377  */
378 CARGRAPHSTATE.prototype.sortPredictionArray = function(point1, point2) {
379 	return point1.x - point2.x;
380 };

382 /**
383  * Remove the prediction annotations from the annotation array
384  * @return
385  */
386 CARGRAPHSTATE.prototype.removePredictionAnnotations = function() {
387 	//loop through all the annotations
388 	for(var x=0; x<this.annotationArray.length; x++) {
389 		var annotation = this.annotationArray[x];
390 		
391 		if(annotation.seriesName.indexOf("prediction") != -1) {
392 			//the annotation is a prediction annotation so we will remove it
393 			this.annotationArray.splice(x, 1);
394 			x--;
395 		}
396 	}
397 };

399 /**
400  * Remove the data point from the prediction array
401  * @param seriesName the name of the series/line
402  * @param dataIndex the index within the prediction array to remove
403  */
404 CARGRAPHSTATE.prototype.removePredictionPoint = function(seriesName, dataIndex) {
405 	//loop through all the prediction objects
406 	for(var x=0; x<this.predictionArray.length; x++) {
407 		//get a prediction object
408 		var prediction = this.predictionArray[x];
409 		
410 		if(prediction != null) {
411 			//make sure the prediction object is for the series we want
412 			if(seriesName == prediction.id) {
413 				//remove the element at the given index
414 				prediction.predictions.splice(dataIndex, 1);
415 			}
416 		}
417 	}
418 };
