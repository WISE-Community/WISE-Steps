 29 /**
 30  * This is the constructor for the object that will perform the logic for
 31  * the step when the students work on it. An instance of this object will
 32  * be created in the .html for this step (look at seasons.html)
 33  * 
 34  * TODO: rename SEASONS
 35  * @constructor
 36  */
 37 function SEASONS(node, view) {
 38 	this.node = node;
 39 	this.view = view;
 40 	this.content = node.getContent().getContentJSON();
 41 	
 42 	if(node.studentWork != null) {
 43 		this.states = node.studentWork; 
 44 	} else {
 45 		this.states = [];  
 46 	};
 47 };

 49 /**
 50  * This function renders everything the student sees when they visit the step.
 51  * This includes setting up the html ui elements as well as reloading any
 52  * previous work the student has submitted when they previously worked on this
 53  * step, if any.
 54  * 
 55  * TODO: rename SEASONS
 56  * 
 57  * note: you do not have to use 'promptDiv' or 'studentResponseTextArea', they
 58  * are just provided as examples. you may create your own html ui elements in
 59  * the .html file for this step (look at seasons.html).
 60  */
 61 SEASONS.prototype.render = function() {
 62 	// get type of model
 63 	var modelType = this.content.modelType;
 64 	
 65 	if (modelType == "whatIstheShapeofEarthsOrbit") {		
 66 		$("#modelIFrame").attr("src","/vlewrapper/vle/node/seasons/earth/seasons1-1.html");
 67 	} else if (modelType == "whatTemperaturePatternsDoYouSee") {		
 68 		$("#modelIFrame").attr("src","/vlewrapper/vle/node/seasons/earth/seasons1-2a.html");
 69 	} else if (modelType == "howDoesEarthsTiltAffectTemperature") {		
 70 		$("#modelIFrame").attr("src","/vlewrapper/vle/node/seasons/earth/seasons1-3.html");
 71 	} else if (modelType == "tiltAndHoursOfDaylight") {		
 72 		$("#modelIFrame").attr("src","/vlewrapper/vle/node/seasons/earth/seasons4.html");
 73 	} 	
 74 };

 78 /**
 79  * This function retrieves the latest student work
 80  * 
 81  * TODO: rename SEASONS
 82  * 
 83  * @return the latest state object or null if the student has never submitted
 84  * work for this step
 85  */
 86 SEASONS.prototype.getLatestState = function() {
 87 	var latestState = null;
 88 	
 89 	//check if the states array has any elements
 90 	if(this.states != null && this.states.length > 0) {
 91 		//get the last state
 92 		latestState = this.states[this.states.length - 1];
 93 	}
 94 	
 95 	return latestState;
 96 };

 98 /**
 99  * This function retrieves the student work from the html ui, creates a state
100  * object to represent the student work, and then saves the student work.
101  * 
102  * TODO: rename SEASONS
103  * 
104  * note: you do not have to use 'studentResponseTextArea', they are just 
105  * provided as examples. you may create your own html ui elements in
106  * the .html file for this step (look at seasons.html).
107  */
108 SEASONS.prototype.save = function() {
109 	//get the answer the student wrote
110 	var seasons_activity = window.document.getElementById("modelIFrame").contentWindow.seasons_activity;
111 	var json_str = JSON.stringify(seasons_activity);
112 	console.log("save, json_str:" + json_str);
113 			
114 	/*
115 	 * create the student state that will store the new work the student
116 	 * just submitted
117 	 * 
118 	 * TODO: rename SEASONSSTATE
119 	 * 
120 	 * make sure you rename SEASONSSTATE to the state object type
121 	 * that you will use for representing student data for this
122 	 * type of step. copy and modify the file below
123 	 * 
124 	 * vlewrapper/WebContent/vle/node/seasons/seasonsstate.js
125 	 * 
126 	 * and use the object defined in your new state.js file instead
127 	 * of SEASONSSTATE. for example if you are creating a new
128 	 * quiz step type you would copy the file above to
129 	 * 
130 	 * vlewrapper/WebContent/vle/node/quiz/quizstate.js
131 	 * 
132 	 * and in that file you would define QUIZSTATE and therefore
133 	 * would change the SEASONSSTATE to QUIZSTATE below
134 	 */
135 	var seasonsState = new SEASONSSTATE(json_str);
136 	
137 	/*
138 	 * fire the event to push this state to the global view.states object.
139 	 * the student work is saved to the server once they move on to the
140 	 * next step.
141 	 */
142 	eventManager.fire('pushStudentWork', seasonsState);
143 
144 	//push the state object into this or object's own copy of states
145 	this.states.push(seasonsState);
146 };
