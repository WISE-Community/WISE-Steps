 90 /**
 91  * Renders the student work into the div. The grading tool will pass in a
 92  * div id to this function and this function will insert the student data
 93  * into the div.
 94  * 
 95  * @param divId the id of the div we will render the student work into
 96  * @param nodeVisit the student work
 97  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
 98  * div ids use this to prevent DOM conflicts such as when the show all work div
 99  * uses the same ids as the show flagged work div
100  * @param workgroupId the id of the workgroup this work belongs to
101  * 
102  * Note: you may need to add code to this function if the student
103  * data for your step is complex or requires additional processing.
104  * look at SensorNode.renderGradingView() as an example of a step that
105  * requires additional processing
106  */
107 SurgeNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
108 	var gradingText = "";
109 	// Get all the trials (ie states) for this nodevisit
110 	var nodeStates = nodeVisit.nodeStates;
111 	
112 	if (nodeStates.length > 0) {
113 		// get the number of trials during this node visit.
114 		gradingText += "This visit has " + nodeStates.length + " trials.<br/><br/>";
115 		
116 		// get the best score
117 		gradingText += "The best medal earned for this level is "+nodeStates[nodeStates.length-1].getStudentWork().response.outcomeAbsoluteText+"<br/><br/>";
118 		
119 		/*
120 		 * loop through the trials from newest to oldest so that
121 		 * the newest displays at the top
122 		 */
123 		for (var i=nodeStates.length - 1; i>=0; i--) {
124 			gradingText += "<b>Trial #"+(i+1)+"</b><br/>"
125 			gradingText += JSON.stringify(nodeStates[i].getStudentWork().response) + "<br/><br/>";
126 		}
127 
128 		//put the student work into the div
129 		$('#' + divId).html(gradingText);
130 	}
131 };

144 /**
145  * Process the student work to see if we need to display a colored
146  * star next to the step in the nav menu
147  * @param studentWork the student's surge state
148  */
149 SurgeNode.prototype.processStudentWork = function(studentWork) {
150 	if(studentWork != null) {
151 		if(studentWork.response != null && studentWork.response != "") {
152 			var className = "";
153 			
154 			//get the top score
155 			var topScore = studentWork.response.topScore;
156 			
157 			if(topScore == 10) {
158 				className = "bronzeStar";
159 			} else if(topScore == 20) {
160 				className = "silverStar";
161 			} else if(topScore == 30) {
162 				className = "goldStar";
163 			}
164 			
165 			//display the star next to the step in the nav menu
166 			eventManager.fire('updateStepRightIcon', [this.id, className]);			
167 		}
168 	}
169 };

181 /**
182  * Get a tag map function given the function name
183  * @param functionName
184  * @return 
185  */
186 SurgeNode.prototype.getTagMapFunctionByName = function(functionName) {
187 	var fun = null;
188 	
189 	//get all the tag map function for this step type
190 	var tagMapFunctions = this.getTagMapFunctions();
191 	
192 	//loop through all the tag map functions
193 	for(var x=0; x<tagMapFunctions.length; x++) {
194 		//get a tag map function
195 		var tagMapFunction = tagMapFunctions[x];
196 		
197 		if(tagMapFunction != null) {
198 			
199 			//check if the function name matches
200 			if(functionName == tagMapFunction.functionName) {
201 				//the function name matches so we have found what we want
202 				fun = tagMapFunction;
203 				break;
204 			}			
205 		}
206 	};
207 	
208 	return fun;
209 };
