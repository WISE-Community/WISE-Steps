 39 /**
 40  * Retrieves the latest student work for this node and returns it in
 41  * a query entry object
 42  * @param vle the vle that this node has been loaded into, this vle
 43  * 		is related to a specific student, so all the work in this vle
 44  * 		is for just one student
 45  * @return a MultipleChoiceQueryEntry that contains the latest student
 46  * 		work for this node. return null if this student has not accessed
 47  * 		this step yet.
 48  */
 49 MultipleChoiceNode.prototype.getLatestWork = function(vle) {
 50 	var latestState = null;
 51 	
 52 	//setup the mc object by loading in the content of the step
 53 	this.mc = new MC(loadXMLString(this.element.getElementsByTagName("jaxbXML")[0].firstChild.nodeValue));
 54 	
 55 	//load the states from the vle into the mc object
 56 	this.mc.loadForTicker(this, vle);
 57 	
 58 	//get the most recent student work for this step
 59 	latestState = this.mc.getLatestState(this.id);
 60 	
 61 	if(latestState == null) {
 62 		//the student has not accessed or completed this step yet
 63 		return null;
 64 	}
 65 	
 66 	//create and return a query entry object
 67 	return new MultipleChoiceQueryEntry(vle.getWorkgroupId(), vle.getUserName(), this.id, this.mc.promptText, latestState.getIdentifier(), this.mc.getCHOICEByIdentifier(latestState.getIdentifier()).text);
 68 };

 70 /**
 71  * Returns the prompt for this node by loading the MC content and then
 72  * obtaining it from the MC
 73  * @return the prompt for this node
 74  */
 75 MultipleChoiceNode.prototype.getPrompt = function() {
 76 	var prompt = "";
 77 	
 78 	if(contentJSON != null) {
 79 		//get the content for the node
 80 		var contentJSON = this.content.getContentJSON();
 81 
 82 		//see if the node content has an assessmentItem
 83 		if(contentJSON != null && contentJSON.assessmentItem != null) {
 84 			//obtain the prompt
 85 			var assessmentItem = contentJSON.assessmentItem;
 86 			var interaction = assessmentItem.interaction;
 87 			prompt = interaction.prompt;	
 88 		}
 89 	}
 90 				
 91 	//return the prompt
 92 	return prompt;
 93 };

 95 /**
 96  * Create a query container that will contain all the query entries
 97  * @param vle the vle that this node has been loaded into, this vle
 98  * 		is related to a specific student, so all the work in this vle
 99  * 		is for just one student
100  * @return a MultipleChoiceQueryContainer that will contain all the
101  * 		query entries for a specific nodeId as well as accumulated 
102  * 		metadata about all those entries such as count totals, etc.
103  */
104 MultipleChoiceNode.prototype.makeQueryContainer = function(vle) {
105 	//setup the mc object by loading in the content of the step
106 	this.mc = new MC(loadXMLString(this.element.getElementsByTagName("jaxbXML")[0].firstChild.nodeValue));
107 	
108 	//load the states from the vle into the mc object
109 	this.mc.loadForTicker(this, vle);
110 	
111 	//create and return a query container object
112 	return new MultipleChoiceQueryContainer(this.id, this.mc.promptText, this.mc.choiceToValueArray);
113 };

132 /**
133  * Renders the student work into the div. The grading tool will pass in a
134  * div id to this function and this function will insert the student data
135  * into the div.
136  * 
137  * @param divId the id of the div we will render the student work into
138  * @param nodeVisit the student work
139  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
140  * div ids use this to prevent DOM conflicts such as when the show all work div
141  * uses the same ids as the show flagged work div
142  * @param workgroupId the id of the workgroup this work belongs to
143  */
144 MultipleChoiceNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
145 	//create the multiple choice object so we can reference the content later
146 	var multipleChoice = new MC(this, this.view);
147 	
148 	//get the latest state
149 	var state = nodeVisit.getLatestWork();
150 	
151 	var studentWork = "";
152 	
153 	//check if there were any choices chosen
154 	if(state.response) {
155 		//loop through the array of choices
156 		for(var x=0; x<state.response.length; x++) {
157 			if(studentWork != "") {
158 				//separate each choice with a comma
159 				studentWork += ", ";
160 			}
161 			
162 			//add the choice to the student work
163 			studentWork += state.response[x];
164 		}
165 		
166 		if(state.score != null){
167 			//get the max score
168 			var maxScore = multipleChoice.getMaxPossibleScore();
169 			
170 			studentWork += "<br><br>";
171 			studentWork += "Auto-Graded Score: " + state.score + "/" + maxScore;
172 		}
173 	}
174 	
175 	$('#' + divId).html(studentWork);
176 };
