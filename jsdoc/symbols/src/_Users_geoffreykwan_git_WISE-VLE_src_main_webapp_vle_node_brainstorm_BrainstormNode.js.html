 14 /**
 15  * @constructor
 16  * @extends Node
 17  * @param nodeType
 18  * @param view
 19  * @returns {BrainstormNode}
 20  */
 21 function BrainstormNode(nodeType, view) {
 22 	this.view = view;
 23 	this.type = nodeType;
 24 	this.audioSupported = true;
 25 	this.serverless = true;
 26 	this.prevWorkNodeIds = [];
 27 };
 28 
 29 /**
 30  * Determines if the this step is using a server back end.
 31  * @return
 32  */
 33 BrainstormNode.prototype.isUsingServer = function() {
 34 	if(this.content.getContentJSON().useServer) {
 35 		//we are using a server back end
 36 		this.serverless = false;
 37 		return true;
 38 	} else {
 39 		//we are not using a server back end
 40 		this.serverless = true;
 41 		return false;
 42 	}
 43 };
 44 
 45 /**
 46  * Takes in a state JSON object and returns a BRAINSTORMSTATE object
 47  * @param nodeStatesJSONObj a state JSON object
 48  * @return a BRAINSTORMSTATE object
 49  */
 50 BrainstormNode.prototype.parseDataJSONObj = function(stateJSONObj) {
 51 	return BRAINSTORMSTATE.prototype.parseDataJSONObj(stateJSONObj);
 52 };
 53 
 54 NodeFactory.addNode('BrainstormNode', BrainstormNode);
 55 
 56 BrainstormNode.prototype.getHTMLContentTemplate = function() {
 57 	var content = null;
 58 	if(this.isUsingServer()) {
 59 		//using server
 60 		content = createContent('node/brainstorm/brainfull.html');
 61 	} else {
 62 		//not using server
 63 		content = createContent('node/brainstorm/brainlite.html');
 64 	}
 65 	return content;
 66 };
