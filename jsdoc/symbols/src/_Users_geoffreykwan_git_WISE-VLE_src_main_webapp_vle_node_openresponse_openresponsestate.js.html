1 /**
  2  * Object for storing state information of OpenResponse item.
  3  * @constructor
  4  * @author Hiroki Terashima
  5  */
  6 function OPENRESPONSESTATE(response, timestamp, locked, submitPeerReview) {
  7 	this.type = "or";
  8 	
  9 	//this is a single element array that contains the response the student wrote
 10 	this.response = response;
 11 	
 12 	if(!timestamp) {
 13 		//if the second argument (timestamp) was ommitted just set it to the current time
 14 		this.timestamp = Date.parse(new Date());
 15 	} else {
 16 		this.timestamp = timestamp;
 17 	}
 18 	this.locked = locked;
 19 	this.submitPeerReview = submitPeerReview;
 20 };
 21 
 22 OPENRESPONSESTATE.prototype.print = function() {
 23 };
 24 
 25 OPENRESPONSESTATE.prototype.getHtml = function() {
 26 	return "timestamp: " + this.timestamp + "<br/>response: " + this.response;
 27 };
 28 
 29 
 30 /**
 31  * Takes in a state JSON object and returns an OPENRESPONSESTATE object
 32  * @param stateJSONObj a state JSON object
 33  * @return a OPENRESPONSESTATE object
 34  */
 35 OPENRESPONSESTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 36 	
 37 	//set the attributes of the OPENRESPONSESTATE object
 38 	var response = stateJSONObj.response;
 39 	var timestamp = stateJSONObj.timestamp;
 40 	var submitPeerReview = stateJSONObj.submitPeerReview;
 41 	var locked = stateJSONObj.locked;
 42 	
 43 	//return the OPENRESPONSESTATE object
 44 	//create a new OPENRESPONSESTATE object
 45 	var orState = new OPENRESPONSESTATE(response, timestamp, locked, submitPeerReview);
 46 	
 47 	return orState;
 48 };
 49 
 50 /**
 51  * Returns what the student typed
 52  * @return the answer the student typed
 53  */
 54 OPENRESPONSESTATE.prototype.getStudentWork = function() {
 55 	var studentWork = this.response;
 56 	
 57 	//check if the response is an array
 58 	if(this.response != null && this.response.constructor.toString().indexOf("Array") != -1) {
 59 		/*
 60 		 * response is an array so we will use the toString() of the array
 61 		 * which should give us just the text within it
 62 		 */
 63 		studentWork = this.response.toString();
 64 	}
 65 	return studentWork;
 66 };
 67 
 68 OPENRESPONSESTATE.prototype.isLocked = function() {
 69 	return this.locked;
 70 };
