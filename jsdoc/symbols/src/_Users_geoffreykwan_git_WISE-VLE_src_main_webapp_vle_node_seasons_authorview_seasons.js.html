 38 /**
 39  * Generates the authoring page. This function will create the authoring
 40  * components such as textareas, radio buttons, check boxes, etc. that
 41  * authors will use to author the step. For example if the step has a
 42  * text prompt that the student will read, this function will create
 43  * a textarea that will allow the author to type the text that the
 44  * student will see. You will also need to populate the textarea with
 45  * the pre-existing prompt if the step has been authored before.
 46  * 
 47  * TODO: rename SeasonsNode
 48  */
 49 View.prototype.SeasonsNode.generatePage = function(view){
 50 	this.view = view;
 51 	
 52 	//get the content of the step
 53 	this.content = this.view.activeContent.getContentJSON();
 54 	
 55 	//get the html element that all the authoring components will be located
 56 	var parent = document.getElementById('dynamicParent');
 57 	
 58 	/*
 59 	 * wipe out the div that contains the authoring components because it
 60 	 * may still be populated with the authoring components from a previous
 61 	 * step the author has been authoring since we re-use the div id
 62 	 */
 63 	parent.removeChild(document.getElementById('dynamicPage'));
 64 
 65 	//create a new div that will contain the authroing components
 66 	var pageDiv = createElement(document, 'div', {id:'dynamicPage', style:'width:100%;height:100%'});
 67 	
 68 	//create the label for the textarea that the author will write the prompt in
 69 	var promptText = document.createTextNode("Select Model Type:");
 70 	
 71 	/*
 72 	 * create the modelType dropdown
 73 	 * 
 74 	 * onchange will fire the 'seasonsModelTypeUpdated' event which will
 75 	 * be handled in the seasonsEvents.js file
 76 	 */
 77 	var modelTypeDropDownHtml = '<select id="modelTypeDropDown" onchange="eventManager.fire(\'seasonsModelTypeUpdated\')">' + 
 78 	'<option value="whatTemperaturePatternsDoYouSee">1 What Temperature Patterns Do You See?</option>' +
 79 	'<option value="whatIstheShapeofEarthsOrbit">2 What is the Shape of Earth\'s Orbit</option>' +
 80 	'<option value="howDoesEarthsTiltAffectTemperature">3 How Does Earth\'s Tilt Affect Temperature?</option>' +
 81 	'</select>';
 82 		
 83 	//var promptTextArea = createElement(document, 'textarea', {id: 'promptTextArea', rows:'20', cols:'85', onkeyup:"eventManager.fire('seasonsUpdatePrompt')"});
 84 	
 85 	//add the authoring components to the page
 86 	pageDiv.appendChild(promptText);
 87 	pageDiv.appendChild(createBreak());
 88 	pageDiv.innerHTML+=modelTypeDropDownHtml;
 89 	pageDiv.appendChild(createBreak());
 90 
 91 	//add the page to the parent
 92 	parent.appendChild(pageDiv);
 93 	
 94 	// show the selected model type in the drop down
 95 	$("#modelTypeDropDown").val(this.content.modelType);
 96 	
 97 	//populate the prompt if this step has been authored before
 98 	//this.populatePrompt();
 99 };
