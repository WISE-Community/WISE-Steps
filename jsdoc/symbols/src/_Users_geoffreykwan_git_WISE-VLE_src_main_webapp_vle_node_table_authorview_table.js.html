 38 /**
 39  * Generates the authoring page. This function will create the authoring
 40  * components such as textareas, radio buttons, check boxes, etc. that
 41  * authors will use to author the step. For example if the step has a
 42  * text prompt that the student will read, this function will create
 43  * a textarea that will allow the author to type the text that the
 44  * student will see. You will also need to populate the textarea with
 45  * the pre-existing prompt if the step has been authored before.
 46  * 
 47  * TODO: rename TemplateNode
 48  */
 49 View.prototype.TableNode.generatePage = function(view){
 50 	this.view = view;
 51 	
 52 	//get the content of the step
 53 	this.content = this.view.activeContent.getContentJSON();

 58 	/*

 76 	 * 
 77 	 * For example if you are creating a quiz step you would look in
 78 	 * your quizEvents.js file.
 79 	 *

 81 	 * new events in the <new step type name>Events.js file and then
 82 	 * create new functions to handle the event

202 /**
203  * Generate the authorable table
204  */
205 View.prototype.TableNode.generateAuthoringTable = function() {
206 	//create the table
207 	var authoringTable = createElement(document, 'table', {id: 'authoringTable', border:'1'});
208 	
209 	//create a header row to display the I (insert) and D (delete) buttons
210 	var headerTR = createElement(document, 'tr');
211 	authoringTable.appendChild(headerTR);
212 	
213 	//add the upper left cell (0,0) that will be empty
214 	var headerTD = createElement(document, 'td');
215 	headerTR.appendChild(headerTD);
216 	
217 	//add a td for each column
218 	for(var x=0; x<this.content.numColumns; x++) {
219 		headerTD = createElement(document, 'td');
220 		
221 		//create the insert and delete buttons
222 		var insertColumnButton = createElement(document, 'input', {type: 'button', id: 'insertColumnButton_' + x, name: 'insertColumnButton_' + x, value: 'I', onclick: 'eventManager.fire("tableInsertColumn", {x:' + x + '})'});
223 		var deleteColumnButton = createElement(document, 'input', {type: 'button', id: 'deleteColumnButton_' + x, name: 'deleteColumnButton_' + x, value: 'D', onclick: 'eventManager.fire("tableDeleteColumn", {x:' + x + '})'});
224 		
225 		headerTD.appendChild(insertColumnButton);
226 		headerTD.appendChild(deleteColumnButton);
227 		
228 		headerTR.appendChild(headerTD);
229 	}
230 	
231 	//loop through all the rows
232 	for(var y=0; y<this.content.numRows; y++) {
233 		
234 		var tr = createElement(document, 'tr');
235 		headerTD = createElement(document, 'td');
236 		
237 		//create the insert and delete buttons
238 		var insertRowButton = createElement(document, 'input', {type: 'button', id: 'insertRowButton_' + y, name: 'insertRowButton_' + y, value: 'I', onclick: 'eventManager.fire("tableInsertRow", {y:' + y + '})'});
239 		var deleteRowButton = createElement(document, 'input', {type: 'button', id: 'deleteRowButton_' + y, name: 'deleteRowButton_' + y, value: 'D', onclick: 'eventManager.fire("tableDeleteRow", {y:' + y + '})'});
240 		
241 		headerTD.appendChild(insertRowButton);
242 		headerTD.appendChild(deleteRowButton);
243 		
244 		tr.appendChild(headerTD);
245 		
246 		//loop through all the columns
247 		for(var x=0; x<this.content.numColumns; x++) {
248 			
249 			var td = createElement(document, 'td');
250 			
251 			var cellText = "";
252 			var cellUneditable = false;
253 			var cellSize = "";
254 			
255 			//get the data for the cell
256 			var cellData = this.getCellData(this.content.tableData, x, y);
257 			
258 			if(cellData != null) {
259 				if(cellData.text != null) {
260 					cellText = cellData.text;					
261 				}
262 				
263 				if(cellData.uneditable != null) {
264 					cellUneditable = cellData.uneditable;					
265 				}
266 				
267 				if(cellData.cellSize != null) {
268 					cellSize = cellData.cellSize;
269 				}
270 			}
271 			
272 			//create the input box for the cell
273 			var cellTextInput = createElement(document, 'input', {type: 'input', id: 'cellTextInput_' + x + '-' + y, name: 'cellTextInput_' + x + '-' + y, value: cellText, size: 10, onkeyup: 'eventManager.fire("tableUpdateCellText", {x:' + x + ', y:' + y + '})'});
274 			
275 			var cellNewline = createElement(document, 'br');
276 						
277 			//label for the 'uneditable' checkbox
278 			var cellCheckBoxText = document.createTextNode('U');
279 			
280 			//create the checkbox for the cell that determines whether the student can edit the cell or not
281 			var cellCheckBox = createElement(document, 'input', {type: 'checkbox', id: 'cellUneditableCheckBox_' + x + '-' + y, name: 'cellUneditableCheckBox_' + x + '-' + y, onchange: 'eventManager.fire("tableUpdateCellUneditable", {x:' + x + ', y:' + y + '})'});
282 			
283 			if(cellUneditable) {
284 				cellCheckBox.checked = true;
285 			}
286 			
287 			//label for the cell size input
288 			var cellSizeText = document.createTextNode(' S ');
289 			
290 			//create the input for the cell size
291 			var cellSizeInput = createElement(document, 'input', {type: 'input', id: 'cellSizeInput_' + x + '-' + y, name: 'cellSizeInput_' + x + '-' + y, value: cellSize, size: 1, onkeyup: 'eventManager.fire("tableUpdateCellSize", {x:' + x + ', y:' + y + '})'});
292 			
293 			//add the elements to the td
294 			td.appendChild(cellTextInput);
295 			td.appendChild(cellNewline);
296 			
297 			//add the checkbox for uneditable
298 			td.appendChild(cellCheckBoxText);
299 			td.appendChild(cellCheckBox);
300 			
301 			//add the input for the cell size
302 			td.appendChild(cellSizeText);
303 			td.appendChild(cellSizeInput);
304 			
305 			tr.appendChild(td);
306 		}
307 		
308 		authoringTable.appendChild(tr);
309 	}
310 	
311 	return authoringTable;
312 };

329 /**
330  * Get the cell data for the given x, y cell
331  * @param tableData the table data
332  * @param x the x coordinate of the cell
333  * @param y the y coordinate of the cell
334  * @return the cell data or null if not found
335  */
336 View.prototype.TableNode.getCellData = function(tableData, x, y) {
337 	var cellData = null;
338 	
339 	//get the column the cell is in
340 	var tableColumn = tableData[x];
341 	
342 	if(tableColumn != null) {
343 		
344 		//get the cell
345 		var tableCell = tableColumn[y];
346 		
347 		if(tableCell != null) {
348 			cellData = tableCell;
349 		}
350 	}
351 	
352 	return cellData;
353 };

355 /**
356  * Update the value of a field for a given cell
357  * @param tableData the table data
358  * @param x the x coordinate of the cell
359  * @param y the y coordinate of the cell
360  * @param fieldName the field name
361  * @param fieldValue the value to set into the field
362  */
363 View.prototype.TableNode.updateCellDataValue = function(tableData, x, y, fieldName, fieldValue) {
364 	//get the column
365 	var tableColumn = tableData[x];
366 	
367 	if(tableColumn == null) {
368 		//column does not exist so we will make it
369 		tableData[x] = [];
370 		
371 		//create a new cell
372 		tableData[x][y] = this.getCellWithDefaultValues();
373 		
374 		//set the value
375 		tableData[x][y][fieldName] = fieldValue;
376 	} else {
377 		var tableCell = tableColumn[y];
378 		
379 		if(tableCell == null) {
380 			//cell does not exist so we will make it
381 			tableColumn[y] = this.getCellWithDefaultValues();
382 			
383 			//set the value
384 			tableColumn[y][fieldName] = fieldValue;
385 		} else {
386 			//set the value
387 			tableCell[fieldName] = fieldValue;
388 		}
389 	}
390 };

392 /**
393  * Called when the author changes the number of rows
394  * @param the new number of rows
395  */
396 View.prototype.TableNode.truncateRows = function(numRows) {
397 	//get the previous number of rows
398 	var tableDataNumRows = this.content.numRows;
399 	
400 	//get the dif in the number of rows
401 	var dif = tableDataNumRows - numRows;
402 	
403 	if(dif > 0) {
404 		//we need to remove some rows
405 		
406 		//loop through all the columns
407 		for(var x=0; x<this.content.tableData.length; x++) {
408 			//get a column
409 			var tableColumn = this.content.tableData[x];
410 			
411 			if(tableColumn != null) {
412 				//truncate the column down to the new number of rows
413 				tableColumn.splice(numRows, dif);
414 			}
415 		}		
416 	}
417 };

513 /**
514  * Update the number of columns since the author has changed the value
515  */
516 View.prototype.TableNode.updateNumColumns = function() {
517 	//get the new number of columns
518 	var numColumns = $('#numColumnsInput').val();
519 	
520 	if(numColumns == '') {
521 		/*
522 		 * do nothing, this is assuming the author is deleting the value to type in a new value after.
523 		 * this function will be called again when the author types in a value
524 		 */
525 	} else {
526 		if(isNaN(numColumns)) {
527 			//message the author that the value is not valid
528 			alert('Error: Invalid Columns value');
529 			
530 			//revert the number of rows value in the text input
531 			$('#numColumnsInput').val(this.content.numColumns);
532 		} else {
533 			numColumns = parseInt(numColumns);
534 			
535 			if(numColumns < this.content.numColumns) {
536 				
537 				var performUpdate = confirm('Are you sure you want to decrease the number of columns? You will lose the data in the columns that will be truncated.');
538 				
539 				if(performUpdate) {
540 					/*
541 					 * the number of columns is less than the previous number
542 					 * of columns so we need to remove some columns
543 					 */
544 					this.truncateColumns(numColumns);
545 				} else {
546 					//do not update
547 					
548 					//revert the number of rows value in the text input
549 					$('#numColumnsInput').val(this.content.numColumns);
550 					
551 					return;
552 				}
553 			}
554 			
555 			//update the number of columns in the content
556 			this.content.numColumns = numColumns;
557 			
558 			//fill in any new cells in the new columns
559 			this.populateNullCells();
560 			
561 			//re-generate the authoring table to reflect the new number of columns
562 			this.updateAuthoringTable();
563 			
564 			//fire source updated event, this will update the preview
565 			this.view.eventManager.fire('sourceUpdated');			
566 		}
567 	}
568 };

627 /**
628  * Update the global cell size in the content
629  */
630 View.prototype.TableNode.updateGlobalCellSize = function() {
631 	//get the global cell size the user has input
632 	var globalCellSize = $('#globalCellSizeInput').val();
633 	
634 	if(globalCellSize == '') {
635 		//the user has input an empty string
636 		
637 		//set the global cell size into the content
638 		this.content.globalCellSize = globalCellSize;
639 		
640 		//save the content

883 /**
884  * Populate any cells with the default cell values if the
885  * cell is null
886  */
887 View.prototype.TableNode.populateNullCells = function() {
888 	//get the table data
889 	var tableData = this.content.tableData;
890 	
891 	//loop through the columns
892 	for(var x=0; x<this.content.numColumns; x++) {
893 		
894 		//try to get the column
895 		if(tableData[x] == null) {
896 			//column does not exist so we will make it
897 			tableData[x] = [];
898 		}
899 		
900 		//loop through all the rows for this column
901 		for(var y=0; y<this.content.numRows; y++) {
902 			//try to get the cell
903 			if(tableData[x][y] == null) {
904 				//the cell does not exist so we will populate it
905 				tableData[x][y] = this.getCellWithDefaultValues();				
906 			}
907 		}
908 	}
909 };
