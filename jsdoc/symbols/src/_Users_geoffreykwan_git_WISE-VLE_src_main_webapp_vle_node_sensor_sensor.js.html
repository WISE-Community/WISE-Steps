  6 /**
  7  * The constructor for the sensor object that we use to peform rendering
  8  * and logic in a sensor step
  9  * @param constructor
 10  * @param node the step we are on
 11  * @param view the vle view
 12  */
 13 function SENSOR(node) {
 14 	//the step we are on
 15 	this.node = node;
 16 	
 17 	//get the view from the node
 18 	this.view = this.node.view;
 19 	
 20 	//the content for the step
 21 	this.content = node.getContent().getContentJSON();
 22 	
 23 	/*
 24 	 * a timestamp used for calculating the amount of time the sensor has
 25 	 * been collecting data. this is only updated in startCollecting() and
 26 	 * dataReceived()
 27 	 */
 28 	this.timeCheck = null;
 29 	
 30 	/*
 31 	 * the amount of time that the sensor has been collecting data. this
 32 	 * will be used as the x value (time) for the sensor data.
 33 	 */
 34 	this.elapsedTime = 0;
 35 	
 36 	//the default time limit for the student to collect data in seconds
 37 	this.dataCollectionTimeLimit = 30;
 38 	
 39 	//the default 
 40 	this.lockPredictionOnCollectionStart = false;
 41 	
 42 	if(node.studentWork != null) {
 43 		//the student has work from previous visits to this step
 44 		this.states = node.studentWork; 
 45 	} else {
 46 		//the student does not have any previous work for this step
 47 		this.states = [];  
 48 	};
 49 	
 50 	//the sensor state that will contain the student work
 51 	this.sensorState = this.getLatestStateCopy();
 52 	
 53 	//flag to keep track of whether the student has change any axis range value this visit
 54 	this.axisRangeChanged = false;
 55 	
 56 	if(this.content != null) {
 57 		//get the graph parameters for displaying the data to the student
 58 		this.graphParams = this.parseGraphParams(this.content.graphParams);
 59 		
 60 		//get the sensor type e.g. 'motion' or 'temperature'
 61 		this.sensorType = this.content.sensorType;
 62 		
 63 		if(this.content.dataCollectionTimeLimit != null) {
 64 			//get the data collection time limit
 65 			var timeLimit = this.content.dataCollectionTimeLimit;
 66 			
 67 			if(!isNaN(parseInt(timeLimit))) {
 68 				//the time limit number is a valid number
 69 				this.dataCollectionTimeLimit = timeLimit;				
 70 			}
 71 		}
 72 		
 73 		if(this.content.lockPredictionOnCollectionStart != null) {
 74 			//get whether to lock the prediction when the student starts collecting data
 75 			this.lockPredictionOnCollectionStart = this.content.lockPredictionOnCollectionStart;			
 76 		}
 77 	}
 78 	
 79 	/*
 80 	 * flag to keep track of whether the student has modified the graph
 81 	 * by retrieving data from the sensor
 82 	 */
 83 	this.graphChanged = false;
 84 	
 85 	//the names for the different types of graphs
 86 	this.graphNames = {
 87 			distance:"distance",
 88 			velocity:"velocity",
 89 			acceleration:"acceleration",
 90 			temperature:"temperature"
 91 	};
 92 	
 93 	//the units for the different types of graphs
 94 	this.graphUnits = {
 95 			distance:"m",
 96 			velocity:"m/s",
 97 			acceleration:"m/s^2",
 98 			temperature:"C",
 99 			time:"s"
100 	};
101 	
102 	/*
103 	 * the color for the graph lines
104 	 * 0 yellow
105 	 * 1 light blue
106 	 * 2 red
107 	 * 3 green
108 	 * 4 purple
109 	 * 5 dark yellow
110 	 * 6 teal blue
111 	 * 7 brown
112 	 * 8 dark green
113 	 * 9 dark purple
114 	 */
115 	this.graphColors = {
116 			distance:2,
117 			velocity:6,
118 			acceleration:4,
119 			temperature:2,
120 			prediction:3
121 	};
122 
123 	//insert the prediction name and unit values into the graphNames and graphUnits arrays
124 	if(this.sensorType == "motion") {
125 		var graphName = "distance prediction";
126 		this.graphNames.prediction = graphName;
127 		this.graphUnits[graphName] = "m";
128 	} else if(this.sensorType == "temperature") {
129 		var graphName = "temperature prediction";
130 		this.graphNames.prediction = graphName;
131 		this.graphUnits[graphName] = "C";
132 	} else {
133 		var graphName = "";
134 		this.graphNames.prediction = graphName;
135 		this.graphUnits[graphName] = "";
136 	}
137 	
138 	/*
139 	 * used to store the plot variable returned from $.plot() so that we can access
140 	 * it from other functions
141 	 */
142 	this.globalPlot = null;
143 	
144 	/*
145 	 * flag to determine whether we need to clear the graph data when the student
146 	 * clicks the Start button. we want to clear the graph when they click the
147 	 * Start button the first time for each new visit of the step. subsequent
148 	 * clicks of the Start button during the same visit will cause the graph 
149 	 * data to append to the existing graph data.
150 	 */
151 	this.clearDataOnStart = true;
152 	
153 	//get the prediction from the previous step if there is a prevWorkNodeIds
154 	this.getPreviousPrediction();
155 	
156 	//default ids for our graph and graph checkboxes divs
157 	this.graphDivId = "graphDiv";
158 	this.graphCheckBoxesDivId = "graphCheckBoxesDiv";
159 	
160 	this.predictionLocked = false;
161 	
162 	if(!this.content.createPrediction || this.sensorState.predictionLocked) {
163 		//the prediction should be locked
164 		this.predictionLocked = true;
165 	}
166 	
167 	//whether the sensor has been loaded or not
168 	this.sensorLoaded = false;
169 	
170 	//the last point the student has clicked on
171 	this.lastPointClicked = null;

174 /**
175  * Render the sensor step
176  */
177 SENSOR.prototype.render = function() {
178 	if(this.content.requirePredictionBeforeEnter && this.node.prevWorkNodeIds.length != 0 && this.sensorState.predictionArray.length == 0) {
179 		/*
180 		 * this step requires a prediction before opening, there is an associated prevWorkNodeId
181 		 * and there is no prediction so we will lock the student out of this step until they
182 		 * create the prediction in the previously associated step
183 		 */
184 		
185 		var prevWorkNodeId = this.node.prevWorkNodeIds[0];
186 		var prevWorkNodeTitle = this.view.getProject().getStepNumberAndTitle(prevWorkNodeId);
187 		
188 		//display the message to tell the student to create a prediction in the previously associated step

190 		this.hideAllInputFields();
191 	} else {
192 		//set the prompt into the step
193 		$('#promptDiv').html(this.content.prompt);
194 
195 		//set the graph title
196 		$('#graphTitle').html(this.content.graphTitle);
197 		
198 		//plot the sensor data from the student's previous visit, if any
199 		this.plotData();
200 		
201 		//add the graph labels
202 		this.setupGraphLabels();
203 		
204 		//set the axis values
205 		this.setupAxisValues();
206 		
207 		//show the graph options if necessary
208 		this.showGraphOptions();
209 		
210 		//display the annotations, if any
211 		this.setupAnnotations();
212 		
213 		/*
214 		 * get the student's previous response, if any, and re-populate
215 		 * the response textarea with it
216 		 */
217 		var response = this.getResponseFromSensorState();
218 		$("#responseTextArea").val(response);
219 		
220 		//set the size of the text area
221 		$("#responseTextArea").attr('rows', this.content.expectedLines);
222 		$("#responseTextArea").attr('cols', 80);
223 		
224 		if(this.content.enableSensor != null && this.content.enableSensor == false) {
225 			//hide the sensor buttons
226 			this.hideSensorButtons();
227 		} else {
228 			/*
229 			 * insert the applet into the html. we need to insert it dynamically
230 			 * because the we need to dynamically determine what type of sensor
231 			 * we expect by looking at the content for this step
232 			 */
233 			this.insertApplet();		
234 		}
235 		
236 		if(!this.content.createPrediction) {
237 			//hide the prediction buttons
238 			this.hidePredictionButtons();
239 		}
240 		
241 		if(this.predictionLocked) {
242 			//disable the prediction buttons
243 			this.disablePredictionButtons();
244 		}
245 		
246 		//display the starter sentence button if necessary
247 		this.displayStarterSentenceButton();
248 		
249 		/*
250 		 * used to determine if the student is click dragging on the graph
251 		 * for use when they are creating a prediction
252 		 */
253 		this.mouseDown = false;
254 		$("#" + this.graphDivId).bind('mousedown', {thisSensor:this}, (function(event) {
255 			event.data.thisSensor.mouseDown = true;
256 		}));
257 		$("#" + this.graphDivId).bind('mouseup', {thisSensor:this}, (function(event) {
258 			event.data.thisSensor.mouseDown = false;
259 		}));
260 
261 		//listen for the keydown event
262 		$(window).bind("keydown", {thisSensor:this}, function(event) {
263 		    event.data.thisSensor.handleKeyDown(event);
264 		});
265 		
266 		//listen for the click event
267 		$(window).bind("click", {thisSensor:this}, function(event) {
268 			//check if the mouse is inside the graph div
269 		    if(!event.data.thisSensor.mouseInsideGraphDiv) {
270 		    	/*
271 		    	 * the mouse is outside the graph div so we will
272 		    	 * set the lastPointClicked to null. we need to do
273 		    	 * this because if the student clicked on a point
274 		    	 * on the graph and then decided to start typing
275 		    	 * their response and typed backspace while typing
276 		    	 * their response, it would delete the 
277 		    	 * lastPointClicked point. so now whenever the
278 		    	 * student clicks outside of the graph div we
279 		    	 * just clear the lastPointClicked so they won't
280 		    	 * accidentally delete a point on the graph.
281 		    	 */
282 		    	event.data.thisSensor.lastPointClicked = null;
283 		    }
284 		});
285 		
286 		//listen for the mouse enter event on the graphDiv
287 		$("#" + this.graphDivId).bind("mouseenter", {thisSensor:this}, function(event) {
288 			event.data.thisSensor.mouseInsideGraphDiv = true;
289 		});
290 		
291 		//listen for the mouse leave event on the graphDiv
292 		$("#" + this.graphDivId).bind("mouseleave", {thisSensor:this}, function(event) {
293 			event.data.thisSensor.mouseInsideGraphDiv = false;
294 		});
295 		
296 		/*
297 		 * used to hide or show the annotation tool tips. if the student has
298 		 * their mouse in the graph div we will hide the annotation tool tips
299 		 * so that they don't block them from clicking on the plot points.
300 		 * when the mouse cursor is outside of the graph div we will show the
301 		 * annotation tool tips for them to view.
302 		 */
303 		$("#" + this.graphDivId).bind('mouseover', (function(event) {
304 			$(".activeAnnotationToolTip").hide();
305 		}));
306 		$("#" + this.graphDivId).bind('mouseleave', (function(event) {
307 			$(".activeAnnotationToolTip").show();
308 		}));
309 	}
310 };

347 /**
348  * This is called when the student has started to collect data
349  */
350 SENSOR.prototype.startCollecting = function() {
351 	//hide the graph message
352 	this.hideGraphMessage();
353 	
354 	if(this.lockPredictionOnCollectionStart && !this.predictionLocked) {
355 		/*
356 		 * this step is set to lock the prediction when the student starts collecting
357 		 * data and it has not been locked yet. we will ask the student if they are
358 		 * sure they want to start collecting data.
359 		 */
360 		var startCollection = confirm('Are you sure you want to start collecting data? You will not be able to change your prediction once you start.');
361 		
362 		if(startCollection) {
363 			//they want to start collecting data
364 			this.sensorState.predictionLocked = true;
365 			this.predictionLocked = true;
366 		} else {
367 			//they do not want to start collecting data yet
368 			return;
369 		}
370 	}
371 	
372 	
373 	if(this.clearDataOnStart) {
374 		/*
375 		 * clear the graph data and annotations because the student
376 		 * is re-collecting the graph data from scratch
377 		 */
378 		this.clearData();
379 		
380 		/*
381 		 * set this to false so that the next time they click the Start button
382 		 * their data won't be cleared. instead the data will be appended
383 		 * and the graph will start back up from where they stopped
384 		 */
385 		this.clearDataOnStart = false;
386 	}
387 	
388 	//get the current date
389 	var currentDate = new Date();
390 	
391 	//get the current time in milliseconds
392 	this.timeCheck = currentDate.getTime();
393 	
394 	if(this.lockPredictionOnCollectionStart) {
395 		//disable the prediction buttons and annotation fields
396 		this.disablePredictionButtons();
397 		this.disablePredictionTextInputAndDeleteButton();
398 	}
399 	
400 	/*
401 	 * get the sensor applet from the html, you won't find it in the html
402 	 * because we dynamically insert it into the sensorAppletDiv since it
403 	 * requires dynamic params such as whether we are using a motion or
404 	 * temperature sensor
405 	 */
406 	var sensorApplet = document.getElementById('sensorApplet');
407 	
408 	//tell the sensor applet to start collecting data
409 	sensorApplet.startCollecting();
410 };

466 /**
467  * This is called when the sensor sends data to the applet
468  * @param type the type of data (not used)
469  * @param count some count value from the sensor (not used)
470  * @param data the data from the sensor
471  */
472 SENSOR.prototype.dataReceived = function(type, count, data) {
473 	//get the current date
474 	var currentDate = new Date();
475 	
476 	//get the current time in milliseconds
477 	var currentTime = currentDate.getTime();
478 	
479 	if(this.sensorState.sensorDataArray.length == 0) {
480 		//this is the first data point so we will set the elapsedTime value to 0
481 		this.elapsedTime = 0;
482 	} else {
483 		//update the amount of time that the sensor has been collecting data
484 		this.elapsedTime += currentTime - this.timeCheck;		
485 	}
486 
487 	if(this.elapsedTime > this.dataCollectionTimeLimit * 1000) {
488 		//we have passed the data collection time limit so we will stop the collection
489 		this.stopCollecting();
490 
491 		return;
492 	}
493 	
494 	//update the time check
495 	this.timeCheck = currentTime;
496 	
497 	if(data != null) {
498 		if(data.constructor.toString().indexOf('Array') != -1) {
499 			/*
500 			 * data is an array so we just want the first element
501 			 * of the array since there is usually only one element.
502 			 * sometimes there are multiple elements but they are very
503 			 * close together in value so we aren't really losing
504 			 * much data. I'm not sure why data sometimes contains
505 			 * more than one element.
506 			 */
507 			data = data[0];
508 		} else {
509 			//data is not an array
510 		}
511 	}
512 	
513 	/*
514 	 * round the x vale to the nearest hundredth. elapsedTime is in milliseconds
515 	 * so we need to divide by 1000 to get seconds
516 	 */
517 	var x = parseFloat((this.elapsedTime / 1000).toFixed(2));
518 	
519 	//round the y value to the nearest hundredth
520 	var y = parseFloat(data.toFixed(2));
521 	
522 	//save the data point into the sensor state.
523 	this.sensorState.dataReceived(x, y);
524 	
525 	//update the graph
526 	this.plotData();
527 	
528 	//update the flag since the graph has changed
529 	this.graphChanged = true;
530 };

532 /**
533  * Generate the data array that we will send to the graph to plot
534  * @param state a SENSORSTATE object
535  * @return an array containing data that will be used to plot
536  * the data onto a graph
537  */
538 SENSOR.prototype.generateGraphDataArray = function(state) {
539 	var dataArray = [];
540 	
541 	if(state != null) {
542 		//get the data array from the state
543 		var sensorDataArray = state.sensorDataArray;
544 		
545 		if(sensorDataArray != null) {
546 
547 			/*
548 			 * the interval that we want to display data points. this is used
549 			 * to smooth out the graphs for the student and to spread out the
550 			 * data points so they aren't all clustered together.
551 			 * e.g. if the sampleInterval is set to 5, we will only show every
552 			 * 5th data point in the graph to the student
553 			 */
554 			var sampleInterval = 3;
555 			
556 			//loop through all the elements in the data array
557 			for(var i=0; i<sensorDataArray.length; i++) {
558 				
559 				/*
560 				 * check if we want to display this data point in regards
561 				 * to the sample interval
562 				 */
563 				if(i % sampleInterval == 0) {
564 					//get the data array element
565 					var sensorData = sensorDataArray[i];
566 					
567 					//get the time
568 					var x = sensorData.x;
569 					
570 					//get the y value. this may be distance or temp or etc.
571 					var y = sensorData.y;
572 					
573 					/*
574 					 * add the x, y data point into the array. flot expects
575 					 * the each element in the array to be an array.
576 					 */
577 					dataArray.push([x, y]);					
578 				}
579 			}
580 		}		
581 	}
582 	
583 	return dataArray;
584 };

616 /**
617  * Parse the graph parameters from the step content
618  * @return an object we can use to pass to the flot graph to
619  * specify how the graph should look
620  */
621 SENSOR.prototype.parseGraphParams = function(contentGraphParams) {
622 	//create the object that will contain the graph params
623 	var graphParams = {};
624 	
625 	//create an xaxis object
626 	graphParams.xaxis = {};
627 	
628 	//create a yaxis object
629 	graphParams.yaxis = {};
630 	
631 	if(contentGraphParams != null) {
632 		if(contentGraphParams.xmin != null && contentGraphParams.xmin != "") {
633 			//set the xmin value
634 			graphParams.xaxis.min = contentGraphParams.xmin;
635 		}
636 
637 		if(contentGraphParams.xmax != null && contentGraphParams.xmax != "") {
638 			//set the xmax value
639 			graphParams.xaxis.max = contentGraphParams.xmax;
640 		}
641 
642 		if(contentGraphParams.ymin != null && contentGraphParams.ymin != "") {
643 			//set the ymin value
644 			graphParams.yaxis.min = contentGraphParams.ymin;
645 		}
646 
647 		if(contentGraphParams.ymax != null && contentGraphParams.ymax != "") {
648 			//set the ymax value
649 			graphParams.yaxis.max = contentGraphParams.ymax;
650 		}
651 	}
652 	
653 	/*
654 	 * if the sensor state contains axis values it will override
655 	 * the axis values from the content
656 	 */
657 	if(this.sensorState != null) {
658 		if(this.sensorState.xMin != null) {
659 			//set the xmin value from the sensor state
660 			graphParams.xaxis.min = this.sensorState.xMin;
661 		}
662 		
663 		if(this.sensorState.xMax != null) {
664 			//set the xmax value from the sensor state
665 			graphParams.xaxis.max = this.sensorState.xMax;
666 		}
667 		
668 		if(this.sensorState.yMin != null) {
669 			//set the ymin value from the sensor state
670 			graphParams.yaxis.min = this.sensorState.yMin;
671 		}
672 		
673 		if(this.sensorState.yMax != null) {
674 			//set the ymax value from the sensor state
675 			graphParams.yaxis.max = this.sensorState.yMax;
676 		}
677 	}
678 	
679 	//turn lines and points on
680 	graphParams.series = {lines:{show:true}, points:{show:true}};
681 	
682 	//allow points to be hoverable and clickable
683 	graphParams.grid = {hoverable:true, clickable:true};
684 	
685 	return graphParams;
686 };

926 /**
927  * Calculate the derivative array by calculating the rate of change
928  * of the points in the given data array
929  * @param dataArray an array containing data points. each element in the
930  * array is also an array and looks like this [<x>,<y>]
931  * @return an array containing the derivative values
932  */
933 SENSOR.prototype.calculateDerivativeArray = function(dataArray) {
934 	//the array we will store the derivative values in
935 	var derivativeArray = [];
936 	
937 	//loop through all the points in the data array
938 	for(var x=0; x<dataArray.length; x++) {
939 		//get the index of the previous point
940 		var previousPointIndex = x - 1;
941 		
942 		//make sure the previous point exists
943 		if(previousPointIndex >= 0) {
944 			//get the previous point
945 			var point1 = dataArray[previousPointIndex];
946 			
947 			//get the current point
948 			var point2 = dataArray[x];
949 			
950 			//get the x and y values of the previous point
951 			var point1x = point1[0];
952 			var point1y = point1[1];
953 			
954 			//get the x and y values of the current point
955 			var point2x = point2[0];
956 			var point2y = point2[1];
957 			
958 			var derivativePoint = [];
959 			
960 			//get the difference between the y values of the points
961 			var derivativeY = point2y - point1y;
962 			
963 			//get the difference between the x values of the points
964 			var derivativeX = point2x - point1x;
965 			
966 			//calculate the rate of change to get the derivative value
967 			var derivativeValue = derivativeY / derivativeX;
968 			
969 			//use the x value of the current point as the x value for the derivative point
970 			derivativePoint[0] = point2x;
971 			
972 			//round the derivative value to the nearest hundredth and set it as the y value for the derivative point
973 			derivativePoint[1] = parseFloat(derivativeValue.toFixed(2));
974 			
975 			//add the derivative point to our array
976 			derivativeArray.push(derivativePoint);
977 		}
978 	}
979 	
980 	return derivativeArray;
981 };
