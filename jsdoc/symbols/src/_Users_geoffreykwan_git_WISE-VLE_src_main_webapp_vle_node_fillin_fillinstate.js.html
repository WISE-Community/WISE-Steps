  6 /**
  7  * For re-creating the student's vle_state from their xml for
  8  * researcher/teacher display
  9  * @constructor
 10  */
 11 function FILLINSTATE(textEntryInteractionIndex, response, timestamp) {
 12 	this.type = "fi";
 13 	this.textEntryInteractionIndex = textEntryInteractionIndex;  // which blank the student answered.
 14 	this.response = response;   // what the student wrote in the blank.
 15 	
 16 	if(arguments.length == 2) {
 17 		//if the third argument (timestamp) was ommitted just set it to the current time
 18 		this.timestamp = Date.parse(new Date());
 19 	} else {
 20 		this.timestamp = timestamp;
 21 	}
 22 }
 23 
 24 FILLINSTATE.prototype.print = function() {
 25 };
 26 
 27 FILLINSTATE.prototype.getDataXML = function() {
 28 	return "<textEntryInteractionIndex>" + this.textEntryInteractionIndex + "</textEntryInteractionIndex><response>" + this.response + "</response><timestamp>" + this.timestamp + "</timestamp>";
 29 };
 30 
 31 FILLINSTATE.prototype.parseDataXML = function(stateXML) {
 32 	var textEntryInteractionIndex = stateXML.getElementsByTagName("textEntryInteractionIndex")[0];
 33 	var response = stateXML.getElementsByTagName("response")[0];
 34 	var timestamp = stateXML.getElementsByTagName("timestamp")[0];
 35 	
 36 	if(textEntryInteractionIndex == undefined || response == undefined || timestamp == undefined) {
 37 		return null;
 38 	} else {
 39 		return new FILLINSTATE(textEntryInteractionIndex.textContent, response.textContent, timestamp.textContent);
 40 	}
 41 };
 42 
 43 
 44 /**
 45  * Takes in a state JSON object and returns an FILLINSTATE object
 46  * @param stateJSONObj a state JSON object
 47  * @return a FILLINSTATE object
 48  */
 49 FILLINSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 50 	//create a new FILLINSTATE object
 51 	var fillinState = new FILLINSTATE();
 52 	
 53 	//set the attributes of the FILLINSTATE object
 54 	fillinState.textEntryInteractionIndex = stateJSONObj.textEntryInteractionIndex;
 55 	fillinState.response = stateJSONObj.response;
 56 	fillinState.timestamp = stateJSONObj.timestamp;
 57 	
 58 	//return the FILLINSTATE object
 59 	return fillinState;
 60 };
 61 
 62 /**
 63  * Returns what the student typed
 64  * @return the answer the student typed
 65  */
 66 FILLINSTATE.prototype.getStudentWork = function() {
 67 	return this.response;
 68 };
