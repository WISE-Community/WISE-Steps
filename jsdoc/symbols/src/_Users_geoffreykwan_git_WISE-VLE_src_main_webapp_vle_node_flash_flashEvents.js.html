  2 /*
  3  * This handles events and calls the appropriate function to handle
  4  * the event.
  5  */
  6 View.prototype.flashDispatcher = function(type,args,obj){
  7 	/*
  8 	 * check to see if the event name matches 
  9 	 */ 
 10 	if(type == 'flashPromptChanged') {
 11 		obj.FlashNode.updatePrompt(args);
 12 	} else if (type == 'flashSwfUrlChanged') {
 13 		obj.FlashNode.updateSwfUrl(args);
 14 	} else if (type == 'flashSwfHeightChanged') {
 15 		obj.FlashNode.updateSwfHeight(args);
 16 	} else if (type == 'flashSwfWidthChanged') {
 17 		obj.FlashNode.updateSwfWidth(args);
 18 	} else if (type == 'flashEnableDataChanged') {
 19 		obj.FlashNode.updateEnableData(args);
 20 	} else if (type == 'flashEnableGradingChanged') {
 21 		obj.FlashNode.updateEnableGrading(args);
 22 	} else if (type == 'flashShowRichText') {
 23 		obj.FlashNode.showRichText(args);
 24 	} else if (type == 'flashHideRichText') {
 25 		obj.FlashNode.hideRichText(args);
 26 	} else if (type == 'flashBrowseClicked') {
 27 		obj.FlashNode.browseFlashAssets(args);
 28 	} else if (type == 'flashGradingTypeChanged') {
 29 		obj.FlashNode.updateGradingType(args);
 30 	} else if (type == 'flashAddFlashvar') {
 31 		obj.FlashNode.createNewFlashvarInput(args);
 32 	} else if (type == 'flashFlashvarsChanged') {
 33 		obj.FlashNode.updateFlashvars(args);
 34 	} else if (type == 'flashDeleteFlashvar') {
 35 		obj.FlashNode.deleteFlashvar(args[0]);
 36 	}
 37 };
 38 
 39 /*
 40  * this is a list of events that can be fired. when the event is fired,
 41  * the dispatcher function above will be called and then call the
 42  * appropriate function to handle the event.
 43  */
 44 var events = [
 45 	'flashPromptChanged',
 46 	'flashSwfUrlChanged',
 47 	'flashSwfHeightChanged',
 48 	'flashSwfWidthChanged',
 49 	'flashEnableDataChanged',
 50 	'flashEnableGradingChanged',
 51 	'flashShowRichText',
 52 	'flashHideRichText',
 53 	'flashBrowseClicked',
 54 	'flashGradingTypeChanged',
 55 	'flashAddFlashvar',
 56 	'flashFlashvarsChanged',
 57 	'flashDeleteFlashvar'
 58 ];
 59 
 60 /*
 61  * add all the events to the vle so the vle will listen for these events
 62  * and call the dispatcher function when the event is fired
 63  */
 64 for(var x=0; x<events.length; x++) {
 65 	componentloader.addEvent(events[x], 'flashDispatcher');
 66 };
 67 
 68 //used to notify scriptloader that this script has finished loading
 69 if(typeof eventManager != 'undefined'){
 70 	eventManager.fire('scriptLoaded', 'vle/node/flash/flashEvents.js');
 71 };
