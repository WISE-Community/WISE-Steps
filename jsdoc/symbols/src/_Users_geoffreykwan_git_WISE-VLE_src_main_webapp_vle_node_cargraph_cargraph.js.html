284 /**
285  * Given xValue, returns yValue of a prediction graph whose points are stored in
286  * the given predictionArray.
287  * 
288  * Note that the predictionArray may not contain the given xValue.  For example, trying to find
289  * the yValue in (3,?) when your predictionArray only contains 
290  * [(1,2),(2,4),(7,3)]. In this case, algebra is used to calculate the y value.
291  * 
292  * @param xValue: x Value in the prediction graph
293  * @param predictionArray: array of points for one prediction graph
294  * @return: corresponding y-Value within the prediction graph.
295  */ 
296 CARGRAPH.prototype.getYValue = function(xValue,predictionArray) {
297     var xSoFar = 0;
298     var ySoFar = 0;
299     for (var i=0; i< predictionArray.length; i++) {
300 	    var prediction = predictionArray[i];  // prediction[0] = x, prediction[1] = y
301 	    if (prediction[0] < xValue) {
302 		    // x value not yet found, set ySoFar in case we'll need it for later
303 		    xSoFar = prediction[0];
304 		    ySoFar = prediction[1];				    
305 	    } else if (prediction[0] == xValue) {
306 		    // x match found, return it
307 		    return prediction[1];
308 	    } else {
309 		    // there was no xValue in the prediction. we've reached a xValue in the prediction array
310 		    // that is greater than xValue. 
311 		    // calculate using the power of Algebra
312 		    // magic formula: y2 = y1 + m(x2-x1)
313 		    // m = (y2-y1) / (x2-x1)
314 		    var slope = (prediction[1] - ySoFar) / (prediction[0] - xSoFar);
315 		    var yValue = ySoFar + slope * (xValue-xSoFar);
316 		    return yValue;
317 	    }
318     }
319     return -1;
320 };

322 /**
323  * Given xValue, returns yValue of a prediction graph whose points are stored in
324  * the given predictionArray.
325  * 
326  * Note that the predictionArray may not contain the given xValue.  For example, trying to find
327  * the yValue in (3,?) when your predictionArray only contains 
328  * [(1,2),(2,4),(7,3)]. In this case, algebra is used to calculate the y value.
329  * 
330  * @param xValue: x Value in the prediction graph
331  * @param predictionArray: array of points for one prediction graph
332  * @return: corresponding y-Value within the prediction graph.
333  */ 
334 CARGRAPH.prototype.getYValueObj = function(xValue,predictionArray) {
335     var xSoFar = 0;
336     var ySoFar = 0;
337     for (var i=0; i< predictionArray.length; i++) {
338 	    var prediction = predictionArray[i];  // prediction[0] = x, prediction[1] = y
339 	    if (prediction.x < xValue) {
340 		    // x value not yet found, set ySoFar in case we'll need it for later
341 		    xSoFar = prediction.x;
342 		    ySoFar = prediction.y;				    
343 	    } else if (prediction.x == xValue) {
344 		    // x match found, return it
345 		    return prediction.y;
346 	    } else {
347 		    // there was no xValue in the prediction. we've reached a xValue in the prediction array
348 		    // that is greater than xValue. 
349 		    // calculate using the power of Algebra
350 		    // magic formula: y2 = y1 + m(x2-x1)
351 		    // m = (y2-y1) / (x2-x1)
352 		    var slope = (prediction.y - ySoFar) / (prediction.x - xSoFar);
353 		    var yValue = ySoFar + slope * (xValue-xSoFar);
354 		    return yValue;
355 	    }
356     }
357     return -1;
358 };

428 /**
429  * Parse the graph parameters from the step content
430  * @return an object we can use to pass to the flot graph to
431  * specify how the graph should look
432  */
433 CARGRAPH.prototype.parseGraphParams = function(contentGraphParams) {
434 	//create the object that will contain the graph params
435 	var graphParams = {};
436 	
437 	//create an xaxis object
438 	graphParams.xaxis = {};
439 	
440 	//create a yaxis object
441 	graphParams.yaxis = {};
442 	
443 	if(contentGraphParams != null) {
444 		if(contentGraphParams.xmin != null && contentGraphParams.xmin != "") {
445 			//set the xmin value
446 			graphParams.xaxis.min = contentGraphParams.xmin;
447 		}
448 
449 		if(contentGraphParams.xmax != null && contentGraphParams.xmax != "") {
450 			//set the xmax value
451 			graphParams.xaxis.max = contentGraphParams.xmax;
452 		}
453 
454 		if(contentGraphParams.ymin != null && contentGraphParams.ymin != "") {
455 			//set the ymin value
456 			graphParams.yaxis.min = contentGraphParams.ymin;
457 		}
458 
459 		if(contentGraphParams.ymax != null && contentGraphParams.ymax != "") {
460 			//set the ymax value
461 			graphParams.yaxis.max = contentGraphParams.ymax;
462 		}
463 		
464 	}
465 	
466 	/*
467 	 * if the carGraph state contains axis values it will override
468 	 * the axis values from the content
469 	 */
470 	if(this.carGraphState != null) {
471 		if(this.carGraphState.xMin != null) {
472 			//set the xmin value from the carGraph state
473 			graphParams.xaxis.min = this.carGraphState.xMin;
474 		}
475 		
476 		if(this.carGraphState.xMax != null) {
477 			//set the xmax value from the carGraph state
478 			graphParams.xaxis.max = this.carGraphState.xMax;
479 		}
480 		
481 		if(this.carGraphState.yMin != null) {
482 			//set the ymin value from the carGraph state
483 			graphParams.yaxis.min = this.carGraphState.yMin;
484 		}
485 		
486 		if(this.carGraphState.yMax != null) {
487 			//set the ymax value from the carGraph state
488 			graphParams.yaxis.max = this.carGraphState.yMax;
489 		}
490 	}
491 	
492 	//turn lines and points on
493 	graphParams.series = {lines:{show:true}, points:{show:true}};
494 	
495 	//allow points to be hoverable and clickable
496 	graphParams.grid = {hoverable:true, clickable:true};
497 	
498 	graphParams.crosshair = { mode: "x" };
499 	return graphParams;
500 };

514 /**
515  * Save the student work for this step. This includes the carGraph
516  * data and the response the student typed. 
517  */
518 CARGRAPH.prototype.save = function() {
519 	//get the response the student typed
520 	var response = ""
521 	
522 	//get the previous student work
523 	var latestState = this.getLatestState();
524 	var previousResponse = '';
525 	
526 	if(latestState != null) {
527 		//get the previous response
528 		var previousResponse = latestState.response;
529 	}
530 	
531 	/*
532 	 * check that the student has changed the response or the graph or any annotations
533 	 */
534 	if(response != previousResponse || this.graphChanged || this.annotationsChanged || this.axisRangeChanged) {
535 		//set the student response into the state
536 		this.carGraphState.response = response;
537 			
538 		//fire the event to push this state to the global view.states object
539 		eventManager.fire('pushStudentWork', this.carGraphState);
540 
541 		//push the state object into the local copy of states
542 		this.states.push(this.carGraphState);		
543 	}
544 	
545 	/*
546 	 * changes have been saved or there were no changes so we will reset
547 	 * the flag back to false
548 	 */
549 	this.graphChanged = false;
550 	this.annotationsChanged = false;
551 	this.axisRangeChanged = false;
552 };

585  * Plot the data onto the graph so the student can see it.
586  * @param graphDivId the id of the div we will use to plot the graph
587  * @param graphCheckBoxesDivId the id of the div we will put the filter check boxes in
588  */
589 CARGRAPH.prototype.plotData = function(graphDivId, graphCheckBoxesDivId) {
590 	if(graphDivId == null) {
591 		//this will be the default graphDivId if none is provided as an argument
592 		graphDivId = this.graphDivId;
593 	}
594 	
595 	//set the graph div id so it can be accessed later by other functions
596 	this.graphDivId = graphDivId;
597 	
598 	if(graphCheckBoxesDivId == null) {
599 		//this will be the default graphCheckBoxesDivId if none is provided as an argument
600 		graphCheckBoxesDivId = "graphCheckBoxesDiv";
601 	}
602 	
603 	//set the graph check boxes div id so it can be accessed later by other functions
604 	this.graphCheckBoxesDivId = graphCheckBoxesDivId;
605 	
606 	//get the graph params
607 	var graphParams = this.getGraphParams();
608 	
609 	//create the data sets array
610 	var dataSets = [];
611 
612 	/*
613 	 * check if we want to show the correct graph. this will be used
614 	 * in the grading tool
615 	 */
616 	if(this.showCorrectGraph) {
617 		//get the cars that are used in this car graph step
618 		var dynamicImages = this.content.dynamicImages;
619 		
620 		//get the expected results
621 		var expectedResults = this.content.expectedResults;
622 		
623 		//loop through all the cars that are used in this step
624 		for(var x=0; x<dynamicImages.length; x++) {
625 			//get a car
626 			var dynamicImage = dynamicImages[x];
627 			
628 			//loop through all the expected results
629 			for(var y=0; y<expectedResults.length; y++) {
630 				//get an expected result
631 				var expectedResult = expectedResults[y];
632 				
633 				if(dynamicImage != null && expectedResult != null) {
634 					//check if we have found an expected result for the car that is used in this step
635 					if(dynamicImage.id == expectedResult.id) {
636 						//get the array of correct points so we can plot them
637 						var expectedPoints = this.convertToPlottableArray(expectedResult.expectedPoints);
638 						
639 						//get the graph line label
640 						var graphLabel = "Correct " + dynamicImage.graphLabel;
641 						
642 						/*
643 						 * get the graph line name. we will just convert the graphLabel to
644 						 * have no white space and change the first character to lower case
645 						 * e.g.
646 						 * Correct Green Car
647 						 * to
