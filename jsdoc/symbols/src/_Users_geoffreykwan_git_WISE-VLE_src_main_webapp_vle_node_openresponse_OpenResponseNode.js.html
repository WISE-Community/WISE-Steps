 12 /**
 13  * @constructor
 14  * @extends Node
 15  * @param nodeType
 16  * @param view
 17  * @returns {OpenResponseNode}
 18  */
 19 function OpenResponseNode(nodeType, view) {
 20 	this.view = view;
 21 	this.type = nodeType;
 22 	this.prevWorkNodeIds = [];
 23 	this.exportableToNodes = new Array(			
 24 			"NoteNode", 
 25 			"OpenResponseNode", 
 26 			"SVGDrawNode");	
 27 	this.importableFileExtensions = new Array(
 28 			"jpg", "png");
 29 };

 31 /**
 32  * Takes in a state JSON object and returns an OPENRESPONSESTATE object
 33  * @param nodeStatesJSONObj a state JSON object
 34  * @return an OPENRESPONSESTATE object

 36 OpenResponseNode.prototype.parseDataJSONObj = function(stateJSONObj) {
 37 	return OPENRESPONSESTATE.prototype.parseDataJSONObj(stateJSONObj);
 38 };

 40 OpenResponseNode.prototype.translateStudentWork = function(studentWork) {
 41 	return studentWork;
 42 };

 44 /**
 45  * Get the prompt for the peer review view
 46  */
 47 OpenResponseNode.prototype.getPeerReviewPrompt = function() {
 48 	if(this.or == null) {
 49 		//create an OPENRESPONSE to load the content
 50 		this.or = new OPENRESPONSE(this);	
 51 	}
 52 	
 53 	//retrieve the prompt from the OPENRESPONSE content
 54 	var peerReviewPrompt = this.or.getPeerReviewPrompt();
 55 	return peerReviewPrompt;
 56 };
 57 
 58 /**
 59  * Get the other student's work that will be peer reviewed.
 60  * We will get the other student's work by parsing the JSON 
 61  * NodeVisit obj that is passed in.
 62  * @param otherStudentWorkJSONObj a JSON object containing a NodeVisit of
 63  * the other student's work
 64  * @return the other student's work
 65  */
 66 OpenResponseNode.prototype.getPeerReviewOtherStudentWork = function(otherStudentWorkJSONObj) {
 67 	if(this.or == null) {
 68 		this.or = new OPENRESPONSE(this);	
 69 	}
 70 	
 71 	var peerReviewOtherStudentWork = this.or.getPeerReviewOtherStudentWork(otherStudentWorkJSONObj);
 72 	return peerReviewOtherStudentWork;
 73 };

 75 /**
 76  * Returns true iff this node can export work to the specified node.
 77  * @param exportToNode node to export work into
 78  * @return true/false
 79  */
 80 OpenResponseNode.prototype.canExportWork = function(exportToNode) {
 81 	return this.exportableToNodes &&
 82 		this.exportableToNodes.indexOf(exportToNode.type) > -1;
 83 };
 84 
 85 /**
 86  * Returns a string of the work so that it can be imported by the specified exportToNode
 87  * @param exportToNode node that will import the return value of this method
 88  * @return null if this node cannot export work to the exportToNode
 89  */
 90 OpenResponseNode.prototype.exportWork = function(exportToNode) {	
 91 	if (this.canExportWork(exportToNode)) {
 92 	    var nodeVisitArray = this.view.state.getNodeVisitsByNodeId(this.id);
 93 	    if (nodeVisitArray.length > 0) {
 94 	        var states = [];
 95 	        var latestNodeVisit = nodeVisitArray[nodeVisitArray.length -1];
 96 	        for (var i = 0; i < nodeVisitArray.length; i++) {
 97 	            var nodeVisit = nodeVisitArray[i];
 98 	            for (var j = 0; j < nodeVisit.nodeStates.length; j++) {
 99 	                states.push(nodeVisit.nodeStates[j]);
100 	            }
101 	        }
102 	        var latestState = states[states.length - 1];
103 	        var studentWork = latestState.getStudentWork();
104 	        
105 	        if (exportToNode.type == "SVGDrawNode") {
106 	        	var svgString = '<text x="250" y="150" font-family="Verdana" font-size="35" fill="black" >'
107 				+ studentWork
108 				+ '</text>';
109 				return svgString;
110 	        } else {
111 	        	return studentWork;
112 	        }
113 	    };			
114 	};
115 	return null;
116 };
117 
118 /**
119  * Imports and inserts the work from the specified importFromNode
120  * @param importFromNode node that will export the data for this node to import
121  * @return
122  */
123 OpenResponseNode.prototype.importWork = function(importFromNode) {
124 	var studentWork = importFromNode.exportWork(this);
125 	if (studentWork != null) {
126 		if(this.contentPanel && this.contentPanel.or) {
127 			this.contentPanel.or.appendResponse(studentWork);
128 		}
129 	};
130 };
131 
132 /**
133  * Returns true iff the given file can be imported 
134  * into this step's work.
135  */
136 OpenResponseNode.prototype.canImportFile = function(filename) {
137 	// can't import if this is not in a rich text editor mode.
138 	if (this.contentPanel.or.content.isRichTextEditorAllowed) {
139 		if (filename.indexOf(".") != -1) {
140 			var fileExt = filename.substr(filename.lastIndexOf(".")+1);	
141 			if (this.importableFileExtensions.indexOf(fileExt.toLowerCase()) != -1) {
142 			return true;
143 			}
144 		}
145 	}
146 	return false;
147 };
148 
149 /**
150  * Imports and inserts the specified file into current response.
151  * @param file to insert into response
152  * @return true iff import is successful
153  */
154 OpenResponseNode.prototype.importFile = function(filename) {
155 	if (this.canImportFile(filename)) {
156 		var importFileHTML = "<img src='"+filename+"'></img>";
157 		this.contentPanel.or.appendResponse(importFileHTML);
158 		return true;
159 	}
160 	return false;
161 };
162 
163 /**
164  * Called when the step is exited. This is used for auto-saving.
165  */
166 OpenResponseNode.prototype.onExit = function() {
167 	//check if the content panel exists
168 	if(this.contentPanel && this.contentPanel.save) {
169 		//tell the content panel to save
170 		this.contentPanel.save();
171 	};
172 };
