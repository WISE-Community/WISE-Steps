  2 /**
  3  * The openResponseDispatcher catches events specific to authoring individual
  4  * open response type steps and delegates them to the appropriate functions for
  5  * this view.
  6  */
  7 View.prototype.openResponseDispatcher = function(type,args,obj){
  8 	if(type=='openResponsePromptChanged'){
  9 		obj.OpenResponseNode.updatePrompt();
 10 	} else if(type=='openResponseStarterOptionChanged'){
 11 		obj.OpenResponseNode.starterChanged();
 12 	} else if(type=='openResponseStarterSentenceUpdated'){
 13 		obj.OpenResponseNode.starterUpdated();
 14 	} else if(type=='openResponseUpdateRichText'){
 15 		obj.OpenResponseNode.updateRichText();
 16 	} else if(type=='openResponseLinesChanged'){
 17 		obj.OpenResponseNode.linesUpdated();
 18 	} else if(type=='openResponsePeerReviewAuthoredWorkUpdated'){
 19 		obj.OpenResponseNode.peerReviewAuthoredWorkUpdated();
 20 	} else if(type=='openResponsePeerReviewPercentageTriggerUpdated'){
 21 		obj.OpenResponseNode.peerReviewPercentageTriggerUpdated();
 22 	} else if(type=='openResponsePeerReviewNumberTriggerUpdated'){
 23 		obj.OpenResponseNode.peerReviewNumberTriggerUpdated();
 24 	} else if(type=='openResponsePeerReviewAuthoredReviewUpdated'){
 25 		obj.OpenResponseNode.peerReviewAuthoredReviewUpdated();
 26 	} else if(type=='openResponsePeerReviewStepNotOpenCustomMessageUpdated'){
 27 		obj.OpenResponseNode.peerReviewStepNotOpenCustomMessageUpdated();
 28 	};
 29 };
 30 
 31 //this list of events
 32 var events = [
 33 	'openResponsePromptChanged',
 34 	'openResponseStarterOptionChanged',
 35 	'openResponseStarterSentenceUpdated',
 36 	'openResponseUpdateRichText',
 37 	'openResponseLinesChanged',
 38 	'openResponsePeerReviewAuthoredWorkUpdated',
 39 	'openResponsePeerReviewPercentageTriggerUpdated',
 40 	'openResponsePeerReviewNumberTriggerUpdated',
 41 	'openResponsePeerReviewAuthoredReviewUpdated',
 42 	'openResponsePeerReviewStepNotOpenCustomMessageUpdated'
 43 ];
 44 
 45 /*
 46  * add all the events to the vle so the vle will listen for these events
 47  * and call the dispatcher function when the event is fired
 48  */
 49 for(var x=0; x<events.length; x++) {
 50 	componentloader.addEvent(events[x], 'openResponseDispatcher');
 51 };
 52 
 53 //used to notify scriptloader that this script has finished loading
 54 if(typeof eventManager != 'undefined'){
 55 	eventManager.fire('scriptLoaded', 'vle/node/openresponse/openResponseEvents.js');
 56 };
