 85 /**
 86  * Overrides Node.smartFilter
 87  * 
 88  * @param stateObj one student's graph points
 89  * @return true iff this state is something that should appear in the smart filter
 90  */
 91 CarGraphNode.prototype.smartFilter = function(stateObj) {
 92 	var returnObj = {
 93 			filtered: false,
 94 			maxError: 0.0,
 95 			avgError: 0.0,
 96 			errMargin: 0.0,
 97 			error: false
 98 		};
 99 	
100 	// no smart filter check if no state.
101 	if (stateObj == null || stateObj == "") {
102 		return returnObj;
103 	}
104 		
105 	// get the configuration JSON object
106 	var objJson = this.content.getContentJSON();
107 	
108 	// Go through the dynamic images and check each one
109 	for (var i=0; i < objJson.dynamicImages.length; i++) {
110 		var carId = objJson.dynamicImages[i].id;
111 	
112 		if (stateObj.getPredictionObjByPredictionId(carId) === null) {
113 			returnObj.error = true;
114 			return returnObj;
115 		}
116 		// get the points drawn by the student for the current car
117 		var carPred = stateObj.getPredictionObjByPredictionId(carId).predictions;
118 			
119 		// retrieve the preconfigured results object for current car from the config JSON
120 		var carExpObj = this.getResultObjByResultId(carId);
121 		
122 		if (carExpObj === null) {
123 			returnObj.error = true;
124 			return returnObj;
125 		}
126 		
127 		// retrieve the allowed error margin
128 		var errMargin = 0.5;
129 		if (carExpObj.maxErrorMargin !== null && carExpObj.maxErrorMargin !== undefined) {
130 			errMargin = carExpObj.maxErrorMargin;
131 		}		
132 		// retrieve the expected points for the car
133 		var carExpPoints = carExpObj.expectedPoints;
134 			
135 		/*
136 		 * We calculate the Y-values for the student graph and the optimal graph
137 		 * at various X-values that we go through on a interval
138 		 */
139 		var carGraphObj = new CARGRAPH(this);
140 		var maxYError = 0.0;
141 		var allError = 0.0;
142 		var filtered = false;
143 	
144 		for (var x = (parseInt(objJson.graphParams.xmin) +1); x < parseInt(objJson.graphParams.xmax); x++) {
145 			var expectedYValue = carGraphObj.getYValueObj(x, carExpPoints);
146 			var studentYValue = carGraphObj.getYValueObj(x, carPred);
147 			var delta = Math.abs(expectedYValue - studentYValue) * 1.0;
148 			
149 			if (delta > returnObj.maxError) {
150 				returnObj.maxError = delta;
151 			}
152 			
153 			allError = allError + delta;
154 			
155 			if (delta > errMargin) {
156 				filtered = true;
157 			}
158 		}
159 		
160 		returnObj.avgError = allError / x;		
161 		returnObj.filtered = filtered;
162 		returnObj.errMargin = errMargin;
163 	}
164 	
165 	return returnObj;	
166 };

169 CarGraphNode.prototype.getResultObjByResultId = function(resultsId) {
170 	for (var i=0; i < this.content.getContentJSON().expectedResults.length; i++) {
171 		if (this.content.getContentJSON().expectedResults[i].id == resultsId) {
172 			return this.content.getContentJSON().expectedResults[i];
173 		}
174 	}
175 	return null;
176 };

179 /**
180  * This function is called when the student exits the step. It is mostly
181  * used for error checking.
182  * 
183  * TODO: rename CarGraphNode
184  * 
185  * Note: In most cases you will not have to change anything here.
186  */
187 CarGraphNode.prototype.onExit = function() {
188 	// run a smart filter and if student has done a bad job, save it in the status
189 	// run smart filter and flag if the smart filter returns true	
190 	var carGraphState = this.view.state.getLatestWorkByNodeId(this.id);
191 	var smartFilterResult = this.smartFilter(carGraphState); // obj
192 
193 	if (this.view.studentStatus == null) {
194 		this.view.studentStatus = new StudentStatus();
195 	}
196 	
197 	if (smartFilterResult.maxError > 10) {
198 		//this.view.studentStatus.maxAlertLevel = "alert";
199 		
200 		if(this.view.studentStatus.alertables == null) {
201 			this.view.studentStatus.alertables = [];			
202 		}
203 		
204 		var alertLevel = 5;
205 		var nodeId = this.id;
206 		var type = "autoScoreResult";
207 		var value = smartFilterResult.maxError;
208 		
209 		var stepNumberAndTitle = this.view.project.getStepNumberAndTitle(this.id);
210 		var nodeType = this.type;
211 		var readableText = 'Autoscore returned '+smartFilterResult.maxError+'. This might mean that the student needs help.';
212 		
213 		var date = new Date();
214 		var timestamp = date.getMilliseconds();
215 		
216 		var alertable = new StudentAlertable(alertLevel, nodeId, type, value, stepNumberAndTitle, nodeType, readableText, timestamp);
217 		
218 		this.view.studentStatus.addAlertable(alertable);
219 	} else {
220 		//this.view.studentStatus.type = "ok";
221 		this.view.studentStatus.removeAlertable(this.id);
222 	}
223 	
224 	//check if the content panel has been set
225 	if(this.contentPanel) {
226 		
227 		if(this.contentPanel.save) {
228 			//tell the content panel to save
229 			this.contentPanel.save();
230 		}
231 		
232 		try {
233 			/*
234 			 * check if the onExit function has been implemented or if we
235 			 * can access attributes of this.contentPanel. if the user
236 			 * is currently at an outside link, this.contentPanel.onExit
237 			 * will throw an exception because we aren't permitted
238 			 * to access attributes of pages outside the context of our
239 			 * server.
240 			 */
241 			if(this.contentPanel.onExit) {
242 				try {
243 					//run the on exit cleanup
244 					this.contentPanel.onExit();					
245 				} catch(err) {
246 					//error when onExit() was called, e.g. mysystem editor undefined
247 				}
248 			}	
249 		} catch(err) {
250 			/*
251 			 * an exception was thrown because this.contentPanel is an
252 			 * outside link. we will need to go back in the history
253 			 * and then trying to render the original node.
254 			 */
255 			history.back();
256 		}
257 	}
258 };

261 /**
262  * Renders the student work into the div. The grading tool will pass in a
263  * div id to this function and this function will insert the student data
264  * into the div.
265  * 
266  * @param divId the id of the div we will render the student work into
267  * @param nodeVisit the student work
268  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
269  * div ids use this to prevent DOM conflicts such as when the show all work div
270  * uses the same ids as the show flagged work div
271  * @param workgroupId the id of the workgroup this work belongs to
272  * 
273  * TODO: rename CarGraphNode
274  * Note: you may need to add code to this function if the student
275  * data for your step is complex or requires additional processing.
276  * look at carGraphNode.renderGradingView() as an example of a step that
277  * requires additional processing
278  */
279 CarGraphNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
280 	//create a carGraph object that we will use to perform all the graphing logic for us
281 	var carGraph = new CARGRAPH(this, this.view);
282 	
283 	if(childDivIdPrefix == null) {
284 		//the default child div id prefix will be "" if none is provided
285 		childDivIdPrefix = "";
286 	}
287 	
288 	//get the step work id from the node visit
289 	var stepWorkId = nodeVisit.id;
290 	
291 	/*
292 	 * get the student work, in this case the student work is
293 	 * the carGraph state
294 	 */
295 	var carGraphState = nodeVisit.getLatestWork();
296 	
297 	//set the carGraph state into our carGraph object
298 	carGraph.carGraphState = carGraphState;
299 		
300 	//get the graph parameters from the content
301 	var graphParams = carGraph.parseGraphParams(this.content.getContentJSON().graphParams);
302 
303 	//create the carGraph graph div that we will use to display the graph
304 	var carGraphGraphDiv = createElement(document, 'div', {id: childDivIdPrefix + 'carGraphGraphDiv_' + stepWorkId, style:'width:400px;height:200px;'});
305 	
306 	//create the div that will display the check boxes to filter the lines (if this graph has multiple lines, if not, this will be empty)
307 	var carGraphGraphCheckBoxesDiv = createElement(document, 'div', {id: childDivIdPrefix + 'carGraphGraphCheckBoxesDiv_' + stepWorkId});
308 	
309 	//create the div that will display the student annotations for the graph
310 	var carGraphAnnotationsDiv = createElement(document, 'div', {id: childDivIdPrefix + 'carGraphAnnotationsDiv_' + stepWorkId});
311 	
312 	//create the response div that we will use to display what the student typed
313 	var carGraphResponseDiv = createElement(document, 'div', {id: childDivIdPrefix + 'carGraphResponseDiv_' + stepWorkId});
314 	
315 	//add all the divs to the main work div 
316 	$('#' + divId).append(carGraphGraphDiv);
317 	$('#' + divId).append(carGraphGraphCheckBoxesDiv);
318 	$('#' + divId).append(carGraphAnnotationsDiv);
319 	$('#' + divId).append(carGraphResponseDiv);
320 	
321 	// run smart filter and flag if the smart filter returns true
322 	var smartFilterResult = this.smartFilter(carGraphState); // obj
323 	if (smartFilterResult.error !== true) {
324 		var carGraphSmartFilterDiv = createElement(document, 'div', {id: "smartFilter_" + stepWorkId, "class":"smartFilter"});
325 		carGraphSmartFilterDiv.innerHTML += "<span class='maxError' style='color:red; display:none'>" + smartFilterResult.maxError + "</span>";
326 		carGraphSmartFilterDiv.innerHTML += "<span class='avgError' style='color:green; display:none'>" + smartFilterResult.avgError + "</span>";
327 		carGraphSmartFilterDiv.innerHTML += "<span class='errMargin' style='color:yellow; display:none'>" + smartFilterResult.errMargin + "</span>";
328 		
329 		//get the max error possible
330 		var maxErrorPossible = 22;
331 		
332 		//get the max auto graded score
333 		var maxAutoGradedScore = 5;
334 		
335 		//calculate the auto graded score for the student
336 		var score = Math.ceil(((maxErrorPossible - smartFilterResult.maxError) / maxErrorPossible) * maxAutoGradedScore);
337 		
338 		//display the auto graded score
339 		carGraphSmartFilterDiv.innerHTML += "Auto-Graded Score: " + score + "/" + maxAutoGradedScore;
340 	}
341 	else {
342 		// some error handling???
343 	}
344 		
345 	$('#' + divId).append(carGraphSmartFilterDiv);
346 	
347 	//tell the car graph to show the correct graph along with the student graph
348 	carGraph.showCorrectGraph = true;
349 	
350 	//plot the graph in the carGraph graph div
351 	carGraph.plotData(carGraphGraphDiv.id, carGraphGraphCheckBoxesDiv.id);
352 	
353 	/*
354 	 * used to hide or show the annotation tool tips. if the teacher has
355 	 * their mouse in the graph div we will hide the annotation tool tips
356 	 * so that they don't block them from viewing the plot points.
357 	 * when the mouse cursor is outside of the graph div we will show the
358 	 * annotation tool tips for them to view.
359 	 */
360 	$("#" + carGraphGraphDiv.id).bind('mouseover', (function(event) {
361 		$(".activeAnnotationToolTip").hide();
362 	}));
363 	$("#" + carGraphGraphDiv.id).bind('mouseleave', (function(event) {
364 		$(".activeAnnotationToolTip").show();
365 	}));
366 	
367 	//get the annotations as a string
368 	var annotationsHtml = carGraphState.getAnnotationsHtml();
369 	
370 	//set the annotations text
371 	$('#' + carGraphAnnotationsDiv.id).html(annotationsHtml);
372 	
373 	//get the student response that was typed
374 	var response = carGraphState.response;
375 	
376 	//replace \n with <br> so that the line breaks are displayed for the teacher
377 	response = this.view.replaceSlashNWithBR(response);
378 	
379 	//insert the response the student typed
380 	$('#' + carGraphResponseDiv.id).html(response);
381 };

383 /**
384  * Handles Smart Filtering in the gradingtool for CarGraphNode.
385  * @return
386  */
387 CarGraphNode.prototype.showSmartFilter = function(doShow) {
388 	var doFilter = function(sliderValue) {
389 		$(".maxError").each(function() {
390 			var maxErrorForOneDrawing = parseFloat($(this).html());
391 			if (maxErrorForOneDrawing < sliderValue) {
392 				$(this).parents(".studentWorkRow").hide();
393 				$(this).parents(".studentWorkRow").removeClass("smartFilterShow");
394 				$(this).parents(".studentWorkRow").addClass("smartFilterHide");
395 			} else {
396 				$(this).parents(".studentWorkRow").show();
397 				$(this).parents(".studentWorkRow").removeClass("smartFilterHide");
398 				$(this).parents(".studentWorkRow").addClass("smartFilterShow");
399 			}
400 		});
401 	};	
402 	
403 	if (doShow) {
404 		if ($("#onlyShowSmartFilteredItemsSlider").length == 0) {
405 			var maxValue = 22;
406 			var defaultValue = 11;
407 			doFilter(defaultValue);
408 			$("#onlyShowSmartFilteredItemsText").append("<div style='width:80%' id='smartFilter'><div style='color:green'><span><- SLIDE LEFT: More errors</span><span style='margin-left:150px'>Showing <span id='currentDisplayedWorkCount'></span> work with highest margin of error more than <span id='currentSliderValue'>"+defaultValue+"</span></span><span style='float:right'>SLIDE RIGHT: Less errors -></span></div><div id='onlyShowSmartFilteredItemsSlider'></div></div>");
409 			$("#onlyShowSmartFilteredItemsSlider").slider({
410 				value:defaultValue,
411 				min:0,
412 				max:maxValue,
413 				slide: function(event, ui) {
414 				var sliderValue = maxValue-ui.value;
415 				// hide anything that is under the threshold (slider value)
416 				doFilter(sliderValue);
417 				
418 				$("#currentSliderValue").html(sliderValue);
419 				if(typeof eventManager != 'undefined'){
420 					// call filterstudentrows again to filter for selected periods
421 					eventManager.fire('filterStudentRows');
422 				}
423 			}
424 			});
425 		};
426 		$(".studentWorkRow.noWork").hide();  // hide rows with no work
427 		$("#smartFilter").show();
428 	} else {
429 		$(".studentWorkRow.noWork").filter(":visible").show();  // show rows with no work that are visible
430 		$("#smartFilter").hide();
431 	}
432 	return true;
433 };
