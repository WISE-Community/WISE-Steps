111 /**
112  * This function is called when the student exits the step. It is mostly
113  * used for error checking.
114  * 
115  * xTODO: rename TemplateNode
116  * 
117  * Note: In most cases you will not have to change anything here.
118  */
119 ExplanationBuilderNode.prototype.onExit = function() {
120 	//check if the content panel has been set
121 	if(this.contentPanel) {
122 		try {
123 			/*
124 			 * check if the onExit function has been implemented or if we
125 			 * can access attributes of this.contentPanel. if the user
126 			 * is currently at an outside link, this.contentPanel.onExit
127 			 * will throw an exception because we aren't permitted
128 			 * to access attributes of pages outside the context of our
129 			 * server.
130 			 */
131 			if(this.contentPanel.save) {
132 				try {
133 					//run the on exit cleanup
134 					this.contentPanel.save();					
135 				} catch(err) {
136 					//error when onExit() was called, e.g. mysystem editor undefined
137 				}
138 			}	
139 		} catch(err) {
140 			/*
141 			 * an exception was thrown because this.contentPanel is an
142 			 * outside link. we will need to go back in the history
143 			 * and then trying to render the original node.
144 			 */
145 			history.back();
146 		}
147 	}
148 };

150 /**
151  * Renders the student work into the div. The grading tool will pass in a
152  * div id to this function and this function will insert the student data
153  * into the div.
154  * 
155  * @param divId the id of the div we will render the student work into
156  * @param nodeVisit the student work
157  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
158  * div ids use this to prevent DOM conflicts such as when the show all work div
159  * uses the same ids as the show flagged work div
160  * 
161  * xTODO: rename TemplateNode
162  * Note: you may need to add code to this function if the student
163  * data for your step is complex or requires additional processing.
164  * look at SensorNode.renderGradingView() as an example of a step that
165  * requires additional processing
166  */
167 ExplanationBuilderNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
168 	//get the step work id
169 	var stepWorkId = nodeVisit.id;
170 	
171 	//create an ExplanationBuilder object so that we can retrieve the prompt an background
172 	var explanationBuilder = new ExplanationBuilder(this, this.view);
173 	explanationBuilder.prompt = explanationBuilder.content.prompt;
174 	explanationBuilder.background = explanationBuilder.content.background;
175 	
176 	var backgroundPath = null;
177 	
178 	if(explanationBuilder.background != null && explanationBuilder.background != "") {
179 		if(explanationBuilder.background.indexOf('http') != -1) {
180 			//background image path is absolute
181 			backgroundPath = explanationBuilder.background;
182 		} else {
183 			//background image is relative so we need to create the full path to the background image
184 			backgroundPath = this.view.getConfig().getConfigParam('getContentBaseUrl') + explanationBuilder.background;			
185 		}
186 	}
187 	
188 	//create the div that will contain the ideas
189 	var explanationBuilderIdeasDivId = childDivIdPrefix + 'explanationBuilderIdeasDiv_' + stepWorkId;
190 	var explanationBuilderIdeasDiv = createElement(document, 'div', {id: explanationBuilderIdeasDivId, style:'width:485px;height:315px;border: 1px solid;position:relative'});
191 	
192 	//add the explanationBuilderIdeasDiv to the grading div
193 	$('#' + divId).append(explanationBuilderIdeasDiv);
194 	
195 	if(backgroundPath != null) {
196 		//set the background attributes
197 		$('#' + explanationBuilderIdeasDivId).css('background-image','url(' + backgroundPath + ')');
198 		$('#' + explanationBuilderIdeasDivId).css('background-repeat','no-repeat');
199 		$('#' + explanationBuilderIdeasDivId).css('background-position','left top');		
200 	}
201 	
202 	//get the idea basket for this student
203 	var ideaBasket = this.view.getIdeaBasketByWorkgroupId(workgroupId);
204 	
205 	/*
206 	 * Get the latest student state object for this step
207 	 * xTODO: rename templateState to reflect your new step type
208 	 * 
209 	 * e.g. if you are creating a quiz step you would change it to quizState
210 	 */
211 	var explanationBuilderState = nodeVisit.getLatestWork();
212 	
213 	//get the explanation ideas the student used
214 	var explanationIdeas = explanationBuilderState.explanationIdeas;
215 	
216 	//get the student text answer
217 	var answer = explanationBuilderState.answer;
218 	
219 	//loop through all the explanation ideas
220 	for(var x=0; x<explanationIdeas.length; x++) {
221 		//get an explanation idea
222 		var explanationIdea = explanationIdeas[x];
223 		
224 		//get the attributes of the explanation idea
225 		var id = explanationIdea.id;
226 		var left = explanationIdea.xpos;
227 		var top  = explanationIdea.ypos;
228 		var currColor = explanationIdea.color;
229 		
230 		var text = "";
231 		
232 		if(ideaBasket != null) {
233 			//get the idea from the basket
234 			var idea = ideaBasket.getIdeaById(id);
235 			
236 			if(idea != null) {
237 				//get the text for the idea
238 				text = idea.text;
239 			}
240 		}
241 		
242 		//create a div for the idea that will be displayed as a rectangle
243 		var explanationIdeaHtml = '<div class="exIdea" class="selected" title="Click and drag to move; Click to change color" id="' + childDivIdPrefix + 'explanationIdea' 
244 			+ id + '_' + stepWorkId + '" style="position:absolute; left:' + left + 'px; top:' + top + 'px; background-color:' + currColor + '">' + text + '</div>';
245 		
246 		//add the idea div to the explanationBuilderIdeasDiv
247 		$('#' + explanationBuilderIdeasDivId).append(explanationIdeaHtml);
248 	}
249 	
250 	//create a div to display the student answer
251 	var explanationBuilderAnswerDiv = childDivIdPrefix + 'explanationBuilderAnswerDiv_' + stepWorkId;
252 	var answerDiv = createElement(document, 'div', {id: explanationBuilderAnswerDiv});
253 	$('#' + divId).append(answerDiv);
254 
255 	//replacen \n with <br>
256 	answer = this.view.replaceSlashNWithBR(answer);
257 	
258 	/*
259 	 * add a <br> before the answer so there will be a new 
260 	 * line between the ideas and this text answer
261 	 */
262 	answer = "<br>" + answer;
263 	
264 	//set the answer in the div
265 	$('#' + explanationBuilderAnswerDiv).html(answer);
266 };

288 /**
289  * Get the prompt for this step
290  * @return the prompt for this step as a string
291  */
292 ExplanationBuilderNode.prototype.getPrompt = function() {
293 	var prompt = "";
294 	
295 	if(this.content != null) {
296 		//get the content for the node
297 		var contentJSON = this.content.getContentJSON();
298 
299 		//see if the node content has a prompt
300 		if(contentJSON != null && contentJSON.prompt != null) {
301 			prompt = contentJSON.prompt;	
302 		}
303 	}
304 	
305 	//return the prompt
306 	return prompt;
307 };

309 /**
310  * Called when the 'ideaBasketChanged' is fired
311  * @param type
312  * @param args
313  * @param obj this node (since this context is not actually in this node
314  * because this function is called from an event fire)
315  */
316 ExplanationBuilderNode.prototype.ideaBasketChanged = function(type,args,obj) {
317 	//get this node
318 	var thisNode = obj;
319 	
320 	/*
321 	 * make sure this node is the current node the student is on because
322 	 * all ExplanationBuilderNode steps will be listening for the event
323 	 * and we only need to update the basket if the student is on 
324 	 * an ExplanationBuilderNode step.
325 	 */
326 	if(thisNode.view.getCurrentNode().id == thisNode.id) {
327 		//update the idea basket within the step
328 		thisNode.contentPanel.explanationBuilder.ideaBasketChanged(thisNode.view.ideaBasket);
329 	}
330 };
