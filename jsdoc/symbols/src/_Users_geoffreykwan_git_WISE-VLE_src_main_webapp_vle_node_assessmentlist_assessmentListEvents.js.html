  2 /**
  3  * The assessmentlist dispatcher catches events specific to authoring individual
  4  * assessmentlist steps and delegates them to the appropriate functions for
  5  * this view.
  6  * @function
  7  */
  8 View.prototype.assessmentlistDispatcher = function(type,args,obj){
  9 	if(type=='assessmentlistPromptChanged'){
 10 		obj.AssessmentListNode.updatePrompt();
 11 	} else if(type=='assessmentlistFieldUpdated'){
 12 		obj.AssessmentListNode.fieldUpdated(args[0], args[1]);
 13 	} else if(type=='assessmentlistRadioItemFieldUpdated'){
 14 		obj.AssessmentListNode.radioItemFieldUpdated(args[0], args[1], args[2]);
 15 	} else if(type=='assessmentlistAddChoice'){
 16 		obj.AssessmentListNode.addChoice(args[0]);
 17 	} else if(type=='assessmentlistRadioItemRemoveChoice'){
 18 		obj.AssessmentListNode.radioItemRemoveChoice(args[0],args[1]);
 19 	} else if(type=='assessmentlistAddNewItem'){
 20 		obj.AssessmentListNode.addNewItem(args[0]);
 21 	} else if(type=='assessmentlistRemoveItem'){
 22 		obj.AssessmentListNode.removeItem(args[0]);
 23 	} else if(type=='assessmentlistStarterSentenceUpdated'){
 24 		obj.AssessmentListNode.starterSentenceChanged(args[0]);
 25 	} else if(type=='assessmentlistStarterOptionChanged'){
 26 		obj.AssessmentListNode.starterOptionChanged(args[0]);
 27 	} else if(type=='assessmentlistUpdateRichText'){
 28 		obj.AssessmentListNode.updateRichText(args[0]);
 29 	} else if(type=='assessmentOptionChanged'){
 30 		obj.AssessmentListNode.optionChanged(args[0]);
 31 	} else if(type=='assessmentListCorrectChoiceChanged'){
 32 		obj.AssessmentListNode.correctChoiceChanged(args[0], args[1]);
 33 	} else if(type=='assessmentListIsAutoScoringEnabledChanged'){
 34 		obj.AssessmentListNode.isAutoScoringEnabledChanged(args[0]);
 35 	} else if(type=='assessmentListUpdateImportantReviewSequencePart'){
 36 		obj.AssessmentListNode.updateImportantReviewSequencePart(args[0]);
 37 	} else if(type=='assessmentListPeerReviewPercentageTriggerUpdated'){
 38 		obj.AssessmentListNode.peerReviewPercentageTriggerUpdated();
 39 	} else if(type=='assessmentListPeerReviewNumberTriggerUpdated'){
 40 		obj.AssessmentListNode.peerReviewNumberTriggerUpdated();
 41 	} else if(type=='assessmentListPeerReviewAuthoredWorkUpdated'){
 42 		obj.AssessmentListNode.peerReviewAuthoredWorkUpdated();
 43 	} else if(type=='assessmentListPeerReviewStepNotOpenCustomMessageUpdated'){
 44 		obj.AssessmentListNode.peerReviewStepNotOpenCustomMessageUpdated();
 45 	}
 46 };
 47 
 48 //this list of events
 49 var events = [
 50 	'assessmentlistPromptChanged',
 51 	'assessmentlistFieldUpdated',
 52 	'assessmentlistRadioItemFieldUpdated',
 53 	'assessmentlistAddNewItem',
 54 	'assessmentlistRemoveItem',
 55 	'assessmentlistAddChoice',
 56 	'assessmentlistRadioItemRemoveChoice',
 57 	'assessmentlistUpdateRichText',
 58 	'assessmentlistStarterOptionChanged',
 59 	'assessmentlistStarterSentenceUpdated',
 60 	'assessmentOptionChanged',
 61 	'assessmentListCorrectChoiceChanged',
 62 	'assessmentListIsAutoScoringEnabledChanged',
 63 	'assessmentListUpdateImportantReviewSequencePart',
 64 	'assessmentListPeerReviewPercentageTriggerUpdated',
 65 	'assessmentListPeerReviewNumberTriggerUpdated',
 66 	'assessmentListPeerReviewAuthoredWorkUpdated',
 67 	'assessmentListPeerReviewStepNotOpenCustomMessageUpdated'
 68 ];
 69 
 70 /*
 71  * add all the events to the vle so the vle will listen for these events
 72  * and call the dispatcher function when the event is fired
 73  */
 74 for(var x=0; x<events.length; x++) {
 75 	componentloader.addEvent(events[x], 'assessmentlistDispatcher');
 76 };
 77 
 78 //used to notify scriptloader that this script has finished loading
 79 if(typeof eventManager != 'undefined'){
 80 	eventManager.fire('scriptLoaded', 'vle/node/assessmentlist/assessmentListEvents.js');
 81 };
