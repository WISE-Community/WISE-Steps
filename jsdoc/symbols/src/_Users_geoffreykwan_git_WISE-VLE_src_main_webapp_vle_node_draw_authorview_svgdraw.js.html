  9 /**
 10  * Generates the authoring page for svg draw node types
 11  */
 12 View.prototype.SVGDrawNode.generatePage = function(view){
 13 	this.view = view;
 14 	this.content = this.view.activeContent.getContentJSON();
 15 	
 16 	this.maxSnaps = 10; // default max allowed snapshots
 17 	
 18 	var parent = document.getElementById('dynamicParent');
 19 	
 20 	/* wipe out old */
 21 	parent.removeChild(document.getElementById('dynamicPage'));
 22 	
 23 	/* create new */
 24 	var pageDiv = createElement(document, 'div', {id:'dynamicPage', style:'width:100%;height:100%'});
 25 	var optDiv = createElement(document, 'div', {id: 'optionsDiv'});
 26 	var toolbarOptionsDiv = createElement(document, 'div', {id: 'toolbarOptionsDiv'});
 27 	var snapshotOptionDiv = createElement(document, 'div', {id: 'snapshotOptionDiv'});
 28 	var snapMaxDiv = createElement(document, 'div', {id: 'snapMaxDiv'});
 29 	var descriptionOptionDiv = createElement(document, 'div', {id: 'descriptionOptionDiv'});
 30 	var backgroundDiv = createElement(document, 'div', {id: 'backgroundDiv'});
 31 	var stampsDiv = createElement(document, 'div', {id:'stampsDiv'});
 32 	
 33 	parent.appendChild(pageDiv);
 34 	pageDiv.appendChild(toolbarOptionsDiv);
 35 	pageDiv.appendChild(snapshotOptionDiv);
 36 	pageDiv.appendChild(snapMaxDiv);
 37 	pageDiv.appendChild(createBreak());
 38 	pageDiv.appendChild(descriptionOptionDiv);
 39 	pageDiv.appendChild(createBreak());
 40 	pageDiv.appendChild(document.createTextNode('Enter instructions for students (optional):'));
 41 	pageDiv.appendChild(createBreak());
 42 	pageDiv.appendChild(createElement(document, 'div', {id: 'promptContainer'}));
 43 	pageDiv.appendChild(createBreak());
 44 	pageDiv.appendChild(backgroundDiv);
 45 	pageDiv.appendChild(createBreak());
 46 	pageDiv.appendChild(stampsDiv);
 47 	pageDiv.appendChild(createBreak());
 48 	
 49 	this.generateToolbarOptions();
 50 	this.generateSnapshotOption();
 51 	this.generateDescriptionOption();
 52 	this.generateBackground();
 53 	this.generateStamps();
 54 };

241 /**
242  * Generates the stamps element for this svg draw node
243  */
244 View.prototype.SVGDrawNode.generateStamps = function(){
245 	var parent = document.getElementById('stampsDiv');
246 	
247 	/* wipe out old */
248 	while(parent.firstChild){
249 		parent.removeChild(parent.firstChild);
250 	};
251 	
252 	/* create new */
253 	var addStampButt = createElement(document, 'input', {type:'button', id:'addStampButton', value:'Add New Stamp', onclick:'eventManager.fire("svgdrawAddNewStamp")'});
254 	parent.appendChild(addStampButt);
255 	parent.appendChild(createBreak());
256 	
257 	/* generate stamp elements for each stamp that is specified in the content */
258 	for(var o=0;o<this.content.stamps.length;o++){
259 		var sText = document.createTextNode('# ' + (o + 1) + ' ');
260 		var sLabelText = document.createTextNode('Title: ');
261 		var sValueText = document.createTextNode('URI: ');
262 		var sWidthText = document.createTextNode('Width: ');
263 		var sHeightText = document.createTextNode('Height: ');
264 		var sLabelInput = createElement(document, 'input', {type:'text', size:'10', id:'stampLabelInput_' + o, value:this.content.stamps[o].title, onchange:'eventManager.fire("svgdrawStampLabelChanged","' + o + '")', onclick:'eventManager.fire("svgdrawStampTitleClicked","' + o + '")'});
265 		var sInput = createElement(document, 'input', {type:'text', size:'15', id:'stampInput_' + o, value:this.content.stamps[o].uri, onchange:'eventManager.fire("svgdrawStampValueChanged","' + o + '")'});
266 		var sWidthInput = createElement(document, 'input', {type:'text', size:'5', id:'stampWidthInput_' + o, value:this.content.stamps[o].width, onchange:'eventManager.fire("svgdrawStampWidthChanged","' + o + '")', onclick:'eventManager.fire("svgdrawStampWidthClicked","' + o + '")'});
267 		var sHeightInput = createElement(document, 'input', {type:'text', size:'5', id:'stampHeightInput_' + o, value:this.content.stamps[o].height, onchange:'eventManager.fire("svgdrawStampHeightChanged","' + o + '")', onclick:'eventManager.fire("svgdrawStampHeightClicked","' + o + '")'});
268 		var removeButt = createElement(document, 'input', {type:'button', id:'removeButt_' + o, value:'remove stamp', onclick:'eventManager.fire("svgdrawRemoveStamp","' + o + '")'});
269 		parent.appendChild(sText);
270 		parent.appendChild(sLabelText);
271 		parent.appendChild(sLabelInput);
272 		parent.appendChild(createSpace());
273 		parent.appendChild(sValueText);
274 		parent.appendChild(sInput);
275 		parent.appendChild(createSpace());
276 		parent.appendChild(sWidthText);
277 		parent.appendChild(sWidthInput);
278 		parent.appendChild(createSpace());
279 		parent.appendChild(sHeightText);
280 		parent.appendChild(sHeightInput);
281 		parent.appendChild(createSpace());
282 		parent.appendChild(removeButt);
283 		parent.appendChild(createBreak());
284 	};
285 };

287 /**
288  * Updates the toolbar_options value of the content to the user specified value
289  * and refreshes the preview.
290  */
291 View.prototype.SVGDrawNode.toolbarOptionsChanged = function(){
292 	var options = document.getElementsByName('toolbarCbx');
293 	
294 	// set initial toolbar_options variable if not yet defined
295 	if(this.content.toolbar_options == null || typeof this.content.toolbar_options == 'undefined'){
296 		this.content.toolbar_options = {};
297 	}
298 	
299 	for(var i=0;i<options.length;i++){
300 		var current = $(options[i]).attr('id').replace('Cbx','');
301 		var isActive = false;
302 		if(options[i].checked){
303 			isActive = true;
304 		}
305 		
306 		if(current == 'pencil'){
307 			this.content.toolbar_options.pencil = isActive;
308 			if(isActive){
309 				if(this.content.snapshots_max > 10){
310 					this.content.snapshots_max = 10;
311 					document.getElementById('snapMaxInput').options[8].selected = true;
312 				}
313 				$('#snapMaxInput option.noPencil').attr('disabled', 'disabled');
314 			} else {
315 				$('#snapMaxInput option.noPencil').removeAttr('disabled');
316 			}
317 		} else if (current == 'line'){
318 			this.content.toolbar_options.line = isActive;
319 		} else if (current == 'connector'){
320 			this.content.toolbar_options.connector = isActive;
321 		} else if (current == 'rectangle'){
322 			this.content.toolbar_options.rectangle = isActive;
323 		} else if (current == 'ellipse'){
324 			this.content.toolbar_options.ellipse = isActive;
325 		} else if (current == 'polygon'){
326 			this.content.toolbar_options.polygon = isActive;
327 		} else if (current == 'text'){
328 			this.content.toolbar_options.text = isActive;
329 		}
330 	}
331 	
332 	/* fire source updated event */
333 	this.view.eventManager.fire('sourceUpdated');
334 };
