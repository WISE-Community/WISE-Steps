 45 // Set applet starting parameters
 46 // waits until applet is active to send starting parameters
 47 MW.prototype.setParams = function(context,data,count){
 48     if (!context.applet.isActive() && count > 0) {
 49        setTimeout( function() { context.setParams( context,data,--count ); }, 2000 );
 50     }
 51     else if (context.applet.isActive()) {
 52     	if (parent.document.getElementById("ifrm") != null) {
 53     		$('#mw_applet').attr('height',parent.document.getElementById("ifrm").offsetHeight-20); // set applet height to fit in iframe
 54 	    	$('#mw_applet').attr('width',parent.document.getElementById("ifrm").offsetWidth-40);
 55     	}
 56     	
 57 		// set starting temp
 58 		if(data.params.temperature){
 59 			context.runScript("mw2d:1:set temperature " + data.params.temperature,context);
 60 		}
 61 		// set autoStart
 62 		if(data.params.auto_start){
 63 			context.runScript("mw2d:1:run",context);
 64 		}
 65     }
 66     else {
 67        alert( 'applet failed to load' ); // applet did not start within 40 seconds
 68     }
 69 };
 70 
 71 // Javascript-MW applet communication
 72 MW.prototype.runScript = function(script,context){
 73 	return context.applet.runMwScript(script);
 74 };
