1 /**
  2  * Object for storing the student's response to the brainstorm
  3  * @constructor
  4  */
  5 function BRAINSTORMSTATE(response, timestamp){
  6 	this.type = "bs";
  7 	this.response = response;
  8 	if(arguments.length == 1) {
  9 		this.timestamp = Date.parse(new Date());
 10 	} else {
 11 		this.timestamp = timestamp;
 12 	};
 13 };
 14 
 15 BRAINSTORMSTATE.prototype.getHtml = function() {
 16 	return "timestamp: " + this.timestamp + "<br/>response: " + this.response;
 17 };
 18 
 19 BRAINSTORMSTATE.prototype.getDataXML = function() {
 20 	return "<response>" + this.response + "</response><timestamp>" + this.timestamp + "</timestamp>";
 21 };
 22 
 23 BRAINSTORMSTATE.prototype.parseDataXML = function(stateXML) {
 24 	var reponse = stateXML.getElementsByTagName("response")[0];
 25 	var timestamp = stateXML.getElementsByTagName("timestamp")[0];
 26 	
 27 	if(reponse == undefined || timestamp == undefined) {
 28 		return null;
 29 	} else {
 30 		return new BRAINSTORMSTATE(reponse.textContent, timestamp.textContent);		
 31 	}
 32 };
 33 
 34 /**
 35  * Takes in a state JSON object and returns an BRAINSTORMSTATE object
 36  * @param stateJSONObj a state JSON object
 37  * @return a BRAINSTORMSTATE object
 38  */
 39 BRAINSTORMSTATE.prototype.parseDataJSONObj = function(stateJSONObj) {
 40 	//create a new BRAINSTORMSTATE object
 41 	var brainState = new BRAINSTORMSTATE();
 42 	
 43 	//set the attributes of the BRAINSTORMSTATE object
 44 	brainState.response = stateJSONObj.response;
 45 	brainState.timestamp = stateJSONObj.timestamp;
 46 	
 47 	//return the BRAINSTORMSTATE object
 48 	return brainState;
 49 }
 50 
 51 BRAINSTORMSTATE.prototype.getStudentWork = function() {
 52 	return this.response;
 53 }
