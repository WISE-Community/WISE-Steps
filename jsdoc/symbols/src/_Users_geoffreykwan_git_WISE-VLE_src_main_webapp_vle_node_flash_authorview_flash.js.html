233 /**
234  * Updates the content's prompt to match that of what the user input
235  */
236 View.prototype.FlashNode.updatePrompt = function(){
237 	/* update content */
238 	if(this.richTextPromptEnabled){
239 		this.content.prompt = this.promptRichText.getContent();
240 	} else {
241 		this.content.prompt = $('#promptTextArea').val();
242 	}
243 	
244 	// strip out any urls with the full project path (and replace with 'assets/file.jpg')
245 	var assetPath = this.projectPath + 'assets/';
246 	var assetPathExp = new RegExp(assetPath,"gi");
247 	this.content.prompt.replace(assetPathExp,"assets/");
248 	
249 	/*
250 	 * fire source updated event, this will update the preview
251 	 */
252 	this.view.eventManager.fire('sourceUpdated');
253 };

314 /**
315  * Populate the advanced options (student data logging, grading)
316  */
317 View.prototype.FlashNode.populateAdvancedOptions = function() {
318 	var context = this;
319 	// get the enable data logging setting from the content and check/uncheck checkbox
320 	$('#enableDataCbx').prop("checked", this.content.enableData);
321 	
322 	// get the enable grading setting from the content and check/uncheck checkbox
323 	$('#enableGradingCbx').prop("checked", this.content.enableGrading);
324 	
325 	// get the grading type setting from the content and select corresponding radio button
326 	$('input[name="gradingType"]').each(function(){
327 		if($(this).val() == context.content.gradingType){
328 			$(this).prop('checked',true);
329 		}
330 	});
331 	
332 	// if data logging is enabled, show grading checkbox
333 	if(this.content.enableData){
334 		$('#gradingDiv').show();
335 		// if grading is enabled, show grading type selectors
336 		if(this.content.enableGrading){
337 			$('#gradingTypeDiv').show();
338 		}
339 	}
340 	// populate any previously stored flashvars
341 	$.each(this.content.flashvars, function(key,value){
342 		context.createNewFlashvarInput(key,value);
343 	});
344 };

405 /**
406  * Add a new flashvar input section (key and value textareas)
407  * @param key The string to use for the key (optional)
408  * @param value The string to use for the value (optional)
409  */
410 View.prototype.FlashNode.createNewFlashvarInput = function(key,value){
411 	var context = this;
412 	this.flashvarCount ++;
413 	// create div for new flashvar
414 	var newFlashvarDiv = $(document.createElement('div')).addClass('flashvar').attr('id','flashvar_' + context.flashvarCount).css('display','none');
415 	// create label for key textarea
416 	var keyLabel = $(document.createElement('span')).text('Key:');
417 	// create the textarea that the author will write the key in
418 	var keyInput = $(createElement(document, 'input', {type:'text', size:'15'})).addClass('key flashvarInput');
419 	// populate key input if key param was sent
420 	if(key && key.replace(/\s/g,"")!=""){
421 		keyInput.val(key);
422 	}
423 	// create label for value textarea
424 	var valueLabel = $(document.createElement('span')).text('Value:');
425 	// create the textarea that the author will write the value in
426 	var valueInput = $(createElement(document, 'input', {type:'text', size:'50'})).addClass('value flashvarInput');
427 	// populate value input if value param was sent
428 	if(value && value.replace(/\s/g,"")!=""){
429 		valueInput.val(value);
430 	}
431 	
432 	// create delete link
433 	var deleteLink = $(document.createElement('a')).text('X').attr('title','Delete').addClass('deleteFlashvar');
434 	var index = context.flashvarCount;
435 	deleteLink.click(function(){
436 		eventManager.fire('flashDeleteFlashvar',index);
437 	});
438 	
439 	newFlashvarDiv.append(keyLabel).append(keyInput).append(valueLabel).append(valueInput).append(deleteLink);
440 	
441 	$('#addFlashvar').before(newFlashvarDiv);
442 	newFlashvarDiv.slideDown('fast');
443 };

445 /**
446  * Updates the flashvar key/value pairs
447  */
448 View.prototype.FlashNode.updateFlashvars = function(){
449 	var context = this;
450 	this.content.flashvars = {};
451 	$('.flashvar').each(function(index){
452 		var key = $('#flashvar_' + index + ' .key').val();
453 		var value = $('#flashvar_' + index + ' .value').val();
454 		if(key.replace(/\s/g,"")!=""){
455 			context.content.flashvars[key] = value;
456 		}
457 	});
458 	
459 	/*
460 	 * fire source updated event, this will update the preview
461 	 */
462 	this.view.eventManager.fire('sourceUpdated');
463 };

578 	var callback = function(field_name, url, type, win){

580 		win.document.getElementById(field_name).value = url;
