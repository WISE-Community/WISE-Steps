1 /**
  2  * @constructor
  3  * @param node
  4  * @param view
  5  * @returns
  6  */
  7 function OPENRESPONSE(node, view) {
  8 	this.node = node;
  9 	this.view = view;
 10 	this.content = node.getContent().getContentJSON();
 11 	
 12 	if(node.studentWork != null) {
 13 		this.states = node.studentWork; 
 14 	} else {
 15 		this.states = [];  
 16 	};
 17 	
 18 	//check if there is an associated node whose work we might display in this step
 19 	if(this.node.associatedStartNode != null) {
 20 		//get the node id for the associated node that the student will be reviewing work for
 21 		this.associatedStartNodeId = this.node.associatedStartNode;
 22 		
 23 		//get the associated node object
 24 		this.associatedStartNode = this.view.getProject().getNodeById(this.associatedStartNodeId);
 25 		
 26 		if(this.associatedStartNode != null) {
 27 			//get the content for the associated node
 28 			this.associatedStartNodeContent = this.associatedStartNode.getContent().getContentJSON();			
 29 		}
 30 	}
 31 	
 32 	//check if there is an associated node whose work we might display in this step
 33 	if(this.node.associatedAnnotateNode != null) {
 34 		//get the node id for the associated node that the student will be reviewing work for
 35 		this.associatedAnnotateNodeId = this.node.associatedAnnotateNode;
 36 		
 37 		//get the associated node object
 38 		this.associatedAnnotateNode = this.view.getProject().getNodeById(this.associatedAnnotateNodeId);
 39 		
 40 		if(this.associatedAnnotateNode != null) {
 41 			//get the content for the associated node
 42 			this.associatedAnnotateNodeContent = this.associatedAnnotateNode.getContent().getContentJSON();			
 43 		}
 44 	}
 45 	
 46 	//check if this step is being used for a peer review
 47 	if(this.node.peerReview != null) {
 48 		//values to be set after we retrieve the other student work that this student will be reviewing
 49 		this.otherStudentWorkgroupId = null;
 50 		this.otherStudentStepWorkId = null;
 51 		this.otherStudentNodeVisit = null;
 52 		this.showAuthorContent = false;
 53 
 54 		//this will store the peer review as an annotation
 55 		this.annotation = null;	
 56 		
 57 		if(this.node.peerReview == 'annotate') {
 58 			this.openPercentageTrigger = this.content.openPercentageTrigger;
 59 			this.openNumberTrigger = this.content.openNumberTrigger;
 60 			this.openLogicTrigger = this.content.openLogicTrigger;
 61 		}
 62 	}
 63 	
 64 	if(this.node.peerReview != null || this.node.teacherReview != null) {
 65 		//tell the node that it is part of a review sequence
 66 		this.node.setIsPartOfReviewSequence();
 67 		
 68 		//get the custom message to display to the student when the step is not open to work on
 69 		this.stepNotOpenCustomMessage = this.content.stepNotOpenCustomMessage;
 70 	}
 71 
 72 	//check if this step is locked
 73 	if(this.view != null && this.view.isLatestNodeStateLocked && this.view.isLatestNodeStateLocked(this.node.id)) {
 74 		//set this flag for future look up
 75 		this.locked = true;
 76 		
 77 		//tell the node that the student has completed it
 78 		this.node.setCompleted();
 79 	}
 80 	
 81 	/*
 82 	 * subscribe this open response to listen for the 'getAnnotationsComplete' event.
 83 	 * this is for when we need to retrieve annotations for the teacher review
 84 	 */
 85 	eventManager.subscribe('getAnnotationsComplete', this.getAnnotationsComplete, this);
 86 };

 88 /**
 89  * Check if this step needs to be locked by looking at the states
 90  * and seeing if any of them contain a state with the locked
 91  * attribute set to true which means the student has previously
 92  * saved and locked their answer.
 93  * @return whether the step is locked or not
 94  */
 95 OPENRESPONSE.prototype.isLocked = function() {
 96 	//loop through all the states
 97 	for(var x=0; x<this.states.length; x++) {
 98 		//get a state
 99 		var state = this.states[x];
100 		
101 		//check if the locked attribute is set to true
102 		if(state.locked) {
103 			return true;
104 		}
105 	}
106 	
107 	return false;
108 };

126 /**
127  * Saves current state of the OpenResponse item
128  * - what the students typed
129  * - timestamp
130  * 
131  * Then disable the textarea and save button and show the edit button
132  */
133 OPENRESPONSE.prototype.save = function(saveAndLock) {
134 	/*
135 	 * check if the save button is available. if it is available
136 	 * it means the student has modified the response. if it
137 	 * is not available, it means the student has not made any
138 	 * changes so we do not to do anything.
139 	 */
140 	if (this.isSaveAvailable() || this.isSaveAndLockAvailable()) {
141 		var response = "";
142 		
143 		/* set html to textarea if richtexteditor exists */
144 		response = this.getResponse();
145 		
146 		//check if the student changed their response
147 		if(this.isResponseChanged() || saveAndLock) {
148 			//response was changed so we will create a new state and save it
149 			var orState = new OPENRESPONSESTATE([response]);
150 
151 			if(saveAndLock) {
152 				//display a confirm message to make sure they want to submit and lock
153 				var lock = confirm("You will not be able to make further edits after submitting this response.  Ready to submit?");
154 				
155 				//check if they answered yes
156 				if(lock) {
157 					if(this.node.peerReview != null && this.node.peerReview == 'start') {
158 						/*
159 						 * set the submitPeerReview attribute in the state so the VLEPostData
160 						 * handles the peer review work correctly
161 						 */
162 						orState.submitForPeerReview = true;
163 					}
164 					
165 					//set the locked attribute in the state
166 					orState.locked = true;
167 					
168 					//disable the response box
169 					this.lockResponseBox();
170 					
171 					//disable the save and lock button
172 					this.setSaveAndLockUnavailable();
173 					
174 					//set the lock flag for future lookup
175 					this.locked = true;
176 					
177 					//tell the node that the student has completed the step
178 					this.node.setCompleted();
179 					
180 					if(this.node.peerReview != null) {
181 						//this is a peer review of a previous step
182 
183 						if(this.node.peerReview == 'annotate' && !this.showAuthorContent) {
184 							//send the annotation for the peer review
185 							this.postAnnotation(response);
186 						}
187 					}
188 				}
189 			}
190 
191 			if(this.node.peerReview == 'revise' || this.node.teacherReview == 'revise') {
192 				/*
193 				 * if the node is a peer/teacher review revise step we will set completed
194 				 * to true because revise steps do not save and lock
195 				 */
196 				
197 				//tell the node that the student has completed it
198 				this.node.setCompleted();
199 			}
200 			
201 			//fire the event to push this state to the global view.states object
202 			eventManager.fire('pushStudentWork', orState);
203 
204 			//push the state object into this or object's own copy of states
205 			this.states.push(orState);
206 		};
207 
208 		//turn the save button off
209 		this.setSaveUnavailable();
210 	}
211 };

213 /**
214  * Send the response back as an annotation. Used for Peer Review. 
215  * @param response
216  */
217 OPENRESPONSE.prototype.postAnnotation = function(response) {
218 	//obtain the parameters needed to post the annotation
219 	var runId = this.view.getConfig().getConfigParam('runId');
220 	var nodeId = this.otherStudentNodeVisit.nodeId;
221 	var toWorkgroup = this.otherStudentWorkgroupId;
222 	var fromWorkgroup = this.view.getUserAndClassInfo().getWorkgroupId();
223 	var type = "comment";
224 	var value = response;
225 	var stepWorkId = this.otherStudentStepWorkId;
226 	var action = "peerReviewAnnotate";
227 	var periodId = this.view.getUserAndClassInfo().getPeriodId();
228 	
229 	//get the url
230 	var postAnnotationsUrl = this.view.getConfig().getConfigParam('postAnnotationsUrl');
231 	
232 	//compile the args into an object for cleanliness
233 	var postAnnotationsUrlArgs = {runId:runId,
234 								  nodeId: nodeId,
235 								  toWorkgroup:toWorkgroup,
236 								  fromWorkgroup:fromWorkgroup,
237 								  annotationType:type,
238 								  value:encodeURIComponent(value),
239 								  stepWorkId: stepWorkId,
240 								  action:action,
241 								  periodId:periodId};
242 	
243 	//create the view's annotations object if it does not exist
244 	if(this.view.annotations == null) {
245 		this.view.annotations = new Annotations();
246 	}
247 	
248 	//create the annotation locally to keep our local copy up to date
249 	var annotation = new Annotation(runId, nodeId, toWorkgroup, fromWorkgroup, type, value, stepWorkId);
250 	
251 	//add the annotation to the view's annotations
252 	this.view.annotations.updateOrAddAnnotation(annotation);
253 	
254 	//a callback function that does nothing
255 	var postAnnotationsCallback = function(text, xml, args) {};
256 	
257 	//post the annotation to the server
258 	this.view.connectionManager.request('POST', 1, postAnnotationsUrl, postAnnotationsUrlArgs, postAnnotationsCallback);
259 };

334 /**
335  * Determines whether the student has changed their response
336  * by comparing the previous state with the current state.
337  * @return whether the student changed their response
338  */
339 OPENRESPONSE.prototype.isResponseChanged = function() {
340 	//obtain the previous state
341 	var previousState = this.states[this.states.length - 1];
342 	
343 	var previousResponse = "";
344 	
345 	if(previousState != null) {
346 		previousResponse = previousState.response;
347 	}
348 	
349 	var currentResponse = this.getResponse();
350 	
351 	//check if there were any changes
352 	if(previousResponse != currentResponse) {
353 		//there were changes
354 		return true;
355 	} else {
356 		//there were no changes
357 		return false;
358 	}
359 };

1284 /**
1285  * Replace \n with <br> so that the new lines are displayed to the
1286  * students
1287  * @param response an array containing the response string or just
1288  * the response string
1289  * @return the response string with all \n replaced with <br>
1290  */
1291 OPENRESPONSE.prototype.replaceSlashNWithBR = function(response) {
1292 	var responseString = '';
1293 	
1294 	//check if the response is an array
1295 	if(response.constructor.toString().indexOf('Array') != -1) {
1296 		//the response is an array so we will obtain the string that is in it
1297 		responseString = response[0];
1298 	} else {
1299 		//the response is a string
1300 		responseString = response;
1301 	}
1302 	
1303 	//replace \n with <br>
1304 	return responseString.replace(/\n/g, '<br>');
1305 };
