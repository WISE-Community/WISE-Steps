321 /**
322  * Generates the starter sentence input options for this open response
323  */
324 View.prototype.SensorNode.generateStarter = function(){
325 	//create div for starterSentence options
326 	var starterDiv = createElement(document, 'div', {id: 'starterDiv'});
327 	
328 	//create starter sentence options
329 	var noStarterInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio', onclick: 'eventManager.fire("openResponseStarterOptionChanged")', value: '0'});
330 	var starterOnClickInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio', onclick: 'eventManager.fire("openResponseStarterOptionChanged")', value: '1'});
331 	var starterImmediatelyInput = createElement(document, 'input', {type: 'radio', name: 'starterRadio', onclick: 'eventManager.fire("openResponseStarterOptionChanged")', value: '2'});
332 	var starterSentenceInput = createElement(document, 'textarea', {id: 'starterSentenceInput', cols: '60', rows: '4', wrap: 'soft', onchange: 'eventManager.fire("openResponseStarterSentenceUpdated")'});
333 	var noStarterInputText = document.createTextNode('Do not use starter sentence');
334 	var starterOnClickInputText = document.createTextNode('Starter sentence available upon request');
335 	var starterImmediatelyInputText = document.createTextNode('Starter sentence shows immediately');
336 	var starterSentenceText = document.createTextNode('Starter sentence: ');
337 	
338 	starterDiv.appendChild(noStarterInput);
339 	starterDiv.appendChild(noStarterInputText);
340 	starterDiv.appendChild(createBreak());
341 	starterDiv.appendChild(starterOnClickInput);
342 	starterDiv.appendChild(starterOnClickInputText);
343 	starterDiv.appendChild(createBreak());
344 	starterDiv.appendChild(starterImmediatelyInput);
345 	starterDiv.appendChild(starterImmediatelyInputText);
346 	starterDiv.appendChild(createBreak());
347 	starterDiv.appendChild(starterSentenceText);
348 	starterDiv.appendChild(starterSentenceInput);
349 	
350 	//set value of textarea
351 	starterSentenceInput.value = this.content.starterSentence.sentence;
352 	
353 	//set appropriate radio button and enable/disable textarea
354 	var displayOption = this.content.starterSentence.display;
355 	
356 	if(displayOption=='0'){
357 		starterSentenceInput.disabled = true;
358 		noStarterInput.checked = true;
359 	} else if(displayOption=='1'){
360 		starterOnClickInput.checked = true;
361 	} else if(displayOption=='2'){
362 		starterImmediatelyInput.checked = true;
363 	};
364 	
365 	return starterDiv;
366 };

447 /**
448  * Updates this content when the starter sentence option has changed.
449  */
450 View.prototype.SensorNode.starterChanged = function(){
451 	var options = document.getElementsByName('starterRadio');
452 	var optionVal;
453 	
454 	/* get the checked option and update the content's starter sentence attribute */
455 	for(var q=0;q<options.length;q++){
456 		if(options[q].checked){
457 			this.content.starterSentence.display = options[q].value;
458 			if(options[q].value=='0'){
459 				document.getElementById('starterSentenceInput').disabled = true;
460 			} else {
461 				document.getElementById('starterSentenceInput').disabled = false;
462 			};
463 		};
464 	};
465 	
466 	/* fire source updated event */
467 	this.view.eventManager.fire('sourceUpdated');
468 };

636

640 		$('#showAccelerationCheckBox').attr('disabled', false);
641 	} else {

645 		
646 		//uncheck the checkboxes
647 		$('#showVelocityCheckBox').attr('checked', false);
648 		$('#showAccelerationCheckBox').attr('checked', false);
649 		
650 		//set the fields to false in the content
651 		this.content.showVelocity = false;
652 		this.content.showAcceleration = false;
653 	}
654 	
655 	//fire source updated event
656 	this.view.eventManager.fire('sourceUpdated');
657 };
