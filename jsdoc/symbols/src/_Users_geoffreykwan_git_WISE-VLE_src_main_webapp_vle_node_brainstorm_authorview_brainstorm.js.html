  9 /**
 10  * Generates the authoring page for brainstorm nodes
 11  */
 12 View.prototype.BrainstormNode.generatePage = function(view){
 13 	this.view = view;
 14 	this.content = this.view.activeContent.getContentJSON();
 15 	this.currentResponse;
 16 	this.nodeUtils = new Node().utils;
 17 	
 18 	var parent = document.getElementById('dynamicParent');
 19 	
 20 	//wipe out old elements
 21 	parent.removeChild(document.getElementById('dynamicPage'));
 22 	
 23 	//create new elements
 24 	var pageDiv = createElement(document, 'div', {id:'dynamicPage', style:'width:100%;height:100%'});
 25 	var optionsDiv = createElement(document, 'div', {id:'optionsDiv'});
 26 	
 27 	parent.appendChild(pageDiv);
 28 	pageDiv.appendChild(optionsDiv);
 29 	
 30 	this.generateOptions();
 31 	pageDiv.appendChild(createElement(document, 'div', {id: 'studentResponseBoxSizeContainer'}));
 32 	pageDiv.appendChild(createElement(document, 'div', {id: 'starterSentenceAuthoringContainer'}));
 33 	pageDiv.appendChild(document.createTextNode("Edit/Enter prompt"));
 34 	pageDiv.appendChild(createElement(document, 'div', {id:'promptContainer'}));
 35 	pageDiv.appendChild(createElement(document, 'div', {id:'cannedResponsesDiv'}));
 36 	this.generateCannedResponses();
 37 };

224 /**
225  * Updates this content and preview when the starter sentence option has changed.
226  */
227 View.prototype.BrainstormNode.starterChanged = function(){
228 	var options = document.getElementsByName('starterRadio');
229 	
230 	for(var q=0;q<options.length;q++){
231 		if(options[q].checked){
232 			this.content.starterSentence.display = options[q].value;
233 			if(options[q].value=='0'){
234 				document.getElementById('starterSentenceInput').disabled = true;
235 			} else {
236 				document.getElementById('starterSentenceInput').disabled = false;
237 			};
238 		};
239 	};
240 	
241 	/* fire source updated event */
242 	this.view.eventManager.fire('sourceUpdated');
243 };

257 /**
258  * Generates the cannedResponses element and removes previous
259  */
260 View.prototype.BrainstormNode.generateCannedResponses = function(){
261 	var parent = document.getElementById('cannedResponsesDiv');
262 	
263 	while(parent.firstChild){
264 		parent.removeChild(parent.firstChild);
265 	};
266 	
267 	var createButton = createElement(document, 'input', {type: 'button', 'class':'button', value: 'Create New Response', onclick: 'eventManager.fire("brainstormCreateNewResponse")'});
268 	var removeButton = createElement(document, 'input', {type: 'button', 'class':'button', value: 'Remove Response', onclick: 'eventManager.fire("brainstormRemoveResponse")'});
269 	var responses = this.content.cannedResponses;
270 	
271 	for(var t=0;t<responses.length;t++){
272 		/* if this is an older response, we will need to set an identifier */
273 		if(!responses[t].identifier){
274 			responses[t].identifier = this.nodeUtils.generateKey();
275 		}
276 		
277 		var responseTitleText = document.createTextNode('Name: ');
278 		var responseTitleInput = createElement(document, 'input', {type: 'text', id: 'responseInput_' + responses[t].identifier, value: responses[t].name, onkeyup: 'eventManager.fire("brainstormResponseNameChanged","' + responses[t].identifier + '")', onclick: 'eventManager.fire("brainstormResponseSelected","' + responses[t].identifier + '")'});
279 		var responseValueText = document.createTextNode('Response: ');
280 		var responseValueInput = createElement(document, 'textarea', {id: 'responseValue_' + responses[t].identifier, onkeyup: 'eventManager.fire("brainstormResponseValueChanged","' + responses[t].identifier + '")', onclick: 'eventManager.fire("brainstormResponseSelected","' + responses[t].identifier + '")'});
281 		responseValueInput.value = responses[t].response;
282 		
283 		parent.appendChild(createBreak());
284 		parent.appendChild(responseTitleText);
285 		parent.appendChild(responseTitleInput);
286 		parent.appendChild(createBreak());
287 		parent.appendChild(responseValueText);
288 		parent.appendChild(responseValueInput);
289 	};
290 	
291 	parent.appendChild(createBreak());
292 	parent.appendChild(createButton);
293 	parent.appendChild(removeButton);
294 };

421 /**
422  * Removes the currently selected response, if none, notifies author
423  */
424 View.prototype.BrainstormNode.removeResponse = function(){
425 	if(this.currentResponse){
426 		this.content.cannedResponses.splice(this.content.cannedResponses.indexOf(this.currentResponse), 1);
427 		this.generateCannedResponses();
428 		
429 		/* fire source updated event */
430 		this.view.eventManager.fire('sourceUpdated');
431 	} else {
432 		this.view.notificationManager.notify('No response is selected for removal. Please select a response and try again.', 3);
433 	};
434 };

487 View.prototype.BrainstormNode.populateStarterSentenceAuthoring = function() {
488 	var displayOption = this.content.starterSentence.display;
489 	
490 	$('input[name=starterRadio]').each(function() {
491 		if($(this).val() == displayOption) {
492 			$(this).attr('checked', true);
493 		}
494 	});
495 	
496 	if(displayOption == 2) {
497 		$('#starterSentenceAuthoringInput').val(this.content.starterSentence.sentence);		
498 	}
