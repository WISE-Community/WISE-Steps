130 /**
131  * Renders the student work into the div. The grading tool will pass in a
132  * div id to this function and this function will insert the student data
133  * into the div.
134  * 
135  * @param divId the id of the div we will render the student work into
136  * @param nodeVisit the student work
137  * @param childDivIdPrefix (optional) a string that will be prepended to all the 
138  * div ids use this to prevent DOM conflicts such as when the show all work div
139  * uses the same ids as the show flagged work div
140  * @param workgroupId the id of the workgroup this work belongs to
141  * 
142  * TODO: rename TemplateNode
143  * Note: you may need to add code to this function if the student
144  * data for your step is complex or requires additional processing.
145  * look at SensorNode.renderGradingView() as an example of a step that
146  * requires additional processing
147  */
148 TemplateNode.prototype.renderGradingView = function(divId, nodeVisit, childDivIdPrefix, workgroupId) {
149 	/*
150 	 * Get the latest student state object for this step
151 	 * TODO: rename templateState to reflect your new step type
152 	 * 
153 	 * e.g. if you are creating a quiz step you would change it to quizState
154 	 */
155 	var templateState = nodeVisit.getLatestWork();
156 	
157 	/*
158 	 * get the step work id from the node visit in case we need to use it in
159 	 * a DOM id. we don't use it in this case but I have retrieved it in case
160 	 * someone does need it. look at SensorNode.js to view an example of
161 	 * how one might use it.
162 	 */
163 	var stepWorkId = nodeVisit.id;
164 	
165 	/*
166 	 * TODO: rename templateState to match the variable name you
167 	 * changed in the previous line above
168 	 */
169 	var studentWork = templateState.getStudentWork();
170 	
171 	//put the student work into the div
172 	$('#' + divId).html(studentWork.response);
173 };
