1 /**
  2  * MultipleChoiceQueryEntry
  3  * 
  4  * An object that contains student work for a multiple choice node
  5  * for a specific student.
  6  */
  7 
  8 MultipleChoiceQueryEntry.prototype = new QueryEntry();
  9 MultipleChoiceQueryEntry.prototype.constructor = MultipleChoiceQueryEntry;
 10 MultipleChoiceQueryEntry.prototype.parent = QueryEntry.prototype;
 11 
 12 /**
 13  * Constructor
 14  * @param dataId the dataId for a specific student
 15  * @param nodeId the nodeId for a specific node
 16  * @param prompt the prompt for this multiple choice node
 17  * @param choiceId the choiceId for this query entry
 18  * @param choiceValue the user displayed value for the query entry
 19  */
 20 function MultipleChoiceQueryEntry(dataId, userName, nodeId, prompt, choiceId, choiceValue) {
 21 	this.dataId = dataId;
 22 	this.userName = userName;
 23 	this.nodeId = nodeId;
 24 	this.prompt = prompt;
 25 	this.choiceId = choiceId;
 26 	this.choiceValue = choiceValue;
 27 }
 28 
 29 /**
 30  * Gets html representation of the student's work
 31  * @return html that displays the student's work
 32  */
 33 MultipleChoiceQueryEntry.prototype.printEntry = function() {
 34 	return "[" + this.userName + "] answered " + this.choiceValue;
 35 }
 36 
 37 /**
 38  * Gets html representation of the student's work for "View Work By Student".
 39  * @return an html string that displays the prompt and the choice the student chose
 40  */
 41 MultipleChoiceQueryEntry.prototype.printStudentEntry = function() {
 42 	var print = "";
 43 	
 44 	print += this.prompt;
 45 	print += "<br>";
 46 	
 47 	print += this.choiceValue;
 48 	print += "<br>";
 49 	
 50 	return print;
 51 }
