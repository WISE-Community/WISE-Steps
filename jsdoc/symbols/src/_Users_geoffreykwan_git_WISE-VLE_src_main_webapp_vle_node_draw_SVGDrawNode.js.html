 38 /**
 39  * Returns the base64 encoded svgstring of the latest work
 40  * @param studentWork json
 41  * @return base64 encoded svgstring
 42  */
 43 SVGDrawNode.prototype.translateStudentWork = function(studentWork) {
 44 	// if the student data has been compressed, decompress it
 45 	if(typeof studentWork == "string"){
 46 		if (studentWork.match(/^--lz77--/)) {
 47 			var lz77 = new LZ77();
 48 			studentWork = studentWork.replace(/^--lz77--/, "");
 49 			studentWork = $.parseJSON(lz77.decompress(studentWork));
 50 		}
 51 	}
 52 	
 53 	var svgString = studentWork.svgString;
 54 	// if the svg has been compressed, decompress it
 55 	if (svgString.match(/^--lz77--/)) {
 56 		var lz77 = new LZ77();
 57 		svgString = svgString.replace(/^--lz77--/, "");
 58 		svgString = lz77.decompress(svgString);
 59 	}
 60 	svgString = Utils.encode64(svgString);
 61 	return svgString;
 62 	
 63 };

 75 /**
 76  * Returns a string of the work so that it can be imported by the specified exportToNode
 77  * @param exportToNode node that will import the return value of this method
 78  * @return null if this node cannot export work to the exportToNode
 79  */
 80 SVGDrawNode.prototype.exportWork = function(exportToNode) {	
 81 	if (exportToNode.type == "SVGDrawNode") {
 82 	    var nodeVisitArray = this.view.state.getNodeVisitsByNodeId(this.id);
 83 	    if (nodeVisitArray.length > 0) {
 84 	        var states = [];
 85 	        var latestNodeVisit = nodeVisitArray[nodeVisitArray.length -1];
 86 	        for (var i = 0; i < nodeVisitArray.length; i++) {
 87 	            var nodeVisit = nodeVisitArray[i];
 88 	            for (var j = 0; j < nodeVisit.nodeStates.length; j++) {
 89 	                states.push(nodeVisit.nodeStates[j]);
 90 	            }
 91 	        }
 92 	        var latestState = states[states.length - 1];
 93 	        var studentWork = latestState.getStudentWork();
 94 	        
 95 	        studentWork = Utils.decode64(this.translateStudentWork(studentWork));
 96 	        
 97 	        // only get the studentlayer. remove newlines and regex for what's inside <title>student</title>...</g>
 98 	        studentWork = studentWork.replace(/[\n\r\t]/g,"").match("<title>student</title>(.*)</g>")[1];
 99 	        
100 			return studentWork;
101 	    };			
102 	};
103 	return null;
104 };

106 /**
107  * Imports and inserts the work from the specified importFromNode
108  * @param importFromNode node that will export the data for this node to import
109  */
110 SVGDrawNode.prototype.importWork = function(importFromNode) {
111 	var studentWorkSVG = importFromNode.exportWork(this);
112 	if (studentWorkSVG != null) {
113 		var svgStringBefore = this.contentPanel.svgCanvas.getSvgString();
114 		var svgStringAfter = svgStringBefore.replace("<title>student</title>", "<title>student</title>" + studentWorkSVG);
115 		this.contentPanel.svgCanvas.setSvgString(svgStringAfter);
116 	}
117 };

119 /**
120  * Returns true iff the given file can be imported 
121  * into this step's work.
122  */
123 SVGDrawNode.prototype.canImportFile = function(filename) {
124 	if (filename.indexOf(".") != -1) {
125 		var fileExt = filename.substr(filename.lastIndexOf(".")+1);	
126 		if (this.importableFileExtensions.indexOf(fileExt.toLowerCase()) != -1) {
127 			return true;
128 		}
129 	}
130 	return false;
131 };

133 /**
134  * Imports and inserts the specified file into current drawing.
135  * @param file to insert into current canvas
136  * @return true if import is successful
137  */
138 SVGDrawNode.prototype.importFile = function(filename) {
139 	if (this.canImportFile(filename)) {
140 		var importFileSVG = '<image x="250" y="150" height="150" width="150" xlink:href="'+filename+'" />';
141 		var svgStringBefore = this.contentPanel.svgCanvas.getSvgString();
142 		var svgStringAfter = svgStringBefore.replace("<title>student</title>", "<title>student</title>" + importFileSVG);
143 			
144 		// xmlns:xlink="http://www.w3.org/1999/xlink" <- make sure this is in xml namespace.
145 		if (svgStringAfter.indexOf('xmlns:xlink="http://www.w3.org/1999/xlink"') == -1) {
146 			svgStringAfter = svgStringAfter.replace("<svg ", "<svg " + 'xmlns:xlink="http://www.w3.org/1999/xlink" ');
147 		}
148 
149 		this.contentPanel.svgCanvas.setSvgString(svgStringAfter);
150 		return true;
151 	}
152 	return false;
153 };
