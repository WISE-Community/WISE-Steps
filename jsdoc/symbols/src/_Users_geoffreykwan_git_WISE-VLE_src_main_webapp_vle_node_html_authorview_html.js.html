 40 /**
 41  * Saves the html content to its file
 42  */
 43 View.prototype.HtmlNode.save = function(close){
 44 	var success = function(t,x,o){
 45 		o.stepSaved = true;
 46 		o.notificationManager.notify('Updated html content on server', 3);
 47 		o.eventManager.fire('setLastEdited');
 48 		if(close){
 49 			o.eventManager.fire('closeOnStepSaved', [true]);
 50 		};
 51 	};
 52 	
 53 	var failure = function(t,x,o){
 54 		o.notificationManager.notify('Failed update of html content on server', 3);
 55 		if(close){
 56 			o.eventManager.fire('closeOnStepSaved', [false]);
 57 		};
 58 	};
 59 	
 60 	// save .ht file and .html file
 61 	/* get json content as string */
 62 	var contentString = encodeURIComponent($.stringify(this.view.activeContent.getContentJSON(),null,3));
 63 
 64 	this.view.connectionManager.request('POST', 3, this.view.requestUrl, {forward:'filemanager', projectId:this.view.portalProjectId, command:'updateFile', fileName:this.view.activeNode.content.getFilename(this.view.getProject().getContentBase()), data:contentString}, success, this.view, failure);
 65 	this.view.connectionManager.request('POST', 3, this.view.requestUrl, {forward:'filemanager', projectId:this.view.portalProjectId, command:'updateFile', fileName:this.view.activeNode.baseHtmlContent.getFilename(this.view.getProject().getContentBase()), data:encodeURIComponent(document.getElementById('promptInput').value)}, success, this.view, failure);
 66 };
